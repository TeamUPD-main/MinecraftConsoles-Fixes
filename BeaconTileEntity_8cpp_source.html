68void BeaconTileEntity::applyEffects()
69{
70 if (isActive && levels > 0 && !level->isClientSide && primaryPower > 0)
71 {
72
73 double range = (levels * 10) + 10;
74 int baseAmp = 0;
75 if (levels >= 4 && primaryPower == secondaryPower)
76 {
77 baseAmp = 1;
78 }
79
80 AABB *bb = AABB::newTemp(x, y, z, x + 1, y + 1, z + 1)->grow(range, range, range);
81 bb->y1 = level->getMaxBuildHeight();
82 vector<shared_ptr<Entity> > *players = level->getEntitiesOfClass(typeid(Player), bb);
83 if ( players )
84 {
85 for (auto& it : *players)
86 {
87 shared_ptr<Player> player = dynamic_pointer_cast<Player>(it);
88 if ( player )
89 player->addEffect(new MobEffectInstance(primaryPower, SharedConstants::TICKS_PER_SECOND * 9, baseAmp, true));
90 }
91
92 if (levels >= 4 && primaryPower != secondaryPower && secondaryPower > 0)
93 {
94 for ( auto& it : *players )
95 {
96 shared_ptr<Player> player = dynamic_pointer_cast<Player>(it);
97 if ( player )
98 player->addEffect(new MobEffectInstance(secondaryPower, SharedConstants::TICKS_PER_SECOND * 9, 0, true));
99 }
100 }
101 delete players;
102 }
103 }

106void BeaconTileEntity::updateShape()
107{
108
109 if (!level->canSeeSky(x, y + 1, z))
110 {
111 isActive = false;
112 levels = 0;
113 }
114 else
115 {
116 isActive = true;
117
118 levels = 0;
119 for (int step = 1; step <= 4; step++)
120 {
121
122 int ly = y - step;
123 if (ly < 0)
124 {
125 break;
126 }
127
128 bool isOk = true;
129 for (int lx = x - step; lx <= x + step && isOk; lx++)
130 {
131 for (int lz = z - step; lz <= z + step; lz++)
132 {
133 int tile = level->getTile(lx, ly, lz);
134 if (tile != Tile::emeraldBlock_Id && tile != Tile::goldBlock_Id && tile != Tile::diamondBlock_Id && tile != Tile::ironBlock_Id)
135 {
136 isOk = false;
137 break;
138 }
139 }
140 }
141 if (isOk)
142 {
143 levels = step;
144 }
145 else
146 {
147 break;
148 }
149 }
150 if (levels == 0)
151 {
152 isActive = false;
153 }
154 }

158float BeaconTileEntity::getAndUpdateClientSideScale()
159{
160
161 if (!isActive)
162 {
163 return 0;
164 }
165
166 int renderDelta = static_cast<int>(level->getGameTime() - clientSideRenderTick);
167 clientSideRenderTick = level->getGameTime();
168 if (renderDelta > 1)
169 {
170 clientSideRenderScale -= (static_cast<float>(renderDelta) / static_cast<float>(SCALE_TIME));
171
172 if (clientSideRenderScale < 0)
173 {
174 clientSideRenderScale = 0;
175 }
176 }
177 clientSideRenderScale += (1.0f / static_cast<float>(SCALE_TIME));
178 if (clientSideRenderScale > 1)
179 {
180 clientSideRenderScale = 1;
181 }
182 return clientSideRenderScale;
183}
