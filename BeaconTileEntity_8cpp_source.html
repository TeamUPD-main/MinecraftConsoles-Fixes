106void BeaconTileEntity::updateShape()
107{
108
109 if (!level->canSeeSky(x, y + 1, z))
110 {
111 isActive = false;
112 levels = 0;
113 }
114 else
115 {
116 isActive = true;
117
118 levels = 0;
119 for (int step = 1; step <= 4; step++)
120 {
121
122 int ly = y - step;
123 if (ly < 0)
124 {
125 break;
126 }
127
128 bool isOk = true;
129 for (int lx = x - step; lx <= x + step && isOk; lx++)
130 {
131 for (int lz = z - step; lz <= z + step; lz++)
132 {
133 int tile = level->getTile(lx, ly, lz);
134 if (tile != Tile::emeraldBlock_Id && tile != Tile::goldBlock_Id && tile != Tile::diamondBlock_Id && tile != Tile::ironBlock_Id)
135 {
136 isOk = false;
137 break;
138 }
139 }
140 }
141 if (isOk)
142 {
143 levels = step;
144 }
145 else
146 {
147 break;
148 }
149 }
150 if (levels == 0)
151 {
152 isActive = false;
153 }
154 }
