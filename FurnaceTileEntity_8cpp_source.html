188void FurnaceTileEntity::tick()
189{
190 bool wasLit = litTime > 0;
191 bool changed = false;
192 if (litTime > 0)
193 {
194 litTime--;
195 }
196
197 if ( level != nullptr && !level->isClientSide)
198 {
199 if (litTime == 0 && canBurn())
200 {
201 litDuration = litTime = getBurnDuration(items[SLOT_FUEL]);
202 if (litTime > 0)
203 {
204 changed = true;
205 if (items[SLOT_FUEL] != nullptr)
206 {
207 // 4J Added: Keep track of whether charcoal was used in production of current stack.
208 if ( items[SLOT_FUEL]->getItem()->id == Item::coal_Id
209 && items[SLOT_FUEL]->getAuxValue() == CoalItem::CHAR_COAL)
210 {
211 m_charcoalUsed = true;
212 }
213
214 items[SLOT_FUEL]->count--;
215 if (items[SLOT_FUEL]->count == 0)
216 {
217 Item *remaining = items[SLOT_FUEL]->getItem()->getCraftingRemainingItem();
218 items[SLOT_FUEL] = remaining != nullptr ? std::make_shared<ItemInstance>(remaining) : nullptr;
219 }
220 }
221 }
222 }
223
224 if (isLit() && canBurn())
225 {
226 tickCount++;
227 if (tickCount == BURN_INTERVAL)
228 {
229 tickCount = 0;
230 burn();
231 changed = true;
232 }
233 }
234 else
235 {
236 tickCount = 0;
237 }
238
239 if (wasLit != litTime > 0)
240 {
241 changed = true;
242 FurnaceTile::setLit(litTime > 0, level, x, y, z);
243 }
244 }
245
246 if (changed) setChanged();
247}

251{
252 if (items[SLOT_INPUT] == nullptr) return false;
253 const ItemInstance *burnResult = FurnaceRecipes::getInstance()->getResult(items[SLOT_INPUT]->getItem()->id);
254 if (burnResult == nullptr) return false;
255 if (items[SLOT_RESULT] == nullptr) return true;

257 if (items[SLOT_RESULT]->count < getMaxStackSize() && items[SLOT_RESULT]->count < items[SLOT_RESULT]->getMaxStackSize()) return true;
258 if (items[SLOT_RESULT]->count < burnResult->getMaxStackSize()) return true;

263void FurnaceTileEntity::burn()
264{
265 if (!canBurn()) return;
266
267 const ItemInstance *result = FurnaceRecipes::getInstance()->getResult(items[SLOT_INPUT]->getItem()->id);
268 if (items[SLOT_RESULT] == nullptr) items[SLOT_RESULT] = result->copy();
269 else if (items[SLOT_RESULT]->id == result->id) items[SLOT_RESULT]->count++;
270
271 items[SLOT_INPUT]->count--;

276int FurnaceTileEntity::getBurnDuration(shared_ptr<ItemInstance> itemInstance)
277{
278 if (itemInstance == nullptr) return 0;
279 int id = itemInstance->getItem()->id;
280
281 Item *item = itemInstance->getItem();
282
283 if (id < 256 && Tile::tiles[id] != nullptr)
284 {
285 Tile *tile = Tile::tiles[id];
286
287 if (tile == Tile::woodSlabHalf)
288 {
289 return BURN_INTERVAL * 3 / 4;
290 }
291
292 if (tile->material == Material::wood)
293 {
294 return BURN_INTERVAL * 3 / 2;
295 }
296
297 if (tile == Tile::coalBlock)
298 {
299 return BURN_INTERVAL * 8 * 10;
300 }
301 }
302
303 if (dynamic_cast<DiggerItem *>(item) && static_cast<DiggerItem *>(item)->getTier() == Item::Tier::WOOD)
304 {
305 return BURN_INTERVAL;
306 }
307 else if (dynamic_cast<WeaponItem *>(item) && static_cast<WeaponItem *>(item)->getTier() == Item::Tier::WOOD)
308 {
309 return BURN_INTERVAL;
310 }
311 else if (dynamic_cast<HoeItem *>(item) && static_cast<HoeItem *>(item)->getTier() == Item::Tier::WOOD)
312 {
313 return BURN_INTERVAL;
314 }
315
316 if (id == Item::stick->id)
317 {
318 return BURN_INTERVAL / 2;
319 }
320
321 if (id == Item::coal->id) return BURN_INTERVAL * 8;
322
323 if (id == Item::bucket_lava->id) return BURN_INTERVAL * 100;
324
325 if (id == Tile::sapling_Id) return BURN_INTERVAL / 2;
326
327 if (id == Item::blazeRod_Id) return BURN_INTERVAL * 12;
328
329 return 0;
330}
