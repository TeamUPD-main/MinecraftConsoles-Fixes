135void TrapDoorTile::neighborChanged(Level *level, int x, int y, int z, int type)
136{
137 if (level->isClientSide) return;
138
139 int data = level->getData(x, y, z);
140 int xt = x;
141 int zt = z;
142 if ((data & 3) == 0) zt++;
143 if ((data & 3) == 1) zt--;

145 if ((data & 3) == 3) xt--;
146
147
148 if (!attachesTo(level->getTile(xt, y, zt)))

150 level->removeTile(x, y, z);
