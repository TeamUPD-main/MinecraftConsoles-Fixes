184static bool ParseIntArg(const char *value, int *outValue)
185{
186 if (value == NULL || *value == 0)
187 return false;
188
189 char *end = NULL;
190 long parsed = strtol(value, &end, 10);
191 if (end == value || *end != 0)
192 return false;
193
194 *outValue = (int)parsed;
195 return true;
196}
197
198static bool ParseInt64Arg(const char *value, __int64 *outValue)
199{
200 if (value == NULL || *value == 0)
201 return false;
202
203 char *end = NULL;
204 __int64 parsed = _strtoi64(value, &end, 10);
205 if (end == value || *end != 0)
206 return false;
207
208 *outValue = parsed;
209 return true;
210}

212static bool ParseCommandLine(int argc, char **argv, DedicatedServerConfig *config)
213{
214 for (int i = 1; i < argc; ++i)
215 {
216 const char *arg = argv[i];
217 if (_stricmp(arg, "-help") == 0 || _stricmp(arg, "--help") == 0 || _stricmp(arg, "-h") == 0)
218 {
219 config->showHelp = true;
220 return true;
221 }
222 else if ((_stricmp(arg, "-port") == 0) && (i + 1 < argc))
223 {
224 int port = 0;
225 if (!ParseIntArg(argv[++i], &port) || port <= 0 || port > 65535)
226 {
227 LogError("startup", "Invalid -port value.");
228 return false;
229 }
230 config->port = port;
231 }
232 else if ((_stricmp(arg, "-ip") == 0 || _stricmp(arg, "-bind") == 0) && (i + 1 < argc))
233 {
234 strncpy_s(config->bindIP, sizeof(config->bindIP), argv[++i], _TRUNCATE);
235 }
236 else if ((_stricmp(arg, "-name") == 0) && (i + 1 < argc))
237 {
238 strncpy_s(config->name, sizeof(config->name), argv[++i], _TRUNCATE);
239 }
240 else if ((_stricmp(arg, "-maxplayers") == 0) && (i + 1 < argc))
241 {
242 int maxPlayers = 0;
243 if (!ParseIntArg(argv[++i], &maxPlayers) || maxPlayers <= 0 || maxPlayers > MINECRAFT_NET_MAX_PLAYERS)
244 {
245 LogError("startup", "Invalid -maxplayers value.");
246 return false;
247 }
248 config->maxPlayers = maxPlayers;
249 }
250 else if ((_stricmp(arg, "-seed") == 0) && (i + 1 < argc))
251 {
252 if (!ParseInt64Arg(argv[++i], &config->seed))
253 {
254 LogError("startup", "Invalid -seed value.");
255 return false;
256 }
257 config->hasSeed = true;
258 }
259 else if ((_stricmp(arg, "-loglevel") == 0) && (i + 1 < argc))
260 {
261 if (!TryParseServerLogLevel(argv[++i], &config->logLevel))
262 {
263 LogError("startup", "Invalid -loglevel value. Use debug/info/warn/error.");
264 return false;
265 }
266 }
267 else
268 {
269 LogErrorf("startup", "Unknown or incomplete argument: %s", arg);
270 return false;
271 }
272 }
273
274 return true;
275}
