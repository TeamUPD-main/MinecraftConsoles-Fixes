1// gdraw_d3d10.cpp - author: Fabian Giesen - copyright 2011 RAD Game Tools
2//
3// This implements the Iggy graphics driver layer for D3D 10.
4
5// GDraw consists of several components that interact fairly loosely with each other;
6// e.g. the resource management, drawing and filtering parts are all fairly independent
7// of each other. If you want to modify some aspect of GDraw - say the texture allocation
8// logic - your best bet is usually to just look for one of the related entry points,
9// e.g. MakeTextureBegin, and take it from there. There's a bunch of code in this file,
10// but none of it is really complicated.
11//
12// The one bit you might want to change that's not that localized is to integrate
13// GDraw with an existing state caching system. The following bits all modify D3D state
14// in some way:
15// - The rendering helpers (set_viewport_raw, set_projection_raw, set_*_renderstate)
16// - RenderTile*/TextureDrawBuffer* may change the active rendertarget and depth/stencil surface,
17// as do D3D1X_(NoMoreGDrawThisFrame) and set_render_target
18// - set_texture
19// - set_renderstate and set_renderstate_full. These are the main places where render state changes occur;
20// you should probably start here.
21// - DrawIndexedTriangles sets the active vertex/index buffers and vertex declaration
22// - Most of the functions in the "filter effects" section modify D3D state, mostly
23// pixel shader constants and textures

83static ID3D10BlendState *create_blend_state(ID3D10Device *dev, BOOL blend, D3D10_BLEND src, D3D10_BLEND dst)
84{
85 D3D10_BLEND_DESC desc = {};
86 desc.BlendEnable[0] = blend;
87 desc.SrcBlend = src;
88 desc.DestBlend = dst;
89 desc.BlendOp = D3D10_BLEND_OP_ADD;
90 desc.SrcBlendAlpha = (src == D3D10_BLEND_DEST_COLOR ) ? D3D10_BLEND_DEST_ALPHA : src;
91 desc.DestBlendAlpha = dst;
92 desc.BlendOpAlpha = D3D10_BLEND_OP_ADD;
93 desc.RenderTargetWriteMask[0] = D3D10_COLOR_WRITE_ENABLE_ALL;
94
95 ID3D10BlendState *res;
96 HRESULT hr = dev->CreateBlendState(&desc, &res);
97 if (FAILED(hr)) {
98 report_d3d_error(hr, "CreateBlendState", "");
99 res = NULL;
100 }
101
102 return res;
103}
