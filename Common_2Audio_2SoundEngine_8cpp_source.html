326void SoundEngine::tick(shared_ptr<Mob> *players, float a)
327{
328 ConsoleSoundEngine::tick();
329
330 // update the listener positions
331 int listenerCount = 0;
332 if( players )
333 {
334 bool bListenerPostionSet = false;
335 for( size_t i = 0; i < MAX_LOCAL_PLAYERS; i++ )
336 {
337 if( players[i] != nullptr )
338 {
339 m_ListenerA[i].bValid=true;
340 F32 x,y,z;
341 x=players[i]->xo + (players[i]->x - players[i]->xo) * a;
342 y=players[i]->yo + (players[i]->y - players[i]->yo) * a;
343 z=players[i]->zo + (players[i]->z - players[i]->zo) * a;
344
345 float yRot = players[i]->yRotO + (players[i]->yRot - players[i]->yRotO) * a;
346 float yCos = (float)cos(yRot * Mth::RAD_TO_GRAD);
347 float ySin = (float)sin(yRot * Mth::RAD_TO_GRAD);
348
349 // store the listener positions for splitscreen
350 m_ListenerA[i].vPosition.x = x;
351 m_ListenerA[i].vPosition.y = y;
352 m_ListenerA[i].vPosition.z = z;
353
354 m_ListenerA[i].vOrientFront.x = -ySin;
355 m_ListenerA[i].vOrientFront.y = 0;
356 m_ListenerA[i].vOrientFront.z = yCos;
357
358 listenerCount++;
359 }
360 else
361 {
362 m_ListenerA[i].bValid=false;
363 }
364 }
365 }
366
367
368 // If there were no valid players set, make up a default listener
369 if( listenerCount == 0 )
370 {
371 m_ListenerA[0].vPosition.x = 0;
372 m_ListenerA[0].vPosition.y = 0;
373 m_ListenerA[0].vPosition.z = 0;
374 m_ListenerA[0].vOrientFront.x = 0;
375 m_ListenerA[0].vOrientFront.y = 0;
376 m_ListenerA[0].vOrientFront.z = 1.0f;
377 listenerCount++;
378 }
379 m_validListenerCount = listenerCount;
380
381#ifdef __PSVITA__
382 // AP - Show that a change has occurred so we know to update the values at the next Mixer callback
383 SoundEngine_Change = true;
384#else

549 MA_SOUND_FLAG_ASYNC,
