277BoundingBox *MineShaftPieces::MineShaftCorridor::findCorridorSize(list<StructurePiece *> *pieces, Random *random, int footX, int footY, int footZ, int direction)
278{
279
280 BoundingBox *box = new BoundingBox(footX, footY, footZ, footX, footY + (DEFAULT_SHAFT_HEIGHT - 1), footZ);
281
282 int corridorLength = random->nextInt(3) + 2;
283 while (corridorLength > 0)
284 {
285 int blockLength = corridorLength * DEFAULT_SHAFT_LENGTH;
286
287 switch (direction)
288 {
289 case Direction::NORTH:
290 box->x1 = footX + (DEFAULT_SHAFT_WIDTH - 1);
291 box->z0 = footZ - (blockLength - 1);
292 break;
293 case Direction::SOUTH:
294 box->x1 = footX + (DEFAULT_SHAFT_WIDTH - 1);
295 box->z1 = footZ + (blockLength - 1);
296 break;
297 case Direction::WEST:
298 box->x0 = footX - (blockLength - 1);
299 box->z1 = footZ + (DEFAULT_SHAFT_WIDTH - 1);
300 break;
301 case Direction::EAST:
302 box->x1 = footX + (blockLength - 1);
303 box->z1 = footZ + (DEFAULT_SHAFT_WIDTH - 1);
304 break;
305 }
306
307 if (StructurePiece::findCollisionPiece(pieces, box) != nullptr)
308 {
309 corridorLength--;
310 }
311 else
312 {
313 break;
314 }
315 }
316
317 if (corridorLength > 0)
318 {
319 return box;
320 }
321 delete box;
322 // unable to place corridor here

506 }
507
508 if (spiderCorridor && !hasPlacedSpider)
509 {
510 int y = getWorldY(y0), newZ = z - 1 + random->nextInt(3);

574BoundingBox *MineShaftPieces::MineShaftCrossing::findCrossing(list<StructurePiece *> *pieces, Random *random, int footX, int footY, int footZ, int direction)
575{
576
577 BoundingBox *box = new BoundingBox(footX, footY, footZ, footX, footY + (DEFAULT_SHAFT_HEIGHT - 1), footZ);
578
579 if (random->nextInt(4) == 0)
580 {
581 box->y1 += DEFAULT_SHAFT_HEIGHT + 1; // two-floored
582 }
583
584 switch (direction)
585 {
586 case Direction::NORTH:
587 box->x0 = footX - 1;
588 box->x1 = footX + DEFAULT_SHAFT_WIDTH;
589 box->z0 = footZ - (DEFAULT_SHAFT_WIDTH + 1);
590 break;
591 case Direction::SOUTH:
592 box->x0 = footX - 1;
593 box->x1 = footX + DEFAULT_SHAFT_WIDTH;
594 box->z1 = footZ + (DEFAULT_SHAFT_WIDTH + 1);
595 break;
596 case Direction::WEST:
597 box->x0 = footX - (DEFAULT_SHAFT_WIDTH + 1);
598 box->z0 = footZ - 1;
599 box->z1 = footZ + DEFAULT_SHAFT_WIDTH;
600 break;
601 case Direction::EAST:
602 box->x1 = footX + (DEFAULT_SHAFT_WIDTH + 1);
603 box->z0 = footZ - 1;
604 box->z1 = footZ + DEFAULT_SHAFT_WIDTH;
605 break;
606 }
607
608 if (StructurePiece::findCollisionPiece(pieces, box) != nullptr)
609 {
610 delete box;
611 return nullptr;
612 }
613
614 return box;

617void MineShaftPieces::MineShaftCrossing::addChildren(StructurePiece *startPiece, list<StructurePiece *> *pieces, Random *random)

657bool MineShaftPieces::MineShaftCrossing::postProcess(Level *level, Random *random, BoundingBox *chunkBB)
658{
659 if (edgesLiquid(level, chunkBB))
660 {
661 return false;
662 }
663
664 // corridor air
665 if (isTwoFloored)
666 {
667 generateBox(level, chunkBB, boundingBox->x0 + 1, boundingBox->y0, boundingBox->z0, boundingBox->x1 - 1, boundingBox->y0 + DEFAULT_SHAFT_HEIGHT - 1, boundingBox->z1, 0, 0, false);
668 generateBox(level, chunkBB, boundingBox->x0, boundingBox->y0, boundingBox->z0 + 1, boundingBox->x1, boundingBox->y0 + DEFAULT_SHAFT_HEIGHT - 1, boundingBox->z1 - 1, 0, 0, false);
669 generateBox(level, chunkBB, boundingBox->x0 + 1, boundingBox->y1 - (DEFAULT_SHAFT_HEIGHT - 1), boundingBox->z0, boundingBox->x1 - 1, boundingBox->y1, boundingBox->z1, 0, 0, false);
670 generateBox(level, chunkBB, boundingBox->x0, boundingBox->y1 - (DEFAULT_SHAFT_HEIGHT - 1), boundingBox->z0 + 1, boundingBox->x1, boundingBox->y1, boundingBox->z1 - 1, 0, 0, false);
671 generateBox(level, chunkBB, boundingBox->x0 + 1, boundingBox->y0 + DEFAULT_SHAFT_HEIGHT, boundingBox->z0 + 1, boundingBox->x1 - 1, boundingBox->y0 + DEFAULT_SHAFT_HEIGHT, boundingBox->z1 - 1, 0, 0, false);
672 }
673 else
674 {
675 generateBox(level, chunkBB, boundingBox->x0 + 1, boundingBox->y0, boundingBox->z0, boundingBox->x1 - 1, boundingBox->y1, boundingBox->z1, 0, 0, false);
676 generateBox(level, chunkBB, boundingBox->x0, boundingBox->y0, boundingBox->z0 + 1, boundingBox->x1, boundingBox->y1, boundingBox->z1 - 1, 0, 0, false);
677 }
678
679 // support pillars
680 generateBox(level, chunkBB, boundingBox->x0 + 1, boundingBox->y0, boundingBox->z0 + 1, boundingBox->x0 + 1, boundingBox->y1, boundingBox->z0 + 1, Tile::wood_Id, 0, false);
681 generateBox(level, chunkBB, boundingBox->x0 + 1, boundingBox->y0, boundingBox->z1 - 1, boundingBox->x0 + 1, boundingBox->y1, boundingBox->z1 - 1, Tile::wood_Id, 0, false);
682 generateBox(level, chunkBB, boundingBox->x1 - 1, boundingBox->y0, boundingBox->z0 + 1, boundingBox->x1 - 1, boundingBox->y1, boundingBox->z0 + 1, Tile::wood_Id, 0, false);
683 generateBox(level, chunkBB, boundingBox->x1 - 1, boundingBox->y0, boundingBox->z1 - 1, boundingBox->x1 - 1, boundingBox->y1, boundingBox->z1 - 1, Tile::wood_Id, 0, false);
684
685 // prevent air floating
686 // note: use world coordinates because the corridor hasn't defined
687 // orientation
688 for (int x = boundingBox->x0; x <= boundingBox->x1; x++)
689 {
690 for (int z = boundingBox->z0; z <= boundingBox->z1; z++)
691 {
692 int block = getBlock(level, x, boundingBox->y0 - 1, z, chunkBB);
693 if (block == 0)
694 {
695 placeBlock(level, Tile::wood_Id, 0, x, boundingBox->y0 - 1, z, chunkBB);
696 }
697 }
698 }
699
700 return true;
701}
