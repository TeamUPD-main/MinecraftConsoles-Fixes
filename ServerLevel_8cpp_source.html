1528int ServerLevel::runUpdate(void* lpParam)
1529{
1530 ShutdownManager::HasStarted(ShutdownManager::eRunUpdateThread,m_updateTrigger);
1531 while(ShutdownManager::ShouldRun(ShutdownManager::eRunUpdateThread))
1532 {
1533 m_updateTrigger->WaitForAll(INFINITE);
1534
1535 if(!ShutdownManager::ShouldRun(ShutdownManager::eRunUpdateThread)) break;
1536
1537 PIXBeginNamedEvent(0,"Updating tiles to be ticked");
1538 // 4J Stu - Grass and Lava ticks currently take up the majority of all tile updates, so I am limiting them
1539 int grassTicks = 0;
1540 int lavaTicks = 0;
1541 for( unsigned int iLev = 0; iLev < 3; ++iLev )
1542 {
1543 EnterCriticalSection(&m_updateCS[iLev]);
1544 for( int i = 0; i < m_updateChunkCount[iLev]; i++ )
1545 {
1546 // 4J - some of these tile ticks will check things in neighbouring tiles, causing chunks to load/create that aren't yet in memory.
1547 // Try and avoid this by limiting the min/max x & z values that we will try and inspect in this chunk according to what surround chunks are loaded
1548 int cx = m_updateChunkX[iLev][i];
1549 int cz = m_updateChunkZ[iLev][i];
1550 int minx = 0;
1551 int maxx = 15;
1552 int minz = 0;
1553 int maxz = 15;
1554 if( !m_level[iLev]->hasChunk(cx, cz) ) continue;
1555
1556 if( !m_level[iLev]->hasChunk(cx + 1, cz + 0) ) { maxx = 11; }
1557 if( !m_level[iLev]->hasChunk(cx + 0, cz + 1) ) { maxz = 11; }
1558 if( !m_level[iLev]->hasChunk(cx - 1, cz + 0) ) { minx = 4; }
1559 if( !m_level[iLev]->hasChunk(cx + 0, cz - 1) ) { minz = 4; }
1560 if( !m_level[iLev]->hasChunk(cx + 1, cz + 1) ) { maxx = 11; maxz = 11; }
1561 if( !m_level[iLev]->hasChunk(cx + 1, cz - 1) ) { maxx = 11; minz = 4; }
1562 if( !m_level[iLev]->hasChunk(cx - 1, cz - 1) ) { minx = 4; minz = 4; }
1563 if( !m_level[iLev]->hasChunk(cx - 1, cz + 1) ) { minx = 4; maxz = 11; }
1564
1565 LevelChunk *lc = m_level[iLev]->getChunk(cx, cz);
1566
1567 for (int j = 0; j < 80; j++)
1568 {
1569 m_randValue[iLev] = m_randValue[iLev] * 3 + m_level[iLev]->addend;
1570 int val = (m_randValue[iLev] >> 2);
1571 int x = (val & 15);
1572 if( ( x < minx ) || ( x > maxx ) ) continue;
1573 int z = ((val >> 8) & 15);
1574 if( ( z < minz ) || ( z > maxz ) ) continue;
1575 int y = ((val >> 16) & (Level::maxBuildHeight - 1));
1576
1577 // This array access is a cache miss pretty much every time
1578 int id = lc->getTile(x,y,z);
1579 if( m_updateTileCount[iLev] >= MAX_UPDATES ) break;
1580
1581 // 4J Stu - Grass and Lava ticks currently take up the majority of all tile updates, so I am limiting them
1582 if( (id == Tile::grass_Id && grassTicks >= MAX_GRASS_TICKS) || (id == Tile::calmLava_Id && lavaTicks >= MAX_LAVA_TICKS) ) continue;
1583
1584 // 4J Stu - Added shouldTileTick as some tiles won't even do anything if they are set to tick and use up one of our updates
1585 if (Tile::tiles[id] != nullptr && Tile::tiles[id]->isTicking() && Tile::tiles[id]->shouldTileTick(m_level[iLev],x + (cx * 16), y, z + (cz * 16) ) )
1586 {
1587 if(id == Tile::grass_Id) ++grassTicks;
1588 else if(id == Tile::calmLava_Id) ++lavaTicks;
1589 m_updateTileX[iLev][m_updateTileCount[iLev]] = x + (cx * 16);
1590 m_updateTileY[iLev][m_updateTileCount[iLev]] = y;
1591 m_updateTileZ[iLev][m_updateTileCount[iLev]] = z + (cz * 16);
1592
1593 m_updateTileCount[iLev]++;
1594 }
1595 }
1596 }
1597 LeaveCriticalSection(&m_updateCS[iLev]);
1598 }
1599 PIXEndNamedEvent();
1600#ifdef __PS3__
1601 Sleep(10);
1602#endif //__PS3__
1603 }
1604
1605 ShutdownManager::HasFinished(ShutdownManager::eRunUpdateThread);
1606
1607 return 0;
