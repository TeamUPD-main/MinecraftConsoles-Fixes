151 inline PlayerUID GeneratePersistentUid()
152 {
153 // Avoid rand_s dependency: mix several Win64 runtime values into a 64-bit seed.
154 FILETIME ft = {};
155 GetSystemTimeAsFileTime(&ft);
156 uint64_t t = (((uint64_t)ft.dwHighDateTime) << 32) | ft.dwLowDateTime;
157
158 LARGE_INTEGER qpc = {};
159 QueryPerformanceCounter(&qpc);
160
161 uint64_t seed = t;
162 seed ^= (uint64_t)qpc.QuadPart;
163 seed ^= ((uint64_t)GetCurrentProcessId() << 32);
164 seed ^= (uint64_t)GetCurrentThreadId();
165 seed ^= (uint64_t)GetTickCount();
166 seed ^= (uint64_t)(size_t)&qpc;
167 seed ^= (uint64_t)(size_t)GetModuleHandleA(NULL);
168
169 uint64_t raw = Mix64(seed) ^ Mix64(seed + 0xA0761D6478BD642FULL);
170 raw ^= 0x8F4B2D6C1A93E705ULL;
171 raw |= 0x8000000000000000ULL;
172
173 PlayerUID xuid = (PlayerUID)raw;
174 if (!IsPersistedUidValid(xuid))
175 {
176 raw ^= 0x0100000000000001ULL;
177 xuid = (PlayerUID)raw;
178 }
179
180 if (!IsPersistedUidValid(xuid))
181 {
182 // Last-resort deterministic fallback for pathological cases.
183 xuid = (PlayerUID)0xD15EA5E000000001ULL;
184 }

190 {

195 // Produces a fully unique 64-bit value with no risk of overlap.

212 inline PlayerUID ResolvePersistentXuid()
213 {
214 // Process-local cache: uid.dat is immutable during runtime and this path is hot.
215 static bool s_cached = false;
216 static PlayerUID s_xuid = INVALID_XUID;
217
218 if (s_cached)
