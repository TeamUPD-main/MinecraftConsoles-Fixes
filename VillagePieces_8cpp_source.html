623void VillagePieces::StraightRoad::addChildren(StructurePiece *startPiece, list<StructurePiece *> *pieces, Random *random)
624{
625 bool hasHouses = false;
626
627 // place left houses
628 int depth = random->nextInt(5);
629 while (depth < length - 8)
630 {
631 StructurePiece *piece = generateHouseNorthernLeft(static_cast<StartPiece *>(startPiece), pieces, random, 0, depth);
632 if (piece != nullptr)
633 {
634 depth += Math::_max(piece->boundingBox->getXSpan(), piece->boundingBox->getZSpan());
635 hasHouses = true;
636 }
637 depth += 2 + random->nextInt(5);
638 }
639
640 // place right houses
641 depth = random->nextInt(5);
642 while (depth < length - 8)
643 {
644 StructurePiece *piece = generateHouseNorthernRight(static_cast<StartPiece *>(startPiece), pieces, random, 0, depth);
645 if (piece != nullptr)
646 {
647 depth += Math::_max(piece->boundingBox->getXSpan(), piece->boundingBox->getZSpan());
648 hasHouses = true;
649 }
650 depth += 2 + random->nextInt(5);
651 }
652
653 if (hasHouses && random->nextInt(3) > 0)
654 {
655 switch (orientation)
656 {
657 case Direction::NORTH:
658 generateAndAddRoadPiece(static_cast<StartPiece *>(startPiece), pieces, random, boundingBox->x0 - 1, boundingBox->y0, boundingBox->z0, Direction::WEST, getGenDepth());
659 break;
660 case Direction::SOUTH:
661 generateAndAddRoadPiece(static_cast<StartPiece *>(startPiece), pieces, random, boundingBox->x0 - 1, boundingBox->y0, boundingBox->z1 - 2, Direction::WEST, getGenDepth());
662 break;
663 case Direction::EAST:
664 generateAndAddRoadPiece(static_cast<StartPiece *>(startPiece), pieces, random, boundingBox->x1 - 2, boundingBox->y0, boundingBox->z0 - 1, Direction::NORTH, getGenDepth());
665 break;
666 case Direction::WEST:
667 generateAndAddRoadPiece(static_cast<StartPiece *>(startPiece), pieces, random, boundingBox->x0, boundingBox->y0, boundingBox->z0 - 1, Direction::NORTH, getGenDepth());
668 break;
669 }
670 }
671 if (hasHouses && random->nextInt(3) > 0)
672 {
673 switch (orientation)
674 {
675 case Direction::NORTH:
676 generateAndAddRoadPiece(static_cast<StartPiece *>(startPiece), pieces, random, boundingBox->x1 + 1, boundingBox->y0, boundingBox->z0, Direction::EAST, getGenDepth());
677 break;
678 case Direction::SOUTH:
679 generateAndAddRoadPiece(static_cast<StartPiece *>(startPiece), pieces, random, boundingBox->x1 + 1, boundingBox->y0, boundingBox->z1 - 2, Direction::EAST, getGenDepth());
680 break;
681 case Direction::EAST:
682 generateAndAddRoadPiece(static_cast<StartPiece *>(startPiece), pieces, random, boundingBox->x1 - 2, boundingBox->y0, boundingBox->z1 + 1, Direction::SOUTH, getGenDepth());
683 break;
684 case Direction::WEST:
685 generateAndAddRoadPiece(static_cast<StartPiece *>(startPiece), pieces, random, boundingBox->x0, boundingBox->y0, boundingBox->z1 + 1, Direction::SOUTH, getGenDepth());
686 break;
687 }
688 }
