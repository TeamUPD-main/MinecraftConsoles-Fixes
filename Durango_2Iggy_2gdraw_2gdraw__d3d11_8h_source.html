30IDOC extern GDrawFunctions * gdraw_D3D11_CreateContext(ID3D11Device *dev, ID3D11DeviceContext *ctx, S32 w, S32 h);
31/* Creates a GDraw context for rendering using D3D. You need to pass in the D3D device,
32 the device context to use for rendering, and the width/height of render target textures.
33
34 The width/height is used solely for sizing internal rendertargets. They will be
35 allocated to the larger of this size and the size of any rendered tiles (with padding).
36 In other words, you can pass in (0,0) and the rendertargets will be allocated to the
37 right size. However, if you draw multiple Iggy files or tiles of different sizes,
38 they might first be allocated too small; it's best to pass in the correct size initially
39 to avoid unnecessary allocation/deallocation of too-small rendertargets.
40
41 There can only be one D3D GDraw context active at any one time.
42
43 If initialization fails for some reason (the main reason would be an out of memory condition),
44 nullptr is returned. Otherwise, you can pass the return value to IggySetGDraw. */

99IDOC extern void RADLINK gdraw_D3D11_GetResourceUsageStats(gdraw_d3d11_resourcetype type, S32 *handles_used, S32 *bytes_used);
100/* D3D only: Get resource usage stats for last frame.
101 This can be used to get an estimate of how much graphics memory got used by GDraw
102 during the last frame.
103
104 For the dynbuffer, this always returns 0 in handles_used and the *size of the largest
105 single allocation* in bytes_used. It needs to be sized so that this allocation fits;
106 make it smaller and it won't work, but if you make it much larger (say more than 2x
107 as big), it's just a waste of memory. That said, we still recommend to make it no
108 smaller than 64k, and the default is 256k.
109
110 Caveat: This counts the number of bytes that GDraw knows about. 3D hardware usually
111 has its own management overhead, alignment requirements, allocation granularity
112 and so on. In short, this is not an accurate estimate of how much memory is actually
113 used by the GPU - it is a lower bound, though, and makes for a useful ballpark estimate. */
