3// This implements the Iggy graphics driver layer for Direct3D 9.
4
5// GDraw consists of several components that interact fairly loosely with each other;
6// e.g. the resource management, drawing and filtering parts are all fairly independent
7// of each other. If you want to modify some aspect of GDraw - say the texture allocation
8// logic - your best bet is usually to just look for one of the related entry points,
9// e.g. MakeTextureBegin, and take it from there. There's a bunch of code in this file,
10// but none of it is really complicated.
11//
12// The one bit you might want to change that's not that localized is to integrate
13// GDraw with an existing state caching system. The following bits all modify D3D state
14// in some way:
15// - The rendering helpers (set_viewport_raw, set_projection_raw, set_*_renderstate)
16// - RenderTile*/TextureDrawBuffer* may change the active rendertarget and depth/stencil surface,
17// as do D3D_NoMoreGDrawThisFrame and set_render_target
18// - set_texture
19// - set_renderstate and set_renderstate_full. These are the main places where render state changes occur;
20// you should probably start here.
21// - DrawIndexedTriangles sets the active vertex/index buffers and vertex declaration
22// - Most of the functions in the "filter effects" section modify D3D state, mostly
23// pixel shader constants and textures
24

283static rrbool RADLINK gdraw_MakeTextureBegin(void *owner, S32 width, S32 height, gdraw_texture_format format, U32 flags, GDraw_MakeTexture_ProcessingInfo *p, GDrawStats *stats)
284{
285 GDrawHandle *t = NULL;
286 D3DFORMAT d3dfmt;
287 S32 bpp;
288
289 if (format == GDRAW_TEXTURE_FORMAT_rgba32) {
290 d3dfmt = D3DFMT_A8R8G8B8;
291 bpp = 4;
292 } else {
293 d3dfmt = D3DFMT_A8;
294 bpp = 1;
295 }
296
297 // compute estimated size of texture in video memory
298 S32 size = width*height*bpp;
299 if (flags & GDRAW_MAKETEXTURE_FLAGS_mipmap)
300 size = size*4/3; // not correct for non-square
301
302 // allocate a handle and make room in the cache for this much data
303 t = gdraw_res_alloc_begin(gdraw->texturecache, size, stats);
304 if (!t)
305 return NULL;
306
307 HRESULT hr = gdraw->d3d_device->CreateTexture(width, height, (flags & GDRAW_MAKETEXTURE_FLAGS_mipmap) ? 0 : 1,
308 0, d3dfmt, D3DPOOL_MANAGED, &t->handle.tex.d3d, NULL);
309
310 if (FAILED(hr)) {
311 gdraw_HandleCacheAllocateFail(t);
312 IggyGDrawSendWarning(NULL, "GDraw CreateTexture() call failed with error code 0x%08x", hr);
313 return false;
314 }
315
316 t->handle.tex.w = width;
317 t->handle.tex.h = height;
318
319 gdraw_HandleCacheAllocateEnd(t, size, owner, (flags & GDRAW_MAKETEXTURE_FLAGS_never_flush) ? GDRAW_HANDLE_STATE_pinned : GDRAW_HANDLE_STATE_locked);
320 stats->nonzero_flags |= GDRAW_STATS_alloc_tex;
321 stats->alloc_tex += 1;
322 stats->alloc_tex_bytes += size;
323
324 p->texture_type = GDRAW_TEXTURE_TYPE_bgra;
325 p->p0 = t;
326
327 if (flags & GDRAW_MAKETEXTURE_FLAGS_mipmap) {
328 rrbool ok;
329 assert(p->temp_buffer != NULL);
330 ok = gdraw_MipmapBegin(&gdraw->mipmap, width, height, t->handle.tex.d3d->GetLevelCount(), bpp, p->temp_buffer, p->temp_buffer_bytes);
331 assert(ok); // this should never trigger unless the temp_buffer is way too small
332
333 p->p1 = &gdraw->mipmap;
334 p->texture_data = gdraw->mipmap.pixels[0];
335 p->num_rows = gdraw->mipmap.bheight;
336 p->stride_in_bytes = gdraw->mipmap.pitch[0];
337 p->i0 = 0; // current output y
338 p->i1 = bpp;
339 } else {
340 D3DLOCKED_RECT z;
341 hr = t->handle.tex.d3d->LockRect(0, &z, NULL, 0);
342 if (FAILED(hr)) {
343 t->handle.tex.d3d->Release();
344 gdraw_HandleCacheAllocateFail(t);
345
346 if (hr == E_OUTOFMEMORY) {
347 IggyGDrawSendWarning(NULL, "GDraw out of texture memory allocating %dx%d (%dbpp) texture", width, height, 8*bpp);
348 return false;
349 } else {
350 IggyGDrawSendWarning(NULL, "GDraw LockRect for texture allocation failed, D3D error 0x%08x\n", hr);
351 return false;
352 }
353 }
354
355 p->p1 = NULL;
356 p->texture_data = (U8 *) z.pBits;
357 p->num_rows = height;
358 p->stride_in_bytes = z.Pitch;
359 }
360
361 return true;
362}

364static rrbool RADLINK gdraw_MakeTextureMore(GDraw_MakeTexture_ProcessingInfo *p)
365{
366 GDrawHandle *t = (GDrawHandle *) p->p0;
367
368 if (p->p1) {
369 GDrawMipmapContext *c = (GDrawMipmapContext *) p->p1;
370 U32 outy = p->i0;
371 U32 bpp = p->i1;
372 U32 width = c->width;
373 U32 height = c->height;
374 U32 bheight = c->bheight;
375 U32 level = 0;
376
377 if (outy >= c->height)
378 return false;
379
380 do {
381 // upload data for this miplevel
382 D3DLOCKED_RECT z;
383 HRESULT hr = t->handle.tex.d3d->LockRect(level, &z, NULL, 0);
384 if (FAILED(hr))
385 return false;
386
387 for (U32 y=0; y < bheight; ++y)
388 memcpy((U8 *) z.pBits + ((outy >> level) + y) * z.Pitch, c->pixels[level] + y * c->pitch[level], width * bpp);
389 t->handle.tex.d3d->UnlockRect(level);
390
391 // prepare next miplevel
392 width = RR_MAX(width >> 1, 1);
393 height = RR_MAX(height >> 1, 1);
394 bheight = RR_MAX(bheight >> 1, 1);
395 } while (gdraw_MipmapAddLines(c, ++level));
396
397 // prepare next chunk
398 p->i0 += p->num_rows;
399 p->texture_data = c->pixels[0];
400 p->num_rows = c->bheight = RR_MIN(c->bheight, c->height - p->i0);
401 return true;
402 } else
403 return false;
404}

436 for (i=0; i < h; i++)

439 s->handle.tex.d3d->UnlockRect(0);
440}

525 failed_call = "CreateIndexBuffer";
526 }

528 hr = vb->handle.vbuf.base->Lock(0, vbuf_size, (void **) &p->vertex_data, 0);
529 failed_call = "Lock";

532 hr = vb->handle.vbuf.indices->Lock(0, ibuf_size, (void **) &p->index_data, 0);
533
534 if (FAILED(hr)) {
535 if (vb->handle.vbuf.base)

539 safe_release(vb->handle.vbuf.indices);

541 gdraw_HandleCacheAllocateFail(vb);

545 }
546
547 p->vertex_data_length = vbuf_size;
548 p->index_data_length = ibuf_size;
549 p->p0 = vb;
550
551 gdraw_HandleCacheAllocateEnd(vb, vbuf_size + ibuf_size, unique_id, GDRAW_HANDLE_STATE_locked);

593static GDrawHandle *get_color_rendertarget(GDrawStats *stats)
594{
595 // try to recycle LRU rendertarget
596 GDrawHandle *t = gdraw_HandleCacheGetLRU(&gdraw->rendertargets);
597 if (t) {
598 gdraw_HandleCacheLock(t, (void *) 1);
599 return t;
600 }
601
602 // ran out of RTs, allocate a new one
603 S32 size = gdraw->frametex_width * gdraw->frametex_height * 4;
604 if (gdraw->rendertargets.bytes_free < size) {
605 IggyGDrawSendWarning(NULL, "GDraw rendertarget allocation failed: hit size limit of %d bytes", gdraw->rendertargets.total_bytes);
606 return NULL;
607 }
608
609 t = gdraw_HandleCacheAllocateBegin(&gdraw->rendertargets);
610 if (!t) {
611 IggyGDrawSendWarning(NULL, "GDraw rendertarget allocation failed: hit handle limit");
612 return t;
613 }
614
615 HRESULT hr = gdraw->d3d_device->CreateTexture(gdraw->frametex_width, gdraw->frametex_height, 1, D3DUSAGE_RENDERTARGET,
616 D3DFMT_A8R8G8B8, D3DPOOL_DEFAULT, &t->handle.tex.d3d, NULL);
617 if (FAILED(hr)) {
618 report_d3d_error(hr, "CreateTexture", " creating rendertarget");
619 gdraw_HandleCacheAllocateFail(t);
620 return NULL;
621 }
622
623 gdraw_HandleCacheAllocateEnd(t, size, (void *) 1, GDRAW_HANDLE_STATE_locked);
624 stats->nonzero_flags |= GDRAW_STATS_alloc_tex;
625 stats->alloc_tex += 1;
626 stats->alloc_tex_bytes += size;
627
628 return t;
629}

743static void set_projection_raw(S32 x0, S32 x1, S32 y0, S32 y1)
744{
745 gdraw->projection[0] = 2.0f / (x1-x0);
746 gdraw->projection[1] = 2.0f / (y1-y0);

750 set_projection_base();
751}

852static void clear_zbuffer(IDirect3DSurface9 *surf, U32 flags)
853{
854 IDirect3DDevice9 *d3d = gdraw->d3d_device;
855 IDirect3DSurface9 *target;
856 D3DSURFACE_DESC desc;
857 surf->GetDesc(&desc);
858
859 // here's where it gets stupid: we need a rendertarget that's
860 // big enough so we can actually clear the full z-buffer.
861 // we don't actually render to it or anything, we just need to
862 // set it...
863 target = gdraw->main_framebuffer;
864 if (surf != gdraw->main_depthbuffer) {
865 // tile surface could, in theory, be larger than main framebuffer, so
866 // check if we have at least one rendertarget matching the rt depth
867 // buffer created and use that while clearing if possible
868 S32 i;
869 for (i=0; i < MAX_RENDER_STACK_DEPTH; ++i)
870 if (gdraw->rendertargets.handle[i].handle.tex.d3d) {
871 gdraw->rendertargets.handle[i].handle.tex.d3d->GetSurfaceLevel(0, &target);
872 break;
873 }
874 }
875
876 D3DVIEWPORT9 vp = { 0, 0, desc.Width, desc.Height, 0.0f, 1.0f };
877 d3d->SetRenderTarget(0, target);
878 d3d->SetDepthStencilSurface(surf);
879 d3d->SetViewport(&vp);
880 d3d->Clear(0, NULL, flags, 0, 1.0f, 0);
881
882 if (target != gdraw->main_framebuffer)
883 target->Release();
884}

1065static rrbool RADLINK gdraw_TextureDrawBufferBegin(gswf_recti *region, gdraw_texture_format /*format*/, U32 flags, void *owner, GDrawStats *stats)
1066{
1067 GDrawFramebufferState *n = gdraw->cur+1;
1068 GDrawHandle *t;
1069 if (gdraw->tw == 0 || gdraw->th == 0)
1070 return false;
1071
1072 if (n >= &gdraw->frame[MAX_RENDER_STACK_DEPTH]) {
1073 IggyGDrawSendWarning(NULL, "GDraw rendertarget nesting exceeds MAX_RENDER_STACK_DEPTH");
1074 return false;
1075 }
1076
1077 if (owner) {
1078 S32 w = region->x1 - region->x0;
1079 S32 h = region->y1 - region->y0;
1080
1081 // allocate a texture handle and free enough texture space
1082 t = gdraw_res_alloc_begin(gdraw->texturecache, w*h*4, stats);
1083 if (!t)
1084 return false;
1085 else {
1086 IDirect3DTexture9 * tex;
1087 HRESULT hr = gdraw->d3d_device->CreateTexture(w,h,1,
1088 D3DUSAGE_RENDERTARGET, D3DFMT_A8R8G8B8,
1089 D3DPOOL_DEFAULT, &tex, NULL);
1090
1091 if (FAILED(hr)) {
1092 if (t)
1093 gdraw_HandleCacheAllocateFail(t);
1094 IggyGDrawSendWarning(NULL, "GDraw D3D error for CreateTexture for cacheAsBitmap rendertarget: 0x%08x", hr);
1095 return false;
1096 }
1097
1098 stats->nonzero_flags |= GDRAW_STATS_alloc_tex;
1099 stats->alloc_tex += 1;
1100 stats->alloc_tex_bytes += w*h*4;
1101
1102 gdraw_HandleCacheAllocateEnd(t, w*h*4, (void *) 1, GDRAW_HANDLE_STATE_locked);
1103 }
1104 } else {
1105 t = get_color_rendertarget(stats);
1106 if (!t)
1107 return false;
1108 }
1109
1110 n->flags = flags;
1111 n->color_buffer = t;
1112 assert(n->color_buffer != NULL); // @GDRAW_ASSERT
1113
1114 ++gdraw->cur;
1115 gdraw->cur->cached = owner != NULL;
1116 if (owner) {
1117 gdraw->cur->base_x = region->x0;
1118 gdraw->cur->base_y = region->y0;
1119 gdraw->cur->width = region->x1 - region->x0;
1120 gdraw->cur->height = region->y1 - region->y0;
1121 }
1122
1123 set_render_target(stats);
1124 assert(gdraw->frametex_width >= gdraw->tw && gdraw->frametex_height >= gdraw->th); // @GDRAW_ASSERT
1125 set_viewport();
1126 set_projection();
1127
1128 int k = (int) (t - gdraw->rendertargets.handle);
1129
1130 if (region) {
1131 D3DRECT r;
1132 S32 ox, oy, pad = 2; // 2 pixels of border on all sides
1133 // 1 pixel turns out to be not quite enough with the interpolator precision we get.
1134
1135 if (gdraw->in_blur)
1136 ox = oy = 0;
1137 else
1138 ox = gdraw->tx0p, oy = gdraw->ty0p;
1139
1140 // clamp region to tile
1141 S32 xt0 = RR_MAX(region->x0 - ox, 0);
1142 S32 yt0 = RR_MAX(region->y0 - oy, 0);
1143 S32 xt1 = RR_MIN(region->x1 - ox, gdraw->tpw);
1144 S32 yt1 = RR_MIN(region->y1 - oy, gdraw->tph);
1145
1146 // but the padding needs to clamp to render target bounds
1147 r.x1 = RR_MAX(xt0 - pad, 0);
1148 r.y1 = RR_MAX(yt0 - pad, 0);
1149 r.x2 = RR_MIN(xt1 + pad, gdraw->frametex_width);
1150 r.y2 = RR_MIN(yt1 + pad, gdraw->frametex_height);
1151
1152 if (r.x2 <= r.x1 || r.y2 <= r.y1) { // region doesn't intersect with current tile
1153 --gdraw->cur;
1154 gdraw_FreeTexture((GDrawTexture *) t, 0, stats);
1155 // note: don't send a warning since this will happen during regular tiled rendering
1156 return false;
1157 }
1158
1159 gdraw->d3d_device->Clear(1, &r, D3DCLEAR_TARGET, 0, 1, 0);
1160 gdraw->rt_valid[k].x0 = xt0;
1161 gdraw->rt_valid[k].y0 = yt0;
1162 gdraw->rt_valid[k].x1 = xt1;
1163 gdraw->rt_valid[k].y1 = yt1;
1164 } else {
1165 gdraw->d3d_device->Clear(0, NULL, D3DCLEAR_TARGET, 0, 1, 0);
1166 gdraw->rt_valid[k].x0 = 0;
1167 gdraw->rt_valid[k].y0 = 0;
1168 gdraw->rt_valid[k].x1 = gdraw->frametex_width;
1169 gdraw->rt_valid[k].y1 = gdraw->frametex_height;
1170 }
1171
1172 return true;
1173}

1176{

1178 GDrawFramebufferState *m = --gdraw->cur;

1297static int set_renderstate_full(S32 vertex_format, GDrawRenderState *r, GDrawStats * /*stats*/)
1298{
1299 IDirect3DDevice9 *d3d = gdraw->d3d_device;
1300 U32 stencil_key, z_key;
1301 F32 depth = depth_from_id(r->id);
1302 VertexVars vvars;
1303 int vvarcount = VVAR_count_world_xoff;
1304
1305 // set vertex shader
1306 d3d->SetVertexShader(gdraw->vert[vertex_format].vshader);
1307
1308 // set vertex shader constants
1309 if (!r->use_world_space)
1310 gdraw_ObjectSpace(vvars.world[0], r->o2w, depth, 0.0f);
1311 else
1312 gdraw_WorldSpace(vvars.world[0], gdraw->world_to_pixel, depth, 0.0f);
1313
1314 memcpy(&vvars.x_off, r->edge_matrix, 4*sizeof(F32));
1315
1316 if (r->texgen0_enabled) {
1317 memcpy(&vvars.texgen_s, r->s0_texgen, 4*sizeof(F32));
1318 memcpy(&vvars.texgen_t, r->t0_texgen, 4*sizeof(F32));
1319 vvarcount = VVAR_count;
1320 }
1321
1322 if (gdraw->use_3d || gdraw->last_was_3d) {
1323 if (gdraw->use_3d) {
1324 vvarcount = VVAR_count3d;
1325 memcpy(&vvars.viewproj, gdraw->xform_3d, 12*sizeof(F32));
1326 } else
1327 set_projection_base();
1328 gdraw->last_was_3d = gdraw->use_3d;
1329 }
1330
1331 d3d->SetVertexShaderConstantF(0, vvars.world[0], vvarcount);
1332
1333 // set the blend mode
1334 int blend_mode = r->blend_mode;
1335 int tex0mode = r->tex0_mode;
1336
1337 static struct gdraw_d3d_blendspec {
1338 BOOL enable;
1339 D3DBLEND src;
1340 D3DBLEND dest;
1341 } blends[ASSERT_COUNT(GDRAW_BLEND__count, 6)] = {
1342 FALSE, D3DBLEND_ONE, D3DBLEND_ZERO, // GDRAW_BLEND_none
1343 TRUE, D3DBLEND_ONE, D3DBLEND_INVSRCALPHA, // GDRAW_BLEND_alpha
1344 TRUE, D3DBLEND_DESTCOLOR, D3DBLEND_INVSRCALPHA, // GDRAW_BLEND_multiply
1345 TRUE, D3DBLEND_ONE, D3DBLEND_ONE, // GDRAW_BLEND_add
1346
1347 FALSE, D3DBLEND_ONE, D3DBLEND_ZERO, // GDRAW_BLEND_filter
1348 FALSE, D3DBLEND_ONE, D3DBLEND_ZERO, // GDRAW_BLEND_special
1349 };
1350 assert(blend_mode >= 0 && blend_mode < sizeof(blends)/sizeof(*blends));
1351
1352 if (blend_mode != gdraw->blend_mode) {
1353 gdraw->blend_mode = blend_mode;
1354 if (blends[blend_mode].enable) {
1355 d3d->SetRenderState(D3DRS_ALPHABLENDENABLE, TRUE);
1356 d3d->SetRenderState(D3DRS_SRCBLEND, blends[blend_mode].src);
1357 d3d->SetRenderState(D3DRS_DESTBLEND, blends[blend_mode].dest);
1358 } else
1359 d3d->SetRenderState(D3DRS_ALPHABLENDENABLE, FALSE);
1360 }
1361
1362 // set the pixel shader
1363 IDirect3DPixelShader9 *pshader;
1364 if (blend_mode != GDRAW_BLEND_special) {
1365 assert(tex0mode >= 0 && tex0mode < sizeof(gdraw->fprog) / sizeof(*gdraw->fprog));
1366
1367 int additive = 0;
1368 if (r->cxf_add) {
1369 additive = 1;
1370 if (r->cxf_add[3]) additive = 2;
1371 }
1372
1373 pshader = gdraw->fprog[tex0mode][additive].pshader;
1374 } else
1375 pshader = gdraw->exceptional_blend[r->special_blend].pshader;
1376
1377 d3d->SetPixelShader(pshader);
1378
1379 // set textures
1380 if (tex0mode != GDRAW_TEXTURE_none) {
1381 if (!r->tex[0]) return 0;
1382 set_d3d_texture(0, ((GDrawHandle *) r->tex[0])->handle.tex.d3d, r->wrap0, r->nearest0);
1383 }
1384
1385 // set pixel shader constants
1386 d3d->SetPixelShaderConstantF(PVAR_cmul, r->color, 1);
1387
1388 if (r->cxf_add) {
1389 F32 temp[4] = { r->cxf_add[0]/255.0f, r->cxf_add[1]/255.0f, r->cxf_add[2]/255.0f, r->cxf_add[3]/255.0f };
1390 d3d->SetPixelShaderConstantF(PVAR_cadd, temp, 1);
1391 }
1392
1393 if (tex0mode == GDRAW_TEXTURE_focal_gradient)
1394 d3d->SetPixelShaderConstantF(PVAR_focal, r->focal_point, 1);
1395
1396 // Set pixel operation states
1397 gdraw->scissor_state = ~0u;
1398 if (r->scissor) {
1399 RECT s;
1400 gdraw->scissor_state = r->scissor;
1401 if (gdraw->cur == gdraw->frame) {
1402 s.left = r->scissor_rect.x0 + gdraw->vx - gdraw->tx0;
1403 s.top = r->scissor_rect.y0 + gdraw->vy - gdraw->ty0;
1404 s.right = r->scissor_rect.x1 + gdraw->vx - gdraw->tx0;
1405 s.bottom = r->scissor_rect.y1 + gdraw->vy - gdraw->ty0;
1406 } else {
1407 s.left = r->scissor_rect.x0 - gdraw->tx0p;
1408 s.top = r->scissor_rect.y0 - gdraw->ty0p;
1409 s.right = r->scissor_rect.x1 - gdraw->tx0p;
1410 s.bottom = r->scissor_rect.y1 - gdraw->ty0p;
1411 }
1412 d3d->SetScissorRect(&s);
1413 d3d->SetRenderState(D3DRS_SCISSORTESTENABLE, TRUE);
1414 } else {
1415 if (r->scissor != gdraw->scissor_state) {
1416 gdraw->scissor_state = r->scissor;
1417 d3d->SetRenderState(D3DRS_SCISSORTESTENABLE, FALSE);
1418 }
1419 }
1420
1421 // stencil changed?
1422 stencil_key = r->stencil_test | (r->stencil_set << 8);
1423 if (stencil_key != gdraw->stencil_key) {
1424 gdraw->stencil_key = stencil_key;
1425
1426 d3d->SetRenderState(D3DRS_STENCILMASK, r->stencil_test);
1427 d3d->SetRenderState(D3DRS_STENCILWRITEMASK, r->stencil_set);
1428 d3d->SetRenderState(D3DRS_STENCILENABLE, (r->stencil_set | r->stencil_test) != 0);
1429 d3d->SetRenderState(D3DRS_COLORWRITEENABLE, r->stencil_set ? 0x0 : 0xf);
1430 }
1431
1432 // z key
1433 z_key = r->set_id | (r->test_id << 1);
1434 if (z_key != gdraw->z_key) {
1435 gdraw->z_key = z_key;
1436 d3d->SetRenderState(D3DRS_ZENABLE, (r->test_id | r->set_id) ? D3DZB_TRUE : D3DZB_FALSE);
1437 d3d->SetRenderState(D3DRS_ZFUNC, r->test_id ? D3DCMP_LESS : D3DCMP_ALWAYS);
1438 d3d->SetRenderState(D3DRS_ZWRITEENABLE, r->set_id);
1439 }
1440
1441 return 1;
1442}

1654static void RADLINK gdraw_FilterQuad(GDrawRenderState *r, S32 x0, S32 y0, S32 x1, S32 y1, GDrawStats *stats)
1655{

1657 F32 tc[4];
1658 gswf_recti s;
1659
1660 // clip to tile boundaries
1661 s.x0 = RR_MAX(x0, gdraw->tx0p);
1662 s.y0 = RR_MAX(y0, gdraw->ty0p);
1663 s.x1 = RR_MIN(x1, gdraw->tx0p + gdraw->tpw);
1664 s.y1 = RR_MIN(y1, gdraw->ty0p + gdraw->tph);
1665 if (s.x1 < s.x0 || s.y1 < s.y0)
1666 return;
1667
1668 tc[0] = (s.x0 - gdraw->tx0p) / (F32) gdraw->frametex_width;
1669 tc[1] = (s.y0 - gdraw->ty0p) / (F32) gdraw->frametex_height;
1670 tc[2] = (s.x1 - gdraw->tx0p) / (F32) gdraw->frametex_width;
1671 tc[3] = (s.y1 - gdraw->ty0p) / (F32) gdraw->frametex_height;
1672
1673 // clear to known render state
1674 d3d->SetRenderState(D3DRS_STENCILENABLE, FALSE);
1675 d3d->SetRenderState(D3DRS_COLORWRITEENABLE, 0xf);
1676
1677 d3d->SetRenderState(D3DRS_ZENABLE, D3DZB_FALSE);
1678 d3d->SetRenderState(D3DRS_ZFUNC, D3DCMP_LESS);
1679 d3d->SetRenderState(D3DRS_ZWRITEENABLE, FALSE);
1680
1681 d3d->SetRenderState(D3DRS_SCISSORTESTENABLE, FALSE);
