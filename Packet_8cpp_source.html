552CompoundTag *Packet::readNbt(DataInputStream *dis)
553{
554 int size = dis->readShort();
555 if (size <= 0) return nullptr;
556
557 const int MAX_NBT_SIZE = 32767;
558 if (size > MAX_NBT_SIZE) return nullptr;
559 byteArray buff(size);
560 if (!dis->readFully(buff))
561 {
562 delete [] buff.data;
563 return nullptr;
564 }
565 CompoundTag *result = (CompoundTag *) NbtIo::decompress(buff);
566 delete [] buff.data;
567 return result;
568}
