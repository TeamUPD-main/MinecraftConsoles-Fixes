17void UILayer::tick()
18{
19 // Delete old scenes - deleting a scene can cause a new scene to be deleted, so we need to make a copy of the scenes that we are going to try and destroy this tick
20 vector<UIScene *>scenesToDeleteCopy;
21 for(auto& scene : m_scenesToDelete)
22 {
23 scenesToDeleteCopy.push_back(scene);
24 }
25 m_scenesToDelete.clear();
26
27 // Delete the scenes in our copy if they are ready to delete, otherwise add back to the ones that are still to be deleted. Actually deleting a scene might also add something back into m_scenesToDelete.
28 for(auto& scene : scenesToDeleteCopy)
29 {
30 if( scene && scene->isReadyToDelete())
31 {
32 delete scene;
33 }
34 else
35 {
36 m_scenesToDelete.push_back(scene);
37 }
38 }
39
40 while (!m_scenesToDestroy.empty())
41 {
42 UIScene *scene = m_scenesToDestroy.back();
43 m_scenesToDestroy.pop_back();
44 scene->destroyMovie();
45 }
46 m_scenesToDestroy.clear();
47
48 for(auto & component : m_components)
49 {
50 component->tick();
51 }
52 // Note: reverse iterator, the last element is the top of the stack
53 int sceneIndex = m_sceneStack.size() - 1;
54 while( sceneIndex >= 0 && sceneIndex < m_sceneStack.size() )
55 {
56 //(*it)->tick();
57 UIScene *scene = m_sceneStack[sceneIndex];
58 scene->tick();
59 --sceneIndex;
60 // TODO: We may wish to ignore ticking the rest of the stack based on this scene
61 }
62}

64void UILayer::render(S32 width, S32 height, C4JRender::eViewportType viewport)
65{
66 if(!ui.IsExpectingOrReloadingSkin())
67 {
68 for(auto& it : m_components)
69 {
70 auto itRef = m_componentRefCount.find(it->getSceneType());
71 if(itRef != m_componentRefCount.end() && itRef->second.second)
72 {
73 if(it->isVisible() )
74 {
75 PIXBeginNamedEvent(0, "Rendering component %d", it->getSceneType() );
76 it->render(width, height,viewport);
77 PIXEndNamedEvent();
78 }
79 }
80 }
81 }
82 if(!m_sceneStack.empty())
83 {
84 int lowestRenderable = m_sceneStack.size() - 1;
85 for(;lowestRenderable >= 0; --lowestRenderable)
86 {
87 if(m_sceneStack[lowestRenderable]->hidesLowerScenes()) break;
88 }
89 if(lowestRenderable < 0) lowestRenderable = 0;
90 for(;lowestRenderable < m_sceneStack.size(); ++lowestRenderable)
91 {
92 if(m_sceneStack[lowestRenderable]->isVisible() && (!ui.IsExpectingOrReloadingSkin() || m_sceneStack[lowestRenderable]->getSceneType()==eUIScene_Timer))
93 {
94 PIXBeginNamedEvent(0, "Rendering scene %d", m_sceneStack[lowestRenderable]->getSceneType() );
95 m_sceneStack[lowestRenderable]->render(width, height,viewport);

448{

587 {
588 --it->second.first;
589
590 if(it->second.first <= 0)

603 compIt = m_components.erase(compIt);
604 }

677// Updates layer focus state if no error message is present (unless this is the error layer)
678bool UILayer::updateFocusState(bool allowedFocus /* = false */)
679{
680 // If haveFocus is false, request it
681 if (!allowedFocus)
682 {
683 // To update focus in this layer we need to request focus from group
684 // Focus will be denied if there's an upper layer that needs focus
685 allowedFocus = m_parentGroup->RequestFocus(this);
686 }
687
688 m_bMenuDisplayed = false;
689 m_bPauseMenuDisplayed = false;
690 m_bContainerMenuDisplayed = false;
691 m_bIgnoreAutosaveMenuDisplayed = false;
692 m_bIgnorePlayerJoinMenuDisplayed = false;
693
694 bool layerFocusSet = false;
695 for (auto it = m_sceneStack.rbegin(); it != m_sceneStack.rend(); ++it)
696 {
697 UIScene *scene = *it;
698
699 // UPDATE FOCUS STATES
700 if(!layerFocusSet && allowedFocus && scene->stealsFocus())
701 {
702 scene->gainFocus();
703 layerFocusSet = true;
704 }
705 else
706 {
707 scene->loseFocus();
708 if(allowedFocus && app.GetGameStarted())
709 {
710 // 4J Stu - This is a memory optimisation so we don't keep scenes loaded in memory all the time
711 // This is required for PS3 (and likely Vita), but I'm removing it on XboxOne so that we can avoid
712 // the scene creation time (which can be >0.5s) since we have the memory to spare
713#ifndef _XBOX_ONE
714 m_scenesToDestroy.push_back(scene);
715#endif
716 }
717
718 if (scene->getSceneType() == eUIScene_SettingsOptionsMenu)
719 {
720 scene->loseFocus();
721 m_scenesToDestroy.push_back(scene);
722 }
723 }
724
726
727 // 4J-PB - this should just be true
728 m_bMenuDisplayed=true;
729
730 EUIScene sceneType = scene->getSceneType();
731 switch(sceneType)
732 {
733 case eUIScene_PauseMenu:
734 m_bPauseMenuDisplayed = true;
735 break;
736 case eUIScene_Crafting2x2Menu:
737 case eUIScene_Crafting3x3Menu:
738 case eUIScene_FurnaceMenu:
739 case eUIScene_ContainerMenu:
740 case eUIScene_LargeContainerMenu:
741 case eUIScene_InventoryMenu:
