386void UIScene_InGameInfoMenu::OnPlayerChanged(void *callbackParam, INetworkPlayer *pPlayer, bool leaving)
387{
388 app.DebugPrintf("<UIScene_InGameInfoMenu::OnPlayerChanged> Player \"%ls\" %s (smallId: %d)\n", pPlayer->GetOnlineName(), leaving ? "leaving" : "joining", pPlayer->GetSmallId());
389
390 UIScene_InGameInfoMenu *scene = static_cast<UIScene_InGameInfoMenu *>(callbackParam);
391 bool playerFound = false;
392 int foundIndex = 0;
393 for(int i = 0; i < scene->m_players.size(); ++i)
394 {
395 if(!playerFound && scene->m_players[i]->m_smallId == pPlayer->GetSmallId() )
396 {
397 if( scene->m_playerList.getCurrentSelection() == scene->m_playerList.getItemCount() - 1 )
398 {
399 scene->m_playerList.setCurrentSelection( scene->m_playerList.getItemCount() - 2 );
400 }
401
402 // Player found
403 playerFound = true;
404 foundIndex = i;
405 }
406 }
407
408 if (leaving && !playerFound) app.DebugPrintf("<UIScene_InGameInfoMenu::OnPlayerChanged> Error: Player \"%ls\" leaving but not found in list\n", pPlayer->GetOnlineName());
409 if (!leaving && playerFound) app.DebugPrintf("<UIScene_InGameInfoMenu::OnPlayerChanged> Error: Player \"%ls\" joining but already in list\n", pPlayer->GetOnlineName());
410
411 // If the player was found remove them (even if they're joining, they'll be added again later)
412 if(playerFound)
413 {
414 app.DebugPrintf("<UIScene_InGameInfoMenu::OnPlayerChanged> Player \"%ls\" found, removing\n", pPlayer->GetOnlineName());
415
416 // Remove player info
417 delete scene->m_players[foundIndex];
418 scene->m_players.erase(scene->m_players.begin() + foundIndex);
419
420 // Remove player from list
421 scene->m_playerList.removeItem(foundIndex);
422 }
423
424 // If the player is joining
425 if(!leaving)
426 {
427 app.DebugPrintf("<UIScene_InGameInfoMenu::OnPlayerChanged> Player \"%ls\" not found, adding\n", pPlayer->GetOnlineName());
428
429 PlayerInfo *info = scene->BuildPlayerInfo(pPlayer);
430 scene->m_players.push_back(info);
431
432 // Note that the tick updates buttons every tick so it's only really important that we
433 // add the button (not the order or content)
434 scene->m_playerList.addItem(info->m_name, info->m_colorState, info->m_voiceStatus);
435 }
