1120bool LocalPlayer::checkInTile(double x, double y, double z)
1121{
1122 int xTile = Mth::floor(x);
1123 int yTile = Mth::floor(y);
1124 int zTile = Mth::floor(z);
1125
1126 double xd = x - xTile;
1127 double zd = z - zTile;
1128
1129 if (isSolidBlock(xTile, yTile, zTile) || isSolidBlock(xTile, yTile + 1, zTile))
1130 {
1131 bool west = !isSolidBlock(xTile - 1, yTile, zTile) && !isSolidBlock(xTile - 1, yTile + 1, zTile);
1132 bool east = !isSolidBlock(xTile + 1, yTile, zTile) && !isSolidBlock(xTile + 1, yTile + 1, zTile);
1133 bool north = !isSolidBlock(xTile, yTile, zTile - 1) && !isSolidBlock(xTile, yTile + 1, zTile - 1);
1134 bool south = !isSolidBlock(xTile, yTile, zTile + 1) && !isSolidBlock(xTile, yTile + 1, zTile + 1);
1135
1136 int dir = -1;
1137 double closest = 9999;
1138 if (west && xd < closest)
1139 {
1140 closest = xd;
1141 dir = 0;
1142 }
1143 if (east && 1 - xd < closest)
1144 {
1145 closest = 1 - xd;
1146 dir = 1;
1147 }
1148 if (north && zd < closest)
1149 {
1150 closest = zd;
1151 dir = 4;
1152 }
1153 if (south && 1 - zd < closest)
1154 {
1155 closest = 1 - zd;
1156 dir = 5;
1157 }
1158
1159 float speed = 0.1f;
1160 if (dir == 0) this->xd = -speed;
1161 if (dir == 1) this->xd = +speed;
1162 if (dir == 4) this->zd = -speed;
1163 if (dir == 5) this->zd = +speed;
1164 }
1165
1166 return false;
1167

1314bool LocalPlayer::creativeModeHandleMouseClick(int button, bool buttonPressed)
1315{
1316 if( buttonPressed )
1317 {
1318 if( lastClickState == lastClick_oldRepeat )
1319 {
1320 return false;
1321 }
1322
1323 // Are we in an auto-repeat situation? - If so only tell the game that we've clicked if we move more than a unit away from our last
1324 // click position in any axis
1325 if( lastClickState != lastClick_invalid )
1326 {
1327 // If we're in disabled mode already (set when sprinting) then don't do anything - if we're sprinting, we don't auto-repeat at all.
1328 // With auto repeat on, we can quickly place fires causing photosensitivity issues due to rapid flashing
1329 if( lastClickState == lastClick_disabled ) return false;
1330 // If we've started sprinting, go into this mode & also don't do anything
1331 // Ignore repeate when sleeping
1332 if( isSprinting() )
1333 {
1334 lastClickState = lastClick_disabled;
1335 return false;
1336 }
1337
1338 // Get distance from last click point in each axis

1356 }

1381 // The new click position is normalised at one unit in the direction of movement, so that we don't gradually drift away if we detect the movement a fraction over
1382 // the unit distance each time
1383

1385 {
1386 dX= ( dX < 0.0f ) ? ceilf(dX) : floorf(dX);
1387 newClick = true;
1388 }
1389 else if( fabsf(dY) >= 1.0f )
1390 {
1391 dY= ( dY < 0.0f ) ? ceilf(dY) : floorf(dY);
1392 newClick = true;
1393 }
1394 else if( fabsf(dZ) >= 1.0f )
1395 {
1396 dZ= ( dZ < 0.0f ) ? ceilf(dZ) : floorf(dZ);
1397 newClick = true;
1398 }
1399
1400 if( ( !newClick ) && ( lastClickTolerance > 0.0f ) )
1401 {
1402 float fTarget = 1.0f - lastClickTolerance;
1403
1404 if( fabsf(dX) >= fTarget ) newClick = true;
1405 if( fabsf(dY) >= fTarget ) newClick = true;
1406 if( fabsf(dZ) >= fTarget ) newClick = true;
1407 }
1408
1409 if( newClick )
1410 {
1411 lastClickX += dX;
1412 lastClickY += dY;
1413 lastClickZ += dZ;
1414
1415 // Get a more accurate pick from the position where the new click should ideally have come from, rather than
1416 // where we happen to be now (ie a rounded number of units from the last Click position)
1417 double oldX = x;
1418 double oldY = y;
1419 double oldZ = z;
1420 x = lastClickX;
1421 y = lastClickY;
1422 z = lastClickZ;
1423
1424 minecraft->gameRenderer->pick(1);
1425
1426 x = oldX;
1427 y = oldY;
1428 z = oldZ;
1429

1431
1432 if( lastClickState == lastClick_stopped )
1433 {
1434 lastClickState = lastClick_init;
1435 lastClickTolerance = 0.0f;
1436 }
1437 else
1438 {
1439 lastClickState = lastClick_moving;
1440 lastClickTolerance = 0.0f;
1441 }
1442 }
1443 }
1444 else
1445 {
1446 // First click - just record position & handle
1447 lastClickX = static_cast<float>(x);
1448 lastClickY = static_cast<float>(y);
1449 lastClickZ = static_cast<float>(z);
1450 // If we actually placed an item, then move into the init state as we are going to be doing the special creative mode auto repeat
1451 bool itemPlaced = handleMouseClick(button);
1452 // If we're sprinting or riding, don't auto-repeat at all. With auto repeat on, we can quickly place fires causing photosensitivity issues due to rapid flashing
1453 // Also ignore repeats when the player is sleeping
1454 if( isSprinting() || isRiding() || isSleeping() )
1455 {
1456 lastClickState = lastClick_disabled;
1457 }
1458 else
1459 {
1460 if( itemPlaced )
1461 {
1462 lastClickState = lastClick_init;
1463 lastClickTolerance = 0.0f;
1464 }
1465 else
1466 {
1467 // Didn't place an item - might actually be activating a switch or door or something - just do a standard auto repeat in this case
1468 lastClickState = lastClick_oldRepeat;
1469 }
1470 }
1471 return true;
1472 }
1473 }
1474 else
1475 {
1476 lastClickState = lastClick_invalid;
1477 }
1478 return false;
