128static void CopyWideArgToAnsi(LPCWSTR source, char* dest, size_t destSize)
129{
130 if (destSize == 0)
131 return;
132
133 dest[0] = 0;
134 if (source == nullptr)
135 return;
136
137 WideCharToMultiByte(CP_ACP, 0, source, -1, dest, static_cast<int>(destSize), nullptr, nullptr);
138 dest[destSize - 1] = 0;
139}

162static bool LoadFullscreenOption()
163{
164 char path[MAX_PATH];
165 GetOptionsFilePath(path, sizeof(path));
166 FILE *f = nullptr;
167 if (fopen_s(&f, path, "r") == 0 && f)
168 {
169 char line[256];
170 while (fgets(line, sizeof(line), f))
171 {
172 int val = 0;
173 if (sscanf_s(line, "fullscreen=%d", &val) == 1)
174 {
175 fclose(f);
176 return val != 0;
177 }
178 }
179 fclose(f);
180 }
181 return false;

717 // g_rScreenWidth/Height updated inside ResizeD3D to backbuffer dims
718 ResizeD3D(newW, newH);
719 }
720 }
721 }
722 break;

845//--------------------------------------------------------------------------------------
846HRESULT InitDevice()
847{
848 HRESULT hr = S_OK;
849
850 RECT rc;
851 GetClientRect( g_hWnd, &rc );
852 UINT width = rc.right - rc.left;
853 UINT height = rc.bottom - rc.top;
854//app.DebugPrintf("width: %d, height: %d\n", width, height);
855 width = g_rScreenWidth;
856 height = g_rScreenHeight;
857//app.DebugPrintf("width: %d, height: %d\n", width, height);
858
859 UINT createDeviceFlags = 0;
860#ifdef _DEBUG
861 createDeviceFlags |= D3D11_CREATE_DEVICE_DEBUG;
862#endif
863
864 D3D_DRIVER_TYPE driverTypes[] =
865 {
866 D3D_DRIVER_TYPE_HARDWARE,
867 D3D_DRIVER_TYPE_WARP,
868 D3D_DRIVER_TYPE_REFERENCE,
869 };
870 UINT numDriverTypes = ARRAYSIZE( driverTypes );
871
872 D3D_FEATURE_LEVEL featureLevels[] =
873 {
874 D3D_FEATURE_LEVEL_11_0,
875 D3D_FEATURE_LEVEL_10_1,
876 D3D_FEATURE_LEVEL_10_0,
877 };
878 UINT numFeatureLevels = ARRAYSIZE( featureLevels );
879
880 DXGI_SWAP_CHAIN_DESC sd;
881 ZeroMemory( &sd, sizeof( sd ) );
882 sd.BufferCount = 1;
883 sd.BufferDesc.Width = width;
884 sd.BufferDesc.Height = height;
885 sd.BufferDesc.Format = DXGI_FORMAT_R8G8B8A8_UNORM;
886 sd.BufferDesc.RefreshRate.Numerator = 60;
887 sd.BufferDesc.RefreshRate.Denominator = 1;
888 sd.BufferUsage = DXGI_USAGE_RENDER_TARGET_OUTPUT | DXGI_USAGE_SHADER_INPUT;
889 sd.OutputWindow = g_hWnd;
890 sd.SampleDesc.Count = 1;
891 sd.SampleDesc.Quality = 0;
892 sd.Windowed = TRUE;
893
894 for( UINT driverTypeIndex = 0; driverTypeIndex < numDriverTypes; driverTypeIndex++ )
895 {
896 g_driverType = driverTypes[driverTypeIndex];
897 hr = D3D11CreateDeviceAndSwapChain( nullptr, g_driverType, nullptr, createDeviceFlags, featureLevels, numFeatureLevels,
898 D3D11_SDK_VERSION, &sd, &g_pSwapChain, &g_pd3dDevice, &g_featureLevel, &g_pImmediateContext );
899 if( HRESULT_SUCCEEDED( hr ) )
900 break;
901 }
902 if( FAILED( hr ) )
903 return hr;
904
905 // Create a render target view
906 ID3D11Texture2D* pBackBuffer = nullptr;
907 hr = g_pSwapChain->GetBuffer( 0, __uuidof( ID3D11Texture2D ), ( LPVOID* )&pBackBuffer );
908 if( FAILED( hr ) )
909 return hr;
910
911 // Create a depth stencil buffer
912 D3D11_TEXTURE2D_DESC descDepth;
913 ZeroMemory(&descDepth, sizeof(descDepth));
914
915 descDepth.Width = width;
916 descDepth.Height = height;
917 descDepth.MipLevels = 1;
918 descDepth.ArraySize = 1;
919 descDepth.Format = DXGI_FORMAT_D24_UNORM_S8_UINT;
920 descDepth.SampleDesc.Count = 1;
921 descDepth.SampleDesc.Quality = 0;
922 descDepth.Usage = D3D11_USAGE_DEFAULT;
923 descDepth.BindFlags = D3D11_BIND_DEPTH_STENCIL;
924 descDepth.CPUAccessFlags = 0;
925 descDepth.MiscFlags = 0;
926 hr = g_pd3dDevice->CreateTexture2D(&descDepth, nullptr, &g_pDepthStencilBuffer);
927
928 D3D11_DEPTH_STENCIL_VIEW_DESC descDSView;
929 ZeroMemory(&descDSView, sizeof(descDSView));
930 descDSView.Format = DXGI_FORMAT_D24_UNORM_S8_UINT;
931 descDSView.ViewDimension = D3D11_DSV_DIMENSION_TEXTURE2D;
932 descDSView.Texture2D.MipSlice = 0;
933
934 hr = g_pd3dDevice->CreateDepthStencilView(g_pDepthStencilBuffer, &descDSView, &g_pDepthStencilView);
935
936 hr = g_pd3dDevice->CreateRenderTargetView( pBackBuffer, nullptr, &g_pRenderTargetView );
937 pBackBuffer->Release();
938 if( FAILED( hr ) )
939 return hr;
940
941 g_pImmediateContext->OMSetRenderTargets( 1, &g_pRenderTargetView, g_pDepthStencilView );
942
943 // Setup the viewport
944 D3D11_VIEWPORT vp;

948 vp.MaxDepth = 1.0f;
949 vp.TopLeftX = 0;
950 vp.TopLeftY = 0;
951 g_pImmediateContext->RSSetViewports( 1, &vp );
952
953 RenderManager.Initialise(g_pd3dDevice, g_pSwapChain);
954
955 PostProcesser::GetInstance().Init();
956
957 return S_OK;
958}

973// Rebuild D3D11 resources after a window resize
974//--------------------------------------------------------------------------------------
975static bool ResizeD3D(int newW, int newH)
976{
977 if (newW <= 0 || newH <= 0) return false;
978 if (!g_pSwapChain) return false;
979 if (!g_bResizeReady) return false;
980
981 int bbW = newW;

985 // 0x10 m_pDevice (ID3D11Device*)
986 // 0x18 m_pDeviceContext (ID3D11DeviceContext*)
987 // 0x20 m_pSwapChain (IDXGISwapChain*)
988 // 0x28 renderTargetView (ID3D11RenderTargetView*) — backbuffer RTV
989 // 0x50 renderTargetShaderResourceView (ID3D11ShaderResourceView*)
990 // 0x98 depthStencilView (ID3D11DepthStencilView*)
991 // 0x5138 backBufferWidth (DWORD) — used by StartFrame() for viewport
992 // 0x513C backBufferHeight (DWORD) — used by StartFrame() for viewport
993 //
994 // Strategy: destroy old swap chain, create new one, patch Renderer's internal
995 // pointers directly. This avoids both ResizeBuffers (outstanding ref issues)
996 // and Initialise() (which wipes the texture table via memset).
997 // The Renderer's old RTV/SRV/DSV are intentionally NOT released — they become
998 // orphaned with the old swap chain. Tiny leak, but avoids fighting unknown refs.
999 char* pRM = (char*)&InternalRenderManager;
1000 ID3D11RenderTargetView** ppRM_RTV = (ID3D11RenderTargetView**)(pRM + 0x28);
1001 ID3D11ShaderResourceView** ppRM_SRV = (ID3D11ShaderResourceView**)(pRM + 0x50);
1002 ID3D11DepthStencilView** ppRM_DSV = (ID3D11DepthStencilView**)(pRM + 0x98);
1003 IDXGISwapChain** ppRM_SC = (IDXGISwapChain**)(pRM + 0x20);
1004 DWORD* pRM_BBWidth = (DWORD*)(pRM + 0x5138);
1005 DWORD* pRM_BBHeight = (DWORD*)(pRM + 0x513C);
1006
1007 // Verify offsets by checking device and swap chain pointers
1008 ID3D11Device** ppRM_Device = (ID3D11Device**)(pRM + 0x10);
1009 if (*ppRM_Device != g_pd3dDevice || *ppRM_SC != g_pSwapChain)
1010 {
1011 app.DebugPrintf("[RESIZE] ERROR: RenderManager offset verification failed! "
1012 "device=%p (expected %p) swapchain=%p (expected %p)\n",
1013 *ppRM_Device, g_pd3dDevice, *ppRM_SC, g_pSwapChain);
1014 return false;
1015 }
1016
1017 // Cross-check backbuffer dimension offsets against swap chain desc
1018 DXGI_SWAP_CHAIN_DESC oldScDesc;
1019 g_pSwapChain->GetDesc(&oldScDesc);
1020 bool bbDimsValid = (*pRM_BBWidth == oldScDesc.BufferDesc.Width &&
1021 *pRM_BBHeight == oldScDesc.BufferDesc.Height);
1022 if (!bbDimsValid)
1023 {
1024 app.DebugPrintf("[RESIZE] WARNING: backBuffer dim offsets wrong: "
1025 "stored=%ux%u, swapchain=%ux%u\n",
1026 *pRM_BBWidth, *pRM_BBHeight, oldScDesc.BufferDesc.Width, oldScDesc.BufferDesc.Height);
1027 }
1028
1029 RenderManager.Suspend();
1030 while (!RenderManager.Suspended()) { Sleep(1); }
1031
1032 PostProcesser::GetInstance().Cleanup();
1033
1034 g_pImmediateContext->ClearState();
1035 g_pImmediateContext->Flush();
1036
1037 // Release OUR views and depth buffer
1038 if (g_pRenderTargetView) { g_pRenderTargetView->Release(); g_pRenderTargetView = NULL; }
1039 if (g_pDepthStencilView) { g_pDepthStencilView->Release(); g_pDepthStencilView = NULL; }
1040 if (g_pDepthStencilBuffer) { g_pDepthStencilBuffer->Release(); g_pDepthStencilBuffer = NULL; }
1041
1042 gdraw_D3D11_PreReset();
1043
1044 // Get IDXGIFactory from the existing device BEFORE destroying the old swap chain.
1045 // If anything fails before we have a new swap chain, we abort without destroying
1046 // the old one — leaving the Renderer in a valid (old-size) state.
1047 IDXGISwapChain* pOldSwapChain = g_pSwapChain;
1048 bool success = false;
1049 HRESULT hr;
1050
1051 IDXGIDevice* dxgiDevice = NULL;
1052 IDXGIAdapter* dxgiAdapter = NULL;
1053 IDXGIFactory* dxgiFactory = NULL;
1054 hr = g_pd3dDevice->QueryInterface(__uuidof(IDXGIDevice), (void**)&dxgiDevice);
1055 if (FAILED(hr)) goto postReset;
1056 hr = dxgiDevice->GetParent(__uuidof(IDXGIAdapter), (void**)&dxgiAdapter);
1057 if (FAILED(hr)) { dxgiDevice->Release(); goto postReset; }
1058 hr = dxgiAdapter->GetParent(__uuidof(IDXGIFactory), (void**)&dxgiFactory);
1059 dxgiAdapter->Release();
1060 dxgiDevice->Release();
1061 if (FAILED(hr)) goto postReset;
1062
1063 // Create new swap chain at backbuffer size
1064 {
1065 DXGI_SWAP_CHAIN_DESC sd = {};
1066 sd.BufferCount = 1;
1067 sd.BufferDesc.Width = bbW;
1068 sd.BufferDesc.Height = bbH;
1069 sd.BufferDesc.Format = DXGI_FORMAT_R8G8B8A8_UNORM;
1070 sd.BufferDesc.RefreshRate.Numerator = 60;
1071 sd.BufferDesc.RefreshRate.Denominator = 1;
1072 sd.BufferUsage = DXGI_USAGE_RENDER_TARGET_OUTPUT | DXGI_USAGE_SHADER_INPUT;
1073 sd.OutputWindow = g_hWnd;
1074 sd.SampleDesc.Count = 1;
1075 sd.SampleDesc.Quality = 0;
1076 sd.Windowed = TRUE;
1077
1078 IDXGISwapChain* pNewSwapChain = NULL;
1079 hr = dxgiFactory->CreateSwapChain(g_pd3dDevice, &sd, &pNewSwapChain);
1080 dxgiFactory->Release();

1082 {
1083 app.DebugPrintf("[RESIZE] CreateSwapChain FAILED hr=0x%08X — keeping old swap chain\n", (unsigned)hr);
1084 goto postReset;
1085 }
1086
1087 // New swap chain created successfully — NOW destroy the old one.
1088 // The Renderer's internal RTV/SRV still reference the old backbuffer —
1089 // those COM objects become orphaned (tiny leak, harmless). We DON'T
1090 // release them because unknown code may also hold refs.
1091 pOldSwapChain->Release();

1093 }
1094
1095 // Patch Renderer's swap chain pointer

1102 if (FAILED(hr)) goto postReset;
1103
1104 // Our RTV
1105 hr = g_pd3dDevice->CreateRenderTargetView(pBackBuffer, NULL, &g_pRenderTargetView);
1106 if (FAILED(hr)) { pBackBuffer->Release(); goto postReset; }
1107
1108 // Renderer's internal RTV (offset 0x28)
1109 hr = g_pd3dDevice->CreateRenderTargetView(pBackBuffer, NULL, ppRM_RTV);
1110 if (FAILED(hr)) { pBackBuffer->Release(); goto postReset; }

1112 // Renderer's SRV: separate texture matching backbuffer dims (used by CaptureThumbnail)
1113 D3D11_TEXTURE2D_DESC backDesc = {};
1114 pBackBuffer->GetDesc(&backDesc);
1115 pBackBuffer->Release();

1121 if (SUCCEEDED(hr))
1122 {
1123 hr = g_pd3dDevice->CreateShaderResourceView(srvTexture, NULL, ppRM_SRV);
1124 srvTexture->Release();
1125 }
1126 if (FAILED(hr)) goto postReset;
1127 }

1132 descDepth.Width = bbW;
1133 descDepth.Height = bbH;
1134 descDepth.MipLevels = 1;
1135 descDepth.ArraySize = 1;

1137 descDepth.SampleDesc.Count = 1;
1138 descDepth.SampleDesc.Quality = 0;

1156 if (bbDimsValid)
1157 {

1159 *pRM_BBHeight = (DWORD)bbH;

1178 g_rScreenHeight = bbH;
1179
1180 success = true;
1181
1182postReset:

1184 {
1185 // Failure recovery: recreate our views from whatever swap chain survived
1186 // so ui.m_pRenderTargetView / m_pDepthStencilView don't dangle.
1187 DXGI_SWAP_CHAIN_DESC recoveryDesc;
1188 g_pSwapChain->GetDesc(&recoveryDesc);
1189 int recW = (int)recoveryDesc.BufferDesc.Width;
1190 int recH = (int)recoveryDesc.BufferDesc.Height;
1191
1192 if (g_pRenderTargetView == NULL)
1193 {
1194 ID3D11Texture2D* pBB = NULL;
1195 if (SUCCEEDED(g_pSwapChain->GetBuffer(0, __uuidof(ID3D11Texture2D), (LPVOID*)&pBB)))
1196 {
1197 g_pd3dDevice->CreateRenderTargetView(pBB, NULL, &g_pRenderTargetView);
1198 pBB->Release();
1199 }
1200 }
1201 if (g_pDepthStencilView == NULL)
1202 {
1203 D3D11_TEXTURE2D_DESC dd = {};
1204 dd.Width = recW; dd.Height = recH; dd.MipLevels = 1; dd.ArraySize = 1;
1205 dd.Format = DXGI_FORMAT_D24_UNORM_S8_UINT;
1206 dd.SampleDesc.Count = 1; dd.Usage = D3D11_USAGE_DEFAULT;
1207 dd.BindFlags = D3D11_BIND_DEPTH_STENCIL;
1208 if (g_pDepthStencilBuffer == NULL)
1209 g_pd3dDevice->CreateTexture2D(&dd, NULL, &g_pDepthStencilBuffer);
1210 if (g_pDepthStencilBuffer != NULL)
1211 {
1212 D3D11_DEPTH_STENCIL_VIEW_DESC dsvd = {};
1213 dsvd.Format = DXGI_FORMAT_D24_UNORM_S8_UINT;
1214 dsvd.ViewDimension = D3D11_DSV_DIMENSION_TEXTURE2D;
1215 g_pd3dDevice->CreateDepthStencilView(g_pDepthStencilBuffer, &dsvd, &g_pDepthStencilView);
1216 }
1217 }
1218 if (g_pRenderTargetView != NULL)
1219 g_pImmediateContext->OMSetRenderTargets(1, &g_pRenderTargetView, g_pDepthStencilView);
1220
1221 ui.updateRenderTargets(g_pRenderTargetView, g_pDepthStencilView);
1222
1223 // If the surviving swap chain is the OLD one, dims are unchanged.
1224 // If it's the NEW one (partial failure after swap), update to new dims.
1225 if (g_pSwapChain != pOldSwapChain)
1226 {
1227 g_rScreenWidth = recW;
1228 g_rScreenHeight = recH;
1229 ui.updateScreenSize(recW, recH);
1230 }
1231
1232 app.DebugPrintf("[RESIZE] FAILED but recovered views at %dx%d\n", g_rScreenWidth, g_rScreenHeight);
1233 }
1234
1235 gdraw_D3D11_PostReset();
1236 gdraw_D3D11_SetRendertargetSize(g_rScreenWidth, g_rScreenHeight);
1237 if (success)
1238 IggyFlushInstalledFonts();
1239 RenderManager.Resume();
1240
1241 if (success)
1242 PostProcesser::GetInstance().Init();
1243
1244 return success;
1245}
