427void UIComponent_TutorialPopup::UpdateInteractScenePosition(bool visible)
428{
429 if( m_interactScene == nullptr ) return;
430
431 // 4J-PB - check this players screen section to see if we should allow the animation
432 bool bAllowAnim=false;
433 bool isCraftingScene = (m_interactScene->getSceneType() == eUIScene_Crafting2x2Menu) || (m_interactScene->getSceneType() == eUIScene_Crafting3x3Menu);
434 bool isCreativeScene = (m_interactScene->getSceneType() == eUIScene_CreativeMenu);
435 bool isTradingScene = (m_interactScene->getSceneType() == eUIScene_TradingMenu);
436 switch(Minecraft::GetInstance()->localplayers[m_iPad]->m_iScreenSection)
437 {
438 case C4JRender::VIEWPORT_TYPE_FULLSCREEN:
439 case C4JRender::VIEWPORT_TYPE_SPLIT_TOP:
440 case C4JRender::VIEWPORT_TYPE_SPLIT_BOTTOM:
441 bAllowAnim=true;
442 break;
443 default:
444 // anim allowed for everything except the crafting 2x2 and 3x3, and the creative menu
445 if(!isCraftingScene && !isCreativeScene && !isTradingScene)
446 {
447 bAllowAnim=true;
448 }
449 break;
450 }
451
452 if(bAllowAnim)
453 {
454 bool movingLeft = visible;
455
456 if( (m_lastInteractSceneMoved != m_interactScene && movingLeft) || ( m_lastInteractSceneMoved == m_interactScene && m_lastSceneMovedLeft != movingLeft ) )
457 {
458 if(movingLeft)
459 {
460 m_interactScene->slideLeft();
461 }
462 else
463 {
464 m_interactScene->slideRight();
465 }
466
467 m_lastInteractSceneMoved = m_interactScene;
468 m_lastSceneMovedLeft = movingLeft;
469 }
470 }

474void UIComponent_TutorialPopup::render(S32 width, S32 height, C4JRender::eViewportType viewport)
475{
476 if(viewport != C4JRender::VIEWPORT_TYPE_FULLSCREEN)
477 {
478 // Derive the viewport origin and fit a 16:9 box inside it (same as UIScene::render),
479 // then apply safezone nudges so the popup stays clear of screen edges.
480 F32 originX, originY, viewW, viewH;
481 GetViewportRect(ui.getScreenWidth(), ui.getScreenHeight(), viewport, originX, originY, viewW, viewH);
482
483 S32 fitW, fitH, offsetX, offsetY;
484 Fit16x9(viewW, viewH, fitW, fitH, offsetX, offsetY);
485
486 S32 xPos = static_cast<S32>(originX) + offsetX;
487 S32 yPos = static_cast<S32>(originY) + offsetY;
488
489 //Adjust for safezone
490 switch( viewport )
491 {
492 case C4JRender::VIEWPORT_TYPE_SPLIT_TOP:
493 case C4JRender::VIEWPORT_TYPE_SPLIT_LEFT:
494 case C4JRender::VIEWPORT_TYPE_SPLIT_RIGHT:
495 case C4JRender::VIEWPORT_TYPE_QUADRANT_TOP_LEFT:
496 case C4JRender::VIEWPORT_TYPE_QUADRANT_TOP_RIGHT:
497 yPos += getSafeZoneHalfHeight();
498 break;
499 default: break;
500 }
501 switch( viewport )
502 {
503 case C4JRender::VIEWPORT_TYPE_SPLIT_TOP:
504 case C4JRender::VIEWPORT_TYPE_SPLIT_BOTTOM:
505 case C4JRender::VIEWPORT_TYPE_SPLIT_RIGHT:
506 case C4JRender::VIEWPORT_TYPE_QUADRANT_TOP_RIGHT:
507 case C4JRender::VIEWPORT_TYPE_QUADRANT_BOTTOM_RIGHT:
508 xPos -= getSafeZoneHalfWidth();
509 break;
510 default: break;
511 }
512 ui.setupRenderPosition(xPos, yPos);
513
514 IggyPlayerSetDisplaySize( getMovie(), fitW, fitH );
515 IggyPlayerDraw( getMovie() );
516 }
517 else

519 UIScene::render(width, height, viewport);
