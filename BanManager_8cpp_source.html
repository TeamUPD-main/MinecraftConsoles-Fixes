26 static bool TryParseUtcTimestamp(const std::string &text, unsigned long long *outFileTime)
27 {
28 if (outFileTime == nullptr)
29 {
30 return false;
31 }
32
33 std::string trimmed = StringUtils::TrimAscii(text);
34 if (trimmed.empty())
35 {
36 return false;
37 }
38
39 unsigned year = 0;
40 unsigned month = 0;
41 unsigned day = 0;
42 unsigned hour = 0;
43 unsigned minute = 0;
44 unsigned second = 0;
45 if (sscanf_s(trimmed.c_str(), "%4u-%2u-%2uT%2u:%2u:%2uZ", &year, &month, &day, &hour, &minute, &second) != 6)
46 {
47 return false;
48 }
49
50 SYSTEMTIME utc = {};
51 utc.wYear = (WORD)year;
52 utc.wMonth = (WORD)month;
53 utc.wDay = (WORD)day;
54 utc.wHour = (WORD)hour;
55 utc.wMinute = (WORD)minute;
56 utc.wSecond = (WORD)second;
57
58 FILETIME fileTime = {};
59 if (!SystemTimeToFileTime(&utc, &fileTime))
60 {
61 return false;
62 }
63
64 ULARGE_INTEGER value = {};
65 value.LowPart = fileTime.dwLowDateTime;
66 value.HighPart = fileTime.dwHighDateTime;
67 *outFileTime = value.QuadPart;
68 return true;
