397bool UIScene_CraftingMenu::handleMouseClick(F32 x, F32 y)
398{
399 if (!m_hSlotBoundsValid || m_hSlotSpacing <= 0)
400 return false;
401
402 // Tab click — tabs sit directly above the H slot row. We derive their
403 // bounds from the H slot positions cached in customDraw, since the Vita
404 // TouchPanel controls are full-screen overlays with unusable bounds.
405 int maxTabs = (m_iContainerType == RECIPE_TYPE_3x3) ? m_iMaxGroup3x3 : m_iMaxGroup2x2;
406 F32 slotHeight = m_hSlotY1 - m_hSlotY0;
407 F32 tabRowY0 = (m_hSlotY0 * 0.75f) - slotHeight * 1.55f;
408 F32 tabRowY1 = tabRowY0 + (slotHeight * 1.7f);
409 F32 tabRowWidth = m_hSlotSpacing * m_iCraftablesMaxHSlotC;
410 F32 tabWidth = tabRowWidth / maxTabs;
411
412 if (tabWidth > 0 && x >= m_hSlotX0 && x < m_hSlotX0 + tabRowWidth &&
413 y >= tabRowY0 && y < tabRowY1)
414 {
415 int iTab = (int)((x - m_hSlotX0) / tabWidth);
416 if (iTab >= 0 && iTab < maxTabs && iTab != m_iGroupIndex)
417 {
418 showTabHighlight(m_iGroupIndex, false);
419 m_iGroupIndex = iTab;
420 showTabHighlight(m_iGroupIndex, true);
421 m_iCurrentSlotHIndex = 0;
422 m_iCurrentSlotVIndex = 1;
423 CheckRecipesAvailable();
424 iVSlotIndexA[0] = CanBeMadeA[m_iCurrentSlotHIndex].iCount - 1;
425 iVSlotIndexA[1] = 0;
426 iVSlotIndexA[2] = 1;
427 ui.PlayUISFX(eSFX_Focus);
428 UpdateVerticalSlots();
429 UpdateHighlight();
430 setGroupText(GetGroupNameText(m_pGroupA[m_iGroupIndex]));
431 }
432 return true;
433 }
434
435 // H slot click — select or craft
436 F32 rowWidth = m_hSlotSpacing * m_iCraftablesMaxHSlotC;
437 if (x >= m_hSlotX0 && x < m_hSlotX0 + rowWidth &&
438 y >= m_hSlotY0 && y < m_hSlotY1)
439 {
440 int iNewSlot = (int)((x - m_hSlotX0) / m_hSlotSpacing);
441 if (iNewSlot >= 0 && iNewSlot < m_iCraftablesMaxHSlotC)
442 {
443 // Only interact with populated slots
444 if (CanBeMadeA[iNewSlot].iCount == 0)
445 return true;
446
447 if (iNewSlot == m_iCurrentSlotHIndex)
448 {
449 // Click on already-selected slot — craft the item
450 handleKeyDown(m_iPad, ACTION_MENU_A, false);
451 }
452 else
453 {
454 int iOldHSlot = m_iCurrentSlotHIndex;
455 m_iCurrentSlotHIndex = iNewSlot;
456 m_iCurrentSlotVIndex = 1;
457 iVSlotIndexA[0] = CanBeMadeA[m_iCurrentSlotHIndex].iCount - 1;
458 iVSlotIndexA[1] = 0;
459 iVSlotIndexA[2] = 1;
460 UpdateVerticalSlots();
461 UpdateHighlight();
462 if (CanBeMadeA[iOldHSlot].iCount > 0)
463 setShowCraftHSlot(iOldHSlot, true);
464 ui.PlayUISFX(eSFX_Focus);
465 }
466 return true;
467 }
468 }
469 // Consume all mouse clicks so misses don't generate ACTION_MENU_A
470 // and accidentally craft. Only blocks mouse-originated presses.
471 return true;
472}

525void UIScene_CraftingMenu::customDraw(IggyCustomDrawCallbackRegion *region)
526{
527 Minecraft *pMinecraft = Minecraft::GetInstance();
528 if(pMinecraft->localplayers[m_iPad] == nullptr || pMinecraft->localgameModes[m_iPad] == nullptr) return;
529
530 shared_ptr<ItemInstance> item = nullptr;
531 int slotId = -1;
532 float alpha = 1.0f;
533 bool decorations = true;
534 bool inventoryItem = false;
535 swscanf(static_cast<wchar_t *>(region->name),L"slot_%d",&slotId);
536 if (slotId == -1)
537 {
538 app.DebugPrintf("This is not the control we are looking for\n");
539 }
540 else if(slotId >= CRAFTING_INVENTORY_SLOT_START && slotId < CRAFTING_INVENTORY_SLOT_END)
541 {
542 int iIndex = slotId - CRAFTING_INVENTORY_SLOT_START;
543 iIndex += m_iMenuInventoryStart;
544 Slot *slot = m_menu->getSlot(iIndex);
545 item = slot->getItem();
546 inventoryItem = true;
547 }
548 else if(slotId >= CRAFTING_HOTBAR_SLOT_START && slotId < CRAFTING_HOTBAR_SLOT_END)
549 {
550 int iIndex = slotId - CRAFTING_HOTBAR_SLOT_START;
551 iIndex += m_iMenuHotBarStart;
552 Slot *slot = m_menu->getSlot(iIndex);
553 item = slot->getItem();
554 inventoryItem = true;
555 }
556 else if(slotId >= CRAFTING_V_SLOT_START && slotId < CRAFTING_V_SLOT_END )
557 {
558 decorations = false;
559 int iIndex = slotId - CRAFTING_V_SLOT_START;
560 if(m_vSlotsInfo[iIndex].show)
561 {
562 item = m_vSlotsInfo[iIndex].item;
563 alpha = static_cast<float>(m_vSlotsInfo[iIndex].alpha)/31.0f;
564 }
565 }
566 else if(slotId >= CRAFTING_H_SLOT_START && slotId < (CRAFTING_H_SLOT_START + m_iCraftablesMaxHSlotC) )
567 {
568 decorations = false;
569 int iIndex = slotId - CRAFTING_H_SLOT_START;
570#ifdef _WINDOWS64
571 // Cache H slot SWF-space positions from the custom draw transform matrix
572 if (iIndex == 0 || iIndex == 1)
573 {
574 F32 mat[16];
575 gdraw_D3D11_CalculateCustomDraw_4J(region, mat);
576 // Matrix to SWF coords (same formula as setupCustomDrawMatrices)
577 F32 sw = (F32)getRenderWidth();
578 F32 sh = (F32)getRenderHeight();
579 F32 swfX = sw * (1.0f + mat[3]) / 2.0f;
580 F32 swfY = sh * (1.0f - mat[7]) / 2.0f;
581 if (iIndex == 0)
582 {
583 m_hSlotX0 = swfX;
584 m_hSlotY0 = swfY;
585 // Slot visual height from matrix scale and region height
586 F32 slotH = sh * (-mat[5]) / 2.0f * region->y1;
587 m_hSlotY1 = swfY + slotH;
588 }
589 else
590 {
591 m_hSlotSpacing = swfX - m_hSlotX0;
592 m_hSlotBoundsValid = (m_hSlotSpacing > 0);

638 }

640 {
641 // 4J Stu - If the aux value is set to match any
642 item->setAuxValue(0);
643 }
