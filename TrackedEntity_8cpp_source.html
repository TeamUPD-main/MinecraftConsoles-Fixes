352void TrackedEntity::broadcast(shared_ptr<Packet> packet)
353{
354 if( Packet::canSendToAnyClient( packet ) )
355 {
356 // 4J-PB - due to the knockback on a player being hit, we need to send to all players, but limit the network traffic here to players that have not already had it sent to their system
357 vector< shared_ptr<ServerPlayer> > sentTo;
358
359 // 4J - don't send to a player we've already sent this data to that shares the same machine.
360 // EntityMotionPacket used to limit themselves to sending once to each machine
361 // by only sending to the primary player on each machine. This was causing trouble for split screen
362 // as only the primary player would get a knockback velocity. Now these packets can be sent to any
363 // player, but we try to restrict the network impact this has by not resending to the one machine
364
365 for( auto& player : seenBy)
366 {
367 bool dontSend = false;
368 if( sentTo.size() )
369 {
370 INetworkPlayer *thisPlayer =player->connection->getNetworkPlayer();
371 if( thisPlayer == nullptr )
372 {
373 dontSend = true;
374 }
375 else
376 {
377 for(auto& player2 : sentTo)
378 {
379 INetworkPlayer *otherPlayer = player2->connection->getNetworkPlayer();
380 if( otherPlayer != nullptr && thisPlayer->IsSameSystem(otherPlayer) )
381 {
382 dontSend = true;
383 }
384 }
385 }
386 }
387 if( dontSend )
388 {
389 continue;
390 }
391
392
393 player->connection->send(packet);
394 sentTo.push_back(player);
395 }
396 }
397 else
398 {
399 // This packet hasn't got canSendToAnyClient set, so just send to everyone here, and it
400
401 for(auto& it : seenBy)
402 {
403 it->connection->send(packet);
404 }
405 }
