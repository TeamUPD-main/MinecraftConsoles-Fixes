68bool MultiPlayerGameMode::destroyBlock(int x, int y, int z, int face)
69{
70 if (localPlayerMode->isAdventureRestricted()) {
71 if (!minecraft->player->mayDestroyBlockAt(x, y, z)) {
72 return false;
73 }
74 }
75
76 if (localPlayerMode->isCreative())
77 {
78 if (minecraft->player->getCarriedItem() != nullptr && dynamic_cast<WeaponItem *>(minecraft->player->getCarriedItem()->getItem()) != nullptr)
79 {
80 return false;
81 }
82 }
83
84 Level *level = minecraft->level;
85 Tile *oldTile = Tile::tiles[level->getTile(x, y, z)];
86
87 if (oldTile == nullptr) return false;
88
89 level->levelEvent(LevelEvent::PARTICLES_DESTROY_BLOCK, x, y, z, oldTile->id + (level->getData(x, y, z) << Tile::TILE_NUM_SHIFT));
90
91 int data = level->getData(x, y, z);
92 bool changed = level->removeTile(x, y, z);
93 if (changed)
94 {
95 oldTile->destroy(level, x, y, z, data);
96 }
97 yDestroyBlock = -1;
98
99 if (!localPlayerMode->isCreative())
100 {
101 shared_ptr<ItemInstance> item = minecraft->player->getSelectedItem();
102 if (item != nullptr)
103 {
104 item->mineBlock(level, oldTile->id, x, y, z, minecraft->player);
105 if (item->count == 0)
106 {
107 minecraft->player->removeSelectedItem();
108 }
109 }
110 }
111
112 return changed;
113}

283bool MultiPlayerGameMode::useItemOn(shared_ptr<Player> player, Level *level, shared_ptr<ItemInstance> item, int x, int y, int z, int face, Vec3 *hit, bool bTestUseOnly, bool *pbUsedItem)
284{
285 if( pbUsedItem ) *pbUsedItem = false; // Did we actually use the held item?
286
287 // 4J-PB - Adding a test only version to allow tooltips to be displayed
288 if(!bTestUseOnly)
289 {
290 ensureHasSentCarriedItem();
291 }
292 float clickX = static_cast<float>(hit->x) - x;
293 float clickY = static_cast<float>(hit->y) - y;
294 float clickZ = static_cast<float>(hit->z) - z;
295 bool didSomething = false;
296
297 if (!player->isSneaking() || player->getCarriedItem() == nullptr)
298 {
299 int t = level->getTile(x, y, z);
300 if (t > 0 && player->isAllowedToUse(Tile::tiles[t]))
301 {
302 if(bTestUseOnly)
303 {
304 switch(t)
305 {
306 case Tile::jukebox_Id:
307 case Tile::bed_Id: // special case for a bed
308 if (Tile::tiles[t]->TestUse(level, x, y, z, player ))
309 {
310 return true;
311 }
312 else if (t==Tile::bed_Id) // 4J-JEV: You can still use items on record players (ie. set fire to them).
313 {
314 // bed is too far away, or something
315 return false;
316 }
317 break;
318 default:
319 if (Tile::tiles[t]->TestUse()) return true;
320 break;
321 }
322 }
323 else
324 {
325 if (Tile::tiles[t]->use(level, x, y, z, player, face, clickX, clickY, clickZ)) didSomething = true;
326 }
327 }
328 }
329
330 if (!didSomething && item != nullptr && dynamic_cast<TileItem *>(item->getItem()))
331 {
332 TileItem *tile = dynamic_cast<TileItem *>(item->getItem());
333 if (!tile->mayPlace(level, x, y, z, face, player, item)) return false;
334 }
335
336 // 4J Stu - In Java we send the use packet before the above check for item being nullptr
337 // so the following never gets executed but the packet still gets sent (for opening chests etc)
338 if(item != nullptr)
339 {
340 if(!didSomething && player->isAllowedToUse(item))
341 {
342 if (localPlayerMode->isCreative())
343 {
344 int aux = item->getAuxValue();
345 int count = item->count;
346 didSomething = item->useOn(player, level, x, y, z, face, clickX, clickY, clickZ, bTestUseOnly);
347 item->setAuxValue(aux);
348 item->count = count;
349 }
350 else
351 {
352 didSomething = item->useOn(player, level, x, y, z, face, clickX, clickY, clickZ, bTestUseOnly);
353 }
354 if( didSomething )
355 {
356 if( pbUsedItem ) *pbUsedItem = true;
357 }
358 }
359 }
360 else
361 {
362 int t = level->getTile(x, y, z);
363 // 4J - Bit of a hack, however seems preferable to any larger changes which would have more chance of causing unwanted side effects.
364 // If we aren't going to be actually performing the use method locally, then call this method with its "soundOnly" parameter set to true.
365 // This is an addition from the java version, and as its name suggests, doesn't actually perform the use locally but just makes any sounds that
366 // are meant to be directly caused by this. If we don't do this, then the sounds never happen as the tile's use method is only called on the
367 // server, and that won't allow any sounds that are directly made, or broadcast back level events to us that would make the sound, since we are
368 // the source of the event.
369 // ---------------------------------------------------------------------------------
370 // Only call soundOnly version if we didn't already call the tile's use method above
371 if( !didSomething && ( t > 0 ) && ( !bTestUseOnly ) && player->isAllowedToUse(Tile::tiles[t]) )
372 {
373 Tile::tiles[t]->use(level, x, y, z, player, face, clickX, clickY, clickZ, true);
374 }
375 }
376
377 // 4J Stu - Do the action before we send the packet, so that our predicted count is sent in the packet and the server
378 // doesn't think it has to update us
379 // Fix for #7904 - Gameplay: Players can dupe torches by throwing them repeatedly into water.
380 if(!bTestUseOnly)
381 {
382 connection->send(std::make_shared<UseItemPacket>(x, y, z, face, player->inventory->getSelected(), clickX, clickY, clickZ));
383 }
384 return didSomething;
385}

387bool MultiPlayerGameMode::useItem(shared_ptr<Player> player, Level *level, shared_ptr<ItemInstance> item, bool bTestUseOnly)
388{
389 if(!player->isAllowedToUse(item)) return false;
390
391 // 4J-PB - Adding a test only version to allow tooltips to be displayed
392 if(!bTestUseOnly)
393 {
394 ensureHasSentCarriedItem();
395 }
396
397 // 4J Stu - Do the action before we send the packet, so that our predicted count is sent in the packet and the server
398 // doesn't think it has to update us, or can update us if we are wrong
399 // Fix for #13120 - Using a bucket of water or lava in the spawn area (centre of the map) causes the inventory to get out of sync
400 bool result = false;
401
402 // 4J-PB added for tooltips to test use only
403 if(bTestUseOnly)
404 {
405 result = item->TestUse(item, level, player);
406 }
407 else
408 {
409 int oldCount = item->count;
410 shared_ptr<ItemInstance> itemInstance = item->use(level, player);
411 if ((itemInstance != nullptr && itemInstance != item) || (itemInstance != nullptr && itemInstance->count != oldCount))
412 {
413 player->inventory->items[player->inventory->selected] = itemInstance;
414 if (itemInstance->count == 0)
415 {
416 player->inventory->items[player->inventory->selected] = nullptr;
417 }
418 result = true;
419 }
420 }
421
422 if(!bTestUseOnly)
423 {
424 connection->send(std::make_shared<UseItemPacket>(-1, -1, -1, 255, player->inventory->getSelected(), 0, 0, 0));
425 }
426 return result;
427}
