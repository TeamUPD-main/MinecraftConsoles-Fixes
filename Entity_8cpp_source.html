41int Entity::getSmallId()
42{
43 unsigned int *puiUsedFlags = entityIdUsedFlags;
44 unsigned int *puiRemovedFlags = nullptr;
45
46 // If we are the server (we should be, if we are allocating small Ids), then check with the server if there are any small Ids which are
47 // still in the ServerPlayer's vectors of entities to be removed - these are used to gather up a set of entities into one network packet
48 // for final notification to the client that the entities are removed. We can't go re-using these small Ids yet, as otherwise we will
49 // potentially end up telling the client that the entity has been removed After we have already re-used its Id and created a new entity.
50 // This ends up with newly created client-side entities being removed by accident, causing invisible mobs.
51 if( ((size_t)TlsGetValue(tlsIdx) != 0 ) )
52 {
53 MinecraftServer *server = MinecraftServer::getInstance();
54 if( server )
55 {
56 // In some attempt to optimise this, flagEntitiesToBeRemoved most of the time shouldn't do anything at all, and in this case it
57 // doesn't even memset the entityIdRemovingFlags array, so we shouldn't use it if the return value is false.
58 bool removedFound = server->flagEntitiesToBeRemoved(entityIdRemovingFlags);
59 if( removedFound )
60 {
61 // Has set up the entityIdRemovingFlags vector in this case, so we should check against this when allocating new ids
62 // app.DebugPrintf("getSmallId: Removed entities found\n");
63 puiRemovedFlags = entityIdRemovingFlags;
64 }
65 }
66 }
67
68 for( int i = 0; i < (2048 / 32 ); i++ )
69 {
70 unsigned int uiFlags = *puiUsedFlags;
71 if( uiFlags != 0xffffffff )
72 {
73 unsigned int uiMask = 0x80000000;
74 for( int j = 0; j < 32; j++ )
75 {
76 // See comments above - now checking (if required) that these aren't newly removed entities that the clients still haven't been told about,
77 // so we don't reuse those ids before we should.
78 if( puiRemovedFlags )
79 {
80 if( puiRemovedFlags[i] & uiMask )
81 {
82 // app.DebugPrintf("Avoiding using ID %d (0x%x)\n", i * 32 + j,puiRemovedFlags[i]);
83 uiMask >>= 1;
84 continue;
85 }
86 }
87 if( ( uiFlags & uiMask ) == 0 )
88 {
89 uiFlags |= uiMask;
90 *puiUsedFlags = uiFlags;
91 return i * 32 + j;
92 }
93 uiMask >>= 1;
94 }
95 }
96 puiUsedFlags++;
97 }

99 app.DebugPrintf("Out of small entity Ids... possible leak?\n");

784

787 bb->move(0, 0, za);
788
789 if (!slide && zaOrg != za)
790 {
791 xa = ya = za = 0;
792 }
793
794 if (footSize > 0 && og && (isPlayerSneaking || ySlideOffset < 0.05f) && ((xaOrg != xa) || (zaOrg != za)))
795 {
796 double xaN = xa;
797 double yaN = ya;
798 double zaN = za;
799
800 xa = xaOrg;
801 ya = footSize;
802 za = zaOrg;
803
804 AABB *normal = bb->copy();
805 bb->set(bbOrg);
806 // 4J - added extra expand, as if we don't move up by footSize by hitting a block above us, then overall we could be trying to move as much as footSize downwards,
807 // so we'd better include cubes under our feet in this list of things we might possibly collide with
808 aABBs = level->getCubes(shared_from_this(), bb->expand(xa, ya, za)->expand(0,-ya,0),false,true);
809
810 if(!level->isClientSide || level->reallyHasChunk(xc, zc))
811 {
812 // 4J Stu - It's horrible that the client is doing any movement at all! But if we don't have the chunk
813 // data then all the collision info will be incorrect as well
814 for ( auto& it : *aABBs )
815 ya = it->clipYCollide(bb, ya);
816 bb->move(0, ya, 0);
817 }
818
819 if (!slide && yaOrg != ya)
820 {
821 xa = ya = za = 0;
822 }
823
824
825 for ( auto& it : *aABBs )
826 xa = it->clipXCollide(bb, xa);
827 bb->move(xa, 0, 0);
828
829 if (!slide && xaOrg != xa)
830 {
831 xa = ya = za = 0;
832 }
833
834 for ( auto& it : *aABBs )
835 za = it->clipZCollide(bb, za);
836 bb->move(0, 0, za);
837
838 if (!slide && zaOrg != za)
839 {
840 xa = ya = za = 0;
841 }
842
843
844 if (!slide && yaOrg != ya)
845 {
846 xa = ya = za = 0;
847 }
848 else
849 {
850 ya = -footSize;
851 // LAND FIRST, then x and z
852 for ( auto& it : *aABBs )
853 ya = it->clipYCollide(bb, ya);
854 bb->move(0, ya, 0);
855 }
856
857 if (xaN * xaN + zaN * zaN >= xa * xa + za * za)
858 {
859 xa = xaN;
860 ya = yaN;
861 za = zaN;
862 bb->set(normal);

864 }
865
866
867 x = (bb->x0 + bb->x1) / 2.0f;
868 y = bb->y0 + heightOffset - ySlideOffset;
869 z = (bb->z0 + bb->z1) / 2.0f;

871 horizontalCollision = (xaOrg != xa) || (zaOrg != za);
872 verticalCollision = !m_ignoreVerticalCollisions && (yaOrg != ya);
873 onGround = !m_ignoreVerticalCollisions && yaOrg != ya && yaOrg < 0;
874 collision = horizontalCollision || verticalCollision;
875 checkFallDamage(ya, onGround);
876
877 if (xaOrg != xa) xd = 0;
878 if (yaOrg != ya) yd = 0;
879 if (zaOrg != za) zd = 0;
880
881 double xm = x - xo;
882 double ym = y - yo;
883 double zm = z - zo;
884
885
886 if (makeStepSound() && !isPlayerSneaking && riding == nullptr)
887 {
888 int xt = Mth::floor(x);
889 int yt = Mth::floor(y - 0.2f - heightOffset);
890 int zt = Mth::floor(z);
891 int t = level->getTile(xt, yt, zt);
892 if (t == 0)
893 {
894 int renderShape = level->getTileRenderShape(xt, yt - 1, zt);
895 if (renderShape == Tile::SHAPE_FENCE || renderShape == Tile::SHAPE_WALL || renderShape == Tile::SHAPE_FENCE_GATE)
896 {
897 t = level->getTile(xt, yt - 1, zt);
898 }
899 }
900 if (t != Tile::ladder_Id)
901 {
902 ym = 0;
903 }
904
905 walkDist += Mth::sqrt(xm * xm + zm * zm) * 0.6;
906 moveDist += Mth::sqrt(xm * xm + ym * ym + zm * zm) * 0.6;
907
908 if (moveDist > nextStep && t > 0)
909 {
910 nextStep = static_cast<int>(moveDist) + 1;
911 if (isInWater())
912 {
913 float speed = Mth::sqrt(xd * xd * 0.2f + yd * yd + zd * zd * 0.2f) * 0.35f;
914 if (speed > 1) speed = 1;
915 playSound(eSoundType_LIQUID_SWIM, speed, 1 + (random->nextFloat() - random->nextFloat()) * 0.4f);
916 }
917 playStepSound(xt, yt, zt, t);
918 Tile::tiles[t]->stepOn(level, xt, yt, zt, shared_from_this());
919 }
920 }
921
922 checkInsideTiles();
923
924
925 bool water = isInWaterOrRain();
926 if (level->containsFireTile(bb->shrink(0.001, 0.001, 0.001)))
927 {
928 burn(1);
929 if (!water)
930 {
931 onFire++;
932 if (onFire == 0) setOnFire(8);
933 }
934 }
935 else
936 {
937 if (onFire <= 0)
938 {
939 onFire = -flameTime;
940 }
941 }
942
943 if (water && onFire > 0)
944 {
945 playSound(eSoundType_RANDOM_FIZZ, 0.7f, 1.6f + (random->nextFloat() - random->nextFloat()) * 0.4f);
