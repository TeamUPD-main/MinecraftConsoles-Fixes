239static void wait_on_fence(GDrawFence fence)
240{
241 // we don't actually wait here, current Orbis GDraw memory management is
242 // non-blocking by design. but the resource manager emits these when it's
243 // about to free (and then reuse) a resource that was previously in use,
244 // so when we get the call, make sure to finish shading first, because
245 // the following commands are likely to stomp over (part of) the resource
246 // pools.
247 if (is_fence_pending(fence)) {
248 void *label = insert_cb_label();
249 gdraw->gfxc->writeAtEndOfShader(Gnm::kEosPsDone, label, 1);
250 gdraw->gfxc->waitOnAddress(label, ~0, Gnm::kWaitCompareFuncEqual, 1);
251 }
252}

302static void gdraw_gpu_memcpy(GDrawHandleCache *c, void *dst, void *src, U32 num_bytes)
303{
304 Gnmx::GfxContext *gfxc = gdraw->gfxc;
305 Gnm::Buffer src_buf, dst_buf;
306 U32 num_16b = (num_bytes + 15) / 16;
307
308 src_buf.initAsDataBuffer(src, Gnm::kDataFormatR32G32B32A32Uint, num_16b);
309 dst_buf.initAsDataBuffer(dst, Gnm::kDataFormatR32G32B32A32Uint, num_16b);
310 src_buf.setResourceMemoryType(Gnm::kResourceMemoryTypeGC);
311 dst_buf.setResourceMemoryType(Gnm::kResourceMemoryTypeGC);
312
313 gfxc->setBuffers(Gnm::kShaderStageCs, 0, 1, &src_buf);
314 gfxc->setRwBuffers(Gnm::kShaderStageCs, 0, 1, &dst_buf);
315 gfxc->dispatch(1, 1, 1);
316
317 // need to sync before the next one can start (because of potential overlaps)
318 // no need to flush L1 because defragment doesn't read any data it just wrote.
319 compute_to_compute_sync();
320}

322static void gdraw_defragment_cache(GDrawHandleCache *c, GDrawStats *stats)
323{
324 if (!gdraw_CanDefragment(c))
325 return;
326
327 gdraw->gfxc->setShaderType(Gnm::kShaderTypeCompute);
328 gdraw->gfxc->setCsShader(gdraw->defragment_cs.cs);
329
330 // actual defragmentation...
331 gdraw_DefragmentMain(c, GDRAW_DEFRAGMENT_may_overlap, stats);
332
333 // go over all handles and adjust pointers.
334 // pointer adjustment is different between textures and vertex buffers
335 if (!c->is_vertex) {
336 for (S32 i=0; i < c->max_handles; i++) {
337 GDrawHandle *h = &c->handle[i];
338 if (gdraw_res_is_managed(h)) {
339 Gnm::Texture *tex = h->handle.tex.gnm;
340 tex->setBaseAddress(h->raw_ptr);
341 }
342 }
343 } else {
344 for (S32 i=0; i < c->max_handles; i++) {
345 GDrawHandle *h = &c->handle[i];
346 if (gdraw_res_is_managed(h)) {
347 SINTa index_offs = (U8 *)h->handle.vbuf.inds - (U8 *)h->handle.vbuf.verts;
348 h->handle.vbuf.verts = h->raw_ptr;
349 h->handle.vbuf.inds = (U8 *)h->raw_ptr + index_offs;
350 }
351 }
352 }
353
354 // synchronize
355 compute_to_graphics_sync();
356 gdraw->gfxc->setCsShader(nullptr);
357 gdraw->gfxc->setShaderType(Gnm::kShaderTypeGraphics);
358
359 // don't need to wait till GPU is done since we never access GPU memory from the
360 // CPU, we always go through the staging buffer.
361}

474static void upload_tex_subrect(Gnm::Texture *dst, U32 dst_mip, U32 dst_x, U32 dst_y, const void *src, U32 pitch, U32 ncols, U32 nrows, U32 bpp)
475{
476 // set up texture descriptions
477 Gnm::Texture dst_tex = *dst;
478 dst_tex.setMipLevelRange(dst_mip, dst_mip);
479 dst_tex.setResourceMemoryType(Gnm::kResourceMemoryTypeGC);
480
481 Gnm::Texture src_tex;
482 Gnm::SizeAlign sa = src_tex.initAs2d(ncols, nrows, 1, dst_tex.getDataFormat(), Gnm::kTileModeDisplay_LinearAligned, Gnm::kNumSamples1);
483
484 // alloc staging buffer for source texture
485 U8 *staging = (U8 *) alloc_staging(sa.m_size, sa.m_align);
486 if (!staging) // if out of staging mem, bail
487 return;
488
489 src_tex.setBaseAddress(staging);
490 src_tex.setResourceMemoryType(Gnm::kResourceMemoryTypeGC);
491
492 // copy source data into staging buffer
493 U32 staging_pitch_bytes = src_tex.getPitch() * bpp;
494 U32 copy_bytes = ncols * bpp;
495 for (U32 y=0; y < nrows; y++)
496 memcpy(staging + y * staging_pitch_bytes, (const U8 *) src + y * pitch, copy_bytes);
497
498 upload_dispatch(dst_tex, src_tex, dst_x, dst_y, ncols, nrows);
499}

501static void gpu_compute_memset(void *ptr, U32 value, U32 size_in_bytes)
502{
503 assert((size_in_bytes & 3) == 0);
504
505 Gnmx::GfxContext *gfxc = gdraw->gfxc;
506 gfxc->setShaderType(Gnm::kShaderTypeCompute);
507 gfxc->setCsShader(gdraw->memset_cs.cs);
508
509 auto para = embed_const_buffer<U32>(Gnm::kShaderStageCs, 0);
510 *para = value;
511
512 // we tag resources as SC (system coherent); i.e. no L1, write to L2,
513 // tag as volatile so we can flush it cheaply.
514
515 Gnm::Buffer buf;
516 if (size_in_bytes >= 16) {
517 U32 nelem = size_in_bytes / 16;
518
519 buf.initAsDataBuffer(ptr, Gnm::kDataFormatR32G32B32A32Uint, nelem);
520 buf.setResourceMemoryType(Gnm::kResourceMemoryTypeSC);
521 gfxc->setRwBuffers(Gnm::kShaderStageCs, 0, 1, &buf);
522 gfxc->dispatch((nelem + 255) / 256, 1, 1);
523 }
524
525 if (size_in_bytes & 0xf) {
526 buf.initAsDataBuffer((U8 *)ptr + (size_in_bytes & ~0xf), Gnm::kDataFormatR32Uint, (size_in_bytes & 0xf) / 4);
527 buf.setResourceMemoryType(Gnm::kResourceMemoryTypeSC);
528 gfxc->setRwBuffers(Gnm::kShaderStageCs, 0, 1, &buf);
529 gfxc->dispatch(1, 1, 1);
530 }
531
532 compute_to_compute_sync();
533
534 // write back L2 volatile here to make sure the values reach memory.
535 // this is all necessary because this function is used to clear HTile
536 // buffers which are written by the CB/DB blocks and not accessed
537 // through the regular caches.
538 gfxc->flushShaderCachesAndWait(Gnm::kCacheActionWriteBackL2Volatile, 0, Gnm::kStallCommandBufferParserDisable);
539 gfxc->setShaderType(Gnm::kShaderTypeGraphics);
540 gfxc->setCsShader(nullptr);
541}

583static rrbool RADLINK gdraw_MakeTextureBegin(void *owner, S32 width, S32 height, gdraw_texture_format gformat, U32 flags, GDraw_MakeTexture_ProcessingInfo *p, GDrawStats *stats)
584{
585 S32 bytes_pixel = 4;
586 GDrawHandle *t = nullptr;
587 Gnm::Texture gt;
588 Gnm::SizeAlign sa;
589
590 Gnm::DataFormat format = Gnm::kDataFormatR8G8B8A8Unorm;
591 if (width > MAX_TEXTURE2D_DIM || height > MAX_TEXTURE2D_DIM) {
592 IggyGDrawSendWarning(nullptr, "GDraw %d x %d texture not supported by hardware (dimension limit %d)", width, height, MAX_TEXTURE2D_DIM);
593 return false;
594 }
595
596 if (gformat == GDRAW_TEXTURE_FORMAT_font) {
597 format = Gnm::kDataFormatR8Unorm;
598 bytes_pixel = 1;
599 }
600
601 // don't do mipmaps for gradients!
602 if (height == 1)
603 flags &= ~GDRAW_MAKETEXTURE_FLAGS_mipmap;
604
605 // determine the number of mipmaps to use and size of resulting surface
606 U32 mipmaps = 0;
607 do
608 mipmaps++;
609 while ((flags & GDRAW_MAKETEXTURE_FLAGS_mipmap) && ((width >> mipmaps) || (height >> mipmaps)));
610
611 sa = gt.initAs2d(width, height, mipmaps, format, (height == 1) ? Gnm::kTileModeDisplay_LinearAligned : Gnm::kTileModeThin_1dThin, Gnm::kNumSamples1);
612
613 if (gformat == GDRAW_TEXTURE_FORMAT_font) // we want an A8 not R8 texture!
614 gt.setChannelOrder(Gnm::kTextureChannelX, Gnm::kTextureChannelX, Gnm::kTextureChannelX, Gnm::kTextureChannelX);
615
616 // Make sure we actually satisfy alignment requirements
617 assert(sa.m_align <= GDRAW_ORBIS_TEXTURE_ALIGNMENT);
618
619 // Determine space requirements for the upload texture and check if there's enough space
620 // do this before gdraw_res_alloc_begin so we don't start freeing resources to make space
621 // only to later discover that we can't proceed due to lack of staging mem anyway.
622 Gnm::SizeAlign sa_up = gdraw->upload_tex.initAs2d(width, height, 1, format, Gnm::kTileModeDisplay_LinearAligned, Gnm::kNumSamples1);
623 if (!can_staging_satisfy(sa_up.m_size, sa_up.m_align)) {
624 track_staging_alloc_attempt(sa_up.m_size, sa_up.m_align);
625 track_staging_alloc_failed();
626 return false;
627 }
628
629 // allocate a handle and make room in the cache for this much data
630 U32 size = sa.m_size;
631 t = gdraw_res_alloc_begin(gdraw->texturecache, size, stats);
632 if (!t)
633 return false;
634
635 t->handle.tex.gnm_ptr = t->raw_ptr;
636 gt.setBaseAddress(t->raw_ptr);
637 *t->handle.tex.gnm = gt;
638
639 // allocate staging texture (we checked that there was enough space earlier)
640 void *upload_ptr = alloc_staging(sa_up.m_size, sa_up.m_align);
641 if (!upload_ptr) {
642 // not supposed to happen - we checked there was enough space earlier!
643 // but if we ever get here, be sure to handle it properly anyway.
644 assert(0);
645 gdraw_HandleCacheAllocateFail(t);
646 return false;
647 }
648 gdraw->upload_tex.setBaseAddress(upload_ptr);
649
650 gdraw_HandleCacheAllocateEnd(t, size, owner, (flags & GDRAW_MAKETEXTURE_FLAGS_never_flush) ? GDRAW_HANDLE_STATE_pinned : GDRAW_HANDLE_STATE_locked);
651 stats->nonzero_flags |= GDRAW_STATS_alloc_tex;
652 stats->alloc_tex += 1;
653 stats->alloc_tex_bytes += size;
654
655 p->texture_type = GDRAW_TEXTURE_TYPE_rgba;
656 p->p0 = t;
657 p->texture_data = (U8 *) upload_ptr;
658 p->num_rows = height; // just send the whole texture at once
659 p->stride_in_bytes = gdraw->upload_tex.getPitch() * bytes_pixel;
660
661 return true;
662}

669static GDrawTexture * RADLINK gdraw_MakeTextureEnd(GDraw_MakeTexture_ProcessingInfo *p, GDrawStats *stats)
670{
671 GDrawHandle *t = (GDrawHandle *) p->p0;
672 Gnm::Texture *gnm_tex = t->handle.tex.gnm;
673 Gnmx::GfxContext *gfxc = gdraw->gfxc;
674 U32 width = gnm_tex->getWidth();
675 U32 height = gnm_tex->getHeight();
676
677 // upload the mip data
678 upload_begin();
679
680 Gnm::Texture dst_tex = *gnm_tex;
681 dst_tex.setResourceMemoryType(Gnm::kResourceMemoryTypeGC);
682 Gnm::Texture src_tex = dst_tex;
683 dst_tex.setMipLevelRange(0, 0);
684 upload_dispatch(dst_tex, gdraw->upload_tex, 0, 0, width, height);
685
686 upload_end();
687
688 // compute the mip maps
689 gfxc->setShaderType(Gnm::kShaderTypeCompute);
690 gfxc->setCsShader(gdraw->mipgen_cs.cs);
691
692 for (U32 mip=1; mip <= gnm_tex->getLastMipLevel(); mip++) {
693 U32 mipw = RR_MAX(gnm_tex->getWidth() >> mip, 1);
694 U32 miph = RR_MAX(gnm_tex->getHeight() >> mip, 1);
695
696 src_tex.setMipLevelRange(mip - 1, mip - 1);
697 dst_tex.setMipLevelRange(mip, mip);
698
699 auto para = embed_const_buffer<MipGenPara>(Gnm::kShaderStageCs, 0);
700 para->sizeMinus1[0] = RR_MAX(gnm_tex->getWidth() >> (mip - 1), 1) - 1;
701 para->sizeMinus1[1] = RR_MAX(gnm_tex->getHeight() >> (mip - 1), 1) - 1;
702
703 gfxc->setTextures(Gnm::kShaderStageCs, 0, 1, &src_tex);
704 gfxc->setRwTextures(Gnm::kShaderStageCs, 0, 1, &dst_tex);
705 gfxc->dispatch((mipw + 7) / 8, (miph + 7) / 8, 1);
706 if (mip < gnm_tex->getLastMipLevel())
707 compute_to_compute_sync();
708 else
709 compute_to_graphics_sync();
710 }
711
712 gfxc->setShaderType(Gnm::kShaderTypeGraphics);
713
714 return (GDrawTexture *) p->p0;
715}

1062static void set_common_renderstate()
1063{
1064 Gnmx::GfxContext *gfxc = gdraw->gfxc;
1065
1066 // clear our state caching
1067 memset(gdraw->active_tex, 0, sizeof(gdraw->active_tex));
1068 gdraw->cur_ps = nullptr;
1069 gdraw->scissor_state = ~0u;
1070 gdraw->blend_mode = -1;
1071
1072 // all the state we won't touch again until we're done rendering
1073 Gnm::ClipControl clip_control;
1074 clip_control.init();
1075 clip_control.setClipSpace(Gnm::kClipControlClipSpaceDX);
1076 gfxc->setClipControl(clip_control);
1077 gfxc->setShaderType(Gnm::kShaderTypeGraphics);
1078 gfxc->setIndexSize(Gnm::kIndexSize16);
1079 gfxc->setStencilClearValue(0);
1080 gfxc->setDepthClearValue(1.0f);
1081 gfxc->setVsShader(gdraw->vs.vs, 0, (void*)0);
1082
1083 Gnm::StencilOpControl stencil_op;
1084 stencil_op.init();
1085 stencil_op.setStencilOps(Gnm::kStencilOpKeep, Gnm::kStencilOpReplaceTest, Gnm::kStencilOpKeep);
1086 gfxc->setStencilOpControl(stencil_op);
1087
1088 Gnm::ViewportTransformControl vt_control;
1089 vt_control.init();
1090 vt_control.setPassThroughEnable(false);
1091 gfxc->setViewportTransformControl(vt_control);
1092
1093 // set up guard band and hardware screen offset once
1094 // we know ahead of time which viewports we're going to set:
1095 // * our viewport top/left corner is always >= (0,0)
1096 // * viewport bottom/right is <= size of the largest render target
1097 S32 min_x = 0;
1098 S32 min_y = 0;
1099 S32 max_x = RR_MAX(gdraw->main_colorbuffer.getWidth(), gdraw->rt_colorbuffer.getWidth());
1100 S32 max_y = RR_MAX(gdraw->main_colorbuffer.getHeight(), gdraw->rt_colorbuffer.getHeight());
1101
1102 F32 offs_x = (F32) (min_x + max_x) * 0.5f;
1103 F32 offs_y = (F32) (min_y + max_y) * 0.5f;
1104 F32 abs_scale_x = (F32) (max_x - min_x) * 0.5f;
1105 F32 abs_scale_y = (F32) (max_y - min_y) * 0.5f;
1106
1107 // set up guard band offset so we're centered around our viewport region
1108 // hardware offset must be a multiple of 16 pixels
1109 S32 hw_offset_x = (S32)floorf(offs_x/16.0f + 0.5f) * 16;
1110 S32 hw_offset_y = (S32)floorf(offs_y/16.0f + 0.5f) * 16;
1111 gfxc->setHardwareScreenOffset(hw_offset_x >> 4, hw_offset_y >> 4);
1112
1113 // set up guard band clip and discard distances
1114 // NB both the values for hw_min and hw_max are slightly smaller than the actual min/max
1115 // (by about 1/256th) to keep a bit of a safety margin for FP round-off error
1116 F32 hw_min = -(F32)(0xff<<16) / (F32)(1<<8);
1117 F32 hw_max = (F32)(0xff<<16) / (F32)(1<<8);
1118 F32 gb_max_x = RR_MIN(hw_max - abs_scale_x - offs_x + hw_offset_x, -abs_scale_x + offs_x - hw_offset_x - hw_min);
1119 F32 gb_max_y = RR_MIN(hw_max - abs_scale_y - offs_y + hw_offset_y, -abs_scale_y + offs_y - hw_offset_y - hw_min);
1120 gfxc->setGuardBandClip(gb_max_x / abs_scale_x, gb_max_y / abs_scale_y);
1121 gfxc->setGuardBandDiscard(1.0f, 1.0f);
1122
1123 assert(gdraw->aa_tex.isTexture()); // if this triggers, your initialization sequence is wrong.
1124 set_gnm_texture(AATEX_SAMPLER, &gdraw->aa_tex, GDRAW_WRAP_clamp, 0);
1125
1126 // states we modify during regular rendering
1127 clear_renderstate();
1128 set_render_target();
1129 set_viewport();
1130 set_projection();
1131}

1157static void clear_whole_zs(bool clear_depth, bool clear_stencil, GDrawStats *stats)
1158{
1159 Gnm::DepthRenderTarget &depthbuf = gdraw->main_depthbuffer;
1160
1161 // to clear both depth and stencil, we can just set up the metadata in HTile with a
1162 // compute shader if a) there's a HTile to begin with and b) it contains stencil metadata.
1163 // if no stencil info in HTile, clearing HTile manually is a net perf loss.
1164 if (clear_depth && clear_stencil && depthbuf.getHtileAccelerationEnable() && !depthbuf.getHtileStencilDisable()) {
1165 gdraw->gfxc->triggerEvent(Gnm::kEventTypeFlushAndInvalidateDbMeta);
1166 gpu_compute_memset(depthbuf.getHtileAddress(), 0xfffc00f0, depthbuf.getHtileSizeInBytes());
1167 } else {
1168 S32 w = gdraw->main_depthbuffer.getWidth();
1169 S32 h = gdraw->main_depthbuffer.getHeight();
1170
1171 Gnmx::GfxContext *gfxc = gdraw->gfxc;
1172 Gnm::DbRenderControl db_control;
1173 db_control.init();
1174 db_control.setDepthClearEnable(clear_depth);
1175 db_control.setStencilClearEnable(clear_stencil);
1176 gfxc->setDbRenderControl(db_control);
1177
1178 Gnm::DepthStencilControl ds_control;
1179 ds_control.init();
1180 ds_control.setDepthControl(clear_depth ? Gnm::kDepthControlZWriteEnable : Gnm::kDepthControlZWriteDisable, Gnm::kCompareFuncAlways);
1181 ds_control.setStencilFunction(Gnm::kCompareFuncAlways);
1182 ds_control.setDepthEnable(clear_depth);
1183 ds_control.setStencilEnable(clear_stencil);
1184 gfxc->setDepthStencilControl(ds_control);
1185
1186 Gnm::StencilControl st_control;
1187 st_control.m_testVal = 255;
1188 st_control.m_mask = 255;
1189 st_control.m_writeMask = 255;
1190 st_control.m_opVal = 0;
1191 gfxc->setStencil(st_control);
1192
1193 set_viewport_raw(0, 0, w, h);
1194 set_projection_raw(0, w, h, 0);
1195 gfxc->setRenderTargetMask(0);

1641static void set_renderstate_full(const GDrawRenderState * RADRESTRICT r, GDrawStats *stats)
1642{
1643 GDraw * RADRESTRICT gd = gdraw;
1644 Gnmx::GfxContext * RADRESTRICT gfxc = gd->gfxc;
1645 F32 depth = depth_from_id(r->id);
1646
1647 // set vertex shader constants
1648 VertexVars * RADRESTRICT vvars = embed_const_buffer<VertexVars>(Gnm::kShaderStageVs, 0);
1649 if (!r->use_world_space)
1650 gdraw_ObjectSpace(vvars->world[0], r->o2w, depth, 0.0f);
1651 else
1652 gdraw_WorldSpace(vvars->world[0], gdraw->world_to_pixel, depth, 0.0f);
1653
1654 __m128 edge = _mm_loadu_ps(r->edge_matrix);
1655 __m128 s0_texgen = _mm_setzero_ps();
1656 __m128 t0_texgen = _mm_setzero_ps();
1657 __m128 viewproj = _mm_load_ps(gd->projection);
1658 if (r->texgen0_enabled) {
1659 s0_texgen = _mm_loadu_ps(&r->s0_texgen[0]);
1660 t0_texgen = _mm_loadu_ps(&r->t0_texgen[0]);
1661 }
1662
1663 _mm_storeu_ps(&vvars->x_off[0], edge);
1664 _mm_storeu_ps(&vvars->texgen_s[0], s0_texgen);
1665 _mm_storeu_ps(&vvars->texgen_t[0], t0_texgen);
1666 _mm_storeu_ps(&vvars->viewproj[0], viewproj);
1667
1668 // set the blend mode
1669 int tex0mode = r->tex0_mode;
1670 int blend_mode = r->blend_mode;
1671 if (blend_mode != gd->blend_mode) {
1672 gd->blend_mode = blend_mode;
1673 gfxc->setBlendControl(0, gd->blend_control[blend_mode]);
1674 }
1675
1676 // color channel write mask: stencil set mode doesn't write color
1677 if (r->stencil_set != gd->writes_masked) {
1678 gd->writes_masked = r->stencil_set;
1679 gfxc->setRenderTargetMask(r->stencil_set ? 0 : 0xf);
1680 }
1681
1682 // set the pixel shader
1683 if (blend_mode != GDRAW_BLEND_special) {
1684 assert(tex0mode >= 0 && tex0mode < sizeof(gd->main_ps) / sizeof(*gd->main_ps));
1685 ShaderCode *ps = gd->basic_ps[tex0mode];
1686
1687 if (r->cxf_add) {
1688 ps++;
1689 if (r->cxf_add[3]) ps++;
1690 }
1691
1692 set_pixel_shader(ps);
1693 } else // special blends have a special pixel shader.
1694 set_pixel_shader(&gd->exceptional_blend[r->special_blend]);
1695
1696 // set textures
1697 if (tex0mode != GDRAW_TEXTURE_none) {
1698 if (!r->tex[0]) // this can happen if some allocs fail. just abort in that case.
1699 return;
1700
1701 if (gd->active_tex[0] != r->tex[0]) {
1702 gd->active_tex[0] = r->tex[0];
1703 set_gnm_texture(0, ((GDrawHandle *) r->tex[0])->handle.tex.gnm, r->wrap0, r->nearest0);
1704 }
1705 }
1706
1707 // pixel shader constants
1708 PixelCommonVars * RADRESTRICT pvars = embed_const_buffer<PixelCommonVars>(Gnm::kShaderStagePs, 0);
1709 __m128 col_mul = _mm_loadu_ps(r->color);
1710 __m128 col_add = _mm_setzero_ps();
1711 __m128 focal = _mm_loadu_ps(r->focal_point);
1712 if (r->cxf_add) {
1713 const float scalef = 1.0f / 255.0f;
1714 col_add = _mm_setr_ps((F32) r->cxf_add[0] * scalef, (F32) r->cxf_add[1] * scalef, (F32) r->cxf_add[2] * scalef, (F32) r->cxf_add[3] * scalef);
1715 }
1716
1717 _mm_storeu_ps(&pvars->color_mul[0], col_mul);
1718 _mm_storeu_ps(&pvars->color_add[0], col_add);
1719 _mm_storeu_ps(&pvars->focal[0], focal);
1720
1721 // set scissor
1722 if (r->scissor) {
1723 S32 xs,ys;
1724 if (gd->cur == gd->frame) {
1725 xs = gd->tx0 - gd->vx;
1726 ys = gd->ty0 - gd->vy;
1727 } else {
1728 xs = gd->tx0p;
1729 ys = gd->ty0p;
1730 }
1731
1732 // clip against viewport
1733 S32 x0 = RR_MAX(r->scissor_rect.x0 - xs, gd->cview.x0);
1734 S32 y0 = RR_MAX(r->scissor_rect.y0 - ys, gd->cview.y0);
1735 S32 x1 = RR_MIN(r->scissor_rect.x1 - xs, gd->cview.x1);
1736 S32 y1 = RR_MIN(r->scissor_rect.y1 - ys, gd->cview.y1);
1737
1738 if (x1 <= x0 || y1 <= y0) {
1739 // dummy scissor rect in case our actual scissor is empty
1740 x0 = x1 = gd->cview.x0;
1741 y0 = y1 = gd->cview.y0;
1742 }
1743
1744 gfxc->setScreenScissor(x0, y0, x1, y1);
1745 gd->scissor_state = 1;
1746 } else if (r->scissor != gd->scissor_state)
1747 disable_scissor(0);
1748
1749 // z/stencil mode changed?
1750 U32 z_stencil_key = r->set_id | (r->test_id << 1) | (r->stencil_test << 16) | (r->stencil_set << 24);

1960static void RADLINK gdraw_FilterQuad(GDrawRenderState *r, S32 x0, S32 y0, S32 x1, S32 y1, GDrawStats *stats)
1961{
1962 F32 tc[4];
1963 gswf_recti s;
1964
1965 // clip to tile boundaries
1966 s.x0 = RR_MAX(x0, gdraw->tx0p);
1967 s.y0 = RR_MAX(y0, gdraw->ty0p);
1968 s.x1 = RR_MIN(x1, gdraw->tx0p + gdraw->tpw);
1969 s.y1 = RR_MIN(y1, gdraw->ty0p + gdraw->tph);
1970 if (s.x1 < s.x0 || s.y1 < s.y0)
1971 return;
1972
1973 // prepare for drawing
1974 tc[0] = (s.x0 - gdraw->tx0p) / (F32) gdraw->frametex_width;
1975 tc[1] = (s.y0 - gdraw->ty0p) / (F32) gdraw->frametex_height;
1976 tc[2] = (s.x1 - gdraw->tx0p) / (F32) gdraw->frametex_width;
1977 tc[3] = (s.y1 - gdraw->ty0p) / (F32) gdraw->frametex_height;
1978
1979 clear_renderstate();
1980
1981 if (r->blend_mode == GDRAW_BLEND_filter) {
1982 switch (r->filter) {
1983 case GDRAW_FILTER_blur: {
1984 GDrawBlurInfo b;
1985 gswf_recti bounds = *get_valid_rect(r->tex[0]);
1986 gdraw_ShiftRect(&s, &s, -gdraw->tx0p, -gdraw->ty0p); // blur uses physical rendertarget coordinates
1987
1988 b.BlurPass = gdraw_DriverBlurPass;
1989 b.w = gdraw->tpw;
1990 b.h = gdraw->tph;
1991 b.frametex_width = gdraw->frametex_width;
1992 b.frametex_height = gdraw->frametex_height;
1993
1994 // blur needs to draw with multiple passes, so set up special state
1995 gdraw->in_blur = true;
1996 set_viewport();
1997 set_projection();
1998
1999 // do the blur
2000 gdraw_Blur(&gdraw_funcs, &b,r, &s, &bounds, stats);
2001
2002 // restore the normal state
2003 gdraw->in_blur = false;
2004 set_viewport();
2005 set_projection();
2006 break;
2007 }
2008
2009 case GDRAW_FILTER_colormatrix:
2010 gdraw_Colormatrix(r, &s, tc, stats);
2011 break;
2012
2013 case GDRAW_FILTER_dropshadow:
2014 gdraw_Filter(r, &s, tc, 0, stats);
2015 break;
2016
2017 case GDRAW_FILTER_bevel:
2018 gdraw_Filter(r, &s, tc, 1, stats);
2019 break;
2020
2021 default:
2022 assert(0);
2023 }
2024 } else {
2025 // for crazy blend modes, we need to read back from the framebuffer
2026 // and do the blending in the pixel shader. because we render as
2027 // a RectList, no quad is ever shaded twice, so we can do this safely
2028 // while texturing from the render target that we're writing to.
2029 if (r->blend_mode == GDRAW_BLEND_special) {
2030 // before we texture from this RT, we need to eliminate fast clears.
2031 eliminate_fast_clear();
2032
2033 // input texture. slightly different logic between the main render
2034 // target and our scratch render targets, because they might have
2035 // different dimensions.
2036 Gnm::Texture rt_tex;
2037
2038 if (gdraw->cur == gdraw->frame) {
2039 rt_tex.initFromRenderTarget(&gdraw->main_colorbuffer, false);
2040 rtt_sync(rt_tex.getBaseAddress(), gdraw->main_colorbuffer.getSizeInBytes() >> 8);
2041 } else {
2042 rt_tex = *gdraw->cur->color_buffer->handle.tex.gnm;
2043 rtt_sync(rt_tex.getBaseAddress(), gdraw->rt_colorbuffer_sa.m_size >> 8);
2044 }
2045
2046 set_gnm_texture(1, &rt_tex, GDRAW_WRAP_clamp, 0);
2047 }
2048
2049 set_renderstate(r, stats);
2050 do_screen_quad(&s, tc, stats);
2051 fence_resources(r->tex[0], r->tex[1]);
2052 }
2053}
