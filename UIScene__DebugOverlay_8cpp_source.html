140{
141 const Minecraft *pMinecraft = Minecraft::GetInstance();
142 if(pMinecraft->localplayers[m_iPad] == nullptr || pMinecraft->localgameModes[m_iPad] == nullptr) return;
143
144 int itemId = -1;
145 swscanf(static_cast<wchar_t *>(region->name),L"item_%d",&itemId);
