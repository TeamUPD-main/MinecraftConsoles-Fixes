132float Sheep::getHeadEatPositionScale(float a)
133{
134 if (eatAnimationTick <= 0)
135 {
136 return 0;

163bool Sheep::mobInteract(shared_ptr<Player> player)
164{
165 shared_ptr<ItemInstance> item = player->inventory->getSelected();
166
167 // 4J-JEV: Fix for #88212,
168 // Untrusted players shouldn't be able to sheer sheep.
169 if (!player->isAllowedToInteract( shared_from_this() ))
170 return false; //Animal::interact(player);
171
172 if (item != nullptr && item->id == Item::shears->id && !isSheared() && !isBaby())
173 {
174 if (!level->isClientSide)
175 {
176 setSheared(true);
177 int count = 1 + random->nextInt(3);
178 for (int i = 0; i < count; i++)
179 {
180 shared_ptr<ItemEntity> ie = spawnAtLocation(std::make_shared<ItemInstance>(Tile::wool_Id, 1, getColor()), 1.0f);
181 ie->yd += random->nextFloat() * 0.05f;
182 ie->xd += (random->nextFloat() - random->nextFloat()) * 0.1f;
183 ie->zd += (random->nextFloat() - random->nextFloat()) * 0.1f;
184 }
185
186 player->awardStat( GenericStats::shearedEntity(eTYPE_SHEEP), GenericStats::param_shearedEntity(eTYPE_SHEEP) );
187 }
188 item->hurtAndBreak(1, player);
189 playSound(eSoundType_MOB_SHEEP_SHEAR, 1, 1);
190 }
191
192 return Animal::mobInteract(player);
193}
