497//--------------------------------------------------------------------------------------
498// Create Direct3D device and swap chain
499//--------------------------------------------------------------------------------------
500HRESULT InitDevice()
501{
502 HRESULT hr = S_OK;
503
504 RECT rc;
505 GetClientRect( g_hWnd, &rc );
506 UINT width = rc.right - rc.left;
507 UINT height = rc.bottom - rc.top;
508
509 UINT createDeviceFlags = 0;
510#ifdef _DEBUG
511 createDeviceFlags |= D3D11_CREATE_DEVICE_DEBUG;
512#endif
513
514 D3D_DRIVER_TYPE driverTypes[] =
515 {
516 D3D_DRIVER_TYPE_HARDWARE,
517 D3D_DRIVER_TYPE_WARP,
518 D3D_DRIVER_TYPE_REFERENCE,
519 };
520 UINT numDriverTypes = ARRAYSIZE( driverTypes );
521
522 D3D_FEATURE_LEVEL featureLevels[] =
523 {
524 D3D_FEATURE_LEVEL_11_0,
525 D3D_FEATURE_LEVEL_10_1,
526 D3D_FEATURE_LEVEL_10_0,
527 };
528 UINT numFeatureLevels = ARRAYSIZE( featureLevels );
529
530 DXGI_SWAP_CHAIN_DESC sd;
531 ZeroMemory( &sd, sizeof( sd ) );
532 sd.BufferCount = 1;
533 sd.BufferDesc.Width = width;
534 sd.BufferDesc.Height = height;
535 sd.BufferDesc.Format = DXGI_FORMAT_R8G8B8A8_UNORM;
536 sd.BufferDesc.RefreshRate.Numerator = 60;
537 sd.BufferDesc.RefreshRate.Denominator = 1;
538 sd.BufferUsage = DXGI_USAGE_RENDER_TARGET_OUTPUT;
539 sd.OutputWindow = g_hWnd;
540 sd.SampleDesc.Count = 1;
541 sd.SampleDesc.Quality = 0;
542 sd.Windowed = TRUE;
543
544 for( UINT driverTypeIndex = 0; driverTypeIndex < numDriverTypes; driverTypeIndex++ )
545 {
546 g_driverType = driverTypes[driverTypeIndex];
547 hr = D3D11CreateDeviceAndSwapChain( nullptr, g_driverType, nullptr, createDeviceFlags, featureLevels, numFeatureLevels,
548 D3D11_SDK_VERSION, &sd, &g_pSwapChain, &g_pd3dDevice, &g_featureLevel, &g_pImmediateContext );
549 if( HRESULT_SUCCEEDED( hr ) )
550 break;
551 }
552 if( FAILED( hr ) )
553 return hr;
554
555 // Create a render target view
556 ID3D11Texture2D* pBackBuffer = nullptr;
557 hr = g_pSwapChain->GetBuffer( 0, __uuidof( ID3D11Texture2D ), ( LPVOID* )&pBackBuffer );
558 if( FAILED( hr ) )
559 return hr;
560
561 // Create a depth stencil buffer
562 D3D11_TEXTURE2D_DESC descDepth;
563
564 descDepth.Width = width;
565 descDepth.Height = height;
566 descDepth.MipLevels = 1;
567 descDepth.ArraySize = 1;
568 descDepth.Format = DXGI_FORMAT_D24_UNORM_S8_UINT;
569 descDepth.SampleDesc.Count = 1;
570 descDepth.SampleDesc.Quality = 0;
571 descDepth.Usage = D3D11_USAGE_DEFAULT;
572 descDepth.BindFlags = D3D11_BIND_DEPTH_STENCIL;
573 descDepth.CPUAccessFlags = 0;
574 descDepth.MiscFlags = 0;
575 hr = g_pd3dDevice->CreateTexture2D(&descDepth, nullptr, &g_pDepthStencilBuffer);
576
577 D3D11_DEPTH_STENCIL_VIEW_DESC descDSView;
578 descDSView.Format = DXGI_FORMAT_D24_UNORM_S8_UINT;
579 descDSView.ViewDimension = D3D11_DSV_DIMENSION_TEXTURE2D;
580 descDSView.Texture2D.MipSlice = 0;
581
582 hr = g_pd3dDevice->CreateDepthStencilView(g_pDepthStencilBuffer, &descDSView, &g_pDepthStencilView);
583
584 hr = g_pd3dDevice->CreateRenderTargetView( pBackBuffer, nullptr, &g_pRenderTargetView );
585 pBackBuffer->Release();
586 if( FAILED( hr ) )
587 return hr;
588
589 g_pImmediateContext->OMSetRenderTargets( 1, &g_pRenderTargetView, g_pDepthStencilView );
590
591 // Setup the viewport
592 D3D11_VIEWPORT vp;
593 vp.Width = (FLOAT)width;
594 vp.Height = (FLOAT)height;
595 vp.MinDepth = 0.0f;
596 vp.MaxDepth = 1.0f;
597 vp.TopLeftX = 0;
598 vp.TopLeftY = 0;
599 g_pImmediateContext->RSSetViewports( 1, &vp );
600
601 RenderManager.Initialise(g_pd3dDevice, g_pSwapChain);
602
603 return S_OK;
604}
