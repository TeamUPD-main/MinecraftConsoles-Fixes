35{
36 if (!potion->hasTag() || !potion->getTag()->contains(L"CustomPotionEffects"))
37 {
38 vector<MobEffectInstance *> *effects = nullptr;
39 auto it = cachedMobEffects.find(potion->getAuxValue());
40 if(it != cachedMobEffects.end()) effects = it->second;

42 {
43 effects = PotionBrewing::getEffects(potion->getAuxValue(), false);
44 cachedMobEffects[potion->getAuxValue()] = effects;

376vector<pair<int, int> > *PotionItem::getUniquePotionValues()
377{
378 if (s_uniquePotionValues.empty())
379 {
380 for (int brew = 0; brew <= PotionBrewing::BREW_MASK; ++brew)
381 {
382 vector<MobEffectInstance *> *effects = PotionBrewing::getEffects(brew, false);
383
384 if (effects != nullptr)
385 {
386 if(!effects->empty())
387 {
388 // 4J Stu - Based on implementation of Java List.hashCode() at http://docs.oracle.com/javase/6/docs/api/java/util/List.html#hashCode()
389 // and adding deleting to clear up as we go
390 int effectsHashCode = 1;
391 for(auto& effect : *effects)
392 {
393 effectsHashCode = 31*effectsHashCode + (effect ? 0 : effect->hashCode());
394 delete effect;
395 }
396
397 bool toAdd = true;
398 for(auto& it : s_uniquePotionValues)
399 {
400 // Some potions hash the same (identical effects) but are throwable so account for that
401 if(it.first == effectsHashCode && !(!isThrowable(it.second) && isThrowable(brew)) )
402 {
403 toAdd = false;
404 break;
405 }
406 }
407 if( toAdd )
408 {
409 s_uniquePotionValues.push_back(pair<int,int>(effectsHashCode, brew) );
410 }
411 }
412 delete effects;
413 }
414 }
415 }
416 return &s_uniquePotionValues;
417}
