130 glColor4f(1.0f, 1.0f, 1.0f, 1);

141

149 }

157void ParticleEngine::renderLit(shared_ptr<Entity> player, float a, int list)
158{
159 // 4J - added. We call this before ParticleEngine::render in the general render per player, so if we
160 // don't set this here then the offsets will be from the previous player - a single frame lag for the
161 // java game, or totally incorrect placement of things for split screen.
162 Particle::xOff = (player->xOld + (player->x - player->xOld) * a);
163 Particle::yOff = (player->yOld + (player->y - player->yOld) * a);
164 Particle::zOff = (player->zOld + (player->z - player->zOld) * a);
165
166 float RAD = PI / 180;
167 float xa = (float) Mth::cos(player->yRot * RAD);
168 float za = (float) Mth::sin(player->yRot * RAD);
169
170 float xa2 = -za * (float) Mth::sin(player->xRot * RAD);
171 float za2 = xa * (float) Mth::sin(player->xRot * RAD);
172 float ya = (float) Mth::cos(player->xRot * RAD);
173
174 int l = level->dimension->id == 0 ? 0 : ( level->dimension->id == -1 ? 1 : 2 );
175 int tt = ENTITY_PARTICLE_TEXTURE;
176
177 if( !particles[l][tt][list].empty() )
178 {
179 Tesselator *t = Tesselator::getInstance();
180 for (unsigned int i = 0; i < particles[l][tt][list].size(); i++)
181 {
182 shared_ptr<Particle> p = particles[l][tt][list][i];
183
184 if (SharedConstants::TEXTURE_LIGHTING) // 4J - change brought forward from 1.8.2
185 {
186 t->tex2(p->getLightColor(a));
187 }
188 p->render(t, a, xa, ya, za, xa2, za2);
