93void RedStoneDustTile::updatePowerStrength(Level *level, int x, int y, int z, int xFrom, int yFrom, int zFrom)
94{
95 int old = level->getData(x, y, z);
96 int target = 0;
97
98 target = checkTarget(level, xFrom, yFrom, zFrom, target);
99
100 shouldSignal = false;
101 int neighborSignal = level->getBestNeighborSignal(x, y, z);
102 shouldSignal = true;
103
104 if (neighborSignal > Redstone::SIGNAL_NONE && neighborSignal > target - 1)
105 {
106 target = neighborSignal;
107 }
108
109 {
110 int newTarget = 0;
111 for (int i = 0; i < 4; i++)
112 {
113 int xt = x;
114 int zt = z;
115 if (i == 0) xt--;
116 if (i == 1) xt++;
117 if (i == 2) zt--;
118 if (i == 3) zt++;
119
120 if (xt != xFrom || zt != zFrom) newTarget = checkTarget(level, xt, y, zt, newTarget);
121 if (level->isSolidBlockingTile(xt, y, zt) && !level->isSolidBlockingTile(x, y + 1, z))
122 {
123 if ((xt != xFrom || zt != zFrom) && y >= yFrom)
124 newTarget = checkTarget(level, xt, y + 1, zt, newTarget);
125 }
126 else if (!level->isSolidBlockingTile(xt, y, zt))
127 {
128 if ((xt != xFrom || zt != zFrom) && y <= yFrom)
129 newTarget = checkTarget(level, xt, y - 1, zt, newTarget);
130 }
131 }
132 if (newTarget > target) target = newTarget - 1;
133 else if (target > 0) target--;
134 else target = 0;
135
136 if (neighborSignal > target - 1)
137 {
138 target = neighborSignal;
139 }
140 }
141
142 if (old != target)
143 {
144 level->setData(x, y, z, target, Tile::UPDATE_CLIENTS);
145
146 {
147 toUpdate.insert(TilePos(x, y, z));
148 toUpdate.insert(TilePos(x - 1, y, z));
149 toUpdate.insert(TilePos(x + 1, y, z));
150 toUpdate.insert(TilePos(x, y - 1, z));
151 toUpdate.insert(TilePos(x, y + 1, z));

263int RedStoneDustTile::getSignal(LevelSource *level, int x, int y, int z, int dir)
264{
265 if (!shouldSignal) return Redstone::SIGNAL_NONE;
266 int data = level->getData(x, y, z);
267 if (data == Facing::DOWN)
268 {
269 return Redstone::SIGNAL_NONE;
270 }
271
272 if (dir == Facing::UP) return data;
273
274 bool w = shouldReceivePowerFrom(level, x - 1, y, z, Direction::WEST)
275 || (!level->isSolidBlockingTile(x - 1, y, z) && shouldReceivePowerFrom(level, x - 1, y - 1, z, Direction::UNDEFINED));
276 bool e = shouldReceivePowerFrom(level, x + 1, y, z, Direction::EAST)
277 || (!level->isSolidBlockingTile(x + 1, y, z) && shouldReceivePowerFrom(level, x + 1, y - 1, z, Direction::UNDEFINED));
278 bool n = shouldReceivePowerFrom(level, x, y, z - 1, Direction::NORTH)
279 || (!level->isSolidBlockingTile(x, y, z - 1) && shouldReceivePowerFrom(level, x, y - 1, z - 1, Direction::UNDEFINED));
280 bool s = shouldReceivePowerFrom(level, x, y, z + 1, Direction::SOUTH)
281 || (!level->isSolidBlockingTile(x, y, z + 1) && shouldReceivePowerFrom(level, x, y - 1, z + 1, Direction::UNDEFINED));
282
283 if (!level->isSolidBlockingTile(x, y + 1, z))
284 {
285 if (level->isSolidBlockingTile(x - 1, y, z) && shouldReceivePowerFrom(level, x - 1, y + 1, z, Direction::UNDEFINED)) w = true;
286 if (level->isSolidBlockingTile(x + 1, y, z) && shouldReceivePowerFrom(level, x + 1, y + 1, z, Direction::UNDEFINED)) e = true;
287 if (level->isSolidBlockingTile(x, y, z - 1) && shouldReceivePowerFrom(level, x, y + 1, z - 1, Direction::UNDEFINED)) n = true;
288 if (level->isSolidBlockingTile(x, y, z + 1) && shouldReceivePowerFrom(level, x, y + 1, z + 1, Direction::UNDEFINED)) s = true;
289 }
290
291 if (!n && !e && !w && !s && (dir >= 2 && dir <= 5)) return data;
292
293 if (dir == 2 && n && (!w && !e)) return data;
294 if (dir == 3 && s && (!w && !e)) return data;
295 if (dir == 4 && w && (!n && !s)) return data;
296 if (dir == 5 && e && (!n && !s)) return data;
297
298 return Redstone::SIGNAL_NONE;
