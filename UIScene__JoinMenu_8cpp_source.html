283void UIScene_JoinMenu::handleInput(int iPad, int key, bool repeat, bool pressed, bool released, bool &handled)
284{
285 if(m_bIgnoreInput) return;
286
287 ui.AnimateKeyPress(m_iPad, key, repeat, pressed, released);
288
289 switch(key)
290 {
291 case ACTION_MENU_CANCEL:
292 if(pressed)
293 {
294 navigateBack();
295 handled = true;
296 }
297 break;
298#ifdef _DURANGO
299 case ACTION_MENU_Y:
300 if(m_selectedSession != nullptr && getControlFocus() == eControl_GamePlayers && m_buttonListPlayers.getItemCount() > 0)
301 {
302 PlayerUID uid = m_selectedSession->searchResult.m_playerXuids[m_buttonListPlayers.getCurrentSelection()];
303 if( uid != INVALID_XUID ) ProfileManager.ShowProfileCard(ProfileManager.GetLockedProfile(),uid);
304 }
305 break;
306#endif
307#ifdef _WINDOWS64
308 case ACTION_MENU_X:
309 if(pressed && m_serverIndex >= 0)
310 {
311 BeginDeleteServer();
312 handled = true;
313 }
314 break;
315 case ACTION_MENU_Y:
316 if(pressed && m_serverIndex >= 0)
317 {
318 BeginEditServer();
319 handled = true;
320 }
321 break;
322#endif
323 case ACTION_MENU_OK:
324 if (getControlFocus() != eControl_GamePlayers)
325 {
326 sendInputToMovie(key, repeat, pressed, released);
327 }
328#ifdef _WINDOWS64
329 else if (pressed && m_serverIndex >= 0)
330 {
331 int sel = m_buttonListPlayers.getCurrentSelection();
332 if (sel == m_editServerButtonIndex)
333 BeginEditServer();
334 else if (sel == m_deleteServerButtonIndex)
335 BeginDeleteServer();
336 }
337#endif
338 handled = true;
339 break;
340#ifdef __ORBIS__
341 case ACTION_MENU_TOUCHPAD_PRESS:
342#endif
343 case ACTION_MENU_UP:
344 case ACTION_MENU_DOWN:
345 case ACTION_MENU_PAGEUP:
346 case ACTION_MENU_PAGEDOWN:
347 sendInputToMovie(key, repeat, pressed, released);
348 handled = true;
349 break;
350 }

458// Shared function to join the game that is the same whether we used the sign-in UI or not
459void UIScene_JoinMenu::JoinGame(UIScene_JoinMenu* pClass)
460{
461 DWORD dwSignedInUsers = 0;
462 bool noPrivileges = false;
463 DWORD dwLocalUsersMask = 0;
464 bool isSignedInLive = true;
465 int iPadNotSignedInLive = -1;
466
467 ProfileManager.SetLockedProfile(0); // TEMP!
468
469 // If we're in SD mode, then only the primary player gets to play
470 if (app.IsLocalMultiplayerAvailable())
471 {
472 for(unsigned int index = 0; index < XUSER_MAX_COUNT; ++index)
473 {
474 if(ProfileManager.IsSignedIn(index))
475 {
476 if (isSignedInLive && !ProfileManager.IsSignedInLive(index))
477 {
478 // Record the first non signed in live pad
479 iPadNotSignedInLive = index;
480 }
481
482 if( !ProfileManager.AllowedToPlayMultiplayer(index) ) noPrivileges = true;
483 dwLocalUsersMask |= CGameNetworkManager::GetLocalPlayerMask(index);
484 isSignedInLive = isSignedInLive && ProfileManager.IsSignedInLive(index);
485 }
486 }
487 }
488 else
489 {
490 if(ProfileManager.IsSignedIn(ProfileManager.GetPrimaryPad()))
491 {
492 if( !ProfileManager.AllowedToPlayMultiplayer(ProfileManager.GetPrimaryPad()) ) noPrivileges = true;
493 dwLocalUsersMask |= CGameNetworkManager::GetLocalPlayerMask(ProfileManager.GetPrimaryPad());
494
495 isSignedInLive = ProfileManager.IsSignedInLive(ProfileManager.GetPrimaryPad());
496#ifdef __PSVITA__
497 if(CGameNetworkManager::usingAdhocMode() && SQRNetworkManager_AdHoc_Vita::GetAdhocStatus())
498 isSignedInLive = true;
499#endif
500
501 }
502 }
503
504 // If this is an online game but not all players are signed in to Live, stop!
505 if (!isSignedInLive)
506 {
507#ifdef __ORBIS__
508 // Check if PSN is unavailable because of age restriction
509 int npAvailability = ProfileManager.getNPAvailability(iPadNotSignedInLive);
510 if (npAvailability == SCE_NP_ERROR_AGE_RESTRICTION)
511 {
512 pClass->m_bIgnoreInput = false;
513 // 4J Stu - This is a bit messy and is due to the library incorrectly returning false for IsSignedInLive if the npAvailability isn't SCE_OK
514 UINT uiIDA[1];
515 uiIDA[0]=IDS_OK;
516 ui.RequestErrorMessage(IDS_ONLINE_SERVICE_TITLE, IDS_CONTENT_RESTRICTION, uiIDA, 1, iPadNotSignedInLive);
517 }
518 else
519#endif
520 {
521 pClass->m_bIgnoreInput=false;
522 UINT uiIDA[1];
523 uiIDA[0]=IDS_CONFIRM_OK;
524 ui.RequestErrorMessage( IDS_PRO_NOTONLINE_TITLE, IDS_PRO_NOTONLINE_TEXT, uiIDA,1,ProfileManager.GetPrimaryPad());
525 }
526 return;
527 }
528
529 // Check if user-created content is allowed, as we cannot play multiplayer if it's not
530 bool noUGC = false;
531 BOOL pccAllowed = TRUE;
532 BOOL pccFriendsAllowed = TRUE;
533
534#if defined(__PS3__) || defined(__PSVITA__)
535 if(isSignedInLive)
536 {
537 ProfileManager.GetChatAndContentRestrictions(ProfileManager.GetPrimaryPad(),false,&noUGC,nullptr,nullptr);
538 }
539#else
540 ProfileManager.AllowedPlayerCreatedContent(ProfileManager.GetPrimaryPad(),false,&pccAllowed,&pccFriendsAllowed);
541 if(!pccAllowed && !pccFriendsAllowed) noUGC = true;
542#endif
543
544
545#ifdef __PSVITA__
546 if( CGameNetworkManager::usingAdhocMode() )
547 {
548 noPrivileges = false;
549 noUGC = false;
550 }
551#endif
552
553 if(noUGC)
554 {
555 pClass->setVisible( true );
556 pClass->m_bIgnoreInput=false;
557
558 int messageText = IDS_NO_USER_CREATED_CONTENT_PRIVILEGE_SINGLE_LOCAL;
559 if(dwSignedInUsers > 1) messageText = IDS_NO_USER_CREATED_CONTENT_PRIVILEGE_ALL_LOCAL;
560
561 ui.RequestUGCMessageBox(IDS_CONNECTION_FAILED, messageText);
562 }
563 else if(noPrivileges)
564 {
565 pClass->setVisible( true );
566 pClass->m_bIgnoreInput=false;
567 UINT uiIDA[1];
568 uiIDA[0]=IDS_CONFIRM_OK;
569 ui.RequestErrorMessage( IDS_NO_MULTIPLAYER_PRIVILEGE_TITLE, IDS_NO_MULTIPLAYER_PRIVILEGE_JOIN_TEXT, uiIDA,1,ProfileManager.GetPrimaryPad());
570 }
571 else
572 {
573#if defined(__ORBIS__) || defined(__PSVITA__)
574 bool chatRestricted = false;
575 ProfileManager.GetChatAndContentRestrictions(ProfileManager.GetPrimaryPad(),false,&chatRestricted,nullptr,nullptr);
576 if(chatRestricted)
577 {
578 ProfileManager.DisplaySystemMessage( SCE_MSG_DIALOG_SYSMSG_TYPE_TRC_PSN_CHAT_RESTRICTION, ProfileManager.GetPrimaryPad() );
579 }
580#endif
581 CGameNetworkManager::eJoinGameResult result = g_NetworkManager.JoinGame( pClass->m_selectedSession, dwLocalUsersMask );
582
583 // Alert the app the we no longer want to be informed of ethernet connections
584 app.SetLiveLinkRequired( false );
585
586 if( result != CGameNetworkManager::JOINGAME_SUCCESS )
587 {
588 int exitReasonStringId = -1;
589 switch(result)
590 {
591 case CGameNetworkManager::JOINGAME_FAIL_SERVER_FULL:
592 exitReasonStringId = IDS_DISCONNECTED_SERVER_FULL;
593 break;
594 }
595
596 if( exitReasonStringId == -1 )
597 {
598 Minecraft* pMinecraft = Minecraft::GetInstance();
599 int primaryPad = ProfileManager.GetPrimaryPad();
600 if( pMinecraft->m_connectionFailed[primaryPad] )
601 {
602 switch( pMinecraft->m_connectionFailedReason[primaryPad] )
603 {
604 case DisconnectPacket::eDisconnect_LoginTooLong:
605 exitReasonStringId = IDS_DISCONNECTED_LOGIN_TOO_LONG;
606 break;
607 case DisconnectPacket::eDisconnect_ServerFull:
608 exitReasonStringId = IDS_DISCONNECTED_SERVER_FULL;
609 break;
610 case DisconnectPacket::eDisconnect_Kicked:
611 exitReasonStringId = IDS_DISCONNECTED_KICKED;
612 break;
613 case DisconnectPacket::eDisconnect_NoUGC_AllLocal:
614 exitReasonStringId = IDS_NO_USER_CREATED_CONTENT_PRIVILEGE_ALL_LOCAL;
615 break;
616 case DisconnectPacket::eDisconnect_NoUGC_Single_Local:
617 exitReasonStringId = IDS_NO_USER_CREATED_CONTENT_PRIVILEGE_SINGLE_LOCAL;
618 break;
619 case DisconnectPacket::eDisconnect_NoFlying:
620 exitReasonStringId = IDS_DISCONNECTED_FLYING;
621 break;
622 case DisconnectPacket::eDisconnect_Quitting:
623 exitReasonStringId = IDS_DISCONNECTED_SERVER_QUIT;
624 break;
625 case DisconnectPacket::eDisconnect_OutdatedServer:
626 exitReasonStringId = IDS_DISCONNECTED_SERVER_OLD;
627 break;
628 case DisconnectPacket::eDisconnect_OutdatedClient:
629 exitReasonStringId = IDS_DISCONNECTED_CLIENT_OLD;
630 break;
631 default:
632 exitReasonStringId = IDS_CONNECTION_LOST_SERVER;
633 break;
634 }
635 }
636 }
637
638 if( exitReasonStringId == -1 )
639 {
640 ui.NavigateBack(pClass->m_iPad);
641 }
642 else
643 {
644 UINT uiIDA[1];

834void UIScene_JoinMenu::UpdateServerInFile(const wstring& newIP, const wstring& newPort, const wstring& newName)
835{
836 char narrowNewIP[256] = {};
837 char narrowNewPort[16] = {};
838 char narrowNewName[256] = {};
839 wcstombs(narrowNewIP, newIP.c_str(), sizeof(narrowNewIP) - 1);
840 wcstombs(narrowNewPort, newPort.c_str(), sizeof(narrowNewPort) - 1);
841 wcstombs(narrowNewName, newName.c_str(), sizeof(narrowNewName) - 1);
842
843 uint16_t newPortNum = (uint16_t)atoi(narrowNewPort);
844
845 struct ServerEntry { std::string ip; uint16_t port; std::string name; };
846 std::vector<ServerEntry> entries;
847
848 FILE* file = fopen("servers.db", "rb");
849 if (file)
850 {
851 char magic[4] = {};
852 if (fread(magic, 1, 4, file) == 4 && memcmp(magic, "MCSV", 4) == 0)
853 {
854 uint32_t version = 0, count = 0;
855 fread(&version, sizeof(uint32_t), 1, file);
856 fread(&count, sizeof(uint32_t), 1, file);
857 if (version == 1)
858 {
859 for (uint32_t s = 0; s < count; s++)
860 {
861 uint16_t ipLen = 0, p = 0, nameLen = 0;
862 if (fread(&ipLen, sizeof(uint16_t), 1, file) != 1) break;
863 if (ipLen == 0 || ipLen > 256) break;
864 char ipBuf[257] = {};
865 if (fread(ipBuf, 1, ipLen, file) != ipLen) break;
866 if (fread(&p, sizeof(uint16_t), 1, file) != 1) break;
867 if (fread(&nameLen, sizeof(uint16_t), 1, file) != 1) break;
868 if (nameLen > 256) break;
869 char nameBuf[257] = {};
870 if (nameLen > 0 && fread(nameBuf, 1, nameLen, file) != nameLen) break;
871 entries.push_back({std::string(ipBuf), p, std::string(nameBuf)});
872 }
873 }
874 }
875 fclose(file);
876 }
877
878 // Find and update the matching entry by original IP and port
879 int idx = m_serverIndex;
880 if (idx >= 0 && idx < (int)entries.size())
881 {
882 entries[idx].ip = std::string(narrowNewIP);
883 entries[idx].port = newPortNum;
884 entries[idx].name = std::string(narrowNewName);
885 }
886
887 file = fopen("servers.db", "wb");
888 if (file)
889 {
890 fwrite("MCSV", 1, 4, file);
891 uint32_t version = 1;
892 uint32_t count = (uint32_t)entries.size();
893 fwrite(&version, sizeof(uint32_t), 1, file);
894 fwrite(&count, sizeof(uint32_t), 1, file);
895
896 for (size_t i = 0; i < entries.size(); i++)
897 {
898 uint16_t ipLen = (uint16_t)entries[i].ip.length();
899 fwrite(&ipLen, sizeof(uint16_t), 1, file);
900 fwrite(entries[i].ip.c_str(), 1, ipLen, file);
901 fwrite(&entries[i].port, sizeof(uint16_t), 1, file);
902 uint16_t nameLen = (uint16_t)entries[i].name.length();
903 fwrite(&nameLen, sizeof(uint16_t), 1, file);
904 fwrite(entries[i].name.c_str(), 1, nameLen, file);
905 }
906 fclose(file);
907 }
908}
