337void CreateDevice()
338{
339 // This flag adds support for surfaces with a different color channel ordering than the API default.
340 // It is recommended usage, and is required for compatibility with Direct2D.
341 UINT creationFlags = D3D11_CREATE_DEVICE_FAST_KICKOFFS;
342#ifndef _CONTENT_PACKAGE
343 creationFlags |= D3D11_CREATE_DEVICE_VALIDATED;
344 creationFlags |= D3D11_CREATE_DEVICE_INSTRUMENTED;
345#endif
346
347 // This array defines the set of DirectX hardware feature levels this app will support.
348 // Note the ordering should be preserved.
349 // Don't forget to declare your application's minimum required feature level in its
350 // description. All applications are assumed to support 9.1 unless otherwise stated.
351 D3D_FEATURE_LEVEL featureLevels[] =
352 {
353 D3D_FEATURE_LEVEL_11_0
354 };
355
356 // Create the DX11 API device object, and get a corresponding context.
357 Microsoft::WRL::ComPtr<ID3D11Device> device;
358 Microsoft::WRL::ComPtr<ID3D11DeviceContext> context;
359
360 DX::ThrowIfFailed(
361 D3D11CreateDevice(
362 nullptr, // specify null to use the default adapter
363 D3D_DRIVER_TYPE_HARDWARE,
364 nullptr, // leave as nullptr unless software device
365 creationFlags, // optionally set debug and Direct2D compatibility flags
366 featureLevels, // list of feature levels this app can support
367 ARRAYSIZE(featureLevels), // number of entries in above list
368 D3D11_SDK_VERSION, // always set this to D3D11_SDK_VERSION
369 &device, // returns the Direct3D device created
370 &g_featureLevel, // returns feature level of device created
371 &context // returns the device immediate context
372 )
373 );
374
375 // Get the DirectX11.1 device by QI off the DirectX11 one.
376 DX::ThrowIfFailed(device.As(&g_d3dDevice));
377
378 // And get the corresponding device context in the same way.
379 DX::ThrowIfFailed(context.As(&g_d3dContext));
380}

382// Allocate all memory resources that change on a window SizeChanged event.
383void CreateResources()
384{
385 // Store the window bounds so the next time we get a SizeChanged event we can
386 // avoid rebuilding everything if the size is identical.
387 g_windowBounds = g_window.Get()->Bounds;
388
389 // If the swap chain already exists, resize it,
390 // otherwise create one.
391 if(g_swapChain != nullptr)
392 {
393 DX::ThrowIfFailed(g_swapChain->ResizeBuffers(2, 0, 0, DXGI_FORMAT_B8G8R8A8_UNORM, 0));
394 }
395 else
396 {
397 // First, retrieve the underlying DXGI Device from the D3D Device
398 Microsoft::WRL::ComPtr<IDXGIDevice1> dxgiDevice;
399 DX::ThrowIfFailed(g_d3dDevice.As(&dxgiDevice));
400
401 // Identify the physical adapter (GPU or card) this device is running on.
402 Microsoft::WRL::ComPtr<IDXGIAdapter> dxgiAdapter;
403 DX::ThrowIfFailed(dxgiDevice->GetAdapter(&dxgiAdapter));
404
405 // And obtain the factory object that created it.
406 Microsoft::WRL::ComPtr<IDXGIFactory2> dxgiFactory;
407 DX::ThrowIfFailed(dxgiAdapter->GetParent(__uuidof(IDXGIFactory2), &dxgiFactory));
408
409 // Create a descriptor for the swap chain.
410 DXGI_SWAP_CHAIN_DESC1 swapChainDesc = {0};
411
412 swapChainDesc.Width = 1920;
413 swapChainDesc.Height = 1080;
414 swapChainDesc.Format = DXGI_FORMAT_R8G8B8A8_UNORM;
415 swapChainDesc.BufferUsage = DXGI_USAGE_RENDER_TARGET_OUTPUT | DXGI_USAGE_SHADER_INPUT;
416 swapChainDesc.BufferCount = 2;
417 swapChainDesc.Stereo = false;
418 swapChainDesc.SampleDesc.Count = 1;
419 swapChainDesc.SampleDesc.Quality = 0;
420 swapChainDesc.Scaling = DXGI_SCALING_STRETCH;
421 swapChainDesc.SwapEffect = DXGI_SWAP_EFFECT_FLIP_SEQUENTIAL;
422 swapChainDesc.Flags = 0;
423
424 // Create a SwapChain from a CoreWindow.
425 DX::ThrowIfFailed(dxgiFactory->CreateSwapChainForCoreWindow(g_d3dDevice.Get(), reinterpret_cast<IUnknown*>(g_window.Get()), &swapChainDesc, nullptr, &g_swapChain));
426 }
427
428 // Obtain the backbuffer for this window which will be the final 3D rendertarget.
429 Microsoft::WRL::ComPtr<ID3D11Texture2D> backBuffer;
430 DX::ThrowIfFailed(g_swapChain->GetBuffer(0, __uuidof(ID3D11Texture2D), &backBuffer));
431
432 // Create a view interface on the rendertarget to use on bind.
433 DX::ThrowIfFailed(g_d3dDevice->CreateRenderTargetView(backBuffer.Get(), nullptr, &g_renderTargetView));
434
435 // Cache the rendertarget dimensions in our helper class for convenient use.
436 D3D11_TEXTURE2D_DESC backBufferDesc = {0};
437 backBuffer->GetDesc(&backBufferDesc);
438
439 // Allocate a 2-D surface as the depth/stencil buffer and
440 // create a DepthStencil view on this surface to use on bind.
441 CD3D11_TEXTURE2D_DESC depthStencilDesc(DXGI_FORMAT_D24_UNORM_S8_UINT, backBufferDesc.Width, backBufferDesc.Height, 1, 1, D3D11_BIND_DEPTH_STENCIL);
442
443 Microsoft::WRL::ComPtr<ID3D11Texture2D> depthStencil;
444 DX::ThrowIfFailed(g_d3dDevice->CreateTexture2D(&depthStencilDesc, nullptr, &depthStencil));
445
446 CD3D11_DEPTH_STENCIL_VIEW_DESC depthStencilViewDesc(D3D11_DSV_DIMENSION_TEXTURE2D);
447 DX::ThrowIfFailed(g_d3dDevice->CreateDepthStencilView(depthStencil.Get(), &depthStencilViewDesc, &g_depthStencilView));
448
449 // Create a viewport descriptor of the full window size.
450 CD3D11_VIEWPORT viewPort(0.0f, 0.0f, static_cast<float>(backBufferDesc.Width), static_cast<float>(backBufferDesc.Height));
451
452 // Set the current viewport using the descriptor.
453 g_d3dContext->RSSetViewports(1, &viewPort);
454 g_d3dContext->OMSetRenderTargets( 1, g_renderTargetView.GetAddressOf(), g_depthStencilView.Get() );
455}
