474HitResult *StairTile::clip(Level *level, int xt, int yt, int zt, Vec3 *a, Vec3 *b)
475{
476 HitResult *results[8];
477 for(unsigned int i = 0; i < 8; ++i)
478 {
479 results[i] = nullptr;
480 }
481 int data = level->getData(xt, yt, zt);
482 int dir = data & 0x3;
483 bool upsideDown = (data & UPSIDEDOWN_BIT) == UPSIDEDOWN_BIT;
484 int *deadSpaces = DEAD_SPACES[dir + (upsideDown ? 4 : 0)];
485
486 isClipping = true;
487 for (int i = 0; i < 8; i++)
488 {
489 clipStep = i;
490
491 for(unsigned int j = 0; j < DEAD_SPACE_COLUMN_COUNT; ++j)
492 {
493 if (deadSpaces[j] == i) continue;
494 }
495
496 results[i] = Tile::clip(level, xt, yt, zt, a, b);
497 }
498
499 for(unsigned int j = 0; j < DEAD_SPACE_COLUMN_COUNT; ++j)
500 {
501 results[deadSpaces[j]] = nullptr;
502 }
503
504 HitResult *closest = nullptr;
505 double closestDist = 0;
506
507 for (unsigned int i = 0; i < 8; ++i)
508 {
509 HitResult *result = results[i];
510 if (result != nullptr)
511 {
512 double dist = result->pos->distanceToSqr(b);
513
514 if (dist > closestDist)
515 {
516 closest = result;
517 closestDist = dist;
518 }
519 }
520 }
521
522 return closest;
523}
