293extern GDrawTexture *gdraw_GLx_(WrappedTextureCreate)(S32 gl_texture_handle, S32 width, S32 height, rrbool has_mipmaps)
294{
295 GDrawStats stats={0};
296 GDrawHandle *p = gdraw_res_alloc_begin(gdraw->texturecache, 0, &stats); // it may need to free one item to give us a handle
297 GLint old;
298
299 glGetIntegerv(GL_TEXTURE_BINDING_2D, &old);
300 glBindTexture(GL_TEXTURE_2D, gl_texture_handle);
301 if (has_mipmaps)
302 glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MIN_FILTER, GL_LINEAR_MIPMAP_LINEAR);
303 else
304 glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MIN_FILTER, GL_LINEAR);
305 glBindTexture(GL_TEXTURE_2D, old);
306
307 p->bytes = 0;
308 p->handle.tex.gl = gl_texture_handle;
309 p->handle.tex.w = width;
310 p->handle.tex.h = height;
311 p->handle.tex.nonpow2 = !(is_pow2(width) && is_pow2(height));
312 gdraw_HandleCacheAllocateEnd(p, 0, nullptr, GDRAW_HANDLE_STATE_user_owned);
313 return (GDrawTexture *) p;
314}

350static rrbool RADLINK gdraw_MakeTextureBegin(void *owner, S32 width, S32 height, gdraw_texture_format format, U32 flags, GDraw_MakeTexture_ProcessingInfo *p, GDrawStats *gstats)
351{
352 S32 size=0, asize, stride;
353 GDrawHandle *t = nullptr;
354 opengl_check();
355
356 stride = width;
357 if (format == GDRAW_TEXTURE_FORMAT_rgba32) stride *= 4;
358 size = stride*height;
359
360 asize = size;
361 if (flags & GDRAW_MAKETEXTURE_FLAGS_mipmap) asize = asize*4/3;
362
363 t = gdraw_res_alloc_begin(gdraw->texturecache, asize, gstats);
364 if (!t)
365 return IGGY_RESULT_Error_GDraw;
366
367 glGenTextures(1, &t->handle.tex.gl);
368
369 p->texture_data = IggyGDrawMalloc(size);
370 if (!p->texture_data) {
371 gdraw_HandleCacheAllocateFail(t);
372 IggyGDrawSendWarning(nullptr, "GDraw malloc for texture data failed");
373 return false;
374 }
375
376 t->handle.tex.w = width;
377 t->handle.tex.h = height;
378 t->handle.tex.nonpow2 = !(is_pow2(width) && is_pow2(height));
379
380 p->num_rows = height;
381 p->p0 = t;
382 p->p1 = owner;
383 p->stride_in_bytes = stride;
384 p->texture_type = GDRAW_TEXTURE_TYPE_rgba;
385 p->i0 = format;
386 p->i1 = flags;
387 p->i2 = width;
388 p->i3 = height;
389 p->i4 = asize;
390 opengl_check();
391 return true;
392}

394static GDrawTexture * RADLINK gdraw_MakeTextureEnd(GDraw_MakeTexture_ProcessingInfo *p, GDrawStats *stats)
395{
396 gdraw_texture_format format = (gdraw_texture_format) p->i0;
397 S32 flags = p->i1;
398 rrbool mipmap = (flags & GDRAW_MAKETEXTURE_FLAGS_mipmap) != 0;
399 S32 width = p->i2, height = p->i3;
400 GLuint z,e;
401 GDrawHandle *t = (GDrawHandle *) p->p0;
402
403 z = t->handle.tex.gl;
404 assert(z != 0);
405
406 make_texture(z);
407 glPixelStorei(GL_UNPACK_ALIGNMENT, 1);
408
409 if (format == GDRAW_TEXTURE_FORMAT_font)
410 glTexImage2D(GL_TEXTURE_2D, 0, GL_ALPHA, width, height, 0, GL_ALPHA, GL_UNSIGNED_BYTE, p->texture_data);
411 else
412 glTexImage2D(GL_TEXTURE_2D, 0, GL_RGBA, width, height, 0, GL_RGBA, GL_UNSIGNED_BYTE, p->texture_data);
413 e = glGetError();
414 break_on_err(e);
415
416 if (mipmap)
417 glGenerateMipmap(GL_TEXTURE_2D);
418 if (!e) e = glGetError();
419
420 if (e != 0) {
421 gdraw_HandleCacheAllocateFail(t);
422 IggyGDrawSendWarning(nullptr, "GDraw OpenGL error creating texture");
423 eat_gl_err();
424 return nullptr;
425 } else {
426 gdraw_HandleCacheAllocateEnd(t, p->i4, p->p1, (flags & GDRAW_MAKETEXTURE_FLAGS_never_flush) ? GDRAW_HANDLE_STATE_pinned : GDRAW_HANDLE_STATE_locked);
427 stats->nonzero_flags |= GDRAW_STATS_alloc_tex;
428 stats->alloc_tex += 1;
429 stats->alloc_tex_bytes += p->i4;
430 }
431
432 // default wrap mode is clamp to edge
433 glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_WRAP_S, GL_CLAMP_TO_EDGE);
434 glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_WRAP_T, GL_CLAMP_TO_EDGE);
435
436 if (mipmap)
437 glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MIN_FILTER, GL_LINEAR_MIPMAP_LINEAR);
438 else
439 glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MIN_FILTER, GL_LINEAR);
440
441 IggyGDrawFree(p->texture_data);
442
443 opengl_check();
444 return (GDrawTexture *) t;
445}

512static rrbool RADLINK gdraw_MakeVertexBufferBegin(void *unique_id, gdraw_vformat vformat, S32 vbuf_size, S32 ibuf_size, GDraw_MakeVertexBuffer_ProcessingInfo *p, GDrawStats *gstats)
513{
514 GLuint e;
515 GDrawHandle *vb;
516 opengl_check();
517 vb = gdraw_res_alloc_begin(gdraw->vbufcache, vbuf_size + ibuf_size, gstats);
518 if (!vb) {
519 IggyGDrawSendWarning(nullptr, "GDraw out of vertex buffer memory");
520 return false;
521 }
522
523 e = glGetError();
524 vb->handle.vbuf.base = 0;
525 vb->handle.vbuf.indices = 0;
526 glGenBuffers(1, &vb->handle.vbuf.base);
527 glGenBuffers(1, &vb->handle.vbuf.indices);
528 glBindBuffer(GL_ARRAY_BUFFER, vb->handle.vbuf.base);
529 glBufferData(GL_ARRAY_BUFFER, vbuf_size, nullptr, GL_STATIC_DRAW);
530 glBindBuffer(GL_ELEMENT_ARRAY_BUFFER, vb->handle.vbuf.indices);
531 glBufferData(GL_ELEMENT_ARRAY_BUFFER, ibuf_size, nullptr, GL_STATIC_DRAW);
532 if (!e) e = glGetError();
533 if (e != GL_NO_ERROR) {
534 glBindBuffer(GL_ARRAY_BUFFER, 0);
535 glBindBuffer(GL_ELEMENT_ARRAY_BUFFER, 0);
536 glDeleteBuffers(1, &vb->handle.vbuf.base);
537 glDeleteBuffers(1, &vb->handle.vbuf.indices);
538 gdraw_HandleCacheAllocateFail(vb);
539 eat_gl_err();
540 IggyGDrawSendWarning(nullptr, "GDraw OpenGL vertex buffer creation failed");
541 return false;
542 }
543
544 p->i0 = vbuf_size;
545 p->i1 = ibuf_size;
546 p->p0 = vb;
547 p->p1 = unique_id;
548
549 if (!gdraw->has_mapbuffer) {
550 p->vertex_data = IggyGDrawMalloc(vbuf_size);
551 p->vertex_data_length = vbuf_size;
552 p->index_data = IggyGDrawMalloc(ibuf_size);
553 p->index_data_length = ibuf_size;
554
555 // check for out of memory conditions
556 if (!p->vertex_data || !p->index_data) {
557 if (p->vertex_data) IggyGDrawFree(p->vertex_data);
558 if (p->index_data) IggyGDrawFree(p->index_data);
559 IggyGDrawSendWarning(nullptr, "GDraw malloc for vertex buffer temporary memory failed");
560 return false;
561 }
562 } else {
563 p->vertex_data = (U8 *)glMapBuffer(GL_ARRAY_BUFFER, GL_WRITE_ONLY);
564 p->vertex_data_length = vbuf_size;
565
566 p->index_data = (U8 *)glMapBuffer(GL_ELEMENT_ARRAY_BUFFER, GL_WRITE_ONLY);
567 p->index_data_length = ibuf_size;
568 }
569
570 opengl_check();
571 return true;
572}

638static GDrawHandle *get_rendertarget_texture(int width, int height, void *owner, GDrawStats *gstats)
639{
640 S32 size;
641 GDrawHandle *t;
642 opengl_check();
643 t = gdraw_HandleCacheGetLRU(&gdraw->rendertargets);
644 if (t) {
645 gdraw_HandleCacheLock(t, (void *) (UINTa) 1);
646 return t;
647 }
648
649 size = gdraw->frametex_width * gdraw->frametex_height * 4;
650 t = gdraw_res_alloc_begin(gdraw->texturecache, size, gstats);
651 if (!t) return t;
652
653 glGenTextures(1, &t->handle.tex.gl);
654 make_rendertarget(t, t->handle.tex.gl, GL_RGBA, GL_RGBA, GL_UNSIGNED_BYTE, width, height, 4);
655 t->handle.tex.w = gdraw->frametex_width;
656 t->handle.tex.h = gdraw->frametex_height;
657 t->handle.tex.nonpow2 = 1; // assume all rendertargets are non-pow2 for consistency
658 gstats->nonzero_flags |= GDRAW_STATS_alloc_tex;
659 gstats->alloc_tex += 1;
660 gstats->alloc_tex_bytes += size;
661 opengl_check();
662 gdraw_HandleCacheAllocateEnd(t, size, owner, GDRAW_HANDLE_STATE_locked);
663
664 return t;
665}

667static GDrawHandle *get_color_rendertarget(GDrawStats *gstats)
668{
669 S32 size;
670 GDrawHandle *t;
671 opengl_check();
672 t = gdraw_HandleCacheGetLRU(&gdraw->rendertargets);
673 if (t) {
674 gdraw_HandleCacheLock(t, (void *) (UINTa) 1);
675 return t;
676 }
677
678 // ran out of RTs, allocate a new one
679 size = gdraw->frametex_width * gdraw->frametex_height * 4;
680 if (gdraw->rendertargets.bytes_free < size) {
681 IggyGDrawSendWarning(nullptr, "GDraw exceeded available rendertarget memory");
682 return nullptr;
683 }
684
685 t = gdraw_HandleCacheAllocateBegin(&gdraw->rendertargets);
686 if (!t) {
687 IggyGDrawSendWarning(nullptr, "GDraw exceeded available rendertarget handles");
688 return t;
689 }
690
691 glGenTextures(1, &t->handle.tex.gl);
692 make_rendertarget(t, t->handle.tex.gl, GL_RGBA, GL_RGBA, GL_UNSIGNED_BYTE, gdraw->frametex_width,gdraw->frametex_height, 4);
693 t->handle.tex.w = gdraw->frametex_width;
694 t->handle.tex.h = gdraw->frametex_height;
695 t->handle.tex.nonpow2 = 1; // assume all rendertargets are non-pow2 for consistency
696
697#ifdef GDRAW_MULTISAMPLING
698 if (gdraw->multisampling) {
699 glGenRenderbuffers(1, &t->handle.tex.gl_renderbuf);
700 glBindRenderbuffer(GL_RENDERBUFFER, t->handle.tex.gl_renderbuf);
701 glRenderbufferStorageMultisample(GL_RENDERBUFFER, gdraw->multisampling, GL_RGBA, gdraw->frametex_width, gdraw->frametex_height);
702 glBindRenderbuffer(GL_RENDERBUFFER, 0);
703 }
704#endif
705 opengl_check();
706
707 gdraw_HandleCacheAllocateEnd(t, size, (void *) (UINTa) 1, GDRAW_HANDLE_STATE_locked);
708 gstats->nonzero_flags |= GDRAW_STATS_alloc_tex;
709 gstats->alloc_tex += gdraw->multisampling ? 2 : 1;
710 gstats->alloc_tex_bytes += (1 + gdraw->multisampling) * size;
711
712 return t;
713}

989static rrbool RADLINK gdraw_TextureDrawBufferBegin(gswf_recti *region, gdraw_texture_format format, U32 flags, void *owner, GDrawStats *gstats)
990{
991 GDrawFramebufferState *n = gdraw->cur+1;
992 GDrawHandle *t;
993 int k;
994 if (gdraw->tw == 0 || gdraw->th == 0) {
995 IggyGDrawSendWarning(nullptr, "GDraw got a request for an empty rendertarget");
996 return false;
997 }
998
999 if (n >= &gdraw->frame[MAX_RENDER_STACK_DEPTH]) {
1000 IggyGDrawSendWarning(nullptr, "GDraw rendertarget nesting exceeded MAX_RENDER_STACK_DEPTH");
1001 return false;
1002 }
1003
1004 if (owner) {
1005 t = get_rendertarget_texture(region->x1 - region->x0, region->y1 - region->y0, owner, gstats);
1006 if (!t) {
1007 IggyGDrawSendWarning(nullptr, "GDraw ran out of rendertargets for cacheAsBItmap");
1008 return false;
1009 }
1010 } else {
1011 t = get_color_rendertarget(gstats);
1012 if (!t) {
1013 IggyGDrawSendWarning(nullptr, "GDraw ran out of rendertargets");
1014 return false;
1015 }
1016 }
1017 n->color_buffer = t;
1018 assert(n->color_buffer != nullptr);
1019
1020 if (n == gdraw->frame+1)
1021 n->stencil_depth = get_depthstencil_renderbuffer(gstats);
1022 else
1023 n->stencil_depth = (n-1)->stencil_depth;
1024 ++gdraw->cur;
1025
1026 gdraw->cur->cached = owner != nullptr;
1027 if (owner) {
1028 gdraw->cur->base_x = region->x0;
1029 gdraw->cur->base_y = region->y0;
1030 gdraw->cur->width = region->x1 - region->x0;
1031 gdraw->cur->height = region->y1 - region->y0;
1032 }
1033
1034 gstats->nonzero_flags |= GDRAW_STATS_rendtarg;
1035
1036 glBindFramebuffer(GL_FRAMEBUFFER, gdraw->framebuffer_stack_object);
1037 set_render_target_state();
1038
1039 assert(glCheckFramebufferStatus(GL_FRAMEBUFFER) == GL_FRAMEBUFFER_COMPLETE);
1040
1041 // viewport for clear (make sure scissor is inside viewport just in case)
1042 glViewport(0, 0, gdraw->frametex_width, gdraw->frametex_height);
1043
1044 k = (int) (n->color_buffer - gdraw->rendertargets.handle);
1045 if (region) {
1046 S32 ox, oy;
1047
1048 // in a perfect world, we'd only need 1 pixel of border on all sides for
1049 // bilinear filtering, which would mean pad = 1. however, texture interpolator
1050 // precision is not that high even on PC parts, and if we only use 1 pixel of
1051 // padding we will often get some un-filled pixels "creeping in" from the sides.
1052 // pad = 2 is fine on recent PC parts, but not old PC parts or even fairly new
1053 // mobile parts, so we play it safe and use 3 pixels which so far gives good
1054 // results everywhere.
1055 S32 pad = 3;
1056
1057 // region.x0,y0 are the top left of the rectangle in display space
1058 // x,y are the *bottom* left of the rectangle in window space
1059 S32 h = gdraw->tph;
1060 S32 xt0, yt0, xt1, yt1;
1061 S32 x0, y0, x1, y1;
1062
1063 if (gdraw->in_blur || !gdraw->tile_enabled)
1064 ox = oy = 0;
1065 else
1066 ox = gdraw->tx0, oy = gdraw->ty0;
1067
1068 // clamp region to tile (in gdraw coords)
1069 xt0 = RR_MAX(region->x0 - ox, 0);
1070 yt0 = RR_MAX(region->y0 - oy, 0);
1071 xt1 = RR_MIN(region->x1 - ox, gdraw->tpw);
1072 yt1 = RR_MIN(region->y1 - oy, gdraw->tph);
1073
1074 // but the padding gets clamped to framebuffer coords! also transfer to window space here.
1075 x0 = RR_MAX( xt0 - pad, 0);
1076 y0 = RR_MAX(h - yt1 - pad, 0);
1077 x1 = RR_MIN( xt1 + pad, gdraw->frametex_width);
1078 y1 = RR_MIN(h - yt0 + pad, gdraw->frametex_height);
1079
1080 if (x1 <= x0 || y1 <= y0) { // region doesn't intersect with current tile
1081 --gdraw->cur;
1082
1083 // remove color and stencil buffers
1084 glFramebufferTexture2D(GL_FRAMEBUFFER, GL_COLOR_ATTACHMENT0 , GL_TEXTURE_2D, 0, 0);
1085 glFramebufferTexture2D(GL_FRAMEBUFFER, GL_STENCIL_ATTACHMENT, GL_TEXTURE_2D, 0, 0);
1086 glFramebufferTexture2D(GL_FRAMEBUFFER, GL_DEPTH_ATTACHMENT , GL_TEXTURE_2D, 0, 0);
1087
1088 // switch render target back
1089 if (gdraw->cur == gdraw->frame)
1090 glBindFramebuffer(GL_FRAMEBUFFER, gdraw->main_framebuffer);
1091 else
1092 set_render_target_state();
1093
1094 set_viewport();
1095 set_projection();
1096 opengl_check();
1097
1098 // free our render target
1099 gdraw_FreeTexture((GDrawTexture *) n->color_buffer, 0, gstats);
1100
1101 // note: don't send a warning since this will happen during regular tiled rendering
1102 return false;
1103 }
1104
1105 glEnable(GL_SCISSOR_TEST);
1106 glScissor(x0, y0, x1 - x0, y1 - y0);
1107 gdraw->rt_valid[k].x0 = xt0;
1108 gdraw->rt_valid[k].y0 = yt0;
1109 gdraw->rt_valid[k].x1 = xt1;
1110 gdraw->rt_valid[k].y1 = yt1;
1111 gstats->cleared_pixels += (x1 - x0) * (y1 - y0);
1112 } else {
1113 glDisable(GL_SCISSOR_TEST);
1114 gdraw->rt_valid[k].x0 = 0;
1115 gdraw->rt_valid[k].y0 = 0;
1116 gdraw->rt_valid[k].x1 = gdraw->frametex_width;
1117 gdraw->rt_valid[k].y1 = gdraw->frametex_height;
1118 gstats->cleared_pixels += gdraw->frametex_width * gdraw->frametex_height;
1119 }
1120
1121 gstats->nonzero_flags |= GDRAW_STATS_clears;
1122 gstats->num_clears += 1;
1123
1124#ifdef GDRAW_FEWER_CLEARS
1125 if (region) {
1126 clear_with_rect(region, n==gdraw->frame+1, gstats);
1127 } else
1128#endif // GDRAW_FEWER_CLEARS
1129 {
1130 glClearColor(0,0,0,0); // must clear destination alpha
1131 glClearStencil(0);
1132 glClearDepth(1);
1133 glStencilMask(255);
1134 glDepthMask(GL_TRUE);
1135 glColorMask(1,1,1,1);
1136 glDisable(GL_STENCIL_TEST);
1137 if (n == gdraw->frame+1)
1138 glClear(GL_COLOR_BUFFER_BIT | GL_STENCIL_BUFFER_BIT | GL_DEPTH_BUFFER_BIT);
1139 else
1140 glClear(GL_COLOR_BUFFER_BIT);
1141 }
1142
1143 set_viewport();
1144 set_projection();
1145
1146 opengl_check();
1147
1148 return true;
1149}

1152static GDrawTexture *RADLINK gdraw_TextureDrawBufferEnd(GDrawStats *gstats)
1153{
1154 GDrawFramebufferState *n = gdraw->cur;
1155 GDrawFramebufferState *m = --gdraw->cur;
1156 if (gdraw->fw == 0 || gdraw->fh == 0) return 0;
1157
1158 if (n >= &gdraw->frame[MAX_RENDER_STACK_DEPTH])
1159 return 0; // already returned a warning in Start...()
1160
1161 assert(m >= gdraw->frame); // bug in Iggy -- unbalanced
1162
1163 if (m != gdraw->frame)
1164 assert(m->color_buffer != nullptr);
1165 assert(n->color_buffer != nullptr);
1166
1167 // remove color and stencil buffers
1168 glFramebufferRenderbuffer(GL_FRAMEBUFFER, GL_COLOR_ATTACHMENT0 , GL_RENDERBUFFER, 0);
1169 glFramebufferRenderbuffer(GL_FRAMEBUFFER, GL_STENCIL_ATTACHMENT, GL_RENDERBUFFER, 0);
1170 glFramebufferRenderbuffer(GL_FRAMEBUFFER, GL_DEPTH_ATTACHMENT , GL_RENDERBUFFER, 0);
1171
1172#ifdef GDRAW_MULTISAMPLING
1173 if (gdraw->multisampling) {
1174 // blit from multisample to texture
1175 if (n->color_buffer->handle.tex.gl_renderbuf) {
1176 GLuint res;
1177 glBindFramebuffer(GL_READ_FRAMEBUFFER, gdraw->framebuffer_copy_to_texture);
1178 glFramebufferRenderbuffer(GL_READ_FRAMEBUFFER, GL_STENCIL_ATTACHMENT, GL_RENDERBUFFER, 0);
1179 glFramebufferRenderbuffer(GL_DRAW_FRAMEBUFFER, GL_STENCIL_ATTACHMENT, GL_RENDERBUFFER, 0);
1180 glFramebufferRenderbuffer(GL_READ_FRAMEBUFFER, GL_DEPTH_ATTACHMENT, GL_RENDERBUFFER, 0);
1181 glFramebufferRenderbuffer(GL_DRAW_FRAMEBUFFER, GL_DEPTH_ATTACHMENT, GL_RENDERBUFFER, 0);
1182 glFramebufferRenderbuffer(GL_READ_FRAMEBUFFER, GL_COLOR_ATTACHMENT0, GL_RENDERBUFFER, n->color_buffer->handle.tex.gl_renderbuf);
1183 glFramebufferTexture2D(GL_DRAW_FRAMEBUFFER, GL_COLOR_ATTACHMENT0, GL_TEXTURE_2D, n->color_buffer->handle.tex.gl, 0);
1184 res = glCheckFramebufferStatus(GL_DRAW_FRAMEBUFFER);
1185 glBlitFramebuffer(0,0,gdraw->tpw,gdraw->tph,0,0,gdraw->tpw,gdraw->tph, GL_COLOR_BUFFER_BIT, GL_NEAREST);
1186 gstats->nonzero_flags |= GDRAW_STATS_blits;
1187 gstats->num_blits += 1;
1188 gstats->num_blit_pixels += (gdraw->tpw * gdraw->tph);
1189
1190 glFramebufferRenderbuffer(GL_DRAW_FRAMEBUFFER, GL_COLOR_ATTACHMENT0, GL_RENDERBUFFER, 0);
1191 glFramebufferRenderbuffer(GL_READ_FRAMEBUFFER, GL_COLOR_ATTACHMENT0, GL_RENDERBUFFER, 0);
1192 glBindFramebuffer(GL_READ_FRAMEBUFFER, 0);
1193 }
1194 }
1195#endif
1196
1197 gstats->nonzero_flags |= GDRAW_STATS_rendtarg;
1198 gstats->rendertarget_changes += 1;
1199
1200 // set the new state
1201 if (m == gdraw->frame) // back to initial framebuffer

1210 opengl_check();
1211
1212 return (GDrawTexture *) n->color_buffer;
1213}

1304static int set_render_state(GDrawRenderState *r, S32 vformat, const int **ovvars, GDrawPrimitive *p, GDrawStats *gstats)
1305{
1306 static struct gdraw_gl_blendspec {
1307 GLboolean enable;
1308 GLenum src;
1309 GLenum dst;
1310 } blends[ASSERT_COUNT(GDRAW_BLEND__count, 6)] = {
1311 { GL_FALSE, GL_ONE, GL_ZERO }, // GDRAW_BLEND_none
1312 { GL_TRUE, GL_ONE, GL_ONE_MINUS_SRC_ALPHA }, // GDRAW_BLEND_alpha
1313 { GL_TRUE, GL_DST_COLOR, GL_ONE_MINUS_SRC_ALPHA }, // GDRAW_BLEND_multiply
1314 { GL_TRUE, GL_ONE, GL_ONE }, // GDRAW_BLEND_add
1315
1316 { GL_FALSE, GL_ONE, GL_ZERO }, // GDRAW_BLEND_filter
1317 { GL_FALSE, GL_ONE, GL_ZERO }, // GDRAW_BLEND_special
1318 };
1319
1320 F32 world[2*4];
1321 ProgramWithCachedVariableLocations *prg;
1322 int *fvars, *vvars;
1323 int blend_mode;
1324
1325 opengl_check();
1326 assert((vformat >= 0 && vformat < GDRAW_vformat__basic_count) || vformat == GDRAW_vformat_ihud1);
1327
1328 if (vformat == GDRAW_vformat_ihud1) {
1329 glEnable(GL_BLEND);
1330 glBlendFunc(GL_ONE, GL_ONE_MINUS_SRC_ALPHA); // premultiplied alpha blend mode
1331 prg = &gdraw->ihud[0];
1332 } else {
1333 // apply the major blend mode
1334 blend_mode = r->blend_mode;
1335 assert(blend_mode >= 0 && blend_mode < sizeof(blends)/sizeof(*blends));
1336 if (blends[blend_mode].enable) {
1337 glEnable(GL_BLEND);
1338 glBlendFunc(blends[blend_mode].src, blends[blend_mode].dst);
1339 } else
1340 glDisable(GL_BLEND);
1341
1342 // set the fragment program if it wasn't set above
1343 if (r->blend_mode != GDRAW_BLEND_special) {
1344 // make sure data has been initialized
1345 int which = r->tex0_mode, additive = 0;
1346
1347 if (r->cxf_add) {
1348 additive = 1;
1349 if (r->cxf_add[3]) additive = 2;
1350 }
1351
1352 prg = &gdraw->fprog[which][additive][vformat];
1353 } else
1354 prg = &gdraw->exceptional_blend[r->special_blend];
1355 }
1356
1357 use_lazy_shader(prg);
1358 opengl_check();
1359 fvars = prg->vars[0];
1360 vvars = prg->vars[1];
1361
1362 if (vformat == GDRAW_vformat_ihud1) {
1363 F32 wv[2][4] = { 1.0f/960,0,0,-1.0, 0,-1.0f/540,0,+1.0 };
1364 glUniform4fv(vvars[VAR_ihudv_worldview], 2, wv[0]);
1365 opengl_check();
1366 glUniform4fv(vvars[VAR_ihudv_material], p->uniform_count, p->uniforms);
1367 opengl_check();
1368 glUniform1f(vvars[VAR_ihudv_textmode], p->drawprim_mode ? 0.0f : 1.0f);
1369 opengl_check();
1370 } else {
1371
1372 // set vertex shader constants
1373 if (!r->use_world_space)
1374 gdraw_ObjectSpace(world, r->o2w, depth_from_id(r->id), 0.0f);
1375 else
1376 gdraw_WorldSpace(world, gdraw->world_to_pixel, depth_from_id(r->id), 0.0f);
1377
1378 #ifdef FLASH_10
1379 set_3d_projection(vvars, world, gdraw->xform_3d);
1380 #else
1381 set_world_projection(vvars, world);
1382 #endif
1383
1384 if (vvars[VVAR_xoff] >= 0)
1385 glUniform4fv(vvars[VVAR_xoff], 1, r->edge_matrix);
1386
1387 if (r->texgen0_enabled) {
1388 assert(vvars[VVAR_texgen_s] >= 0 && vvars[VVAR_texgen_t] >= 0);
1389 glUniform4fv(vvars[VVAR_texgen_s], 1, r->s0_texgen);
1390 glUniform4fv(vvars[VVAR_texgen_t], 1, r->t0_texgen);
1391 }
1392 }
1393
1394 // texture stuff
1395 set_texture(0, r->tex[0]);
1396 opengl_check();
1397
1398 if (r->tex[0] && gdraw->has_conditional_non_power_of_two && ((GDrawHandle*) r->tex[0])->handle.tex.nonpow2) {
1399 // only wrap mode allowed in conditional nonpow2 is clamp; this should
1400 // have been set when the texture was created, but to be on the safe side...
1401 glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_WRAP_S, GL_CLAMP_TO_EDGE);
1402 glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_WRAP_T, GL_CLAMP_TO_EDGE);
1403 } else switch (r->wrap0) {
1404 case GDRAW_WRAP_repeat:
1405 glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_WRAP_S, GL_REPEAT);
1406 glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_WRAP_T, GL_REPEAT);
1407 break;
1408 case GDRAW_WRAP_clamp:
1409 glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_WRAP_S, GL_CLAMP_TO_EDGE);
1410 glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_WRAP_T, GL_CLAMP_TO_EDGE);
1411 break;
1412 case GDRAW_WRAP_mirror:
1413 glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_WRAP_S, GL_MIRRORED_REPEAT);
1414 glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_WRAP_T, GL_MIRRORED_REPEAT);
1415 break;
1416 }
1417
1418 glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MAG_FILTER, r->nearest0 ? GL_NEAREST : GL_LINEAR);
1419
1420 // fragment shader constants
1421
1422 if (fvars[VAR_cmul] >= 0)
1423 glUniform4f(fvars[VAR_cmul], r->color[0], r->color[1], r->color[2], r->color[3]);
1424 if (fvars[VAR_cadd] >= 0)
1425 if (r->cxf_add)
1426 glUniform4f(fvars[VAR_cadd], r->cxf_add[0]/255.0f, r->cxf_add[1]/255.0f, r->cxf_add[2]/255.0f, r->cxf_add[3]/255.0f);
1427 else
1428 glUniform4f(fvars[VAR_cadd], 0,0,0,0);
1429 if (fvars[VAR_focal] >= 0)
1430 glUniform4fv(fvars[VAR_focal], 1, r->focal_point);
1431
1432 glActiveTexture(GL_TEXTURE0);
1433
1434 // Set pixel operation states
1435
1436 if (r->scissor) {
1437 S32 x0,y0,x1,y1;
1438 // scissor.x0,y0 are the top left of the rectangle in display space
1439 // x,y are the *bottom* left of the rectangle in window space
1440 x0 = r->scissor_rect.x0;
1441 y0 = r->scissor_rect.y1;
1442 x1 = r->scissor_rect.x1;
1443 y1 = r->scissor_rect.y0;
1444 // convert into tile-relative coordinates
1445 if (gdraw->tile_enabled) {
1446 x0 -= gdraw->tx0;
1447 y0 -= gdraw->ty0;
1448 x1 -= gdraw->tx0;
1449 y1 -= gdraw->ty0;
1450 }
1451 // convert bottom-most edge to bottom-relative
1452 y0 = (gdraw->th) - y0;
1453 y1 = (gdraw->th) - y1;
1454 if (gdraw->cur == gdraw->frame) {
1455 // move into viewport space
1456 x0 += gdraw->vx;
1457 y0 += gdraw->vy;
1458 x1 += gdraw->vx;
1459 y1 += gdraw->vy;
1460 }
1461 glScissor(x0,y0,x1-x0,y1-y0);
1462 glEnable(GL_SCISSOR_TEST);
1463 } else
1464 glDisable(GL_SCISSOR_TEST);
1465
1466 glStencilOp(GL_KEEP, GL_KEEP, GL_REPLACE);
1467 glStencilMask(r->stencil_set);
1468 glStencilFunc(GL_EQUAL, 255, r->stencil_test);
1469 if (r->stencil_set | r->stencil_test)
1470 glEnable(GL_STENCIL_TEST);
1471 else
1472 glDisable(GL_STENCIL_TEST);
1473
1474 if (r->stencil_set)
1475 glColorMask(0,0,0,0);
1476 else
1477 glColorMask(1,1,1,1);
1478
1479 if (r->test_id) {
1480 glEnable(GL_DEPTH_TEST);
1481 glDepthFunc(GL_LESS);
1482 } else {
1483 glDisable(GL_DEPTH_TEST);
1484 glDepthFunc(GL_LESS);
1485 }
1486
1487 if (r->set_id)
1488 glDepthMask(GL_TRUE);
1489 else
1490 glDepthMask(GL_FALSE);
1491
1492 opengl_check();
1493 if (ovvars)
1494 *ovvars = vvars;
1495
1496 return 1;
1497}

1563{
1564 2,4,4,5
1565};
1566
1567static void RADLINK gdraw_DrawIndexedTriangles(GDrawRenderState *r, GDrawPrimitive *p, GDrawVertexBuffer *buf, GDrawStats *gstats)
1568{
1569 GDrawHandle *vb = (GDrawHandle *) buf;
1570 if (vb) {
1571 glBindBuffer(GL_ARRAY_BUFFER, vb->handle.vbuf.base);
1572 glBindBuffer(GL_ELEMENT_ARRAY_BUFFER, vb->handle.vbuf.indices);
1573 } else {
1574 glBindBuffer(GL_ARRAY_BUFFER, 0);
1575 glBindBuffer(GL_ELEMENT_ARRAY_BUFFER, 0);
1576 }
1577
1578 if (!set_render_state(r,p->vertex_format, nullptr, p, gstats)) return;
1579 gstats->nonzero_flags |= GDRAW_STATS_batches;
1580 gstats->num_batches += 1;
1581 gstats->drawn_indices += p->num_indices;
1582 gstats->drawn_vertices += p->num_vertices;
1583
1584 if (vb || p->indices) { // regular path
1585 set_vertex_format(p->vertex_format, p->vertices);
1586 glDrawElements(GL_TRIANGLES, p->num_indices, GL_UNSIGNED_SHORT, p->indices);
1587 } else { // dynamic quads
1588 S32 pos = 0;
1589 U32 stride = vformat_stride[p->vertex_format]; // in units of sizeof(F32)
1590 glBindBuffer(GL_ELEMENT_ARRAY_BUFFER, gdraw->quad_ib);
1591 assert(p->num_vertices % 4 == 0);
1592
1593 while (pos < p->num_vertices) {
1594 S32 vert_count = RR_MIN(p->num_vertices - pos, QUAD_IB_COUNT * 4);
1595 set_vertex_format(p->vertex_format, p->vertices + pos*stride);
1596 glDrawElements(GL_TRIANGLES, (vert_count >> 2) * 6, GL_UNSIGNED_SHORT, nullptr);
1597 pos += vert_count;
1598 }
1599
1600 glBindBuffer(GL_ELEMENT_ARRAY_BUFFER, 0);
1601 }
1602 reset_vertex_format(p->vertex_format);
1603
1604 if (vb) {
1605 glBindBuffer(GL_ARRAY_BUFFER, 0);
1606 glBindBuffer(GL_ELEMENT_ARRAY_BUFFER, 0);
1607 }
1608
1609 opengl_check();
1610 tag_resources(vb, r->tex[0], r->tex[1]);
1611}

1631 vert[3][0] = px0; vert[3][1] = py1; vert[3][2] = s0; vert[3][3] = t1;

1639 set_vertex_format(GDRAW_vformat_v2tc2, vert[0]);

1648 gstats->num_batches += 1;
1649 gstats->drawn_vertices += 4;
1650 gstats->drawn_indices += 6;
1651
1652 opengl_check();
1653}

1772static void RADLINK gdraw_FilterQuad(GDrawRenderState *r, S32 x0, S32 y0, S32 x1, S32 y1, GDrawStats *gstats)
1773{
1774 F32 tc[4];
1775 gswf_recti s;
1776
1777 // clip to tile boundaries
1778 s.x0 = RR_MAX(x0, gdraw->tx0p);
1779 s.y0 = RR_MAX(y0, gdraw->ty0p);
1780 s.x1 = RR_MIN(x1, gdraw->tx0p + gdraw->tpw);
1781 s.y1 = RR_MIN(y1, gdraw->ty0p + gdraw->tph);
1782 if (s.x1 <= s.x0 || s.y1 <= s.y0)
1783 return;
1784
1785 // if it's a rendertarget, it's inverted from our design because OpenGL is bottom-left 0,0
1786 // and we have to compensate for scaling
1787 tc[0] = (s.x0 - gdraw->tx0p) / (F32) gdraw->frametex_width;
1788 tc[1] = (gdraw->tph - (s.y0 + gdraw->ty0p)) / (F32) gdraw->frametex_height;
1789 tc[2] = (s.x1 - gdraw->tx0p) / (F32) gdraw->frametex_width;
1790 tc[3] = (gdraw->tph - (s.y1 - gdraw->ty0p)) / (F32) gdraw->frametex_height;
1791
1792 glUseProgram(0);
1793 set_texture(0, 0);
1794 set_texture(1, 0);
1795 set_texture(2, 0);
1796
1797 glStencilOp(GL_KEEP, GL_KEEP, GL_KEEP);
1798 glStencilMask(255);
1799 glDisable(GL_STENCIL_TEST);
1800 glColorMask(1,1,1,1);
1801 glDisable(GL_BLEND);
1802 glDisable(GL_DEPTH_TEST);
1803 opengl_check();
1804
1805 if (r->blend_mode == GDRAW_BLEND_filter) {
1806 switch (r->filter) {
1807 case GDRAW_FILTER_blur: {
1808 GDrawBlurInfo b;
1809 gswf_recti bounds = *get_valid_rect(r->tex[0]);
1810 gdraw_ShiftRect(&s, &s, -gdraw->tx0p, -gdraw->ty0p); // blur uses physical rendertarget coordinates
1811
1812 b.BlurPass = gdraw_DriverBlurPass;
1813 b.w = gdraw->tpw;
1814 b.h = gdraw->tph;
1815 b.frametex_width = gdraw->frametex_width;
1816 b.frametex_height = gdraw->frametex_height;
1817
1818 // blur passes must override the viewport/ortho projection
1819
1820 gdraw->in_blur = true; // prevent viewport/projection munging in start/end texture
1821 set_viewport();
1822 set_projection();
1823 gdraw_Blur(&gdraw_funcs, &b,r, &s, &bounds, gstats);
1824
1825 gdraw->in_blur = false;
1826
1827 set_viewport();
1828 set_projection();
1829 break;
1830 }
1831
1832 case GDRAW_FILTER_colormatrix:
1833 gdraw_Colormatrix(r, &s, tc, gstats);
1834 break;
1835
1836 case GDRAW_FILTER_dropshadow:
1837 gdraw_Filter(r, &s, tc, 0, gstats);
1838 break;
1839
1840 case GDRAW_FILTER_bevel:
1841 gdraw_Filter(r, &s, tc, 1, gstats);
1842 break;
1843
1844 default:
1845 assert(0);
1846 }
1847 } else {
1848 GDrawTexture *blend_tex = nullptr;

1852 // and do the blending in the pixel shader. we do this with CopyTexSubImage,

1854 // @TODO: propogate the rectangle down and only copy what we need, like in 360

1857 blend_tex = (GDrawTexture *) get_color_rendertarget(gstats);

1859 if (gdraw->cur != gdraw->frame)

1863
1864 set_texture(1, blend_tex);
1865 }
1866
1867 if (!set_render_state(r, GDRAW_vformat_v2tc2, &vvars, nullptr, gstats))
1868 return;
1869 do_screen_quad(&s, tc, vvars, gstats, 0);
1870 tag_resources(r->tex[0],r->tex[1],0);

1872 gdraw_FreeTexture(blend_tex, 0, gstats);
