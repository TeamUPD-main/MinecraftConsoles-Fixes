142GameRuleDefinition *LevelGenerationOptions::addChild(ConsoleGameRules::EGameRuleType ruleType)
143{
144 GameRuleDefinition *rule = nullptr;
145 if(ruleType == ConsoleGameRules::eGameRuleType_ApplySchematic)
146 {
147 rule = new ApplySchematicRuleDefinition(this);
148 m_schematicRules.push_back(static_cast<ApplySchematicRuleDefinition *>(rule));
149 }
150 else if(ruleType == ConsoleGameRules::eGameRuleType_GenerateStructure)
151 {
152 rule = new ConsoleGenerateStructure();
153 m_structureRules.push_back(static_cast<ConsoleGenerateStructure *>(rule));
154 }
155 else if(ruleType == ConsoleGameRules::eGameRuleType_BiomeOverride)
156 {
157 rule = new BiomeOverride();
158 m_biomeOverrides.push_back(static_cast<BiomeOverride *>(rule));
159 }
160 else if(ruleType == ConsoleGameRules::eGameRuleType_StartFeature)
161 {
162 rule = new StartFeature();
163 m_features.push_back(static_cast<StartFeature *>(rule));
164 }
165 else
166 {
167#ifndef _CONTENT_PACKAGE
168 wprintf(L"LevelGenerationOptions: Attempted to add invalid child rule - %d\n", ruleType );
169#endif
170 }
171 return rule;
172}

256void LevelGenerationOptions::processSchematics(LevelChunk *chunk)
257{
258 PIXBeginNamedEvent(0,"Processing schematics for chunk (%d,%d)", chunk->x, chunk->z);
259 AABB *chunkBox = AABB::newTemp(chunk->x*16,0,chunk->z*16,chunk->x*16 + 16,Level::maxBuildHeight,chunk->z*16 + 16);
260 for( ApplySchematicRuleDefinition *rule : m_schematicRules )
261 rule->processSchematic(chunkBox, chunk);
262
263 int cx = (chunk->x << 4);
264 int cz = (chunk->z << 4);
265
266 for ( ConsoleGenerateStructure *structureStart : m_structureRules )
267 {
268 if (structureStart->getBoundingBox()->intersects(cx, cz, cx + 15, cz + 15))
269 {
270 BoundingBox *bb = new BoundingBox(cx, cz, cx + 15, cz + 15);
271 structureStart->postProcess(chunk->level, nullptr, bb);
272 delete bb;
273 }
274 }
275 PIXEndNamedEvent();
276}

289bool LevelGenerationOptions::checkIntersects(int x0, int y0, int z0, int x1, int y1, int z1)
290{
291 PIXBeginNamedEvent(0,"Check Intersects");
292
293 // As an optimisation, we can quickly discard things below a certain y which makes most ore checks faster due to
294 // a) ores generally being below ground/sea level and b) tutorial world additions generally being above ground/sea level
295 if(!m_bHaveMinY)
296 {
297 for ( ApplySchematicRuleDefinition *rule : m_schematicRules )
298 {
299 int minY = rule->getMinY();
300 if(minY < m_minY) m_minY = minY;
301 }
302
303 for ( ConsoleGenerateStructure *structureStart : m_structureRules )
304 {
305 int minY = structureStart->getMinY();
306 if(minY < m_minY) m_minY = minY;
307 }
308
309 m_bHaveMinY = true;
310 }
311
312 // 4J Stu - We DO NOT intersect if our upper bound is below the lower bound for all schematics
313 if( y1 < m_minY ) return false;
314
315 bool intersects = false;
316 for( ApplySchematicRuleDefinition *rule : m_schematicRules )
317 {
318 intersects = rule->checkIntersects(x0,y0,z0,x1,y1,z1);
319 if(intersects) break;
320 }
321
322 if(!intersects)
323 {
324 for( ConsoleGenerateStructure *structureStart : m_structureRules )
325 {
326 intersects = structureStart->checkIntersects(x0,y0,z0,x1,y1,z1);
327 if(intersects) break;
328 }
329 }
330 PIXEndNamedEvent();
331 return intersects;
332}

344{
345 // If we have already loaded this, just return
346 auto it = m_schematics.find(filename);
347 if(it != m_schematics.end())

352 it->second->incrementRefCount();
353 return it->second;

355
356 ConsoleSchematicFile *schematic = nullptr;
357 byteArray data(pbData,dwLen);
358 ByteArrayInputStream bais(data);
359 DataInputStream dis(&bais);

361 schematic->load(&dis);
362 m_schematics[filename] = schematic;
363 bais.reset();
364 return schematic;
