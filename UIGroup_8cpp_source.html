62 int highestRenderable = 0;

241bool UIGroup::RequestFocus(UILayer* layerPtr)
242{
243 // Find the layer
244 unsigned int layerIndex = GetLayerIndex(layerPtr);
245
246 // Top layer is always allowed focus
247 if (layerIndex == 0) return true;
248
249 // Check layers above to see if any of them have focus
250 for (int i = layerIndex-1; i >= 0; i--)
251 {
252 if (m_layers[i]->m_hasFocus) return false;
253 }
254
255 return true;
256}
