77Path *PathFinder::findPath(Entity *e, Node *from, Node *to, Node *size, float maxDist)
78{
79 from->g = 0;
80 from->h = from->distanceToSqr(to);
81 from->f = from->h;
82
83 openSet.clear();
84 openSet.insert(from);
85
86 Node *closest = from;
87
88 while (!openSet.isEmpty())
89 {
90 Node *x = openSet.pop();
91
92 if (x->equals(to))
93 {
94 return reconstruct_path(from, to);
95 }
96
97 if (x->distanceToSqr(to) < closest->distanceToSqr(to))
98 {
99 closest = x;
100 }
101 x->closed = true;
102
103 int neighborCount = getNeighbors(e, x, size, to, maxDist);
104 for (int i = 0; i < neighborCount; i++)
105 {
106 Node *y = neighbors->data[i];
107
108 float tentative_g_score = x->g + x->distanceToSqr(y);
109 if (!y->inOpenSet() || tentative_g_score < y->g)
110 {
111 y->cameFrom = x;
112 y->g = tentative_g_score;
113 y->h = y->distanceToSqr(to);
114 if (y->inOpenSet())
115 {
116 openSet.changeCost(y, y->g + y->h);
117 }
118 else
119 {
120 y->f = y->g + y->h;
121 openSet.insert(y);
122 }
123 }
124 }
125 }
126
127 if (closest == from) return nullptr;
128 return reconstruct_path(from, closest);
129}

131int PathFinder::getNeighbors(Entity *entity, Node *pos, Node *size, Node *target, float maxDist)
132{
133 int p = 0;
134
135 int jumpSize = 0;
136 if (isFree(entity, pos->x, pos->y + 1, pos->z, size) == TYPE_OPEN) jumpSize = 1;
137
138 Node *n = getNode(entity, pos->x, pos->y, pos->z + 1, size, jumpSize);
139 Node *w = getNode(entity, pos->x - 1, pos->y, pos->z, size, jumpSize);
140 Node *e = getNode(entity, pos->x + 1, pos->y, pos->z, size, jumpSize);
141 Node *s = getNode(entity, pos->x, pos->y, pos->z - 1, size, jumpSize);
142
143 if (n != nullptr && !n->closed && n->distanceTo(target) < maxDist) neighbors->data[p++] = n;
144 if (w != nullptr && !w->closed && w->distanceTo(target) < maxDist) neighbors->data[p++] = w;
145 if (e != nullptr && !e->closed && e->distanceTo(target) < maxDist) neighbors->data[p++] = e;
146 if (s != nullptr && !s->closed && s->distanceTo(target) < maxDist) neighbors->data[p++] = s;
147
148 return p;

151Node *PathFinder::getNode(Entity *entity, int x, int y, int z, Node *size, int jumpSize)
152{
153 Node *best = nullptr;
154 int pathType = isFree(entity, x, y, z, size);
155 if (pathType == TYPE_WALKABLE) return getNode(x, y, z);
156 if (pathType == TYPE_OPEN) best = getNode(x, y, z);
157 if (best == nullptr && jumpSize > 0 && pathType != TYPE_FENCE && pathType != TYPE_TRAP && isFree(entity, x, y + jumpSize, z, size) == TYPE_OPEN)
158 {
159 best = getNode(x, y + jumpSize, z);
160 y += jumpSize;
161 }
162
163 if (best != nullptr)
164 {
165 int drop = 0;
166 int cost = 0;
167 while (y > 0)
168 {
169 cost = isFree(entity, x, y - 1, z, size);
170 if (avoidWater && cost == TYPE_WATER) return nullptr;
171 if (cost != TYPE_OPEN) break;
172 // fell too far?
173 if (++drop >= 4) return nullptr; // 4J - rolling this back to pre-java 1.6.4 version as we're suspicious of the performance implications of this
174// if (drop++ >= entity->getMaxFallDistance()) return nullptr;
175 y--;
176
177 if (y > 0) best = getNode(x, y, z);
178 }
179 // fell into lava?
180 if (cost == TYPE_LAVA) return nullptr;
181 }
182
183 return best;
184}

210int PathFinder::isFree(Entity *entity, int x, int y, int z, Node *size, bool avoidWater, bool canOpenDoors, bool canPassDoors)
211{
212 bool walkable = false;
213 for (int xx = x; xx < x + size->x; xx++)
214 for (int yy = y; yy < y + size->y; yy++)
215 for (int zz = z; zz < z + size->z; zz++)
216 {
217 int tileId = entity->level->getTile(xx, yy, zz);
218 if(tileId <= 0) continue;
219 if (tileId == Tile::trapdoor_Id) walkable = true;
220 else if (tileId == Tile::water_Id || tileId == Tile::calmWater_Id)
221 {
222 if (avoidWater) return TYPE_WATER;
223 else walkable = true;
224 }
225 else if (!canPassDoors && tileId == Tile::door_wood_Id)
226 {
227 return TYPE_BLOCKED;
228 }
229
230 Tile *tile = Tile::tiles[tileId];
231
232 // 4J Stu - Use new getTileRenderShape passing in the tileId we have already got
233 if (entity->level->getTileRenderShape(tileId) == Tile::SHAPE_RAIL)
234 {
235 int xt = Mth::floor(entity->x);
236 int yt = Mth::floor(entity->y);
237 int zt = Mth::floor(entity->z);
238 if (entity->level->getTileRenderShape(xt, yt, zt) == Tile::SHAPE_RAIL
239 || entity->level->getTileRenderShape(xt, yt - 1, zt) == Tile::SHAPE_RAIL)
240 {
241 continue;
242 }
243 else
244 {
245 return TYPE_FENCE;
246 }
247 }
248
249 if (tile->isPathfindable(entity->level, xx, yy, zz)) continue;
250 if (canOpenDoors && tileId == Tile::door_wood_Id) continue;
251
252 int renderShape = tile->getRenderShape();
253 if (renderShape == Tile::SHAPE_FENCE || tileId == Tile::fenceGate_Id || renderShape == Tile::SHAPE_WALL) return TYPE_FENCE;
254 if (tileId == Tile::trapdoor_Id) return TYPE_TRAP;
255 Material *m = tile->material;
256 if (m == Material::lava)
257 {
258 if (entity->isInLava()) continue;
259 return TYPE_LAVA;
260 }
261 return TYPE_BLOCKED;
262 }
263
264 return walkable ? TYPE_WALKABLE : TYPE_OPEN;
265}

268Path *PathFinder::reconstruct_path(Node *from, Node *to)
269{
270 int count = 1;
271 Node *n = to;
272 while (n->cameFrom != nullptr)
273 {
274 count++;
275 n = n->cameFrom;
276 }
277
278 NodeArray nodes = NodeArray(count);
279 n = to;
280 nodes.data[--count] = n;
281 while (n->cameFrom != nullptr)
282 {
283 n = n->cameFrom;
284 nodes.data[--count] = n;
285 }
286 Path *ret = new Path(nodes);
287 delete [] nodes.data;
288 return ret;
