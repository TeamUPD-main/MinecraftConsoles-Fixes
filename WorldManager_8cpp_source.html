87static bool EnsureDirectoryExists(const std::wstring &directoryPath, bool *outCreated)
88{
89 if (outCreated != NULL)
90 {
91 *outCreated = false;
92 }
93 if (directoryPath.empty())
94 {
95 return false;
96 }
97
98 DWORD attrs = GetFileAttributesW(directoryPath.c_str());
99 if (attrs != INVALID_FILE_ATTRIBUTES)
100 {
101 if ((attrs & FILE_ATTRIBUTE_DIRECTORY) != 0)
102 {
103 return true;
104 }
105
106 LogErrorf("world-io", "path exists but is not a directory: %s", WideToUtf8(directoryPath).c_str());
107 return false;
108 }
109
110 if (CreateDirectoryW(directoryPath.c_str(), NULL))
111 {
112 if (outCreated != NULL)
113 {
114 *outCreated = true;
115 }
116 return true;
117 }
118
119 DWORD error = GetLastError();
120 if (error == ERROR_ALREADY_EXISTS)
121 {
122 attrs = GetFileAttributesW(directoryPath.c_str());
123 if (attrs != INVALID_FILE_ATTRIBUTES && ((attrs & FILE_ATTRIBUTE_DIRECTORY) != 0))
124 {
125 return true;
126 }
127 }
128
129 LogErrorf(
130 "world-io",
131 "failed to create directory %s (error=%lu)",
132 WideToUtf8(directoryPath).c_str(),
133 (unsigned long)error);
134 return false;
135}

365static EWorldSaveLoadResult PrepareWorldSaveData(
366 const std::wstring &targetWorldName,
367 const std::string &targetSaveFilename,
368 int actionPad,
369 WorldManagerTickProc tickProc,
370 LoadSaveDataThreadParam **outSaveData,
371 std::string *outResolvedSaveFilename)
372{
373 if (outSaveData == NULL)
374 {
375 return eWorldSaveLoad_Failed;
376 }
377 *outSaveData = NULL;
378 if (outResolvedSaveFilename != NULL)
379 {
380 outResolvedSaveFilename->clear();
381 }
382
383 LogWorldIO("enumerating saves for configured world");
384 StorageManager.ClearSavesInfo();
385
386 SaveInfoQueryContext infoContext;
387 int infoState = StorageManager.GetSavesInfo(actionPad, &GetSavesInfoCallbackProc, &infoContext, "save");
388 if (infoState == C4JStorage::ESaveGame_Idle)
389 {
390 infoContext.done = true;
391 infoContext.success = true;
392 infoContext.details = StorageManager.ReturnSavesInfo();
393 }
394 else if (infoState != C4JStorage::ESaveGame_GetSavesInfo)
395 {
396 LogWorldIO("GetSavesInfo failed to start");
397 return eWorldSaveLoad_Failed;
398 }
399
400 if (!WaitForSaveInfoResult(&infoContext, 10000, tickProc))
401 {
402 LogWorldIO("timed out waiting for save list");
403 return eWorldSaveLoad_Failed;
404 }
405
406 if (infoContext.details == NULL)
407 {
408 infoContext.details = StorageManager.ReturnSavesInfo();
409 }
410 if (infoContext.details == NULL)
411 {
412 LogWorldIO("failed to retrieve save list");
413 return eWorldSaveLoad_Failed;
414 }
415
416 int matchedIndex = -1;
417 if (!targetSaveFilename.empty())
418 {
419 // 1) If save ID is provided, search by it first
420 // This is the most stable way to reuse the same world target
421 for (int i = 0; i < infoContext.details->iSaveC; ++i)
422 {
423 LogEnumeratedSaveInfo(i, infoContext.details->SaveInfoA[i]);
424 if (SaveInfoMatchesSaveFilename(infoContext.details->SaveInfoA[i], targetSaveFilename))
425 {
426 matchedIndex = i;
427 break;
428 }
429 }
430 }
431
432 if (matchedIndex < 0 && targetSaveFilename.empty())
433 {
434 for (int i = 0; i < infoContext.details->iSaveC; ++i)
435 {
436 LogEnumeratedSaveInfo(i, infoContext.details->SaveInfoA[i]);
437 }
438 }
439
440 for (int i = 0; i < infoContext.details->iSaveC; ++i)
441 {
442 // 2) If no save matched by ID, try compatibility fallback
443 // Match worldName against save title or save filename
444 if (matchedIndex >= 0)
445 {
446 break;
447 }
448 if (SaveInfoMatchesWorldName(infoContext.details->SaveInfoA[i], targetWorldName))
449 {
450 matchedIndex = i;
451 break;
452 }
453 }
454
455 if (matchedIndex < 0)
456 {
457 LogWorldIO("no save matched configured world name");
458 return eWorldSaveLoad_NotFound;
459 }
460
461 std::wstring matchedTitle = Utf8ToWide(infoContext.details->SaveInfoA[matchedIndex].UTF8SaveTitle);
462 if (matchedTitle.empty())
463 {
464 matchedTitle = targetWorldName;
465 }
466 LogWorldName("matched save title", matchedTitle);
467 SAVE_INFO *matchedSaveInfo = &infoContext.details->SaveInfoA[matchedIndex];
468 std::wstring matchedFilename = Utf8ToWide(matchedSaveInfo->UTF8SaveFilename);
469 if (!matchedFilename.empty())
470 {
471 LogWorldName("matched save filename", matchedFilename);
472 }
473
474 ApplyWorldStorageTarget(targetWorldName, targetSaveFilename);
475
476 std::string resolvedSaveFilename;
477 if (matchedSaveInfo->UTF8SaveFilename[0] != 0)
478 {
479 // Prefer the save ID that was actually matched, then keep using it for future saves
480 resolvedSaveFilename = matchedSaveInfo->UTF8SaveFilename;
481 SetStorageSaveUniqueFilename(resolvedSaveFilename);
482 }
483 else if (!targetSaveFilename.empty())
484 {
485 resolvedSaveFilename = targetSaveFilename;
486 }
487
488 if (outResolvedSaveFilename != NULL)
489 {
490 *outResolvedSaveFilename = resolvedSaveFilename;
491 }
492
493 SaveDataLoadContext loadContext;
494 int loadState = StorageManager.LoadSaveData(matchedSaveInfo, &LoadSaveDataCallbackProc, &loadContext);
495 if (loadState != C4JStorage::ESaveGame_Load && loadState != C4JStorage::ESaveGame_Idle)
496 {
497 LogWorldIO("LoadSaveData failed to start");
498 return eWorldSaveLoad_Failed;
499 }
500
501 if (loadState == C4JStorage::ESaveGame_Load)
502 {
503 if (!WaitForSaveLoadResult(&loadContext, 15000, tickProc))
504 {
505 LogWorldIO("timed out waiting for save data load");
506 return eWorldSaveLoad_Failed;
507 }
508 if (loadContext.isCorrupt)
509 {
510 LogWorldIO("target save is corrupt; aborting load");
511 return eWorldSaveLoad_Failed;
512 }
513 }
514
515 unsigned int saveSize = StorageManager.GetSaveSize();
516 if (saveSize == 0)
517 {
518 // Treat zero-byte payload as failure even when load API reports success
519 LogWorldIO("loaded save has zero size");
520 return eWorldSaveLoad_Failed;
521 }
522
523 byteArray loadedSaveData(saveSize, false);
524 unsigned int loadedSize = saveSize;
525 StorageManager.GetSaveData(loadedSaveData.data, &loadedSize);
526 if (loadedSize == 0)
527 {
528 LogWorldIO("failed to copy loaded save data from storage manager");
529 return eWorldSaveLoad_Failed;
530 }
531
532 *outSaveData = new LoadSaveDataThreadParam(loadedSaveData.data, loadedSize, matchedTitle);
533 LogWorldIO("prepared save data payload for server startup");
534 return eWorldSaveLoad_Loaded;
535}

546WorldBootstrapResult BootstrapWorldForServer(
547 const ServerPropertiesConfig &config,
548 int actionPad,
549 WorldManagerTickProc tickProc)
550{
551 WorldBootstrapResult result;
552 if (!EnsureGameHddRootExists())
553 {
554 LogWorldIO("failed to prepare Windows64\\GameHDD storage root");
555 return result;
556 }
557
558 std::wstring targetWorldName = config.worldName;
559 std::string targetSaveFilename = config.worldSaveId;
560 if (targetWorldName.empty())
561 {
562 targetWorldName = L"world";
563 }
564
565 LogWorldName("configured level-name", targetWorldName);
566 if (!targetSaveFilename.empty())
567 {
568 LogSaveFilename("configured level-id", targetSaveFilename);
569 }
570
571 ApplyWorldStorageTarget(targetWorldName, targetSaveFilename);
572
573 std::string loadedSaveFilename;
574 EWorldSaveLoadResult worldLoadResult = PrepareWorldSaveData(
575 targetWorldName,
576 targetSaveFilename,
577 actionPad,
578 tickProc,
579 &result.saveData,
580 &loadedSaveFilename);
581 if (worldLoadResult == eWorldSaveLoad_Loaded)
582 {
583 result.status = eWorldBootstrap_Loaded;
584 result.resolvedSaveId = loadedSaveFilename;
585 LogStartupStep("loading configured world from save data");
586 }
587 else if (worldLoadResult == eWorldSaveLoad_NotFound)
588 {
589 // Create a new context only when no matching save exists
590 // Fix saveId here so the next startup writes to the same location
591 result.status = eWorldBootstrap_CreatedNew;
592 result.resolvedSaveId = targetSaveFilename;
593 LogStartupStep("configured world not found; creating new world");
594 LogWorldIO("creating new world save context");
595 StorageManager.ResetSaveData();
596 ApplyWorldStorageTarget(targetWorldName, targetSaveFilename);
597 }
598 else
599 {
600 result.status = eWorldBootstrap_Failed;
601 }
602
603 return result;
604}
