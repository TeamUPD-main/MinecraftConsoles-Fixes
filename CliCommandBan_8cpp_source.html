64 bool CliCommandBan::Execute(const ServerCliParsedLine &line, ServerCliEngine *engine)
65 {
66 if (line.tokens.size() < 2)
67 {
68 engine->LogWarn("Usage: ban <player> [reason ...]");
69 return false;
70 }
71 if (!ServerRuntime::Access::IsInitialized())
72 {
73 engine->LogWarn("Access manager is not initialized.");
74 return false;
75 }
76
77 const auto target = engine->FindPlayerByNameUtf8(line.tokens[1]);
78 if (target == nullptr)
79 {
80 engine->LogWarn("Unknown player: " + line.tokens[1] + " (this server build can only ban players that are currently online).");
81 return false;
82 }
83
84 std::vector<PlayerUID> xuids;
85 CollectPlayerBanXuids(target, &xuids);
86 if (xuids.empty())
87 {
88 engine->LogWarn("Cannot ban that player because no valid XUID is available.");
89 return false;
90 }
91
92 const bool hasUnbannedIdentity = std::any_of(
93 xuids.begin(),
94 xuids.end(),
95 [](PlayerUID xuid) { return !ServerRuntime::Access::IsPlayerBanned(xuid); });
96 if (!hasUnbannedIdentity)
97 {
98 engine->LogWarn("That player is already banned.");
99 return false;
100 }
101
102 ServerRuntime::Access::BanMetadata metadata = ServerRuntime::Access::BanManager::BuildDefaultMetadata("Console");
103 metadata.reason = StringUtils::JoinTokens(line.tokens, 2);
104 if (metadata.reason.empty())
105 {
106 metadata.reason = "Banned by an operator.";
107 }
108
109 const std::string playerName = StringUtils::WideToUtf8(target->getName());
110 for (const auto xuid : xuids)
111 {
112 if (ServerRuntime::Access::IsPlayerBanned(xuid))
113 {
114 continue;
115 }
116
117 if (!ServerRuntime::Access::AddPlayerBan(xuid, playerName, metadata))
118 {
119 engine->LogError("Failed to write player ban.");
120 return false;
121 }
122 }
123
124 if (target->connection != nullptr)
125 {
126 target->connection->disconnect(DisconnectPacket::eDisconnect_Banned);
127 }
128
129 engine->LogInfo("Banned player " + playerName + ".");
130 return true;
131 }
