32static wstring ReadLevelNameFromSaveFile(const wstring& filePath)
33{
34 // Check for a worldname.txt sidecar written by the rename feature first
35 size_t slashPos = filePath.rfind(L'\\');
36 if (slashPos != wstring::npos)
37 {
38 wstring sidecarPath = filePath.substr(0, slashPos + 1) + L"worldname.txt";
39 FILE *fr = nullptr;
40 if (_wfopen_s(&fr, sidecarPath.c_str(), L"r") == 0 && fr)
41 {
42 char buf[128] = {};
43 if (fgets(buf, sizeof(buf), fr))
44 {
45 int len = static_cast<int>(strlen(buf));
46 while (len > 0 && (buf[len-1] == '\n' || buf[len-1] == '\r' || buf[len-1] == ' '))
47 buf[--len] = '\0';
48 fclose(fr);
49 if (len > 0)
50 {
51 wchar_t wbuf[128] = {};
52 mbstowcs(wbuf, buf, 127);
53 return wstring(wbuf);
54 }
55 }
56 else fclose(fr);
57 }
58 }
59
60 HANDLE hFile = CreateFileW(filePath.c_str(), GENERIC_READ, FILE_SHARE_READ, nullptr, OPEN_EXISTING, FILE_FLAG_SEQUENTIAL_SCAN, nullptr);
61 if (hFile == INVALID_HANDLE_VALUE) return L"";
62
63 DWORD fileSize = GetFileSize(hFile, nullptr);
64 if (fileSize < 12 || fileSize == INVALID_FILE_SIZE) { CloseHandle(hFile); return L""; }
65
66 unsigned char *rawData = new unsigned char[fileSize];
67 DWORD bytesRead = 0;
68 if (!ReadFile(hFile, rawData, fileSize, &bytesRead, nullptr) || bytesRead != fileSize)
69 {
70 CloseHandle(hFile);
71 delete[] rawData;
72 return L"";
73 }
74 CloseHandle(hFile);
75
76 unsigned char *saveData = nullptr;
77 unsigned int saveSize = 0;
78 bool freeSaveData = false;
79
80 if (*(unsigned int*)rawData == 0)
81 {
82 // Compressed format: bytes 0-3=0, bytes 4-7=decompressed size, bytes 8+=compressed data
83 unsigned int decompSize = *(unsigned int*)(rawData + 4);
84 if (decompSize == 0 || decompSize > 128 * 1024 * 1024)
85 {
86 delete[] rawData;
87 return L"";
88 }
89 saveData = new unsigned char[decompSize];
90 Compression::getCompression()->Decompress(saveData, &decompSize, rawData + 8, fileSize - 8);
91 saveSize = decompSize;
92 freeSaveData = true;
93 }
94 else
95 {
96 saveData = rawData;
97 saveSize = fileSize;
98 }
99
100 wstring result = L"";
101 if (saveSize >= 12)
102 {
103 unsigned int headerOffset = *(unsigned int*)saveData;
104 unsigned int numEntries = *(unsigned int*)(saveData + 4);
105 const unsigned int entrySize = sizeof(FileEntrySaveData);
106
107 if (headerOffset < saveSize && numEntries > 0 && numEntries < 10000 &&
108 headerOffset + numEntries * entrySize <= saveSize)
109 {
110 FileEntrySaveData *table = (FileEntrySaveData *)(saveData + headerOffset);
111 for (unsigned int i = 0; i < numEntries; i++)
112 {
113 if (wcscmp(table[i].filename, L"level.dat") == 0)
114 {
115 unsigned int off = table[i].startOffset;
116 unsigned int len = table[i].length;
117 if (off >= 12 && off + len <= saveSize && len > 0 && len < 4 * 1024 * 1024)
118 {
119 byteArray ba;
120 ba.data = (byte*)(saveData + off);
121 ba.length = len;
122 CompoundTag *root = NbtIo::decompress(ba);
123 if (root != nullptr)
124 {
125 CompoundTag *dataTag = root->getCompound(L"Data");
126 if (dataTag != nullptr)
127 result = dataTag->getString(L"LevelName");
128 delete root;
129 }
130 }
131 break;
132 }
133 }
134 }
135 }
136
137 if (freeSaveData) delete[] saveData;
138 delete[] rawData;
139 // "world" is the engine default - it means no real name was ever set,
140 // so return empty to let the caller fall back to the save filename (timestamp).
141 if (result == L"world") result = L"";
142 return result;
143}
