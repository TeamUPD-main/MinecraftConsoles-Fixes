176void FallingTile::causeFallDamage(float distance)
177{
178 if (hurtEntities)
179 {
180 int dmg = Mth::ceil(distance - 1);
181 if (dmg > 0)
182 {
183 // 4J: Copy vector since it might be modified when we hurt the entities (invalidating our iterator)
184 vector<shared_ptr<Entity> > *entities = new vector<shared_ptr<Entity> >(*level->getEntities(shared_from_this(), bb));
185 DamageSource *source = tile == Tile::anvil_Id ? DamageSource::anvil : DamageSource::fallingBlock;
186 if ( source )
187 {
188 for (auto& it : *entities)
189 {
190 it->hurt(source, static_cast<float>(std::min<int>(Mth::floor(dmg * fallDamageAmount), fallDamageMax)));
191 }
192 delete entities;
193 }
194
195 if (tile == Tile::anvil_Id && random->nextFloat() < 0.05f + (dmg * 0.05))
196 {
197 int damage = data >> 2;
198 int dir = data & 3;
199
200 if (++damage > 2)
201 {
202 cancelDrop = true;
203 }
204 else
205 {
206 data = dir | (damage << 2);
207 }
208 }
209 }
