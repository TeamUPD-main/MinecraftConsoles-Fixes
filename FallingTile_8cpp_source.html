85void FallingTile::tick()
86{
87 if (tile == 0)
88 {
89 remove();
90 return;
91 }
92
93 xo = x;
94 yo = y;
95 zo = z;
96 time++;
97
98 yd -= 0.04f;
99 move(xd, yd, zd);
100 xd *= 0.98f;
101 yd *= 0.98f;
102 zd *= 0.98f;
103
104 if(!level->isClientSide)
105 {
106
107 int xt = Mth::floor(x);
108 int yt = Mth::floor(y);
109 int zt = Mth::floor(z);
110 if(time == 1)
111 {
112 if (level->getTile(xt, yt, zt) == tile)
113 {
114 level->removeTile(xt, yt, zt);
115 }
116 else
117 {
118 remove();
119 return;
120 }
121 }
122
123 if (onGround)
124 {
125 xd *= 0.7f;
126 zd *= 0.7f;
127 yd *= -0.5f;
128
129 if (level->getTile(xt, yt, zt) != Tile::pistonMovingPiece_Id)
130 {
131 remove();
132 if (!cancelDrop && level->mayPlace(tile, xt, yt, zt, true, 1, nullptr, nullptr) && !HeavyTile::isFree(level, xt, yt - 1, zt) && level->setTileAndData(xt, yt, zt, tile, data, Tile::UPDATE_ALL))
133 {
134 HeavyTile *hv = dynamic_cast<HeavyTile *>(Tile::tiles[tile]);
135 if (hv)
136 {
137 hv->onLand(level, xt, yt, zt, data);
138 }
139 if (tileData != nullptr && Tile::tiles[tile]->isEntityTile())
140 {
141 shared_ptr<TileEntity> tileEntity = level->getTileEntity(xt, yt, zt);
142
143 if (tileEntity != nullptr)
144 {
145 CompoundTag *swap = new CompoundTag();
146 tileEntity->save(swap);
147 vector<Tag *> *allTags = tileData->getAllTags();
148 if ( allTags )
149 {
150 for ( Tag* tag : *allTags )
151 {
152 if (tag->getName().compare(L"x") == 0 || tag->getName().compare(L"y") == 0 || tag->getName().compare(L"z") == 0) continue;
153 swap->put(tag->getName(), tag->copy());
154 }
155 delete allTags;
156 }
157 tileEntity->load(swap);
158 tileEntity->setChanged();
159 }
160 }
161 }
162 else
163 {
164 if(dropItem && !cancelDrop) spawnAtLocation(std::make_shared<ItemInstance>(tile, 1, Tile::tiles[tile]->getSpawnResourcesAuxValue(data)), 0);
165 }
166 }
167 }
168 else if ( (time > 20 * 5 && !level->isClientSide && (yt < 1 || yt > Level::maxBuildHeight)) || (time > 20 * 30))
169 {
170 if(dropItem) spawnAtLocation(std::make_shared<ItemInstance>(tile, 1, Tile::tiles[tile]->getSpawnResourcesAuxValue(data)), 0);
171 remove();
172 }
173 }
174}

176void FallingTile::causeFallDamage(float distance)
177{
178 if (hurtEntities)
179 {
180 int dmg = Mth::ceil(distance - 1);
181 if (dmg > 0)
182 {
183 // 4J: Copy vector since it might be modified when we hurt the entities (invalidating our iterator)
184 vector<shared_ptr<Entity> > *entities = new vector<shared_ptr<Entity> >(*level->getEntities(shared_from_this(), bb));
185 DamageSource *source = tile == Tile::anvil_Id ? DamageSource::anvil : DamageSource::fallingBlock;
186 if ( source )
187 {
188 for (auto& it : *entities)
189 {
190 it->hurt(source, static_cast<float>(std::min<int>(Mth::floor(dmg * fallDamageAmount), fallDamageMax)));
191 }
192 delete entities;
193 }
194
195 if (tile == Tile::anvil_Id && random->nextFloat() < 0.05f + (dmg * 0.05))
196 {
197 int damage = data >> 2;
198 int dir = data & 3;
199
200 if (++damage > 2)
201 {
202 cancelDrop = true;
203 }
204 else
205 {
206 data = dir | (damage << 2);
207 }
208 }
209 }
210 }
