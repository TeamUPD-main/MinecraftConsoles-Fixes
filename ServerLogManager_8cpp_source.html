101 // Split one debug payload into individual lines so each line becomes a prompt-safe server log entry.
102 static void ForwardClientDebugMessage(const char *source, const char *message)
103 {
104 if (message == NULL || message[0] == 0)
105 {
106 return;
107 }
108
109 const char *cursor = message;
110 while (*cursor != 0)
111 {
112 const char *lineStart = cursor;
113 while (*cursor != 0 && *cursor != '\r' && *cursor != '\n')
114 {
115 ++cursor;
116 }
117
118 // Split multi-line client debug output into prompt-safe server log entries.
119 if (cursor > lineStart)
120 {
121 EmitClientDebugLogLine(source, std::string(lineStart, (size_t)(cursor - lineStart)));
122 }
123
124 while (*cursor == '\r' || *cursor == '\n')
125 {
126 ++cursor;
127 }
128 }
129 }
