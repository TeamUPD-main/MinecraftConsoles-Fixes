167void UIScene_Keyboard::tick()
168{
169 UIScene::tick();
170
171 // Sync our buffer from Flash so we pick up changes made via controller/on-screen buttons.
172 // Without this, switching between controller and keyboard would use stale text.
173 // In PC mode we own the buffer — skip sync to preserve cursor position.
174 if (!m_bPCMode)
175 {
176 const wchar_t* flashText = m_KeyboardTextInput.getLabel();
177 if (flashText)
178 m_win64TextBuffer = flashText;
179 }
180
181 // Accumulate physical keyboard chars into our own buffer, then push to Flash via setLabel.
182 // This bypasses Iggy's focus system (char events only route to the focused element).
183 // The char buffer is cleared on open so Enter/clicks from the triggering action don't leak in.
184 wchar_t ch;
185 bool changed = false;
186 while (g_KBMInput.ConsumeChar(ch))
187 {
188 if (ch == 0x08) // backspace
189 {
190 if (m_bPCMode)
191 {
192 if (m_iCursorPos > 0)
193 {
194 m_win64TextBuffer.erase(m_iCursorPos - 1, 1);
195 m_iCursorPos--;
196 changed = true;
197 }
198 }
199 else if (!m_win64TextBuffer.empty())
200 {
201 m_win64TextBuffer.pop_back();
202 changed = true;
203 }
204 }
205 else if (ch == 0x0D) // enter - confirm
206 {
207 if (!m_bKeyboardDonePressed)
208 {
209 addTimer(KEYBOARD_DONE_TIMER_ID, KEYBOARD_DONE_TIMER_TIME);
210 m_bKeyboardDonePressed = true;
211 }
212 }
213 else if (static_cast<int>(m_win64TextBuffer.length()) < m_win64MaxChars)
214 {
215 if (m_bPCMode)
216 {
217 m_win64TextBuffer.insert(m_iCursorPos, 1, ch);
218 m_iCursorPos++;
219 }
220 else
221 {
222 m_win64TextBuffer += ch;
223 }
224 changed = true;
225 }
226 }
227
228 // Paste from clipboard
229 if (g_KBMInput.IsKeyPressed('V') && g_KBMInput.IsKeyDown(VK_CONTROL))
230 {
231 wstring pasted = Screen::getClipboard();
232 wstring sanitized;
233 sanitized.reserve(pasted.length());
234
235 for (wchar_t pc : pasted)
236 {
237 if (pc >= 0x20) // Keep printable characters
238 {
239 if (static_cast<int>(m_win64TextBuffer.length() + sanitized.length()) >= m_win64MaxChars)
240 break;
241 sanitized += pc;
242 }
243 }
244
245 if (!sanitized.empty())
246 {
247 if (m_bPCMode)
248 {
249 m_win64TextBuffer.insert(m_iCursorPos, sanitized);
250 m_iCursorPos += (int)sanitized.length();
251 }
252 else
253 {
254 m_win64TextBuffer += sanitized;
255 }
256 changed = true;
257 }
258 }
259
260 if (m_bPCMode)
261 {
262 // Arrow keys, Home, End, Delete for cursor movement
263 if (g_KBMInput.IsKeyPressed(VK_LEFT) && m_iCursorPos > 0)
264 m_iCursorPos--;
265 if (g_KBMInput.IsKeyPressed(VK_RIGHT) && m_iCursorPos < (int)m_win64TextBuffer.length())
266 m_iCursorPos++;
267 if (g_KBMInput.IsKeyPressed(VK_HOME))
268 m_iCursorPos = 0;
269 if (g_KBMInput.IsKeyPressed(VK_END))
270 m_iCursorPos = (int)m_win64TextBuffer.length();
271 if (g_KBMInput.IsKeyPressed(VK_DELETE) && m_iCursorPos < (int)m_win64TextBuffer.length())
272 {
273 m_win64TextBuffer.erase(m_iCursorPos, 1);
274 changed = true;
275 }
276 }
277
278 if (changed)
279 m_KeyboardTextInput.setLabel(m_win64TextBuffer.c_str(), true /*instant*/);
280
281 if (m_bPCMode)
282 {
283 m_KeyboardTextInput.setCaretVisible(true);
284 m_KeyboardTextInput.setCaretIndex(m_iCursorPos);
285 }
286}

289void UIScene_Keyboard::handleInput(int iPad, int key, bool repeat, bool pressed, bool released, bool &handled)
290{
291 IggyDataValue result;
292 IggyResult out;
293
294 if(repeat || pressed)
295 {
296 switch(key)
297 {
298 case ACTION_MENU_CANCEL:
299#ifdef _WINDOWS64
300 {
301 // Cache before navigateBack() destroys this scene
302 int(*cb)(LPVOID, const bool) = m_win64Callback;
303 LPVOID cbParam = m_win64CallbackParam;

313 case ACTION_MENU_X: // X
314 case ACTION_MENU_PAGEUP: // LT
315 case ACTION_MENU_Y: // Y

317 case ACTION_MENU_LEFT_SCROLL: // LB
318 case ACTION_MENU_RIGHT_SCROLL: // RB

320
321 if (m_bPCMode)
322 {
323 handled = true;

326#endif

329 else if (key == ACTION_MENU_PAGEUP)

331 else if (key == ACTION_MENU_Y)
332 out = IggyPlayerCallMethodRS ( getMovie() , &result, IggyPlayerRootPath( getMovie() ), m_funcSpaceButtonPressed, 0 , nullptr);
333 else if (key == ACTION_MENU_STICK_PRESS)

337 else if (key == ACTION_MENU_RIGHT_SCROLL)
338 out = IggyPlayerCallMethodRS ( getMovie() , &result, IggyPlayerRootPath( getMovie() ), m_funcCursorRightButtonPressed, 0 , nullptr);
339 handled = true;

341 case ACTION_MENU_PAUSEMENU: // Start
342 if(!m_bKeyboardDonePressed)
343 {
344 out = IggyPlayerCallMethodRS ( getMovie() , &result, IggyPlayerRootPath( getMovie() ), m_funcDoneButtonPressed, 0 , nullptr );
345
346 // kick off done timer
347 addTimer(KEYBOARD_DONE_TIMER_ID,KEYBOARD_DONE_TIMER_TIME);
348 m_bKeyboardDonePressed = true;
349 }
350 handled = true;
351 break;
352 }
353 }
354
355 switch(key)
356 {
357 case ACTION_MENU_OK:
358#ifdef _WINDOWS64
359 if (m_bPCMode)
360 {
361 // pressing enter sometimes causes a "y" to be entered.
362 handled = true;
363 break;
364 }
365#endif
366 // fall through for controller mode
367 case ACTION_MENU_LEFT:
368 case ACTION_MENU_RIGHT:
369 case ACTION_MENU_UP:
370 case ACTION_MENU_DOWN:
371#ifdef _WINDOWS64
372 if (!m_bPCMode)
373#endif
374 sendInputToMovie(key, repeat, pressed, released);
375 handled = true;
376 break;
377 }

409#ifdef _WINDOWS64
