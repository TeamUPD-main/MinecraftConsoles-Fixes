20 static bool CompareWhitelistEntries(const ServerRuntime::Access::WhitelistedPlayerEntry &left, const ServerRuntime::Access::WhitelistedPlayerEntry &right)
21 {
22 const auto leftName = StringUtils::ToLowerAscii(left.name);
23 const auto rightName = StringUtils::ToLowerAscii(right.name);

115 bool CliCommandWhitelist::Execute(const ServerCliParsedLine &line, ServerCliEngine *engine)
116 {
117 if (line.tokens.size() < 2)
118 {
119 engine->LogWarn(std::string("Usage: ") + kWhitelistUsage);
120 return false;
121 }
122 if (!ServerRuntime::Access::IsInitialized())
123 {
124 engine->LogWarn("Access manager is not initialized.");
125 return false;
126 }
127
128 const auto subcommand = StringUtils::ToLowerAscii(line.tokens[1]);
129 if (subcommand == "on" || subcommand == "off")
130 {
131 if (line.tokens.size() != 2)
132 {
133 engine->LogWarn("Usage: whitelist <on|off>");
134 return false;
135 }
136
137 const bool enabled = (subcommand == "on");
138 if (!PersistWhitelistToggle(enabled))
139 {
140 engine->LogError("Failed to persist whitelist mode to server.properties.");
141 return false;
142 }
143
144 ServerRuntime::Access::SetWhitelistEnabled(enabled);
145 engine->LogInfo(std::string("Whitelist ") + (enabled ? "enabled." : "disabled."));
146 return true;
147 }
148
149 if (subcommand == "list")
150 {
151 if (line.tokens.size() != 2)
152 {
153 engine->LogWarn("Usage: whitelist list");
154 return false;
155 }
156
157 return LogWhitelistEntries(engine);
158 }
159
160 if (subcommand == "reload")
161 {
162 if (line.tokens.size() != 2)
163 {
164 engine->LogWarn("Usage: whitelist reload");
165 return false;
166 }
167 if (!ServerRuntime::Access::ReloadWhitelist())
168 {
169 engine->LogError("Failed to reload whitelist.");
170 return false;
171 }
172
173 const auto config = LoadServerPropertiesConfig();
174 ServerRuntime::Access::SetWhitelistEnabled(config.whiteListEnabled);
175 engine->LogInfo("Reloaded whitelist from disk.");
176 LogWhitelistMode(engine);
177 return true;
178 }
179
180 if (subcommand == "add")
181 {
182 if (line.tokens.size() < 3)
183 {
184 engine->LogWarn("Usage: whitelist add <xuid> [name ...]");
185 return false;
186 }
187
188 PlayerUID xuid = INVALID_XUID;
189 if (!TryParseWhitelistXuid(line.tokens[2], engine, &xuid))
190 {
191 return false;
192 }
193
194 if (ServerRuntime::Access::IsPlayerWhitelisted(xuid))
195 {
196 engine->LogWarn("That XUID is already whitelisted.");
197 return false;
198 }
199
200 const auto metadata = ServerRuntime::Access::WhitelistManager::BuildDefaultMetadata("Console");
201 const auto name = StringUtils::JoinTokens(line.tokens, 3);
202 if (!ServerRuntime::Access::AddWhitelistedPlayer(xuid, name, metadata))
203 {
204 engine->LogError("Failed to write whitelist entry.");
205 return false;
206 }
207
208 std::string message = "Whitelisted XUID " + ServerRuntime::Access::FormatXuid(xuid) + ".";
209 if (!name.empty())
210 {
211 message += " Name: " + name;
212 }
213 engine->LogInfo(message);
214 return true;
215 }
216
217 if (subcommand == "remove")
218 {
219 if (line.tokens.size() != 3)
220 {
221 engine->LogWarn("Usage: whitelist remove <xuid>");
222 return false;
223 }
224
225 PlayerUID xuid = INVALID_XUID;
226 if (!TryParseWhitelistXuid(line.tokens[2], engine, &xuid))
227 {
228 return false;
229 }
230
231 if (!ServerRuntime::Access::IsPlayerWhitelisted(xuid))
232 {
233 engine->LogWarn("That XUID is not whitelisted.");
234 return false;
235 }
236
237 if (!ServerRuntime::Access::RemoveWhitelistedPlayer(xuid))
238 {
239 engine->LogError("Failed to remove whitelist entry.");
240 return false;
241 }
242
243 engine->LogInfo("Removed XUID " + ServerRuntime::Access::FormatXuid(xuid) + " from the whitelist.");
244 return true;
245 }
246
247 engine->LogWarn(std::string("Usage: ") + kWhitelistUsage);
248 return false;
249 }

251 void CliCommandWhitelist::Complete(const ServerCliCompletionContext &context, const ServerCliEngine *engine, std::vector<std::string> *out) const
252 {
253 (void)engine;
254 if (out == nullptr)
255 {
256 return;
257 }
258
259 if (context.currentTokenIndex == 1)
260 {

263 SuggestLiteral("list", context, out);
264 SuggestLiteral("add", context, out);
265 SuggestLiteral("remove", context, out);
266 SuggestLiteral("reload", context, out);
267 return;
268 }
269
270 if (context.currentTokenIndex == 2 && context.parsed.tokens.size() >= 2 && StringUtils::ToLowerAscii(context.parsed.tokens[1]) == "remove")
271 {
272 std::vector<ServerRuntime::Access::WhitelistedPlayerEntry> entries;
273 if (!ServerRuntime::Access::SnapshotWhitelistedPlayers(&entries))
274 {
275 return;
276 }
277
278 for (const auto &entry : entries)
279 {
280 SuggestLiteral(entry.xuid, context, out);
281 }
282 }
