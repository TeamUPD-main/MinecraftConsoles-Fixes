63 bool CliCommandPardon::Execute(const ServerCliParsedLine &line, ServerCliEngine *engine)
64 {
65 if (line.tokens.size() != 2)
66 {
67 engine->LogWarn("Usage: pardon <player>");
68 return false;
69 }
70 if (!ServerRuntime::Access::IsInitialized())
71 {
72 engine->LogWarn("Access manager is not initialized.");
73 return false;
74 }
75
76 std::vector<PlayerUID> xuidsToRemove;
77 std::vector<std::string> matchedNames;
78 std::shared_ptr<ServerPlayer> onlineTarget = engine->FindPlayerByNameUtf8(line.tokens[1]);
79 if (onlineTarget != nullptr)
80 {
81 if (ServerRuntime::Access::IsPlayerBanned(onlineTarget->getXuid()))
82 {
83 AppendUniqueXuid(onlineTarget->getXuid(), &xuidsToRemove);
84 }
85 if (ServerRuntime::Access::IsPlayerBanned(onlineTarget->getOnlineXuid()))
86 {
87 AppendUniqueXuid(onlineTarget->getOnlineXuid(), &xuidsToRemove);
88 }
89 }
90
91 std::vector<ServerRuntime::Access::BannedPlayerEntry> entries;
92 if (!ServerRuntime::Access::SnapshotBannedPlayers(&entries))
93 {
94 engine->LogError("Failed to read banned players.");
95 return false;
96 }
97
98 const std::string loweredTarget = StringUtils::ToLowerAscii(line.tokens[1]);
99 for (const auto &entry : entries)
100 {
101 if (StringUtils::ToLowerAscii(entry.name) == loweredTarget)
102 {
103 PlayerUID parsedXuid = INVALID_XUID;
104 if (ServerRuntime::Access::TryParseXuid(entry.xuid, &parsedXuid))
105 {
106 AppendUniqueXuid(parsedXuid, &xuidsToRemove);
107 }
108 AppendUniqueText(entry.name, &matchedNames);
109 }
110 }
111
112 if (xuidsToRemove.empty())
113 {
114 engine->LogWarn("That player is not banned.");
115 return false;
116 }
117
118 for (const auto xuid : xuidsToRemove)
119 {
120 if (!ServerRuntime::Access::RemovePlayerBan(xuid))
121 {
122 engine->LogError("Failed to remove player ban.");
123 return false;
124 }
125 }
126
127 std::string playerName = line.tokens[1];
128 if (!matchedNames.empty())
129 {
130 playerName = matchedNames[0];
131 }
132 else if (onlineTarget != nullptr)
133 {
134 playerName = StringUtils::WideToUtf8(onlineTarget->getName());
135 }
136
137 engine->LogInfo("Unbanned player " + playerName + ".");
138 return true;
139 }
