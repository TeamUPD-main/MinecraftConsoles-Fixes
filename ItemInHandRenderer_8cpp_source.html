904void ItemInHandRenderer::tick()
905{
906 oHeight = height;
907
908
909 shared_ptr<Player> player = minecraft->player;
910 shared_ptr<ItemInstance> nextTile = player->inventory->getSelected();
911
912 bool matches = lastSlot == player->inventory->selected && nextTile == selectedItem;
913 if (selectedItem == nullptr && nextTile == nullptr)
914 {
915 matches = true;
916 }
917 if (nextTile != nullptr && selectedItem != nullptr && nextTile != selectedItem && nextTile->id == selectedItem->id && nextTile->getAuxValue() == selectedItem->getAuxValue())
918 {
919 selectedItem = nextTile;
920 matches = true;
921 }
922
923 float max = 0.4f;
924 float tHeight = matches ? 1.0f : 0;
925 float dd = tHeight - height;
926 if (dd < -max) dd = -max;
927 if (dd > max) dd = max;
928
929 height += dd;
930 if (height < 0.1f)
931 {
932 selectedItem = nextTile;
933 lastSlot = player->inventory->selected;
934 }
935
