139 // 4J Stu - If one player on a system is updated, then make sure they all are as they all have their
140 // range extended to include entities visible by any other player on the system
141 // Fix for #11194 - Gameplay: Host player and their split-screen avatars can become invisible and invulnerable to client.
142 MinecraftServer *server = MinecraftServer::getInstance();
143 for( unsigned int i = 0; i < server->getPlayers()->players.size(); i++ )
144 {
145 shared_ptr<ServerPlayer> ep = server->getPlayers()->players[i];
146 if( ep->dimension != level->dimension->id ) continue;
147
148 if( ep->connection == nullptr ) continue;
149 INetworkPlayer *thisPlayer = ep->connection->getNetworkPlayer();
150 if( thisPlayer == nullptr ) continue;
151
152 bool addPlayer = false;
153 for (unsigned int j = 0; j < movedPlayers.size(); j++)
154 {
155 shared_ptr<ServerPlayer> sp = movedPlayers[j];
156
157 if( sp == ep ) break;
158
159 if(sp->connection == nullptr) continue;
160 INetworkPlayer *otherPlayer = sp->connection->getNetworkPlayer();
161 if( otherPlayer != nullptr && thisPlayer->IsSameSystem(otherPlayer) )
162 {
163 addPlayer = true;
164 break;
165 }
166 }
167 if( addPlayer ) movedPlayers.push_back( ep );
168 }
169
170 for (unsigned int i = 0; i < movedPlayers.size(); i++)
171 {
172 shared_ptr<ServerPlayer> player = movedPlayers[i];
173 if(player->connection == nullptr) continue;
174 for( auto& te : entities )
175 {
176 if ( te && te->e != player)
177 {
178 te->updatePlayer(this, player);
179 }
180 }
181 }

184 for (auto& it : level->players )
