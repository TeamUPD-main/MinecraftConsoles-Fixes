21 [[maybe_unused]]
22 int FovToSliderValue(float fov)
23 {
24 const int clampedFov = ClampFov(static_cast<int>(fov + 0.5f));
25 return ((clampedFov - FOV_MIN) * FOV_SLIDER_MAX) / (FOV_MAX - FOV_MIN);

28 int sliderValueToFov(int sliderValue)
29 {
30 if (sliderValue < 0) sliderValue = 0;
31 if (sliderValue > FOV_SLIDER_MAX) sliderValue = FOV_SLIDER_MAX;
32 return FOV_MIN + ((sliderValue * (FOV_MAX - FOV_MIN)) / FOV_SLIDER_MAX);
33 }
