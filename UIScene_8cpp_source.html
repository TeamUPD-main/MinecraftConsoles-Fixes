507bool UIScene::handleMouseClick(F32 x, F32 y)
508{
509 S32 panelOffsetX = 0, panelOffsetY = 0;
510 UIControl *pMainPanel = GetMainPanel();
511 if (pMainPanel)
512 {
513 pMainPanel->UpdateControl();
514 panelOffsetX = pMainPanel->getXPos();
515 panelOffsetY = pMainPanel->getYPos();
516 }
517
518 // Click-outside-to-deselect: confirm any active direct edit if
519 // the click landed outside the editing text input.
520 {
521 vector<UIControl_TextInput*> deInputs;
522 getDirectEditInputs(deInputs);
523 for (size_t i = 0; i < deInputs.size(); i++)
524 {
525 if (!deInputs[i]->isDirectEditing())
526 continue;
527 deInputs[i]->UpdateControl();
528 S32 cx = deInputs[i]->getXPos() + panelOffsetX;
529 S32 cy = deInputs[i]->getYPos() + panelOffsetY;
530 S32 cw = deInputs[i]->getWidth();
531 S32 ch = deInputs[i]->getHeight();
532 if (!(cw > 0 && ch > 0 && x >= cx && x <= cx + cw && y >= cy && y <= cy + ch))
533 {
534 deInputs[i]->confirmDirectEdit();
535 onDirectEditFinished(deInputs[i], UIControl_TextInput::eDirectEdit_Confirmed);
536 }
537 }
538 }
539
540 vector<UIControl *> *controls = GetControls();
541 if (!controls) return false;
542
543 // Hit-test controls and pick the smallest-area match to handle
544 // overlapping Flash bounds correctly without sacrificing precision.
545 int bestId = -1;
546 S32 bestArea = INT_MAX;
547 UIControl *bestCtrl = NULL;
548
549 for (size_t i = 0; i < controls->size(); ++i)
550 {
551 UIControl *ctrl = (*controls)[i];
552 if (!ctrl || ctrl->getHidden() || !ctrl->getVisible() || ctrl->getId() < 0)
553 continue;
554
555 UIControl::eUIControlType type = ctrl->getControlType();
556 if (type != UIControl::eButton && type != UIControl::eTextInput &&
557 type != UIControl::eCheckBox)
558 continue;
559
560 if (pMainPanel && ctrl->getParentPanel() != pMainPanel)
561 continue;
562
563 ctrl->UpdateControl();
564 S32 cx = ctrl->getXPos() + panelOffsetX;
565 S32 cy = ctrl->getYPos() + panelOffsetY;
566 S32 cw = ctrl->getWidth();
567 S32 ch = ctrl->getHeight();
568 if (cw <= 0 || ch <= 0)
569 continue;
570
571 if (x >= cx && x <= cx + cw && y >= cy && y <= cy + ch)
572 {
573 S32 area = cw * ch;
574 if (area < bestArea)
575 {
576 bestArea = area;
577 bestId = ctrl->getId();
578 bestCtrl = ctrl;
579 }
580 }
581 }
582
583 if (bestId >= 0 && bestCtrl)
584 {
585 if (bestCtrl->getControlType() == UIControl::eCheckBox)
586 {
587 UIControl_CheckBox *cb = static_cast<UIControl_CheckBox*>(bestCtrl);
588 if (cb->IsEnabled())
589 {
590 bool newState = !cb->IsChecked();
591 cb->setChecked(newState);
592 handleCheckboxToggled((F64)bestId, newState);
593 }
594 }
595 else
596 {
597 handlePress((F64)bestId, 0);
598 }
599 return true;
600 }
601 return false;
602}

866void UIScene::_customDrawSlotControl(CustomDrawData *region, int iPad, shared_ptr<ItemInstance> item, float fAlpha, bool isFoil, bool bDecorations, bool usingCommandBuffer)
867{
868 Minecraft *pMinecraft=Minecraft::GetInstance();
869
870 float bwidth,bheight;
871 bwidth = region->x1 - region->x0;
872 bheight = region->y1 - region->y0;
873
874 float x = region->x0;
875 float y = region->y0;
876
877 // Base scale on height of this control, compared to height of what the item renderer normally renders (16 pixels high). Potentially
878 // we might want separate x & y scales here
879
880 float scaleX = bwidth / 16.0f;
881 float scaleY = bheight / 16.0f;
882
883 glEnable(GL_RESCALE_NORMAL);
884 glPushMatrix();
885 glRotatef(120, 1, 0, 0);
886 Lighting::turnOn();
887 glPopMatrix();
888
889 float pop = item->popTime;
890 if (pop > 0)
891 {
892 glPushMatrix();
893 float squeeze = 1 + pop / static_cast<float>(Inventory::POP_TIME_DURATION);
894 float sx = x;
895 float sy = y;
896 float sxoffs = 8 * scaleX;
897 float syoffs = 12 * scaleY;
898 glTranslatef((float)(sx + sxoffs), (float)(sy + syoffs), 0);
899 glScalef(1 / squeeze, (squeeze + 1) / 2, 1);
900 glTranslatef((float)-(sx + sxoffs), (float)-(sy + syoffs), 0);
901 }
902
903 PIXBeginNamedEvent(0,"Render and decorate");
904 if(m_pItemRenderer == nullptr) m_pItemRenderer = new ItemRenderer();
905 RenderManager.StateSetBlendEnable(true);
906 RenderManager.StateSetBlendFunc(GL_SRC_ALPHA, GL_ONE_MINUS_SRC_ALPHA);
907 RenderManager.StateSetBlendFactor(0xffffffff);
908 m_pItemRenderer->renderAndDecorateItem(pMinecraft->font, pMinecraft->textures, item, x, y,scaleX,scaleY,fAlpha,isFoil,false, !usingCommandBuffer);
909 PIXEndNamedEvent();
910
911 if (pop > 0)
912 {
913 glPopMatrix();
914 }
915
916 if(bDecorations)
917 {
918 if((scaleX!=1.0f) ||(scaleY!=1.0f))
919 {
920 glPushMatrix();
921 glScalef(scaleX, scaleY, 1.0f);
922 int iX= static_cast<int>(0.5f + ((float)x) / scaleX);
923 int iY= static_cast<int>(0.5f + ((float)y) / scaleY);
924
925 m_pItemRenderer->renderGuiItemDecorations(pMinecraft->font, pMinecraft->textures, item, iX, iY, fAlpha);
926 glPopMatrix();
927 }
928 else
929 {
930 m_pItemRenderer->renderGuiItemDecorations(pMinecraft->font, pMinecraft->textures, item, static_cast<int>(x), static_cast<int>(y), fAlpha);
931 }
932 }
933
934 Lighting::turnOff();
935 glDisable(GL_RESCALE_NORMAL);
936}

1055{
1056 if(!swf) return;

1058 int iggyKeyCode = convertGameActionToIggyKeycode(key);

1066#ifdef _WINDOWS64

1068 // available one so arrow keys work even when the mouse is over empty space.

1086 IggyMakeEventKey( &keyEvent, pressed?IGGY_KEYEVENT_Down:IGGY_KEYEVENT_Up, static_cast<IggyKeycode>(iggyKeyCode), IGGY_KEYLOC_Standard );
