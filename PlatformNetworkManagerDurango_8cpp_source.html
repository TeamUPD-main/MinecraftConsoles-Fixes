13 "DNM_STATE_INITIALISING",

37 m_bLeavingGame = false;

39 m_bHostChanged = false;
40 g_NetworkManager.StateChange_AnyToJoining();

43 {
44 g_NetworkManager.StateChange_JoiningToIdle(JOIN_FAILED_NONSPECIFIC);
45 }
46 else if( newState == DQRNetworkManager::DNM_STATE_IDLE && oldState == DQRNetworkManager::DNM_STATE_HOSTING )
47 {
48 m_bLeavingGame = true;
49 }
50 else if( newState == DQRNetworkManager::DNM_STATE_STARTING )
51 {
52 m_lastPlayerEventTimeStart = app.getAppTime();

54 g_NetworkManager.StateChange_AnyToStarting();
55 }

57 // 4J Stu - If the client joins just as the host is exiting, then they can skip to leaving without passing through ending
58 else if( newState == DQRNetworkManager::DNM_STATE_ENDING )
59 {
60 g_NetworkManager.StateChange_AnyToEnding( oldState == DQRNetworkManager::DNM_STATE_PLAYING );
61
62 if( m_pDQRNet->IsHost() )
63 {
64 m_bLeavingGame = true;
65 }
66 }
67

671// Add a player to the per system flag storage - if we've already got a player from that system, copy its flags over
672void CPlatformNetworkManagerDurango::SystemFlagAddPlayer(INetworkPlayer *pNetworkPlayer)
673{
674 PlayerFlags *newPlayerFlags = new PlayerFlags( pNetworkPlayer, m_flagIndexSize);
675 // If any of our existing players are on the same system, then copy over flags from that one
676 for( unsigned int i = 0; i < m_playerFlags.size(); i++ )
677 {
678 if( pNetworkPlayer->IsSameSystem(m_playerFlags[i]->m_pNetworkPlayer) )
679 {
680 memcpy( newPlayerFlags->flags, m_playerFlags[i]->flags, m_playerFlags[i]->count / 8 );
681 break;
682 }
683 }
684 m_playerFlags.push_back(newPlayerFlags);
685}

711// Set a per system flag - this is done by setting the flag on every player that shares that system
712void CPlatformNetworkManagerDurango::SystemFlagSet(INetworkPlayer *pNetworkPlayer, int index)
713{
714 if( ( index < 0 ) || ( index >= m_flagIndexSize ) ) return;

757{
758 if( m_bSearchPending )
759 {
760 if( !m_pDQRNet->FriendPartyManagerIsBusy() )
761 {
762 m_searchResultsCount = m_pDQRNet->FriendPartyManagerGetCount();
763 delete [] m_pSearchResults;

767 {
768 m_pDQRNet->FriendPartyManagerGetSessionInfo(i, &m_pSearchResults[i] );
769 }
770 m_bSearchPending = false;
771
772 if( m_SessionsUpdatedCallback != nullptr ) m_SessionsUpdatedCallback(m_pSearchParam);
773 }
774 }
775 else
776 {
777 if( !m_pDQRNet->FriendPartyManagerIsBusy() )
778 {
779 // Don't start searches unless we have registered a callback
780 if( m_SessionsUpdatedCallback != nullptr && (m_lastSearchStartTime + MINECRAFT_DURANGO_PARTY_SEARCH_DELAY_MILLISECONDS) < GetTickCount() )
781 {
782 if( m_pDQRNet->FriendPartyManagerSearch() );
783 {
784 m_bSearchPending = true;

786 }
787 }
788 }
