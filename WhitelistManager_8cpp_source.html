58 bool WhitelistManager::LoadPlayers(std::vector<WhitelistedPlayerEntry> *outEntries) const
59 {
60 if (outEntries == nullptr)
61 {
62 return false;
63 }
64 outEntries->clear();
65
66 std::string text;
67 const std::string path = GetWhitelistFilePath();
68 if (!FileUtils::ReadTextFile(path, &text))
69 {
70 LogErrorf("access", "failed to read %s", path.c_str());
71 return false;
72 }
73
74 if (text.empty())
75 {
76 text = "[]";
77 }
78
79 OrderedJson root;
80 try
81 {
82 root = OrderedJson::parse(StringUtils::StripUtf8Bom(text));
83 }
84 catch (const nlohmann::json::exception &e)
85 {
86 LogErrorf("access", "failed to parse %s: %s", path.c_str(), e.what());
87 return false;
88 }
89
90 if (!root.is_array())
91 {
92 LogErrorf("access", "failed to parse %s: root json value is not an array", path.c_str());
93 return false;
94 }
95
96 for (const auto &object : root)
97 {
98 if (!object.is_object())
99 {
100 LogWarnf("access", "skipping whitelist entry that is not an object in %s", path.c_str());
101 continue;
102 }
103
104 std::string rawXuid;
105 if (!AccessStorageUtils::TryGetStringField(object, "xuid", &rawXuid))
106 {
107 LogWarnf("access", "skipping whitelist entry without xuid in %s", path.c_str());
108 continue;
109 }
110
111 WhitelistedPlayerEntry entry;
112 entry.xuid = AccessStorageUtils::NormalizeXuid(rawXuid);
113 if (entry.xuid.empty())
114 {
115 LogWarnf("access", "skipping whitelist entry with empty xuid in %s", path.c_str());
116 continue;
117 }
118
119 AccessStorageUtils::TryGetStringField(object, "name", &entry.name);
120 AccessStorageUtils::TryGetStringField(object, "created", &entry.metadata.created);
121 AccessStorageUtils::TryGetStringField(object, "source", &entry.metadata.source);
122 NormalizeMetadata(&entry.metadata);
123
124 outEntries->push_back(entry);
125 }
126
127 return true;
128 }

130 bool WhitelistManager::SavePlayers(const std::vector<WhitelistedPlayerEntry> &entries) const
131 {
132 OrderedJson root = OrderedJson::array();
133 for (const auto &entry : entries)
134 {

136 object["xuid"] = AccessStorageUtils::NormalizeXuid(entry.xuid);
137 object["name"] = entry.name;
138 object["created"] = entry.metadata.created;
139 object["source"] = entry.metadata.source;
140 root.push_back(object);

142
143 const std::string path = GetWhitelistFilePath();
144 const std::string json = root.empty() ? std::string("[]\n") : (root.dump(2) + "\n");
145 if (!FileUtils::WriteTextFileAtomic(path, json))
146 {
147 LogErrorf("access", "failed to write %s", path.c_str());
148 return false;
149 }
