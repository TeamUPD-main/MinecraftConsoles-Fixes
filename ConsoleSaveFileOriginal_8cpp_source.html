27ConsoleSaveFileOriginal::ConsoleSaveFileOriginal(const wstring &fileName, LPVOID pvSaveData /*= nullptr*/, DWORD dFileSize /*= 0*/, bool forceCleanSave /*= false*/, ESavePlatform plat /*= SAVE_FILE_PLATFORM_LOCAL*/)
28{
29 InitializeCriticalSectionAndSpinCount(&m_lock,5120);
30
31 // One time initialise of static stuff required for our storage
32 if( pvHeap == nullptr )
33 {
34 // Reserve a chunk of 64MB of virtual address space for our saves, using 64KB pages.
35 // We'll only be committing these as required to grow the storage we need, which will
36 // the storage to grow without having to use realloc.
37
38 // AP - The Vita doesn't have virtual memory so a pretend system has been implemented in PSVitaStubs.cpp.
39 // All access to the memory must be done via the access function as the pointer returned from VirtualAlloc
40 // can't be used directly.
41 pvHeap = VirtualAlloc(nullptr, MAX_PAGE_COUNT * CSF_PAGE_SIZE, RESERVE_ALLOCATION, PAGE_READWRITE );
42 }
43
44 pvSaveMem = pvHeap;
45 m_fileName = fileName;
46
47 DWORD fileSize = dFileSize;
48
49 // Load a save from the game rules
50 bool bLevelGenBaseSave = false;
51 LevelGenerationOptions *levelGen = app.getLevelGenerationOptions();
52 if( pvSaveData == nullptr && levelGen != nullptr && levelGen->requiresBaseSave())
53 {
54 pvSaveData = levelGen->getBaseSaveData(fileSize);
55 if(pvSaveData && fileSize != 0) bLevelGenBaseSave = true;
56 }
57
58 if( pvSaveData == nullptr || fileSize == 0)
59 fileSize = StorageManager.GetSaveSize();
60
61 if( forceCleanSave )
62 fileSize = 0;
63
64 DWORD heapSize = max( fileSize, (DWORD)(1024 * 1024 * 2)); // 4J Stu - Our files are going to be bigger than 2MB so allocate high to start with
65
66 // Initially committ enough room to store headSize bytes (using CSF_PAGE_SIZE pages, so rounding up here). We should only ever have one save file at a time,
67 // and the pages should be decommitted in the dtor, so pages committed should always be zero at this point.
68 if( pagesCommitted != 0 )
69 {
70#ifndef _CONTENT_PACKAGE
71 __debugbreak();
72#endif
73 }
74
75 unsigned int pagesRequired = ( heapSize + (CSF_PAGE_SIZE - 1 ) ) / CSF_PAGE_SIZE;
76
77 void *pvRet = VirtualAlloc(pvHeap, pagesRequired * CSF_PAGE_SIZE, COMMIT_ALLOCATION, PAGE_READWRITE);
78 if( pvRet == nullptr )
79 {
80#ifndef _CONTENT_PACKAGE
81 // Out of physical memory
82 __debugbreak();
83#endif
84 }
85 pagesCommitted = pagesRequired;
86
87 if( fileSize > 0)
88 {
89 bool AllocData = false;
90 if(pvSaveData != nullptr)
91 {
92#ifdef __PSVITA__
93 // AP - use this to access the virtual memory
94 VirtualCopyTo(pvSaveMem, pvSaveData, fileSize);
95#else
96 memcpy(pvSaveMem, pvSaveData, fileSize);
97 if(bLevelGenBaseSave)
98 {
99 levelGen->deleteBaseSaveData();
100 }
101#endif
102 }
103 else
104 {
105 unsigned int storageLength;
106#ifdef __PSVITA__
107 // create a buffer to hold the compressed data
108 pvSaveData = malloc(fileSize);
109 AllocData = true;
110 StorageManager.GetSaveData( pvSaveData, &storageLength );
111#else
112 StorageManager.GetSaveData( pvSaveMem, &storageLength );
113#endif
114#ifdef __PS3__
115 StorageManager.FreeSaveData();
116#endif
117 app.DebugPrintf("Filesize - %d, Adjusted size - %d\n",fileSize,storageLength);
118 fileSize = storageLength;
119 }
120
121#ifdef __PSVITA__
122 if(plat == SAVE_FILE_PLATFORM_PSVITA)
123 {
124 // AP - decompress via the access function. This uses a special RLE format
125 VirtualDecompress((unsigned char *)pvSaveData+8, fileSize-8 );
126 if( AllocData )
127 {
128 // free the compressed data buffer if required
129 free( pvSaveData );
130 }
131 else if(bLevelGenBaseSave)
132 {
133 levelGen->deleteBaseSaveData();
134 }
135 }
136 else
137#endif
138 {
139#ifdef __PSVITA__
140 void* pvSourceData = pvSaveData;
141#else
142 void* pvSourceData = pvSaveMem;
143#endif
144 int compressed = *static_cast<int *>(pvSourceData);
145 if( compressed == 0 )
146 {
147 unsigned int decompSize = *( static_cast<int *>(pvSourceData)+1 );
148 if(isLocalEndianDifferent(plat)) System::ReverseULONG(&decompSize);
149
150 // An invalid save, so clear the memory and start from scratch
151 if(decompSize == 0)
152 {
153 // 4J Stu - Saves created between 2/12/2011 and 7/12/2011 will have this problem
154 app.DebugPrintf("Invalid save data format\n");
155 ZeroMemory( pvSourceData, fileSize );
156 // Clear the first 8 bytes that reference the header
157 header.WriteHeader( pvSourceData );
158 }
159 else
160 {
161 unsigned char *buf = new unsigned char[decompSize];
162#ifndef _XBOX
163 if(plat == SAVE_FILE_PLATFORM_PSVITA)
164 {
165 Compression::VitaVirtualDecompress(buf, &decompSize, static_cast<unsigned char *>(pvSourceData)+8, fileSize-8 );
166 }
167 else
168#endif
169 {
170 Compression::getCompression()->SetDecompressionType(plat); // if this save is from another platform, set the correct decompression type
171 Compression::getCompression()->Decompress(buf, &decompSize, static_cast<unsigned char *>(pvSourceData)+8, fileSize-8 );
172 Compression::getCompression()->SetDecompressionType(SAVE_FILE_PLATFORM_LOCAL); // and then set the decompression back to the local machine's standard type
173 }
174
175 // Only ReAlloc if we need to (we might already have enough) and align to 512 byte boundaries
176 DWORD currentHeapSize = pagesCommitted * CSF_PAGE_SIZE;
177
178 DWORD desiredSize = decompSize;
179
180 if( desiredSize > currentHeapSize )
181 {
182 unsigned int pagesRequired = ( desiredSize + (CSF_PAGE_SIZE - 1 ) ) / CSF_PAGE_SIZE;
183 void *pvRet = VirtualAlloc(pvHeap, pagesRequired * CSF_PAGE_SIZE, COMMIT_ALLOCATION, PAGE_READWRITE);
184 if( pvRet == nullptr )
185 {
186 // Out of physical memory
187 __debugbreak();
188 }
189 pagesCommitted = pagesRequired;
190 }
191#ifdef __PSVITA__
192 VirtualCopyTo(pvSaveMem, buf, decompSize);
193#else
194 memcpy(pvSaveMem, buf, decompSize);
195#endif
196 delete[] buf;
197 }
198 }
199 }
200
201 header.ReadHeader( pvSaveMem, plat );
202
203 }
204 else
205 {
206 // Clear the first 8 bytes that reference the header
207 header.WriteHeader( pvSaveMem );
208 }
209}

236void ConsoleSaveFileOriginal::deleteFile( FileEntry *file )
237{
238 if( file == nullptr ) return;
239
240 LockSaveAccess();
241
242 DWORD numberOfBytesRead = 0;
243 DWORD numberOfBytesWritten = 0;
244
245 const int bufferSize = 4096;
246 int amountToRead = bufferSize;
247 byte buffer[bufferSize];
248 DWORD bufferDataSize = 0;
249
250
251 char *readStartOffset = static_cast<char *>(pvSaveMem) + file->data.startOffset + file->getFileSize();
252
253 char *writeStartOffset = static_cast<char *>(pvSaveMem) + file->data.startOffset;
254
255 char *endOfDataOffset = static_cast<char *>(pvSaveMem) + header.GetStartOfNextData();
256
257 while(true)
258 {
259 // Fill buffer from file
260 if( readStartOffset + bufferSize > endOfDataOffset )
261 {
262 amountToRead = static_cast<int>(endOfDataOffset - readStartOffset);
263 }
264 else
265 {
266 amountToRead = bufferSize;
267 }
268
269 if( amountToRead == 0 )
270 break;
271
272#ifdef __PSVITA__
273 // AP - use this to access the virtual memory
274 VirtualCopyFrom( buffer, readStartOffset, amountToRead );
275#else
276 memcpy( buffer, readStartOffset, amountToRead );
277#endif
278 numberOfBytesRead = amountToRead;
279
280 bufferDataSize = amountToRead;
281 readStartOffset += numberOfBytesRead;
282
283 // Write buffer to file
284#ifdef __PSVITA__
285 // AP - use this to access the virtual memory
286 VirtualCopyTo( (void *)writeStartOffset, buffer, bufferDataSize );
287#else
288 memcpy( (void *)writeStartOffset, buffer, bufferDataSize );
289#endif
290 numberOfBytesWritten = bufferDataSize;
291
292 writeStartOffset += numberOfBytesWritten;
293 }

295 header.RemoveFile( file );
296

317void ConsoleSaveFileOriginal::PrepareForWrite( FileEntry *file, DWORD nNumberOfBytesToWrite )
318{
319 int bytesToGrowBy = ( (file->currentFilePointer - file->data.startOffset) + nNumberOfBytesToWrite) - file->getFileSize();
320 if( bytesToGrowBy <= 0 )
321 return;
322
323 // 4J Stu - Not forcing a minimum size, it is up to the caller to write data in sensible amounts
324 // This lets us keep some of the smaller files small
325 //if( bytesToGrowBy < 1024 )
326 // bytesToGrowBy = 1024;
327
328 // Move all the data beyond us
329 MoveDataBeyond(file, bytesToGrowBy);
330
331 // Update our length
332 if( file->data.length < 0 )
333 file->data.length = 0;
334 file->data.length += bytesToGrowBy;
335
336 // Write the header with the updated data
337 finalizeWrite();
338}

463void ConsoleSaveFileOriginal::finalizeWrite()
464{
465 LockSaveAccess();
466
467 // Ensure buffer is large enough for the full file including header table.
468 // New file entries (e.g. from RegionFile creation) increase GetFileSize()
469 // without triggering MoveDataBeyond, so the committed pages may be short.
470 DWORD currentHeapSize = pagesCommitted * CSF_PAGE_SIZE;
471 DWORD desiredSize = header.GetFileSize();
472 if( desiredSize > currentHeapSize )
473 {
474 unsigned int pagesRequired = ( desiredSize + (CSF_PAGE_SIZE - 1 ) ) / CSF_PAGE_SIZE;
475 void *pvRet = VirtualAlloc(pvHeap, pagesRequired * CSF_PAGE_SIZE, COMMIT_ALLOCATION, PAGE_READWRITE);
476 if( pvRet == NULL )
477 {
478 __debugbreak();
479 }
480 pagesCommitted = pagesRequired;
481 }
482
483 header.WriteHeader( pvSaveMem );
484 ReleaseSaveAccess();
485}

487void ConsoleSaveFileOriginal::MoveDataBeyond(FileEntry *file, DWORD nNumberOfBytesToWrite)
488{
489 DWORD numberOfBytesRead = 0;
490 DWORD numberOfBytesWritten = 0;
491
492 const DWORD bufferSize = 4096;
493 DWORD amountToRead = bufferSize;
494 //assert( nNumberOfBytesToWrite <= bufferSize );
495 static byte buffer1[bufferSize];
496 static byte buffer2[bufferSize];
497 DWORD buffer1Size = 0;
498 DWORD buffer2Size = 0;
499
500 // Only ReAlloc if we need to (we might already have enough) and align to 512 byte boundaries
501 DWORD currentHeapSize = pagesCommitted * CSF_PAGE_SIZE;
502
503 DWORD desiredSize = header.GetFileSize() + nNumberOfBytesToWrite;
504
505 if( desiredSize > currentHeapSize )
506 {
507 unsigned int pagesRequired = ( desiredSize + (CSF_PAGE_SIZE - 1 ) ) / CSF_PAGE_SIZE;
508 void *pvRet = VirtualAlloc(pvHeap, pagesRequired * CSF_PAGE_SIZE, COMMIT_ALLOCATION, PAGE_READWRITE);
509 if( pvRet == nullptr )
510 {
511 // Out of physical memory
512 __debugbreak();
513 }
514 pagesCommitted = pagesRequired;
515 }
516
517 // This is the start of where we want the space to be, and the start of the data that we need to move
518 char *spaceStartOffset = static_cast<char *>(pvSaveMem) + file->data.startOffset + file->getFileSize();
519
520 // This is the end of where we want the space to be
521 char *spaceEndOffset = spaceStartOffset + nNumberOfBytesToWrite;
522
523 // This is the current end of the data that we want to move
524 char *beginEndOfDataOffset = static_cast<char *>(pvSaveMem) + header.GetStartOfNextData();
525
526 // This is where the end of the data is going to be
527 char *finishEndOfDataOffset = beginEndOfDataOffset + nNumberOfBytesToWrite;
528
529 // This is where we are going to read from (with the amount we want to read subtracted before we read)
530 char *readStartOffset = beginEndOfDataOffset;
531
532 // This is where we can safely write to (with the amount we want write subtracted before we write)
533 char *writeStartOffset = finishEndOfDataOffset;
534
535 //printf("\n******* MOVEDATABEYOND *******\n");
536 //printf("Space start: %d, space end: %d\n", spaceStartOffset - (char *)pvSaveMem, spaceEndOffset - (char *)pvSaveMem);
537 //printf("Current end of data: %d, new end of data: %d\n", beginEndOfDataOffset - (char *)pvSaveMem, finishEndOfDataOffset - (char *)pvSaveMem);
538
539 // Optimisation for things that are being moved in whole region file sector (4K chunks). We could generalise this a bit more but seems safest at the moment to identify this particular type
540 // of move and code explicitly for this situation
541 if( ( nNumberOfBytesToWrite & 4095 ) == 0 )
542 {
543 if( nNumberOfBytesToWrite > 0 )
544 {
545 // Get addresses for start & end of the region we are copying from as uintptr_t, for easier maths
546 uintptr_t uiFromStart = (uintptr_t)spaceStartOffset;
547 uintptr_t uiFromEnd = (uintptr_t)beginEndOfDataOffset;
548
549 // Round both of these values to get 4096 byte chunks that we will need to at least partially move
550 uintptr_t uiFromStartChunk = uiFromStart & ~static_cast<uintptr_t>(4095);
551 uintptr_t uiFromEndChunk = (uiFromEnd - 1 ) & ~static_cast<uintptr_t>(4095);
552
553 // Loop through all the affected source 4096 chunks, going backwards so we don't overwrite anything we'll need in the future
554 for( uintptr_t uiCurrentChunk = uiFromEndChunk; uiCurrentChunk >= uiFromStartChunk; uiCurrentChunk -= 4096 )
555 {
556 // Establish chunk we'll need to copy
557 uintptr_t uiCopyStart = uiCurrentChunk;
558 uintptr_t uiCopyEnd = uiCurrentChunk + 4096;
559 // Clamp chunk to the bounds of the full region we are trying to copy
560 if( uiCopyStart < uiFromStart )
561 {
562 // Needs to be clampged against the start of our region
563 uiCopyStart = uiFromStart;
564 }
565 if ( uiCopyEnd > uiFromEnd )
566 {
567 // Needs to be clamped to the end of our region
568 uiCopyEnd = uiFromEnd;
569 }
570#ifdef __PSVITA__
571 // AP - use this to access the virtual memory
572 VirtualMove( (void *)(uiCopyStart + nNumberOfBytesToWrite), ( void *)uiCopyStart, uiCopyEnd - uiCopyStart);
573#else
574 XMemCpy( (void *)(uiCopyStart + nNumberOfBytesToWrite), ( void *)uiCopyStart, uiCopyEnd - uiCopyStart );
575#endif
576 }
577 }
578 }
579 else
580 {
581 while(true)
582 {
583 // Copy buffer 1 to buffer 2
584 memcpy( buffer2, buffer1, buffer1Size);
585 buffer2Size = buffer1Size;
586
587 // Fill buffer 1 from file
588 if( (readStartOffset - bufferSize) < spaceStartOffset )
589 {
590 amountToRead = static_cast<DWORD>(readStartOffset - spaceStartOffset);
591 }
592 else
593 {
594 amountToRead = bufferSize;
595 }
596
597 // Push the read point back by the amount of bytes that we are going to read
598 readStartOffset -= amountToRead;
599
600 //printf("About to read %u from %d\n", amountToRead, readStartOffset - (char *)pvSaveMem );
601#ifdef __PSVITA__
602 // AP - use this to access the virtual memory
603 VirtualCopyFrom(buffer1, readStartOffset, amountToRead);
604#else
605 memcpy( buffer1, readStartOffset, amountToRead );
606#endif
607 numberOfBytesRead = amountToRead;
608
609 buffer1Size = amountToRead;
610
611 // Move back the write pointer by the amount of bytes we are going to write
612 writeStartOffset -= buffer2Size;
613
614 // Write buffer 2 to file
615 if( (writeStartOffset + buffer2Size) <= finishEndOfDataOffset)
616 {
617 //printf("About to write %u to %d\n", buffer2Size, writeStartOffset - (char *)pvSaveMem );
618#ifdef __PSVITA__
619 // AP - use this to access the virtual memory
620 VirtualCopyTo((void *)writeStartOffset, buffer2, buffer2Size);
621#else
622 memcpy( (void *)writeStartOffset, buffer2, buffer2Size );
623#endif
624 numberOfBytesWritten = buffer2Size;
625 }
626 else
627 {
628 assert((writeStartOffset + buffer2Size) <= finishEndOfDataOffset);
629 numberOfBytesWritten = 0;
630 }
631
632 if( numberOfBytesRead == 0 )
633 {
634 //printf("\n************** MOVE COMPLETED *************** \n\n");
635 assert( writeStartOffset == spaceEndOffset );
636 break;
637 }
638 }
639 }
640
641 header.AdjustStartOffsets( file, nNumberOfBytesToWrite );
642}

653void ConsoleSaveFileOriginal::Flush(bool autosave, bool updateThumbnail )
654{
655 LockSaveAccess();
656
657#ifdef __PSVITA__
658 // On Vita we've had problems with saves being corrupted on rapid save/save-exiting so seems prudent to wait for idle
659 while( StorageManager.GetSaveState() != C4JStorage::ESaveGame_Idle )
660 {
661 app.DebugPrintf("Flush wait\n");
662 Sleep(10);
663 }
664#endif
665
666 finalizeWrite();
667
668 // Get the frequency of the timer
669 LARGE_INTEGER qwTicksPerSec, qwTime, qwNewTime, qwDeltaTime;
670 float fElapsedTime = 0.0f;
671 QueryPerformanceFrequency( &qwTicksPerSec );
672 float fSecsPerTick = 1.0f / static_cast<float>(qwTicksPerSec.QuadPart);
673
674 unsigned int fileSize = header.GetFileSize();
675
676 // Assume that the compression will make it smaller so initially attempt to allocate the current file size
677 // We add 4 bytes to the start so that we can signal compressed data
678 // And another 4 bytes to store the decompressed data size
679 unsigned int compLength = fileSize+8;
680
681 // 4J Stu - Added TU-1 interim
682
683#ifdef __PS3__
684 // On PS3, don't compress the data as we can't really afford the extra memory this requires for the output buffer. Instead we'll be writing
685 // directly from the save data.
686 StorageManager.SetSaveData(pvSaveMem,fileSize);
687 byte *compData = (byte *)pvSaveMem;
688#else
689 // Attempt to allocate the required memory
690 // We do not own this, it belongs to the StorageManager
691 byte *compData = static_cast<byte *>(StorageManager.AllocateSaveData(compLength));
692
693#ifdef __PSVITA__
694 // AP - make sure we always allocate just what is needed so it will only SAVE what is needed.
695 // If we don't do this the StorageManager will save a file of uncompressed size unnecessarily.
696 compData = nullptr;
697#endif
698
699 // If we failed to allocate then compData will be nullptr
700 // Pre-calculate the compressed data size so that we can attempt to allocate a smaller buffer
701 if(compData == nullptr)
702 {
703 // Length should be 0 here so that the compression call knows that we want to know the length back
704 compLength = 0;
705
706 // Pre-calculate the buffer size required for the compressed data
707 PIXBeginNamedEvent(0,"Pre-calc save compression");
708 // Save the start time
709 QueryPerformanceCounter( &qwTime );

722 PIXEndNamedEvent();

724 // We add 4 bytes to the start so that we can signal compressed data

791
792#ifdef _XBOX
793 StorageManager.SaveSaveData( compLength+8,pbThumbnailData,dwThumbnailDataSize,bTextMetadata,iTextMetadataBytes );

802 }
803#endif
804 }
805 else
806 {
807 // We have failed to allocate the memory required to save this file. Now what?
808 }
809
810 ReleaseSaveAccess();
811#elif (defined __PS3__ || defined __ORBIS__ || defined __PSVITA__ || defined _DURANGO || defined _WINDOWS64)
812 // set the icon and save image
813 StorageManager.SetSaveImages(pbThumbnailData,dwThumbnailDataSize,pbDataSaveImage,dwDataSizeSaveImage,bTextMetadata,iTextMetadataBytes);
814 app.DebugPrintf("Save thumbnail size %d\n",dwThumbnailDataSize);
815
816 // save the data
817 StorageManager.SaveSaveData( &ConsoleSaveFileOriginal::SaveSaveDataCallback, this );
818#ifndef _CONTENT_PACKAGE
819 if( app.DebugSettingsOn())
820 {

825 }
826#endif

828 }
829#else
830 }
