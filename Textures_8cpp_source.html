423void Textures::bindTextureLayers(ResourceLocation *resource)
424{
425 assert(resource->isPreloaded());
426
427 // Hack: 4JLibs on Windows does not currently reproduce Minecraft's layered horse texture path reliably.
428 // Merge the layers on the CPU and bind the cached result as a normal single texture instead.
429 wstring cacheKey = L"%layered%";
430 int layers = resource->getTextureCount();
431 for( int i = 0; i < layers; i++ )
432 {
433 cacheKey += std::to_wstring(resource->getTexture(i));
434 cacheKey += L"/";
435 }
436
437 int id = -1;
438 bool inMap = ( idMap.find(cacheKey) != idMap.end() );
439 if( inMap )
440 {
441 id = idMap[cacheKey];
442 }
443 else
444 {
445 // Cache by layer signature so the merge cost is only paid once per horse texture combination.
446 intArray mergedPixels;
447 int mergedWidth = 0;
448 int mergedHeight = 0;
449 bool hasMergedPixels = false;
450
451 for( int i = 0; i < layers; i++ )
452 {
453 TEXTURE_NAME textureName = resource->getTexture(i);
454 if( textureName == static_cast<_TEXTURE_NAME>(-1) )
455 {
456 continue;
457 }
458
459 wstring resourceName = wstring(preLoaded[textureName]) + L".png";
460 BufferedImage *image = readImage(textureName, resourceName);
461 if( image == nullptr )
462 {
463 continue;
464 }
465
466 int width = image->getWidth();
467 int height = image->getHeight();
468 intArray layerPixels = loadTexturePixels(image);
469 delete image;
470
471 if( !hasMergedPixels )
472 {
473 mergedWidth = width;
474 mergedHeight = height;
475 mergedPixels = intArray(width * height);
476 memcpy(mergedPixels.data, layerPixels.data, width * height * sizeof(int));
477 hasMergedPixels = true;
478 }
479 else if( width == mergedWidth && height == mergedHeight )
480 {
481 for( int p = 0; p < width * height; p++ )
482 {
483 int dst = mergedPixels[p];
484 int src = layerPixels[p];
485
486 float srcAlpha = ((src >> 24) & 0xff) / 255.0f;
487 if( srcAlpha <= 0.0f )
488 {
489 continue;
490 }
491
492 float dstAlpha = ((dst >> 24) & 0xff) / 255.0f;
493 float outAlpha = srcAlpha + dstAlpha * (1.0f - srcAlpha);
494 if( outAlpha <= 0.0f )
495 {
496 mergedPixels[p] = 0;
497 continue;
498 }
499
500 float srcFactor = srcAlpha / outAlpha;
501 float dstFactor = (dstAlpha * (1.0f - srcAlpha)) / outAlpha;
502
503 int outA = static_cast<int>(outAlpha * 255.0f + 0.5f);
504 int outR = static_cast<int>((((src >> 16) & 0xff) * srcFactor) + (((dst >> 16) & 0xff) * dstFactor) + 0.5f);
505 int outG = static_cast<int>((((src >> 8) & 0xff) * srcFactor) + (((dst >> 8) & 0xff) * dstFactor) + 0.5f);
506 int outB = static_cast<int>(((src & 0xff) * srcFactor) + ((dst & 0xff) * dstFactor) + 0.5f);
507 mergedPixels[p] = (outA << 24) | (outR << 16) | (outG << 8) | outB;
508 }
509 }
510
511 delete[] layerPixels.data;
512 }
513
514 if( hasMergedPixels )
515 {
516 BufferedImage *mergedImage = new BufferedImage(mergedWidth, mergedHeight, BufferedImage::TYPE_INT_ARGB);
517 memcpy(mergedImage->getData(), mergedPixels.data, mergedWidth * mergedHeight * sizeof(int));
518 delete[] mergedPixels.data;
519 id = getTexture(mergedImage, C4JRender::TEXTURE_FORMAT_RxGyBzAw, false);
520 }
521 else
522 {
523 id = 0;
524 }
525
526 idMap[cacheKey] = id;
527 }
528
529 RenderManager.TextureBind(id);
530}
