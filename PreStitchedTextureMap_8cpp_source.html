201void PreStitchedTextureMap::makeTextureAnimated(TexturePack *texturePack, StitchedTexture *tex)
202{
203 if(!tex->hasOwnData())

205 animatedTextures.push_back(tex);
206 return;
207 }
208
209 wstring textureFileName = tex->m_fileName;
210
211 wstring animString = texturePack->getAnimationString(textureFileName, path, true);
212
213 if(!animString.empty())
214 {
215 wstring filename = path + textureFileName + extension;
216
217 // TODO: [EB] Put the frames into a proper object, not this inside out hack
218 vector<Texture *> *frames = TextureManager::getInstance()->createTextures(filename, m_mipMap);
219 if (frames == nullptr || frames->empty())
220 {
221 return; // Couldn't load a texture, skip it
222 }
223
224 Texture *first = frames->at(0);
225
226#ifndef _CONTENT_PACKAGE
227 if(first->getWidth() != tex->getWidth() || first->getHeight() != tex->getHeight())
228 {
229 app.DebugPrintf("%ls - first w - %d, h - %d, tex w - %d, h - %d\n",textureFileName.c_str(),first->getWidth(),tex->getWidth(),first->getHeight(),tex->getHeight());

231 }
232#endif
233
234 tex->init(stitchResult, frames, tex->getX(), tex->getY(), first->getWidth(), first->getHeight(), false);
235
236 if (frames->size() > 1)
237 {

240 tex->loadAnimationFrames(animString);
