669void MultiPlayerLevel::animateTick(int xt, int yt, int zt)
670{
671 // Get 8x8x8 chunk (ie not like the renderer or game chunks... maybe we need another word here...) that the player is in
672 // We then want to add a 3x3 region of chunks into a set that we'll be ticking over. Set is stored as unsigned ints which encode
673 // this chunk position
674 int cx = xt >> 3;
675 int cy = yt >> 3;
676 int cz = zt >> 3;
677
678 for( int xx = -1; xx <= 1; xx++ )
679 for( int yy = -1; yy <= 1; yy++ )
680 for( int zz = -1; zz <= 1; zz++ )
681 {
682 if( ( cy + yy ) < 0 ) continue;
683 if( ( cy + yy ) > 15 ) continue;
684 // Note - LEVEL_MAX_WIDTH is in game (16) tile chunks, and so our level goes from -LEVEL_MAX_WIDTH to LEVEL_MAX_WIDTH of our half-sized chunks
685 if( ( cx + xx ) >= LEVEL_MAX_WIDTH ) continue;
686 if( ( cx + xx ) < -LEVEL_MAX_WIDTH ) continue;
687 if( ( cz + zz ) >= LEVEL_MAX_WIDTH ) continue;
688 if( ( cz + zz ) < -LEVEL_MAX_WIDTH ) continue;
689 chunksToAnimate.insert( ( ( ( cx + xx ) & 0xff ) << 16 ) | ( ( ( cy + yy ) & 0xff ) << 8 ) | ( ( ( cz + zz ) & 0xff ) ) );
690 }
691}

693// 4J - the game used to tick 1000 tiles in a random region +/- 16 units round the player. We've got a 3x3 region of 8x8x8 chunks round each
694// player. So the original game was ticking 1000 things in a 32x32x32 region ie had about a 1 in 32 chance of updating any one tile per tick.
695// We're not dealing with quite such a big region round each player (24x24x24) but potentially we've got 4 players. Ultimately, we could end
696// up ticking anywhere between 432 and 1728 tiles depending on how many players we've got, which seems like a good tradeoff from the original.
697void MultiPlayerLevel::animateTickDoWork()
698{
699 const int ticksPerChunk = 16; // This ought to give us roughly the same 1000/32768 chance of a tile being animated as the original
700
701 // Horrible hack to communicate with the level renderer, which is just attached as a listener to this level. This let's the particle
702 // rendering know to use this level (rather than try to work it out from the current player), and to not bother distance clipping particles
703 // which would again be based on the current player.
704 Minecraft::GetInstance()->animateTickLevel = this;
705
706 MemSect(31);
707 Random *animateRandom = new Random();
708 MemSect(0);
709
710 for( int i = 0; i < ticksPerChunk; i++ )
711 {
712 for(int packed : chunksToAnimate)
713 {
714 int cx = ( packed << 8 ) >> 24;
715 int cy = ( packed << 16 ) >> 24;
716 int cz = ( packed << 24 ) >> 24;
717 cx <<= 3;
718 cy <<= 3;
719 cz <<= 3;
720 int x = cx + random->nextInt(8);
721 int y = cy + random->nextInt(8);
722 int z = cz + random->nextInt(8);
723 int t = getTile(x, y, z);
724 if (random->nextInt(8) > y && t == 0 && dimension->hasBedrockFog()) // 4J - test for bedrock fog brought forward from 1.2.3
725 {
726 addParticle(eParticleType_depthsuspend, x + random->nextFloat(), y + random->nextFloat(), z + random->nextFloat(), 0, 0, 0);
727 }
728 else if (t > 0)
729 {
730 Tile::tiles[t]->animateTick(this, x, y, z, animateRandom);
731 }
732 }
733 }
734
735 Minecraft::GetInstance()->animateTickLevel = nullptr;
736 delete animateRandom;
737
738 chunksToAnimate.clear();
739
740}

747void MultiPlayerLevel::playLocalSound(double x, double y, double z, int iSound, float volume, float pitch, bool distanceDelay/*= false */, float fClipSoundDist)
748{
749 //float dd = 16;

754 for( int i = 0; i < XUSER_MAX_COUNT; i++ )
755 {
756 if( minecraft->localplayers[i] )

762 }
763 }
764 }
765
766 if (minDistSq < fClipSoundDist * fClipSoundDist)
767 {
768 if (distanceDelay && minDistSq > 10 * 10)
769 {
770 // exhaggerate sound speed effect by making speed of sound ~=
771 // 40 m/s instead of 300 m/s
772 double delayInSeconds = sqrt(minDistSq) / 40.0;
773 minecraft->soundEngine->schedule(iSound, static_cast<float>(x), static_cast<float>(y), static_cast<float>(z), volume, pitch, static_cast<int>(Math::round(delayInSeconds * SharedConstants::TICKS_PER_SECOND)));
774 }
775 else
776 {
777 minecraft->soundEngine->play(iSound, static_cast<float>(x), static_cast<float>(y), static_cast<float>(z), volume, pitch);
