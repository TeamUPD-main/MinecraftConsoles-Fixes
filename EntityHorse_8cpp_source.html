1357void EntityHorse::travel(float xa, float ya)
1358{
1359 // If the entity is not ridden by Player, then execute the normal
1360 // Entityliving code
1361 if (rider.lock() == nullptr || !isSaddled())
1362 {
1363 footSize = .5f;

1366 return;

1369 yRotO = yRot = rider.lock()->yRot;

1374 shared_ptr<LivingEntity> livingRider = dynamic_pointer_cast<LivingEntity>(rider.lock());
1375 xa = livingRider->xxa * .5f;
1376 ya = livingRider->yya;

1381 ya *= .25f;

1400 hasImpulse = true;

1408 zd += 0.4f * cos * playerJumpPendingScale;
