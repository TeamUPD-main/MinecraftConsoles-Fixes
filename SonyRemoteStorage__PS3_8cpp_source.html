442int SonyRemoteStorage_PS3::LoadCompressCallback(void *pParam,bool bIsCorrupt, bool bIsOwner)
443{
444 SonyRemoteStorage_PS3* pRS = static_cast<SonyRemoteStorage_PS3 *>(pParam);
445 int origFilesize = StorageManager.GetSaveSize();
446 void* pOrigSaveData = malloc(origFilesize);
447 unsigned int retFilesize;
448 StorageManager.GetSaveData( pOrigSaveData, &retFilesize );
449 // check if this save file is already compressed
450 if(*static_cast<int *>(pOrigSaveData) != 0)
451 {
452 app.DebugPrintf("compressing save data\n");
453
454 // Assume that the compression will make it smaller so initially attempt to allocate the current file size
455 // We add 4 bytes to the start so that we can signal compressed data
456 // And another 4 bytes to store the decompressed data size
457 unsigned int compLength = origFilesize+8;
458 byte *compData = static_cast<byte *>(malloc(compLength));
459 Compression::UseDefaultThreadStorage();
460 Compression::getCompression()->Compress(compData+8,&compLength,pOrigSaveData,origFilesize);
461 ZeroMemory(compData,8);
462 int saveVer = 0;
463 memcpy( compData, &saveVer, sizeof(int) );
464 memcpy( compData+4, &origFilesize, sizeof(int) );
465
466 StorageManager.FreeSaveData();
467 StorageManager.SetSaveData(compData,compLength+8);
468 pRS->m_compressedSaveState = e_state_CompressedSave;
469 }
470 else
471 {
472 // already compressed, do nothing
473 pRS->m_compressedSaveState = e_state_Idle;
474 }
475
476 return 0;
477}
