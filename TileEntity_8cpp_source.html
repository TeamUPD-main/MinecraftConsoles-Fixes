98shared_ptr<TileEntity> TileEntity::loadStatic(CompoundTag *tag)
99{
100 shared_ptr<TileEntity> entity = nullptr;
101 auto it = idCreateMap.find(tag->getString(L"id"));
102 if (it != idCreateMap.end())
103 entity = shared_ptr<TileEntity>(it->second());
104 if (entity != nullptr)
