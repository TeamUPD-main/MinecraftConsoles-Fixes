147//Callback function type for returning horizontal metrics for each glyph
148IggyGlyphMetrics * UIBitmapFont::GetGlyphMetrics(S32 glyph,IggyGlyphMetrics *metrics)
149{
150 // 4J-JEV: Information about 'Glyph Metrics'.
151 // http://freetype.sourceforge.net/freetype2/docs/glyphs/glyphs-3.html - Overview.
152 // http://en.wikipedia.org/wiki/Kerning#Kerning_values - 'Font Units'
153
154 //Description
155 // Horizontal metrics for a glyph
156 //Members
157 // x0 y0 x1 y1 - bounding box
158 // advance - horizontal distance to move character origin after drawing this glyph
159
160
161 /* 4J-JEV: *IMPORTANT*
162 *
163 * I believe these are measured wrt the scale mentioned in GetGlyphBitmap
164 * i.e. 1.0f == pixel_scale,
165 *
166 * However we do not have that information here, then all these values need to be
167 * the same for every scale in this font.
168 *
169 * We have 2 scales of bitmap glyph, and we can only scale these up by powers of 2
170 * otherwise the fonts will become blurry. The appropriate glyph is chosen in
171 * 'GetGlyphBitmap' however we need to set the horizontal sizes here.
172 */
173
174 float glyphAdvance = m_cFontData->getAdvance(glyph);
175
176 // 4J-JEV: Anything outside this measurement will be
177 // cut off if it's at the start or end of the row.
178 metrics->x0 = 0.0f;
179
180 if ( m_cFontData->glyphIsWhitespace(glyph) )
181 metrics->x1 = 0.0f;
182 else
183 metrics->x1 = glyphAdvance;
184
185 // The next Glyph just starts right after this one.
186 metrics->advance = glyphAdvance;
187
188 //app.DebugPrintf("[UIBitmapFont] GetGlyphMetrics:\n\tmetrics->advance == %f,\n", metrics->advance);
189
190 // These don't do anything either.
191 metrics->y0 = 0.0f; metrics->y1 = 1.0f;
192
193 return metrics;
194}

221// Description
222// Callback function type used for getting the bitmap for a given glyph
223// Parameters
224// glyph The glyph to compute/get the bitmap for
225// pixel_scale The scale factor (pseudo point size) requested by the textfield,adjusted for display resolution
226// bitmap The structure to store the bitmap into
227rrbool UIBitmapFont::GetGlyphBitmap(S32 glyph,F32 pixel_scale,IggyBitmapCharacter *bitmap)
228{
229 //Description
230 // Data structure used to return to Iggy the bitmap to use for a glyph
231 //Members
232 // pixels_one_per_byte - pixels startin with the top-left-most; 0 is transparent and 255 is opaque
233 // width_in_pixels - this is the width of the bitmap data
234 // height_in_pixels - this is the height of the bitmap data
235 // stride_in_bytes - the distance from one row to the next
236 // oversample - this is the amount of oversampling (0 or 1 = not oversample,2 = 2x oversampled,4 = 4x oversampled)
237 // point_sample - if true,the bitmap will be drawn with point sampling; if false,it will be drawn with bilinear
238 // top_left_x - the offset of the top left corner from the character origin
239 // top_left_y - the offset of the top left corner from the character origin
240 // pixel_scale_correct - the pixel_scale at which this character should be displayed at displayed_width_in_pixels
241 // pixel_scale_min - the smallest pixel_scale to allow using this character (scaled down)
242 // pixel_scale_max - the largest pixels cale to allow using this character (scaled up)
243 // user_context_for_free - you can use this to store data to access on the corresponding free call
244
245 int row = 0,col = 0;
246 m_cFontData->getPos(glyph,row,col);
247
248 // Skip to glyph start.
249 bitmap->pixels_one_per_byte = m_cFontData->topLeftPixel(row,col);
250
251 // Choose a reasonable glyph scale.
252 float glyphScale = 1.0f, truePixelScale = 1.0f / m_cFontData->getFontData()->m_fAdvPerPixel;
253 while ( (0.5f + glyphScale) * truePixelScale < pixel_scale)
254 glyphScale++;
255
256 // Debug: log each unique (font, pixel_scale) pair
257 {
258 static std::unordered_set<int> s_loggedScaleKeys;
259 // Encode font pointer + quantized scale into a key to log each combo once
260 int scaleKey = (int)(pixel_scale * 100.0f) ^ (int)(uintptr_t)m_cFontData;
261 if (s_loggedScaleKeys.find(scaleKey) == s_loggedScaleKeys.end() && s_loggedScaleKeys.size() < 50) {
262 s_loggedScaleKeys.insert(scaleKey);
263 float tps = truePixelScale;
264 app.DebugPrintf("[FONT-DBG] GetGlyphBitmap: font=%s glyph=%d pixel_scale=%.3f truePixelScale=%.1f glyphScale=%.0f\n",
265 m_cFontData->getFontName().c_str(), glyph, pixel_scale, tps, glyphScale);
266 }
267 }

270#if (!defined _CONTENT_PACKAGE) && (VERBOSE_FONT_OUTPUT > 0)

310 }

313 // It is not necessary to shrink the glyph width here
314 // as its already been done in 'GetGlyphMetrics' by:
315 // > metrics->x1 = m_kerningTable[glyph] * ratio;
316 bitmap->width_in_pixels = m_cFontData->getFontData()->m_uiGlyphWidth;
317 bitmap->height_in_pixels = m_cFontData->getFontData()->m_uiGlyphHeight;
318
319 /* 4J-JEV: This is to do with glyph placement,
320 * and not the position in the archive.
321 * I don't know why the 0.65 is needed, or what it represents,
322 * although it doesn't look like its the baseline.
323 */
324 bitmap->top_left_x = 0;
325
326 // 4J-PB - this was chopping off the top of the characters, so accented ones were losing a couple of pixels at the top
327 // DaveK has reduced the height of the accented capitalised characters, and we've dropped this from 0.65 to 0.64
328 bitmap->top_left_y = -static_cast<S32>(m_cFontData->getFontData()->m_uiGlyphHeight) * m_cFontData->getFontData()->m_fAscent;
329
330 bitmap->oversample = 0;
331
332#ifdef _WINDOWS64
333 // On Windows64 the window can be any size, producing fractional
334 // pixel_scale values that don't align to integer multiples of
335 // truePixelScale. The original console code cached glyphs with a
336 // broad [truePixelScale, 99] range in the "normal" branch, which
337 // works on consoles (fixed 1080p — font sizes are exact multiples)
338 // but causes cache pollution on Windows: the first glyph cached in
339 // that range sets pixel_scale_correct for ALL subsequent requests,
340 // so different font sizes get scaled by wrong ratios, producing
341 // mixed letter sizes on screen.
342 //
343 // Fix: always use pixel_scale_correct = truePixelScale so every
344 // cache entry is consistent. Two ranges: downscale (bilinear for
345 // smooth reduction) and upscale (point_sample for crisp pixel-art).
346 bitmap->pixel_scale_correct = truePixelScale;
347 if (pixel_scale < truePixelScale)
348 {
349 bitmap->pixel_scale_min = 0.0f;
350 bitmap->pixel_scale_max = truePixelScale;
351 bitmap->point_sample = false;
352 }
353 else
354 {
355 bitmap->pixel_scale_min = truePixelScale;
356 bitmap->pixel_scale_max = 99.0f;
357 bitmap->point_sample = true;
358 }
359#else
360 if (glyphScale <= 1 && pixel_scale < truePixelScale)
361 {
362 // Small display: pixel_scale is less than the native glyph size.
363 // Report the bitmap at its true native scale so Iggy downscales it
364 // to match the layout metrics (bilinear for smooth downscaling).
365 bitmap->pixel_scale_correct = truePixelScale;
366 bitmap->pixel_scale_min = 0.0f;
367 bitmap->pixel_scale_max = truePixelScale * 1.001f;
368 bitmap->point_sample = false;
369 }
370 else
371 {
372 // Normal/upscale case: integer-multiple scaling for pixel-art look.
373 // Console-only — fixed resolution means pixel_scale values are exact
374 // integer multiples of truePixelScale, so cache sharing is safe.
375 float actualScale = pixel_scale / glyphScale;
376 bitmap->pixel_scale_correct = actualScale;
377 bitmap->pixel_scale_min = truePixelScale;
378 bitmap->pixel_scale_max = 99.0f;
379 bitmap->point_sample = true;
380 }
381#endif
382
383 // 4J-JEV: Nothing to do with glyph placement,
384 // entirely to do with cropping your glyph out of an archive.
385 bitmap->stride_in_bytes = m_cFontData->getFontData()->m_uiGlyphMapX;
386
387 // 4J-JEV: Additional information needed to release memory afterwards.
388 bitmap->user_context_for_free = nullptr;
389
390 return true;
391}
