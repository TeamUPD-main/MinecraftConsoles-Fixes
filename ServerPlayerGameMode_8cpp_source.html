246bool ServerPlayerGameMode::destroyBlock(int x, int y, int z)
247{

251 {
252 return false;
253 }
254 }
255
256 if (gameModeForPlayer->isCreative())

259 {

269 // 4J - In creative mode, the point where we need to tell the renderer that we are about to destroy a tile via destroyingTileAt is quite complicated.
270 // If the player being told is remote, then we always want the client to do it as it does the final update. If the player being told is local,
271 // then we need to update the renderer Here if we are sharing data between host & client as this is the final point where the original data is still intact.
272 // If the player being told is local, and we aren't sharing data between host & client, then we can just treat it as if it is a remote player and
273 // it can update the renderer.
274 bool clientToUpdateRenderer = false;
275 if( isCreative() )
276 {
277 clientToUpdateRenderer = true;
278 if( dynamic_pointer_cast<ServerPlayer>(player)->connection->isLocal() )
279 {
280 // Establish whether we are sharing this chunk between client & server
281 MultiPlayerLevel *clientLevel = Minecraft::GetInstance()->getLevel(level->dimension->id);
282 if( clientLevel )
283 {
284 LevelChunk *lc = clientLevel->getChunkAt( x, z );
285#ifdef SHARING_ENABLED
286 if( lc->sharingTilesAndData )
287 {
288 // We are sharing - this is the last point we can tell the renderer
289 Minecraft::GetInstance()->levelRenderer->destroyedTileManager->destroyingTileAt( clientLevel, x, y, z );
290
291 // Don't need to ask the client to do this too
292 clientToUpdateRenderer = false;
293 }
294#endif
295 }
296 }
297 }
298
299 bool changed = superDestroyBlock(x, y, z);
300
301 if (isCreative())
302 {
303 shared_ptr<TileUpdatePacket> tup = std::make_shared<TileUpdatePacket>(x, y, z, level);
304 // 4J - a bit of a hack here, but if we want to tell the client that it needs to inform the renderer of a block being destroyed, then send a block 255 instead of a 0. This is handled in ClientConnection::handleTileUpdate
305 if( tup->block == 0 )
306 {
307 if( clientToUpdateRenderer ) tup->block = 255;
308 }
309 player->connection->send( tup );
310 }
311 else
312 {
313 shared_ptr<ItemInstance> item = player->getSelectedItem();
314 bool canDestroy = player->canDestroy(Tile::tiles[t]);
315 if (item != nullptr)
316 {
317 item->mineBlock(level, t, x, y, z, player);
318 if (item->count == 0)
319 {
320 player->removeSelectedItem();
321 }
322 }
323 if (changed && canDestroy)
324 {
325 Tile::tiles[t]->playerDestroy(level, player, x, y, z, data);
326 }
327 }
328 return changed;
