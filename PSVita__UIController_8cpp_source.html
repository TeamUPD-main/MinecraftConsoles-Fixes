12void ConsoleUIController::init(S32 w, S32 h)
13{
14#ifdef _ENABLEIGGY
15
16 // Shared init
17 preInit(w,h);
18
19 /* Now that Iggy is ready, we need to create a set of draw
20 callbacks for it to use to draw things on the target platform.
21 We're using the GXM GDraw implementation that's included with
22 the Iggy SDK. To get it set up, we need to provide it with
23 memory for the different resource type. GDraw needs memory
24 for textures and vertex buffers. There's also the dynamic buffer;
25 we'll get to that in a bit.
26
27 We recommend that all resource pools be allocated in uncached memory for
28 performance reasons. They also *NEED* to be in GPU-writeable memory;
29 passing in memory that is only GPU-readable will cause a fault!
30 Typically, it is not necessary to allocate textures in CDRAM; LPDDR will
31 do (Iggy UI rendering doesn't need a lot of memory bandwidth for
32 textures).
33
34 The texture and vertex buffer pools contain multiple textures and
35 vertex buffers. Each object requires a bit of main memory for management
36 overhead. We need to specify how many object handles can be in use at
37 the same time in any given pool. If Iggy runs out of GDraw handles, it
38 won't crash, but instead throw out old cache data in a LRU fashion.
39 If the number of handles is too small, this will happen frequently, and
40 textures will need to decoded multiple times (or vertex buffers re-generated
41 multiple times), which causes a performance hit. */
42
43 // Size of the various memory pools. All these values are intentionally
44 // picked to be way larger than necessary for typical Flash files.
45 S32 tex_mem_size = 48*1024*1024; // Texture pool
46 S32 vert_mem_size = 4*1024*1024; // Vertex buffer pool
47
48 S32 tex_handles = 300;
49 S32 vert_handles = 1000;
50
51 /* SetResourceMemory is the call that actually sets up the location and size
52 of a resource pool. */
53 gdraw_psp2_SetResourceMemory(GDRAW_PSP2_RESOURCE_texture, tex_handles,
54 RenderManager.allocGPUCDMem(tex_mem_size, GDRAW_PSP2_TEXTURE_ALIGNMENT), tex_mem_size);
55 gdraw_psp2_SetResourceMemory(GDRAW_PSP2_RESOURCE_vertexbuffer, vert_handles,
56 RenderManager.allocGPURWMem(vert_mem_size, GDRAW_PSP2_VERTEXBUFFER_ALIGNMENT), vert_mem_size);
57
58 /* There's one more resource-related buffer we need to allocate: the dynamic
59 buffer. This is where GDraw writes dynamic vertex data and uniform buffers.
60
61 The dynamic buffer should be allocated in uncached memory that is mapped
62 GPU-readable. GPU writes are not required for this memory.
63
64 In theory you can use only one dynamic buffer, but if you did that, GDraw
65 would have to wait for the GPU to finish rendering the previous frame
66 before it could start CPU work on the next. This is clearly not a good idea
67 from a performance standpoint.
68
69 Hence, we recommend that you use two or three dynamic buffers in a round-robin
70 fashion. This example uses as many dynamic buffers as it does back buffers
71 (display queue entries), which is a good general rule of thumb.
72 */
73 S32 dynamic_buf_size = 1*1024*1024; // Again, very generous!
74 m_currentBackBuffer = 0;
75 m_dynamicBuffer = new gdraw_psp2_dynamic_buffer[RenderManager.getDisplayBufferCount()];
76 for (int i=0; i < RenderManager.getDisplayBufferCount(); i++)
77 gdraw_psp2_InitDynamicBuffer(&m_dynamicBuffer[i], RenderManager.allocGPUROMem(dynamic_buf_size, 4), dynamic_buf_size);
78
79 /* Whew! Now that the GDraw memory configuration is set up, we need to initialize
80 GDraw. But first, we need to allocate yet another block of video memory for
81 GDraw to store some global data in - a few buffers that are shared between
82 all Iggys. Unlike the rest of the memory configuration, the work area is reserved
83 to GDraw and can't be freed or relocated afterwards without shutting GDraw and all
84 attached Iggys down completely. But it's fairly small (16k as of this writing) so
85 this shouldn't be much of a problem. It should be allocted in uncached,
86 GPU-readable memory. */
87 void *gdraw_context_mem = RenderManager.allocGPUROMem(GDRAW_PSP2_CONTEXT_MEM_SIZE, 1);
88
89 /* GDraw also needs a single GPU notification label for its synchronization work.
90 Since there's no other rendering in this example, we just use the first label. */
91 volatile uint32_t *notify_region = RenderManager.getNotificationRegion(); //sceGxmGetNotificationRegion();
92
93 /* Finally, we can actually create the GDraw GXM driver and pass it to Iggy. */
94 gdraw_funcs = gdraw_psp2_CreateContext(
95 RenderManager.getGXMShaderPatcher(),
96 gdraw_context_mem,
97 notify_region + 0,
98 SCE_GXM_OUTPUT_REGISTER_FORMAT_UCHAR4);
99 IggySetGDraw(gdraw_funcs);
100
101 /* Flash content can have audio embedded. We'd like to be able
102 to play back any audio there is in the Flash that we load,
103 but in this tutorial we don't care about processing the sound
104 ourselves. So we call $IggyAudioUseDefault to tell Iggy
105 to go ahead and use the default sound driver. */
106 //IggyAudioUseDefault();
107
108 // Shared init
109 postInit();
110#endif
111}

113void ConsoleUIController::render()
114{
115#ifdef _ENABLEIGGY
116 /* Tell GDraw that we're about to start Iggy rendering.
117 There may only be one pair of $gdraw_psp2_Begin / $gdraw_psp2_End
118 per scene. Pass the same color surface and depth/stencil surface
119 you passed to sceGxmBeginScene.
120
121 You can do other rendering before you call $gdraw_psp2_Begin.
122 */
123
124 gdraw_psp2_Begin(RenderManager.getGXMContext(), RenderManager.getCurrentDisplaySurface(), RenderManager.getCurrentDepthSurface(), &m_dynamicBuffer[m_currentBackBuffer]);
125 ++m_currentBackBuffer;
126 m_currentBackBuffer %= RenderManager.getDisplayBufferCount();
127
128 /* We need to tell GDraw which surface to render to.
129 This tutorial just uses our main color and depth buffers,
130 but for in-game UI usage, you might want to use another
131 render target (like a texture) instead. */
132 gdraw_psp2_SetTileOrigin(0, 0);
133
134 renderScenes();
135
136 /* That's it with the Iggy rendering for this scene, so tell
137 GDraw we're done rendering. $gdraw_psp2_End returns a notification
138 that you have to pass as fragment notification to sceGxmEndScene.
139 This is *crucial* or rendering artifacts will occur in certain
140 cases!
141
142 After this call, you may resume other (non-Iggy) rendering for this
143 scene. */
144 SceGxmNotification notify = gdraw_psp2_End();
145
146 /* End the GXM scene. Note that we pass "notify" (returned from
147 $gdraw_psp2_End) as the fragment notification. */
148 //sceGxmEndScene(gxm, nullptr, &notify);
149 RenderManager.setFragmentNotification(notify);
150#endif
151}
