255
256 if(player == nullptr) return;

259
260 // 4J - TomK setup time update value array size to update the active effects
261 int iValue = 0;
262 IggyDataValue *UpdateValue = new IggyDataValue[activeEffects->size()*2];
263
264 for(auto& effect : *activeEffects)
265 {
266 if(effect->getDuration() >= m_bEffectTime[effect->getId()])
267 {

269 if (effect->getAmplifier() > 0)

284 potencyString += app.GetString( IDS_POTION_POTENCY_3 );
285 break;
286 default:
287 potencyString = app.GetString( IDS_POTION_POTENCY_0 );
288 break;
289 }
290 effectString += potencyString;
291 }
292 int icon = 0;
293 MobEffect *mobEffect = MobEffect::effects[effect->getId()];
294 if (mobEffect->hasIcon())
295 {
296 icon = mobEffect->getIcon();
297 }
298 IggyDataValue result;
299 IggyDataValue value[3];

306 value[1].type = IGGY_DATATYPE_string_UTF16;
307 value[1].string16 = stringVal;
308

310 value[2].type = IGGY_DATATYPE_number;
311 value[2].number = seconds;
312 IggyResult out = IggyPlayerCallMethodRS ( getMovie() , &result, IggyPlayerRootPath( getMovie() ), m_funcAddEffect , 3 , value );
313 }
314

316 {
317 // 4J - TomK set ids and remaining duration so we can update the timers accurately in one call! (this prevents performance related timer sync issues, especially on PSVita)
318 UpdateValue[iValue].type = IGGY_DATATYPE_number;
319 UpdateValue[iValue].number = MobEffect::effects[effect->getId()]->getIcon();
320 UpdateValue[iValue + 1].type = IGGY_DATATYPE_number;
321 UpdateValue[iValue + 1].number = (int)(effect->getDuration() / SharedConstants::TICKS_PER_SECOND);
322 iValue+=2;
323 }
324

326 }
327
328 IggyDataValue result;
329 IggyResult out = IggyPlayerCallMethodRS ( getMovie() , &result, IggyPlayerRootPath( getMovie() ), m_funcUpdateEffects , activeEffects->size()*2 , UpdateValue );
330
331 delete activeEffects;
332}
