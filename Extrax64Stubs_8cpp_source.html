280 !m_player[dwUserIndex].m_isRemote &&
281 Win64_IsActivePlayer(&m_player[dwUserIndex], dwUserIndex))

294 return &m_player[idx];
295 return nullptr;
296 }
297 // Offline: scan for first local player
298 for (DWORD i = 0; i < s_playerCount; i++)
299 {
300 if (!m_player[i].m_isRemote && Win64_IsActivePlayer(&m_player[i], i))
301 return &m_player[i];
302 }
303 return nullptr;
304 }
305 // Split-screen pads 1-3: the player is at m_player[dwUserIndex] with isRemote=false
306 if (dwUserIndex < MINECRAFT_NET_MAX_PLAYERS &&
307 !m_player[dwUserIndex].m_isRemote &&
308 Win64_IsActivePlayer(&m_player[dwUserIndex], dwUserIndex))
309 return &m_player[dwUserIndex];
310 return nullptr;
