52void Minimap::reloadColours()
53{
54 ColourTable *colourTable = Minecraft::GetInstance()->getColourTable();
55 // 4J note that this code has been extracted pretty much as it was in Minimap::render, although with some byte order changes
56 for( int i = 0; i < (14 * 4); i++ ) // 14 material colours currently, 4 brightnesses of each
57 {
58 if (i / 4 == 0)
59 {
60 // 4J - changed byte order to save having to reorder later
61#ifdef __ORBIS__
62 LUT[i] = 0;
63#else
64 LUT[i] = (((i + i / w) & 1) * 8 + 16);
65#endif
66 //pixels[i] = (((i + i / w) & 1) * 8 + 16) << 24;
67 }
68 else
69 {
70 int color = colourTable->getColor( MaterialColor::colors[i / 4]->col );
71 int brightness = i & 3;
72
73 int br = 220;
74 if (brightness == 2) br = 255;
75 if (brightness == 0) br = 180;
76
77 int r = ((color >> 16) & 0xff) * br / 255;
78 int g = ((color >> 8) & 0xff) * br / 255;
79 int b = ((color) & 0xff) * br / 255;
80
81 // 4J - changed byte order to save having to reorder later
82#if ( defined _DURANGO || defined _WIN64 || __PSVITA__ )
83 LUT[i] = 255 << 24 | b << 16 | g << 8 | r;
84#elif defined _XBOX
85 LUT[i] = 255 << 24 | r << 16 | g << 8 | b;
86#elif defined __ORBIS__
87 r >>= 3; g >>= 3; b >>= 3;
88 LUT[i] = 1 << 15 | ( r << 10 ) | ( g << 5 ) | b;
89#else
90 LUT[i] = r << 24 | g << 16 | b << 8 | 255;
91#endif
92
93 //pixels[i] = (255) << 24 | r << 16 | g << 8 | b;
94 }
95
96 }
97 genLUT = false;
98}
