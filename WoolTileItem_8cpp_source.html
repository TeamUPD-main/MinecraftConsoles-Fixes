137unsigned int WoolTileItem::getDescriptionId(shared_ptr<ItemInstance> instance)
138{
139 int tileId = getTileId();
140 switch(getTileId())

142 case Tile::stained_glass_Id:

150 case Tile::wool_Id:

152 return COLOR_DESCS[ColoredTile::getTileDataForItemAuxValue(instance->getAuxValue())];
153 };
