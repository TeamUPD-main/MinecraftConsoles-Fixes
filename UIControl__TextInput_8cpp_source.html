96void UIControl_TextInput::setCaretVisible(bool visible)
97{
98 if (!m_parentScene || !m_parentScene->getMovie())
99 return;
100
101 // Check once whether this SWF's FJ_TextInput actually has a m_mcCaret child.
102 // IggyValuePathMakeNameRef always succeeds (creates a ref to undefined),
103 // so we validate by trying to read a property from the resolved path.
104 if (!m_bCaretChecked)
105 {
106 IggyValuePath caretPath;
107 if (IggyValuePathMakeNameRef(&caretPath, getIggyValuePath(), "m_mcCaret"))
108 {
109 rrbool test = false;
110 IggyResult res = IggyValueGetBooleanRS(&caretPath, m_nameVisible, NULL, &test);
111 m_bHasCaret = (res == 0);
112 }
113 else
114 {
115 m_bHasCaret = false;
116 }
117 m_bCaretChecked = true;
118 }
119 if (!m_bHasCaret)
120 return;
121
122 IggyValuePath caretPath;
123 if (IggyValuePathMakeNameRef(&caretPath, getIggyValuePath(), "m_mcCaret"))
124 {
125 IggyValueSetBooleanRS(&caretPath, m_nameVisible, NULL, visible);
126 }
127}

158UIControl_TextInput::EDirectEditResult UIControl_TextInput::tickDirectEdit()
159{
160 if (m_iDirectEditCooldown > 0)
161 m_iDirectEditCooldown--;
162
163 if (!m_bDirectEditing)
164 {
165 setCaretVisible(false);
166 return eDirectEdit_Continue;
167 }
168
169 // Enforce caret visibility and position every tick — setLabel() and Flash
170 // focus changes can reset both at any time.
171 setCaretVisible(true);
172 setCaretIndex(m_iCursorPos);
173
174 // For SWFs without m_mcCaret, insert '_' at the cursor position.
175 // All characters remain visible — '_' sits between them like a cursor.
176 if (!m_bHasCaret)
177 {
178 wstring display = m_editBuffer;
179 display.insert(m_iCursorPos, 1, L'_');
180 setLabel(display.c_str());
181 }
182
183 EDirectEditResult result = eDirectEdit_Continue;
184 bool changed = false;
185
186 // Consume typed characters from the KBM buffer
187 wchar_t ch;
188 while (g_KBMInput.ConsumeChar(ch))
189 {
190 if (ch == 0x08) // Backspace
191 {
192 if (m_iCursorPos > 0)
193 {
194 m_editBuffer.erase(m_iCursorPos - 1, 1);
195 m_iCursorPos--;
196 changed = true;
197 }
198 }
199 else if (ch == 0x0D) // Enter — confirm edit
200 {
201 m_bDirectEditing = false;
202 m_iDirectEditCooldown = 4;
203 setLabel(m_editBuffer.c_str(), true);
204 setCaretVisible(false);
205 return eDirectEdit_Confirmed;
206 }
207 else if (m_iCharLimit <= 0 || (int)m_editBuffer.length() < m_iCharLimit)
208 {
209 m_editBuffer.insert(m_iCursorPos, 1, ch);
210 m_iCursorPos++;
211 changed = true;
212 }
213 }
214
215 // Paste from clipboard
216 if (g_KBMInput.IsKeyPressed('V') && g_KBMInput.IsKeyDown(VK_CONTROL))
217 {
218 wstring pasted = Screen::getClipboard();
219 wstring sanitized;
220 sanitized.reserve(pasted.length());
221
222 for (wchar_t pc : pasted)
223 {
224 if (pc >= 0x20) // Keep printable characters
225 {
226 if (m_iCharLimit > 0 && (m_editBuffer.length() + sanitized.length()) >= (size_t)m_iCharLimit)
227 break;
228 sanitized += pc;
229 }
230 }
231
232 if (!sanitized.empty())
233 {
234 m_editBuffer.insert(m_iCursorPos, sanitized);
235 m_iCursorPos += (int)sanitized.length();
236 changed = true;
237 }
238 }
239
240 // Arrow keys, Home, End, Delete for cursor movement
241 if (g_KBMInput.IsKeyPressed(VK_LEFT) && m_iCursorPos > 0)
242 {
243 m_iCursorPos--;
244 setCaretIndex(m_iCursorPos);
245 }
246 if (g_KBMInput.IsKeyPressed(VK_RIGHT) && m_iCursorPos < (int)m_editBuffer.length())
247 {
248 m_iCursorPos++;
249 setCaretIndex(m_iCursorPos);
250 }
251 if (g_KBMInput.IsKeyPressed(VK_HOME))
252 {
253 m_iCursorPos = 0;
254 setCaretIndex(m_iCursorPos);
255 }
256 if (g_KBMInput.IsKeyPressed(VK_END))
257 {
258 m_iCursorPos = (int)m_editBuffer.length();
259 setCaretIndex(m_iCursorPos);
260 }
261 if (g_KBMInput.IsKeyPressed(VK_DELETE) && m_iCursorPos < (int)m_editBuffer.length())
262 {
263 m_editBuffer.erase(m_iCursorPos, 1);
264 changed = true;
265 }
266
267 // Escape — cancel edit and restore original text
268 if (g_KBMInput.IsKeyPressed(VK_ESCAPE))
269 {
270 m_editBuffer = m_textBeforeEdit;
271 m_bDirectEditing = false;
272 m_iDirectEditCooldown = 4;
273 setLabel(m_editBuffer.c_str());
274 setCaretVisible(false);
275 return eDirectEdit_Cancelled;
276 }
277
278 if (changed)
279 {
280 if (m_bHasCaret)
281 {
282 setLabel(m_editBuffer.c_str());
283 setCaretIndex(m_iCursorPos);
284 }
285 // SWFs without caret: the cursor block above already updates the label every tick
286 }
287
288 return eDirectEdit_Continue;
289}
