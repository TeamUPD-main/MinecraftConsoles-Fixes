91// Temporary implementation of lock free stack with quite a bit more locking than you might expect
92template <typename T> class XLockFreeStack
93{
94 std::vector<T *> intStack;
95public:
96 XLockFreeStack() {
97#ifdef __ORBIS__
98 OrbisInit(); // For PS4, we need to make sure Ult is set up for the critical sections to be able to initialise
99#endif
100 InitializeCriticalSectionAndSpinCount(&m_cs,5120);
101 }
102 ~XLockFreeStack() { DeleteCriticalSection( &m_cs ); }
103 void Initialize() {}
104 void Push(T *data)
105 {
106 EnterCriticalSection(&m_cs);
107 intStack.push_back(data);
108 LeaveCriticalSection(&m_cs);
109 }
110 T *Pop()
111 {
112 EnterCriticalSection(&m_cs);
113 if( intStack.size() )
114 {
115 T* ret = intStack.back();
116 intStack.pop_back();
117 LeaveCriticalSection(&m_cs);
118 return ret;
119 }
120 else
121 {
122 LeaveCriticalSection(&m_cs);
123 return nullptr;
124 }
125 }
126private:
127 CRITICAL_SECTION m_cs;
128};
