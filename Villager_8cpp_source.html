96void Villager::serverAiMobStep()
97{
98 if (--villageUpdateInterval <= 0)
99 {
100 level->villages->queryUpdateAround(Mth::floor(x), Mth::floor(y), Mth::floor(z));
101 villageUpdateInterval = 70 + random->nextInt(50);
102
103 shared_ptr<Village> _village = level->villages->getClosestVillage(Mth::floor(x), Mth::floor(y), Mth::floor(z), Villages::MaxDoorDist);
104 village = _village;
105 if (_village == nullptr) clearRestriction();
106 else
107 {
108 Pos *center = _village->getCenter();
109 restrictTo(center->x, center->y, center->z, static_cast<int>((float)_village->getRadius() * 0.6f));
110 if (rewardPlayersOnFirstVillage)
111 {
112 rewardPlayersOnFirstVillage = false;
113 _village->rewardAllPlayers(5);
114 }
115 }
116 }
117
118 if (!isTrading() && updateMerchantTimer > 0)
119 {
120 updateMerchantTimer--;
121 if (updateMerchantTimer <= 0)
122 {
123 if (addRecipeOnUpdate)
124 {
125 // improve max uses for all obsolete recipes
126 if (offers->size() > 0)
127 {
128 //for (MerchantRecipe recipe : offers)
129 for(auto& recipe : *offers)
130 {
131 if (recipe->isDeprecated())
132 {
133 recipe->increaseMaxUses(random->nextInt(6) + random->nextInt(6) + 2);
134 }
135 }
136 }
137 addOffers(1);
138 addRecipeOnUpdate = false;
139
140 if (village.lock() != nullptr && !lastPlayerTradeName.empty())
141 {
142 level->broadcastEntityEvent(shared_from_this(), EntityEvent::VILLAGER_HAPPY);
143 village.lock()->modifyStanding(lastPlayerTradeName, 1);
144 }
145 }
146 addEffect(new MobEffectInstance(MobEffect::regeneration->id, SharedConstants::TICKS_PER_SECOND * 10, 0));
147 }
148 }
149
150 AgableMob::serverAiMobStep();
151}

282void Villager::die(DamageSource *source)
283{
284 shared_ptr<Village> _village = village.lock();
285 if (_village != nullptr)
286 {
287 shared_ptr<Entity> sourceEntity = source->getEntity();
288 if (sourceEntity != nullptr)
289 {
290 if ( sourceEntity->instanceof(eTYPE_PLAYER) )
291 {
292 _village->modifyStanding( dynamic_pointer_cast<Player>(sourceEntity)->getName(), -2 );
293 }
294 else if ( sourceEntity->instanceof(eTYPE_ENEMY) )
295 {
296 _village->resetNoBreedTimer();
297 }
298 }
299 else if (sourceEntity == nullptr)
300 {
301 // if the villager was killed by the world (such as lava or falling), blame
302 // the nearest player by not reproducing for a while
303 shared_ptr<Player> nearestPlayer = level->getNearestPlayer(shared_from_this(), 16.0f);
304 if (nearestPlayer != nullptr)
305 {
306 _village->resetNoBreedTimer();
307 }
308 }
309 }
310
311 // Make the gui close if the villager die while trading
312 if (auto currentTrader = tradingPlayer.lock())
313 {
314 if (currentTrader->containerMenu != nullptr)
315 {
316 auto menu = currentTrader->containerMenu;
317 menu->removed(currentTrader);
318 }
319
320 tradingPlayer.reset();
321 }
322
323 AgableMob::die(source);
324}
