215// We call this twice a frame, either side of the render call so is a good place to "tick" things
216void CPlatformNetworkManagerStub::DoWork()
217{
218#ifdef _WINDOWS64
219 extern QNET_STATE _iQNetStubState;
220 if (_iQNetStubState == QNET_STATE_SESSION_STARTING && app.GetGameStarted())
221 {
222 _iQNetStubState = QNET_STATE_GAME_PLAY;
223 if (m_pIQNet->IsHost())
224 WinsockNetLayer::UpdateAdvertiseJoinable(true);
225 }
226 if (_iQNetStubState == QNET_STATE_IDLE)
227 TickSearch();
228 if (_iQNetStubState == QNET_STATE_GAME_PLAY && m_pIQNet->IsHost())
229 {
230 BYTE disconnectedSmallId;
231 while (WinsockNetLayer::PopDisconnectedSmallId(&disconnectedSmallId))
232 {
233 IQNetPlayer* qnetPlayer = m_pIQNet->GetPlayerBySmallId(disconnectedSmallId);
234 if (qnetPlayer != nullptr && qnetPlayer->m_smallId == disconnectedSmallId)
235 {
236 NotifyPlayerLeaving(qnetPlayer);
237 qnetPlayer->m_smallId = 0;
238 qnetPlayer->m_isRemote = false;
239 qnetPlayer->m_isHostPlayer = false;
240 qnetPlayer->m_resolvedXuid = INVALID_XUID;
241 qnetPlayer->m_gamertag[0] = 0;
242 qnetPlayer->SetCustomDataValue(0);
243 //while (IQNet::s_playerCount > 1 && IQNet::m_player[IQNet::s_playerCount - 1].GetCustomDataValue() == 0)
244 // Recalculate s_playerCount as the highest active slot + 1.
245 // A blind decrement would hide players at higher-indexed slots when a
246 // lower-indexed player disconnects first: GetPlayerBySmallId scans
247 // [0, s_playerCount) so any slot at or above the decremented count
248 // becomes invisible, causing its disconnect to be missed (ghost player).
249 while (IQNet::s_playerCount > 1 &&
250 IQNet::m_player[IQNet::s_playerCount - 1].GetCustomDataValue() == 0)
251 IQNet::s_playerCount--;
252 }
253 // NOTE: Do NOT call PushFreeSmallId here. The old PlayerConnection's
254 // write thread may still be alive (it dies in PlayerList::tick when
255 // m_smallIdsToClose is processed). If we recycle the smallId now,
256 // AcceptThread can reuse it for a new connection, and the old write
257 // thread's getPlayer() lookup will resolve to the NEW player, sending
258 // stale game packets to the new client's TCP socket — corrupting its
259 // login handshake (bad packet id crash). PushFreeSmallId and
260 // ClearSocketForSmallId are called from PlayerList::tick after the
261 // old Connection threads are dead.
262 //
263 // Clear chunk visibility flags for this system so rejoin gets fresh chunk state.
264 SystemFlagRemoveBySmallId(disconnectedSmallId);
265 }
266 }
267
268 // Client-side host disconnect detection:
269 // if TCP is gone, propagate through normal network-disconnect flow so UI returns to menus.
270 // The processing from the Xbox version will be reused.
271 if (_iQNetStubState == QNET_STATE_GAME_PLAY && !m_pIQNet->IsHost() && !m_bLeavingGame)
272 {
273 if (!WinsockNetLayer::IsConnected())
274 {
275 if (!m_bLeaveGameOnTick)
276 {
277 m_bLeaveGameOnTick = true;
278 g_NetworkManager.HandleDisconnect(false);
279 }
280 }
281 else
282 {
283 m_bLeaveGameOnTick = false;
284 }
285 }
286#endif
287}

693{
694 PlayerFlags *newPlayerFlags = new PlayerFlags( pNetworkPlayer, m_flagIndexSize);
