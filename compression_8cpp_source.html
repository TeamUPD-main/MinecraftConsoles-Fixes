57HRESULT Compression::CompressLZXRLE(void *pDestination, unsigned int *pDestSize, void *pSource, unsigned int SrcSize)
58{
59 EnterCriticalSection(&rleCompressLock);
60 //static unsigned char rleBuf[1024*100];
61
62 // RLE can expand data (each 0xFF byte becomes 2 bytes), so worst case is 2*SrcSize.
63 // Use the static buffer when it fits, dynamically allocate otherwise.
64 static const unsigned int staticCompressSize = 1024*100;
65 unsigned int rleBufSize = SrcSize * 2;
66 unsigned char *dynamicRleBuf = nullptr;
67 unsigned char *rleBuf;
68
69 if(rleBufSize <= staticCompressSize)
70 {
71 rleBuf = rleCompressBuf;
72 rleBufSize = staticCompressSize;
73 }
74 else
75 {
76 dynamicRleBuf = new unsigned char[rleBufSize];
77 rleBuf = dynamicRleBuf;
78 }
79
80 const unsigned char *pucIn = static_cast<unsigned char*>(pSource);
81 const unsigned char *pucEnd = pucIn + SrcSize;
82 unsigned char *pucOut = rleBuf;
83
84 // Compress with RLE first:
85 // 0 - 254 - encodes a single byte
86 // 255 followed by 0, 1, 2 - encodes a 1, 2, or 3 255s
87 // 255 followed by 3-255, followed by a byte - encodes a run of n + 1 bytes
88 PIXBeginNamedEvent(0,"RLE compression");
89 do
90 {
91 const unsigned char thisOne = *pucIn++;
92
93 unsigned int count = 1;
94 while( ( pucIn != pucEnd ) && ( *pucIn == thisOne ) && ( count < 256 ) )
95 {
96 pucIn++;
97 count++;
98 }
99
100 if( count <= 3 )
101 {
102 if( thisOne == 255 )
103 {
104 *pucOut++ = 255;
105 *pucOut++ = count - 1;
106 }
107 else
108 {
109 for( unsigned int i = 0; i < count ; i++ )
110 {
111 *pucOut++ = thisOne;
112 }
113 }
114 }
115 else
116 {
117 *pucOut++ = 255;
118 *pucOut++ = count - 1;
119 *pucOut++ = thisOne;
120 }
121 } while (pucIn != pucEnd);
122 const unsigned int rleSize = static_cast<unsigned int>(pucOut - rleBuf);
123 PIXEndNamedEvent();
124
125 PIXBeginNamedEvent(0,"Secondary compression");
126 Compress(pDestination, pDestSize, rleBuf, rleSize);
127 PIXEndNamedEvent();
128
129 if(dynamicRleBuf != nullptr) delete [] dynamicRleBuf;
130
131 LeaveCriticalSection(&rleCompressLock);
132// printf("Compressed from %d to %d to %d\n",SrcSize,rleSize,*pDestSize);
133
134 return S_OK;
135}

475HRESULT Compression::DecompressWithType(void *pDestination, unsigned int *pDestSize, void *pSource, unsigned int SrcSize)
476{
477 switch(m_decompressType)
478 {
479 case eCompressionType_RLE: // 4J-JEV, RLE is just that; don't want to break here though.
480 case eCompressionType_None:
481 memcpy(pDestination,pSource,SrcSize);
482 *pDestSize = SrcSize;
483 return S_OK;
484 case eCompressionType_LZXRLE:
485 {
486#if (defined _XBOX || defined _DURANGO || defined _WIN64)
487 SIZE_T destSize = (SIZE_T)(*pDestSize);
488 const HRESULT res = XMemDecompress(decompressionContext, pDestination, (SIZE_T *)&destSize, pSource, SrcSize);
489 *pDestSize = static_cast<unsigned int>(destSize);
490 return res;
491#else
492 assert(0);
493#endif
494 }
495 break;
496 case eCompressionType_ZLIBRLE:
497#if (defined __ORBIS__ || defined __PS3__ || defined _DURANGO || defined _WIN64)
498 if (pDestination != nullptr)
499 return ::uncompress(static_cast<PBYTE>(pDestination), (unsigned long *) pDestSize, static_cast<PBYTE>(pSource), SrcSize); // Decompress
500 else break; // Cannot decompress when destination is nullptr
501#else
502 assert(0);
503 break;
504#endif
505 case eCompressionType_PS3ZLIB:
506#if (defined __ORBIS__ || defined __PSVITA__ || defined _DURANGO || defined _WIN64)
507 // Note that we're missing the normal zlib header and footer so we'll use inflate to
508 // decompress the payload and skip all the CRC checking, etc
509 if (pDestination != nullptr)
510 {
511 // Read big-endian srcize from array
512 const PBYTE pbDestSize = (PBYTE) pDestSize;
513 const PBYTE pbSource = static_cast<PBYTE>(pSource);
514 for (int i = 3; i >= 0; i--) {
515 pbDestSize[3-i] = pbSource[i];
516 }
517
518 const byteArray uncompr = byteArray(*pDestSize);
519
520 // Build decompression stream
521 z_stream strm;
522 strm.zalloc = Z_NULL;
523 strm.zfree = Z_NULL;
524 strm.opaque = Z_NULL;
525 strm.next_out = uncompr.data;
526 strm.avail_out = uncompr.length;
527 // Skip those first 4 bytes
528 strm.next_in = static_cast<PBYTE>(pSource) + 4;
529 strm.avail_in = SrcSize - 4;
530
531 int hr = inflateInit2(&strm, -15);
532
533 // Run inflate() on input until end of stream
534 do {
535 hr = inflate(&strm, Z_NO_FLUSH);
536
537 // Check
538 switch (hr) {
539 case Z_NEED_DICT:
540 case Z_DATA_ERROR:
541 case Z_MEM_ERROR:
542 case Z_STREAM_ERROR:
543 (void)inflateEnd(&strm);
544 assert(false);
545 }
546 } while (hr != Z_STREAM_END);
547
548 inflateEnd(&strm); // MGH - added, to clean up zlib, was causing a leak on Vita when dowloading a PS3 save
549
550 // Copy the uncompressed data to the destination
551 memcpy(pDestination, uncompr.data, uncompr.length);
552 *pDestSize = uncompr.length;
553
554 // Delete uncompressed data
555 delete uncompr.data;
556 return S_OK;
557 }
558 else break; // Cannot decompress when destination is nullptr
559#else
560 assert(0);
561#endif
562 }
563
564 assert(false);
565 return -1;
566}
