119void StructureStart::moveBelowSeaLevel(Level *level, Random *random, int offset)
120{
121 const int MAX_Y = level->seaLevel - offset;
122
123 // set lowest possible position (at bedrock)
124 int y1Pos = boundingBox->getYSpan() + 1;
125 // move up randomly within the available span
126 if (y1Pos < MAX_Y)
127 {
128 y1Pos += random->nextInt(MAX_Y - y1Pos);
129 }
130
131 // move all bounding boxes
132 int dy = y1Pos - boundingBox->y1;
133 boundingBox->move(0, dy, 0);
134 for(auto& piece : pieces)
135 {
