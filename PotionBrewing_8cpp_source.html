267int PotionBrewing::constructParsedValue(bool isNot, bool hasMultiplier, bool isNeg, int countCompare, int valuePart, int multiplierPart, int brew)
268{
269 int value = 0;
270 if (isNot)
271 {
272 value = isNotBit(brew, valuePart);
273 }
274#if !(_SIMPLIFIED_BREWING)
275 else if (countCompare != NO_COUNT) // Never true for simplified brewing
276 {
277 if (countCompare == EQUAL_COUNT && countOnes(brew) == valuePart)
278 {
279 value = 1;
280 }
281 else if (countCompare == GREATER_COUNT && countOnes(brew) > valuePart)
282 {
283 value = 1;
284 }
285 else if (countCompare == LESS_COUNT && countOnes(brew) < valuePart)
286 {
287 value = 1;
288 }
289 }
290#endif
291 else
292 {
293 value = isBit(brew, valuePart);
294 }
295#if !(_SIMPLIFIED_BREWING)
296 if (hasMultiplier) // Always false for simplified brewing
297 {
298 value *= multiplierPart;
299 }
300#endif
301 if (isNeg)
302 {
303 value *= -1;
304 }
305 return value;
306}

553vector<MobEffectInstance *> *PotionBrewing::getEffects(int brew, bool includeDisabledEffects)
554{
555 vector<MobEffectInstance *> *list = nullptr;
556
557 //for (MobEffect effect : MobEffect.effects)
558 for(unsigned int i = 0; i < MobEffect::NUM_EFFECTS; ++i)
559 {
560 MobEffect *effect = MobEffect::effects[i];
561 if (effect == nullptr || (effect->isDisabled() && !includeDisabledEffects))
562 {
563 continue;
564 }
565 //wstring durationString = potionEffectDuration.get(effect->getId());
566 auto effIt = potionEffectDuration.find(effect->getId());
567 if ( effIt == potionEffectDuration.end() )
568 {
569 continue;
570 }
571 wstring durationString = effIt->second;
572
573 int duration = parseEffectFormulaValue(durationString, 0, static_cast<int>(durationString.length()), brew);
574 if (duration > 0)
575 {
576 int amplifier = 0;
577 auto ampIt = potionEffectAmplifier.find(effect->getId());
578 if (ampIt != potionEffectAmplifier.end())
579 {
580 wstring amplifierString = ampIt->second;
581 amplifier = parseEffectFormulaValue(amplifierString, 0, static_cast<int>(amplifierString.length()), brew);
582 if (amplifier < 0)
583 {
584 amplifier = 0;
585 }
586 }
587
588 if (effect->isInstantenous())
589 {
590 duration = 1;
591 }
592 else
593 {
594 // 3, 8, 13, 18.. minutes
595 duration = (SharedConstants::TICKS_PER_SECOND * 60) * (duration * 3 + (duration - 1) * 2);
596 duration >>= amplifier;
597 duration = static_cast<int>(Math::round((double)duration * effect->getDurationModifier()));
598
599 if ((brew & THROWABLE_MASK) != 0)
600 {
601 duration = static_cast<int>(Math::round((double)duration * .75 + .5));
602 }
603 }
604
605 if (list == nullptr)
606 {
607 list = new vector<MobEffectInstance *>();
608 }
609 MobEffectInstance *instance = new MobEffectInstance(effect->getId(), duration, amplifier);
610 if ((brew & THROWABLE_MASK) != 0) instance->setSplash(true);
611 list->push_back(instance);
612 }
613 }
614
615 return list;
616}

619int PotionBrewing::boil(int brew)
620{
621 if ((brew & 1) == 0)
622 {
623 return brew;
624 }
625
626 // save highest bit
627 int savedBit = NUM_BITS - 1;
628 while ((brew & (1 << savedBit)) == 0 && savedBit >= 0)
629 {
630 savedBit--;
631 }
632 // it's not possible to boil if there are no "empty slots" in front of
633 // the last bit
634 if (savedBit < 2 || (brew & (1 << (savedBit - 1))) != 0)
635 {
636 return brew;
637 }
638 if (savedBit >= 0)
639 {
640 brew &= ~(1 << savedBit);
641 }
642
643 brew <<= 1;
644
645 if (savedBit >= 0)
646 {
647 brew |= (1 << savedBit);

654int PotionBrewing::shake(int brew)
655{
656
657 // save highest bit
658 int savedBit = NUM_BITS - 1;
659 while ((brew & (1 << savedBit)) == 0 && savedBit >= 0)
660 {
661 savedBit--;
662 }
663 if (savedBit >= 0)
664 {
665 brew &= ~(1 << savedBit);
666 }
667
668 int currentResult = 0;
669 int nextResult = brew;
670
671 while (nextResult != currentResult)
672 {
673 nextResult = brew;
674 currentResult = 0;
675 // evaluate each bit
676 for (int bit = 0; bit < NUM_BITS; bit++)
677 {
678
679 bool on = isWrappedLit(brew, bit);
680 if (on)
681 {
682 if (!isWrappedLit(brew, bit + 1) && isWrappedLit(brew, bit + 2))
683 {
684 on = false;
685 }
686 else if (!isWrappedLit(brew, bit - 1) && isWrappedLit(brew, bit - 2))
687 {
688 on = false;
689 }
690 }
691 else
692 {
693 // turn on if both neighbors are on
694 on = isWrappedLit(brew, bit - 1) && isWrappedLit(brew, bit + 1);
695 }
696 if (on)
697 {
698 currentResult |= (1 << bit);
699 }
700 }
701 brew = currentResult;
702 }
703
704 if (savedBit >= 0)
705 {
706 currentResult |= (1 << savedBit);
707 }
708
709 return currentResult & BREW_MASK;
710}

722int PotionBrewing::applyBrewBit(int currentBrew, int bit, bool isNeg, bool isNot, bool isRequired)
723{
724 if (isRequired)
725 {
726 // 4J-JEV: I wanted to be able to specify that a
727 // bit is required to be false.
728 if (isLit(currentBrew, bit) == isNot)
729 {
730 return 0;
731 }
732 }
733 else if (isNeg)
734 {
735 currentBrew &= ~(1 << bit);
736 }
737 else if (isNot)
738 {
739 if ((currentBrew & (1 << bit)) == 0)
740 {
741 currentBrew |= (1 << bit);
742 }
743 else
744 {
745 currentBrew &= ~(1 << bit);
746 }
747 }
748 else
749 {
750 currentBrew |= (1 << bit);
751 }
752 return currentBrew;

755int PotionBrewing::applyBrew(int currentBrew, const wstring &formula)
756{
757
758 int start = 0;
759 int end = static_cast<int>(formula.length());
760
761 bool hasValue = false;
762 bool isNot = false;
763 bool isNeg = false;
764 bool isRequired = false;
765 int valuePart = 0;
766 for (int i = start; i < end; i++)
767 {
768 char current = formula.at(i);
769 if (current >= L'0' && current <= L'9')
770 {
771 valuePart *= 10;
772 valuePart += (int) (current - L'0');
773 hasValue = true;
774 }
775 else if (current == L'!')
776 {
777 if (hasValue)
778 {
779 currentBrew = applyBrewBit(currentBrew, valuePart, isNeg, isNot, isRequired);
780 hasValue = isNeg = isNot = isRequired = false;
781 valuePart = 0;
782 }
783
784 isNot = true;
785 }
786 else if (current == L'-')
787 {
788 if (hasValue)
789 {
790 currentBrew = applyBrewBit(currentBrew, valuePart, isNeg, isNot, isRequired);
791 hasValue = isNeg = isNot = isRequired = false;
792 valuePart = 0;
793 }
794
795 isNeg = true;
796 }
797 else if (current == L'+')
798 {
799 if (hasValue)
800 {
801 currentBrew = applyBrewBit(currentBrew, valuePart, isNeg, isNot, isRequired);
802 hasValue = isNeg = isNot = isRequired = false;
803 valuePart = 0;
804 }
805 }
806 else if (current == L'&')
807 {
808 if (hasValue)
809 {
810 currentBrew = applyBrewBit(currentBrew, valuePart, isNeg, isNot, isRequired);
811 hasValue = isNeg = isNot = isRequired = false;
812 valuePart = 0;
813 }
814 isRequired = true;
815 }
816 }
817 if (hasValue)
818 {
819 currentBrew = applyBrewBit(currentBrew, valuePart, isNeg, isNot, isRequired);
820 }
821
822 return currentBrew & BREW_MASK;
823}
