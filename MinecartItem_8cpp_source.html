11{

15 // Spawn the minecart 'just' outside the dispenser, it overlaps 2 'pixels' now.
16 // Also at half-block-height so it can connect with sloped rails
17 double spawnX = source->getX() + facing->getStepX() * (1 + 2.0f / 16);
18 double spawnY = source->getY() + facing->getStepY() * (1 + 2.0f / 16);
19 double spawnZ = source->getZ() + facing->getStepZ() * (1 + 2.0f / 16);
20
21 int frontX = source->getBlockX() + facing->getStepX();
22 int frontY = source->getBlockY() + facing->getStepY();
23 int frontZ = source->getBlockZ() + facing->getStepZ();
24 int inFront = world->getTile(frontX, frontY, frontZ);
25
26 // 4J: If we're at limit, just dispense item (instead of adding minecart)
27 if (world->countInstanceOf(eTYPE_MINECART, false) >= Level::MAX_CONSOLE_MINECARTS)
28 {
29 outcome = DISPENCED_ITEM;
30 return defaultDispenseItemBehavior.dispense(source, dispensed);
31 }
32
33 double yOffset;
34 if (BaseRailTile::isRail(inFront))
35 {
36 yOffset = 0;
37 }
38 else if (inFront == 0 && BaseRailTile::isRail(world->getTile(frontX, frontY - 1, frontZ)))
39 {
40 yOffset = -1;
41 }
42 else
43 {
44 outcome = DISPENCED_ITEM;
45 return defaultDispenseItemBehavior.dispense(source, dispensed);
46 }
47
48 outcome = ACTIVATED_ITEM;
49
50 shared_ptr<Minecart> minecart = Minecart::createMinecart(world, spawnX, spawnY + yOffset, spawnZ, static_cast<MinecartItem *>(dispensed->getItem())->type);
51 if (dispensed->hasCustomHoverName())
52 {
53 minecart->setCustomName(dispensed->getHoverName());
54 }
55 world->addEntity(minecart);
56
57 dispensed->remove(1);
58 return dispensed;
59}
