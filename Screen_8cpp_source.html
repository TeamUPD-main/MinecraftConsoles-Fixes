63{

67 size_t len = (str.length() + 1) * sizeof(wchar_t);
68 HGLOBAL h = GlobalAlloc(GMEM_MOVEABLE, len);

124void Screen::updateEvents()
125{
126#ifdef _WINDOWS64
127 // Poll mouse button state and dispatch click/release events
128 for (int btn = 0; btn < 3; btn++)
129 {
130 if (g_KBMInput.IsMouseButtonPressed(btn))
131 {
132 int xm = Mouse::getX() * width / minecraft->width;
133 int ym = height - Mouse::getY() * height / minecraft->height - 1;
134 mouseClicked(xm, ym, btn);
135 }
136 if (g_KBMInput.IsMouseButtonReleased(btn))
137 {
138 int xm = Mouse::getX() * width / minecraft->width;
139 int ym = height - Mouse::getY() * height / minecraft->height - 1;
140 mouseReleased(xm, ym, btn);
141 }
142 }
143
144 // Only drain WM_CHAR when this screen wants text input (e.g. ChatScreen); otherwise we'd steal keys from the game
145 if (dynamic_cast<ChatScreen*>(this) != nullptr)
146 {
147 wchar_t ch;
148 while (g_KBMInput.ConsumeChar(ch))
149 {
150 if (ch >= 0x20)
151 keyPressed(ch, -1);
152 else if (ch == 0x08)
153 keyPressed(0, Keyboard::KEY_BACK);
154 else if (ch == 0x0D)
155 keyPressed(0, Keyboard::KEY_RETURN);
156 }
157 }
158
159 // Arrow key repeat: deliver on first press (when key down and last==0) and while held (throttled)
160 static DWORD s_arrowLastTime[2] = { 0, 0 };
161 static bool s_arrowFirstRepeat[2] = { false, false };
162 const DWORD ARROW_REPEAT_DELAY_MS = 250;
163 const DWORD ARROW_REPEAT_INTERVAL_MS = 50;
164 DWORD now = GetTickCount();
165
166 // Poll keyboard events (special keys that may not come through WM_CHAR, e.g. Escape, arrows)
167 for (int vk = 0; vk < 256; vk++)
168 {
169 bool deliver = g_KBMInput.IsKeyPressed(vk);
170 if (vk == VK_LEFT || vk == VK_RIGHT)
171 {
172 int idx = (vk == VK_LEFT) ? 0 : 1;
173 if (!g_KBMInput.IsKeyDown(vk))
174 {
175 s_arrowLastTime[idx] = 0;
176 s_arrowFirstRepeat[idx] = false;
177 }
178 else
179 {
180 DWORD last = s_arrowLastTime[idx];
181 if (last == 0)
182 deliver = true;
183 else if (!deliver)
184 {
185 DWORD interval = s_arrowFirstRepeat[idx] ? ARROW_REPEAT_INTERVAL_MS : ARROW_REPEAT_DELAY_MS;
186 if ((now - last) >= interval)
187 {
188 deliver = true;
189 s_arrowFirstRepeat[idx] = true;
190 }
191 }
192 if (deliver)
193 s_arrowLastTime[idx] = now;
194 }
195 }
196 // Escape: deliver when key is down so we don't miss it (IsKeyPressed can be one frame late)
197 if (vk == VK_ESCAPE && g_KBMInput.IsKeyDown(VK_ESCAPE))
198 deliver = true;
199 if (!deliver) continue;
200
201 if (dynamic_cast<ChatScreen*>(this) != nullptr &&
202 (vk >= 'A' && vk <= 'Z' || vk >= '0' && vk <= '9' || vk == VK_SPACE || vk == VK_RETURN || vk == VK_BACK))
203 continue;
204 // Map to Screen::keyPressed
205 int mappedKey = -1;
206 wchar_t ch = 0;
207 //if (vk == VK_ESCAPE) mappedKey = Keyboard::KEY_ESCAPE;
208 if (vk == VK_ESCAPE) mappedKey = Keyboard::KEY_ESCAPE;
209 else if (vk == VK_RETURN) mappedKey = Keyboard::KEY_RETURN;
210 else if (vk == VK_BACK) mappedKey = Keyboard::KEY_BACK;
211 else if (vk == VK_UP) mappedKey = Keyboard::KEY_UP;
212 else if (vk == VK_DOWN) mappedKey = Keyboard::KEY_DOWN;
213 else if (vk == VK_LEFT) mappedKey = Keyboard::KEY_LEFT;
214 else if (vk == VK_RIGHT) mappedKey = Keyboard::KEY_RIGHT;
215 else if (vk == VK_LSHIFT || vk == VK_RSHIFT) mappedKey = Keyboard::KEY_LSHIFT;
216 else if (vk == VK_TAB) mappedKey = Keyboard::KEY_TAB;
217 else if (vk >= 'A' && vk <= 'Z')
218 {
219 ch = static_cast<wchar_t>(vk - 'A' + L'a');
220 if (g_KBMInput.IsKeyDown(VK_LSHIFT) || g_KBMInput.IsKeyDown(VK_RSHIFT)) ch = static_cast<wchar_t>(vk);
221 }
222 else if (vk >= '0' && vk <= '9') ch = static_cast<wchar_t>(vk);
223 else if (vk == VK_SPACE) ch = L' ';
224
225 if (mappedKey != -1) keyPressed(ch, mappedKey);
226 else if (ch != 0) keyPressed(ch, -1);
227 }
228#else
229 /* 4J - TODO
230 while (Mouse.next()) {
231 mouseEvent();
232 }
233
234 while (Keyboard.next()) {
235 keyboardEvent();
236 }
237 */
238#endif
239}
