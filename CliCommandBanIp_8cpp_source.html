93 bool CliCommandBanIp::Execute(const ServerCliParsedLine &line, ServerCliEngine *engine)
94 {
95 if (line.tokens.size() < 2)
96 {
97 engine->LogWarn("Usage: ban-ip <address|player> [reason ...]");
98 return false;
99 }
100 if (!ServerRuntime::Access::IsInitialized())
101 {
102 engine->LogWarn("Access manager is not initialized.");
103 return false;
104 }
105
106 const std::string targetToken = line.tokens[1];
107 std::string remoteIp;
108 // Match Java Edition behavior by accepting either a literal IP or an online player name.
109 const auto targetPlayer = engine->FindPlayerByNameUtf8(targetToken);
110 if (targetPlayer != nullptr)
111 {
112 if (!TryGetPlayerRemoteIp(targetPlayer, &remoteIp))
113 {
114 engine->LogWarn("Cannot ban that player's IP because no current remote IP is available.");
115 return false;
116 }
117 }
118 else if (NetworkUtils::IsIpLiteral(targetToken))
119 {
120 remoteIp = StringUtils::TrimAscii(targetToken);
121 }
122 else
123 {
124 engine->LogWarn("Unknown player or invalid IP address: " + targetToken);
125 return false;
126 }
127
128 // Refuse duplicate bans so operators get immediate feedback instead of rewriting the same entry.
129 if (ServerRuntime::Access::IsIpBanned(remoteIp))
130 {
131 engine->LogWarn("That IP address is already banned.");
132 return false;
133 }
134
135 ServerRuntime::Access::BanMetadata metadata = ServerRuntime::Access::BanManager::BuildDefaultMetadata("Console");
136 metadata.reason = StringUtils::JoinTokens(line.tokens, 2);
137 if (metadata.reason.empty())
138 {
139 metadata.reason = "Banned by an operator.";
140 }
141
142 // Publish the ban before disconnecting players so reconnect attempts are rejected immediately.
143 if (!ServerRuntime::Access::AddIpBan(remoteIp, metadata))
144 {
145 engine->LogError("Failed to write IP ban.");
146 return false;
147 }
148
149 const int disconnectedCount = DisconnectPlayersByRemoteIp(remoteIp);
150 // Report the resolved IP rather than the original token so player-name targets are explicit in the console.
151 engine->LogInfo("Banned IP address " + remoteIp + ".");
152 if (disconnectedCount > 0)
153 {
154 engine->LogInfo("Disconnected " + std::to_string(disconnectedCount) + " player(s) with that IP.");
155 }
156 return true;
