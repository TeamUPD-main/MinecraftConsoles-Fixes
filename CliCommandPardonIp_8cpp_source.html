32 bool CliCommandPardonIp::Execute(const ServerCliParsedLine &line, ServerCliEngine *engine)
33 {
34 if (line.tokens.size() != 2)
35 {
36 engine->LogWarn("Usage: pardon-ip <address>");
37 return false;
38 }
39 if (!ServerRuntime::Access::IsInitialized())
40 {
41 engine->LogWarn("Access manager is not initialized.");
42 return false;
43 }
44
45 // Java Edition pardon-ip only operates on a literal address, so do not resolve player names here.
46 const std::string ip = StringUtils::TrimAscii(line.tokens[1]);
47 if (!NetworkUtils::IsIpLiteral(ip))
48 {
49 engine->LogWarn("Invalid IP address: " + line.tokens[1]);
50 return false;
51 }
52 // Distinguish invalid input from a valid but currently unbanned address for clearer operator feedback.
53 if (!ServerRuntime::Access::IsIpBanned(ip))
54 {
55 engine->LogWarn("That IP address is not banned.");
56 return false;
57 }
58 if (!ServerRuntime::Access::RemoveIpBan(ip))
59 {
60 engine->LogError("Failed to remove IP ban.");
61 return false;
62 }
63
64 engine->LogInfo("Unbanned IP address " + ip + ".");
65 return true;
66 }
