105 vector<shared_ptr<Entity> > entities(levelEntities->begin(), levelEntities->end() );
106 Vec3 *center = Vec3::newTemp(x, y, z);
107
108 for ( auto& e : entities )
109 {
110 // 4J Stu - If the entity is not in a block that would be blown up, then they should not be damaged
111 // Fix for #46606 - TU5: Content: Gameplay: The player can be damaged and killed by explosions behind obsidian walls

123 if (dist <= 1)

126 double ya = e->y + e->getHeadHeight() - y;
127 double za = e->z - z;
128
129 double da = sqrt(xa * xa + ya * ya + za * za);
130
131 // 4J Stu - Added this check to remove divide by zero errors (or rather the issues caused by
132 // the values being set to NaN and used in comparisons a bit later on e.g. fireball)
133 if( da == 0 )

139 xa /= da;
140 ya /= da;
141 za /= da;
142 }
143

145 double pow = (1 - dist) * sp;
146 if(canDamage) e->hurt(DamageSource::explosion(this), static_cast<int>((pow * pow + pow) / 2 * 8 * r + 1));
147
148 double kbPower = ProtectionEnchantment::getExplosionKnockbackAfterDampener(e, pow);
149 e->xd += xa *kbPower;
150 e->yd += ya *kbPower;
151 e->zd += za *kbPower;
152
153
154 if (e->instanceof(eTYPE_PLAYER))
155 {
156 shared_ptr<Player> player = dynamic_pointer_cast<Player>(e);
157 //app.DebugPrintf("Adding player knockback (%f,%f,%f)\n", xa * pow, ya * pow, za * pow);

159 }
160 }
161 }
