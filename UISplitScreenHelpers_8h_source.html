36// Fit a 16:9 rectangle inside the given dimensions.
37inline void Fit16x9(F32 viewW, F32 viewH, S32 &fitW, S32 &fitH, S32 &offsetX, S32 &offsetY)
38{
39 const F32 kAspect = 16.0f / 9.0f;
40 if(viewW / viewH > kAspect)

42 fitH = (S32)viewH;
43 fitW = (S32)(viewH * kAspect);

61// Compute the uniform scale and tileYStart for split-screen tile rendering.
62// Used by HUD, Chat, and Tooltips to scale the SWF movie to cover the viewport tile.
63inline void ComputeTileScale(S32 tileWidth, S32 tileHeight, S32 movieWidth, S32 movieHeight,
64 bool needsYTile, F32 &outScale, S32 &outTileYStart)
65{
66 F32 scaleW = (F32)tileWidth / (F32)movieWidth;
67 F32 scaleH = (F32)tileHeight / (F32)movieHeight;
68 F32 scale = (scaleW > scaleH) ? scaleW : scaleH;
69 if(scale < 1.0f) scale = 1.0f;
70
71 outTileYStart = 0;
72 if(needsYTile)
73 {
74 S32 dispH = (S32)(movieHeight * scale);
75 outTileYStart = dispH - tileHeight;
76 if(outTileYStart < 0) outTileYStart = 0;
77 scaleH = (F32)(outTileYStart + tileHeight) / (F32)movieHeight;
78 scale = (scaleW > scaleH) ? scaleW : scaleH;
79 if(scale < 1.0f) scale = 1.0f;
80 }
81
82 outScale = scale;
83}
