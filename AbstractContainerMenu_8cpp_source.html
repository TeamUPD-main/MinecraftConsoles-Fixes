606bool AbstractContainerMenu::moveItemStackTo(shared_ptr<ItemInstance> itemStack, int startSlot, int endSlot, bool backwards)
607{
608 bool anythingChanged = false;
609
610 int destSlot = startSlot;
611 if (backwards)
612 {
613 destSlot = endSlot - 1;
614 }
615
616 // find stackable slots first
617 if (itemStack->isStackable())
618 {
619 while (itemStack->count > 0 && ((!backwards && destSlot < endSlot) || (backwards && destSlot >= startSlot)))
620 {
621
622 Slot *slot = slots.at(destSlot);
623 shared_ptr<ItemInstance> target = slot->getItem();
624 if (target != nullptr && target->id == itemStack->id && (!itemStack->isStackedByData() || itemStack->getAuxValue() == target->getAuxValue())
625 && ItemInstance::tagMatches(itemStack, target) )
626 {
627 int totalStack = target->count + itemStack->count;
628 if (totalStack <= itemStack->getMaxStackSize())
629 {
630 itemStack->count = 0;
631 target->count = totalStack;
632 slot->setChanged();
633 anythingChanged = true;
634 }
635 else if (target->count < itemStack->getMaxStackSize())
636 {
637 itemStack->count -= (itemStack->getMaxStackSize() - target->count);
638 target->count = itemStack->getMaxStackSize();
639 slot->setChanged();
640 anythingChanged = true;
641 }
642 }
643
644 if (backwards)
645 {
646 destSlot--;
647 }
648 else
649 {
650 destSlot++;
651 }
652 }
653 }
654
655 // find empty slot
656 if (itemStack->count > 0)
657 {
658 if (backwards)
659 {
660 destSlot = endSlot - 1;
661 }
662 else
663 {
664 destSlot = startSlot;
665 }
666 while ((!backwards && destSlot < endSlot) || (backwards && destSlot >= startSlot))
667 {
668 Slot *slot = slots.at(destSlot);
669 shared_ptr<ItemInstance> target = slot->getItem();
670
671 if (target == nullptr)
672 {
673 slot->set(itemStack->copy());
674 slot->setChanged();
