744void IUIScene_CreativeMenu::ScrollBar(UIVec2D pointerPos)
745{
746 UIVec2D pos;
747 UIVec2D size;
748 GetItemScreenData(eSectionInventoryCreativeSlider, 0, &pos, &size);
749 float fPosition = ((float)pointerPos.y - pos.y) / size.y;
750
751 // clamp
752 if(fPosition > 1)
753 fPosition = 1.0f;
754 else if(fPosition < 0)
755 fPosition = 0.0f;
756
757 // calculate page position according to page count
758 int iCurrentPage = Math::round(fPosition * (specs[m_curTab]->getPageCount() - 1));
759
760 // set tab page
761 m_tabPage[m_curTab] = iCurrentPage;
762
763 // update tab
764 switchTab(m_curTab);
765}
