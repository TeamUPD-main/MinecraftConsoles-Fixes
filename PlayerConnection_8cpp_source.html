458 }

472 int z = packet->z;

477 // position, so add 1.5 blocks
478 double yDist = player->y - (y + 0.5) + 1.5;
479 double zDist = player->z - (z + 0.5);
480 double dist = xDist * xDist + yDist * yDist + zDist * zDist;

483 return;
484 }
485 if (y >= server->getMaxBuildHeight())
486 {
487 return;
488 }
489 }
490

493 /* if (true) player->gameMode->startDestroyBlock(x, y, z, packet->face); // 4J - condition was !server->isUnderSpawnProtection(level, x, y, z, player) (from Java 1.6.4) but putting back to old behaviour

505 }
506 else if (packet->action == PlayerActionPacket::ABORT_DESTROY_BLOCK)

508 player->gameMode->abortDestroyBlock(x, y, z);
509 if (level->getTile(x, y, z) != 0) player->connection->send(std::make_shared<TileUpdatePacket>(x, y, z, level));
510 }
