132//necessary but nonexistent parent directories. Note that if this
133//operation fails it may have succeeded in creating some of the necessary

136//@return <code>true</code> if and only if the directory was created,

152 std::vector<std::wstring> path = stringSplit( m_abstractPathName, pathSeparator );

154 std::wstring pathToHere = L"";

166 // if not exists
