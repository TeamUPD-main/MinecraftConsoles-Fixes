154IDOC typedef enum gdraw_blendspecial
155{
156 GDRAW_BLENDSPECIAL_layer, // s
157 GDRAW_BLENDSPECIAL_multiply, // s*d
158 GDRAW_BLENDSPECIAL_screen, // sa*da - (da-d)*(sa-s)
159 GDRAW_BLENDSPECIAL_lighten, // max(sa*d,s*da)
160 GDRAW_BLENDSPECIAL_darken, // min(sa*d,s*da)
161 GDRAW_BLENDSPECIAL_add, // min(d+s,1.0)
162 GDRAW_BLENDSPECIAL_subtract, // max(d-s,0.0)
163 GDRAW_BLENDSPECIAL_difference, // abs(sa*d-s*da)
164 GDRAW_BLENDSPECIAL_invert, // sa*(da-d)
165 GDRAW_BLENDSPECIAL_overlay, // d < da/2.0 ? (2.0*s*d) : (sa*da - 2.0*(da-d)*(sa-s))
166 GDRAW_BLENDSPECIAL_hardlight, // s < sa/2.0 ? (2.0*s*d) : (sa*da - 2.0*(da-d)*(sa-s))
167
168 // these do extra-special math on the output alpha
169 GDRAW_BLENDSPECIAL_erase, // d*(1.0-sa)
170 GDRAW_BLENDSPECIAL_alpha_special, // d*sa
171
172 GDRAW_BLENDSPECIAL__count,
173} gdraw_blendspecial;
174/* Specifies a type of "special" blend mode, which is defined as one
175 that has to read from the framebuffer to compute its effect.
176
177 These modes are only used with a 1-to-1 textured quad containing
178 the exact output data in premultiplied alpha. They all need to
179 read from the framebuffer to compute their effect, so a GDraw
180 implementation will usually need a custom path to handle that.
181 Users will not warn in advance whether you're going to need this
182 operation, so implementations either need to always render to a
183 texture in case it happens, or copy the framebuffer to a texture
184 when it does.
185
186 Note that $(gdraw_blendspecial::GDRAW_BLENDSPECIAL_erase) and
187 $(gdraw_blendspecial::GDRAW_BLENDSPECIAL_alpha_special) are unique
188 among $gdraw_blendspecial modes in that they may not actually need
189 to be implemented with the destination input as a texture if
190 the destination buffer doesn't have an alpha channel. */
191
