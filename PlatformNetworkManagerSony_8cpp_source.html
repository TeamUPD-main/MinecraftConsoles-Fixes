438int CPlatformNetworkManagerSony::CorrectErrorIDS(int IDS)
439{
440 // Attempts to remap the following messages to provide something that PS3 TCRs are happier with
441 //
442 // IDS_CONNECTION_LOST - "Connection lost"
443 // IDS_CONNECTION_FAILED - "Connection failed"
444 // IDS_CONNECTION_LOST_LIVE - "Connection to "PSN" was lost. Exiting to the main menu."
445 // IDS_CONNECTION_LOST_LIVE_NO_EXIT - "Connection to "PSN" was lost."
446 // IDS_CONNECTION_LOST_SERVER - "Connection to the server was lost. Exiting to the main menu."
447 //
448 // Map to:
449 //
450 // IDS_ERROR_NETWORK - "A network error has occurred"
451 // IDS_ERROR_NETWORK_TITLE - "Network Error"
452 // IDS_ERROR_NETWORK_EXIT - "A network error has occurred. Exiting to Main Menu."
453 // IDS_ERROR_PSN_SIGN_OUT - You have been signed out from the "PSN".
454 // IDS_ERROR_PSN_SIGN_OUT_EXIT - You have been signed out from the "PSN". Exiting to Main Menu
455
456 // Determine if we'd prefer to present errors as a signing out issue, rather than a network issue, based on whether we have a network connection at all or not
457 bool preferSignoutError = false;
458 int state;
459
460#if defined __PSVITA__ // MGH - to fix devtrack #6258
461 if(!ProfileManager.IsSignedInPSN(ProfileManager.GetPrimaryPad()))
462 preferSignoutError = true;
463#elif defined __ORBIS__
464 if(!ProfileManager.isSignedInPSN(ProfileManager.GetPrimaryPad()))
465 preferSignoutError = true;
466#elif defined __PS3__
467 int ret = cellNetCtlGetState( &state );
468 int IPObtainedState = CELL_NET_CTL_STATE_IPObtained;
469 if( ret == 0 )
470 {
471 if( state == IPObtainedState )
472 {
473 preferSignoutError = true;
474 }
475 }
476#endif
477
478#ifdef __PSVITA__
479 // If we're in ad-hoc mode this problem definitely wasn't PSN related
480 if (usingAdhocMode()) preferSignoutError = false;
481#endif
482
483 // If we're the host we haven't lost connection to the server
484 if (IDS == IDS_CONNECTION_LOST_SERVER && g_NetworkManager.IsHost())
485 {
486 IDS = IDS_CONNECTION_LOST_LIVE;
487 }
488
489 switch(IDS)
490 {
491 case IDS_CONNECTION_LOST:
492 case IDS_CONNECTION_FAILED:
493 return IDS_ERROR_NETWORK_TITLE;
494 case IDS_CONNECTION_LOST_LIVE:
495 if( preferSignoutError )
496 {
497 return IDS_ERROR_PSN_SIGN_OUT_EXIT;
498 }
499 else
500 {
501 return IDS_ERROR_NETWORK_EXIT;
502 }
503 case IDS_CONNECTION_LOST_LIVE_NO_EXIT:
504 if( preferSignoutError )
505 {
506 return IDS_ERROR_PSN_SIGN_OUT;
507 }
508 else
509 {
510 return IDS_ERROR_NETWORK_TITLE;
511 }
512 break;
513#ifdef __PSVITA__
514 case IDS_CONNECTION_LOST_SERVER:
515 if(preferSignoutError)
516 {
517 if(ProfileManager.IsSignedInPSN(ProfileManager.GetPrimaryPad()) == false)
518 return IDS_ERROR_PSN_SIGN_OUT_EXIT;
519 }
520#endif
521 default:
522 return IDS;
523 }
524
525
