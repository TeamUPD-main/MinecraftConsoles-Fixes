252static void *start_write_dyn(DynBuffer *buf, U32 size)
253{
254 U8 *ptr = nullptr;
255
256 if (size > buf->size) {
257 IggyGDrawSendWarning(nullptr, "GDraw dynamic vertex buffer usage of %d bytes in one call larger than buffer size %d", size, buf->size);
258 return nullptr;
259 }
260
261 // update statistics
262 gdraw->dyn_maxalloc = RR_MAX(gdraw->dyn_maxalloc, size);
263
264 // invariant: current alloc_pos is in [0,size]
265 assert(buf->alloc_pos <= buf->size);
266
267 // wrap around when less than "size" bytes left in buffer
268 buf->write_pos = ((buf->size - buf->alloc_pos) < size) ? 0 : buf->alloc_pos;
269
270 // discard buffer whenever the current write position is 0;
271 // done this way so that if a DISCARD Map() were to fail, we would
272 // just keep retrying the next time around.
273 ptr = (U8 *) map_buffer(gdraw->d3d_context, buf->buffer, buf->write_pos == 0);
274 if (ptr) {
275 ptr += buf->write_pos; // we return pointer to write position in buffer
276 buf->alloc_pos = buf->write_pos + size; // bump alloc position
277 assert(buf->alloc_pos <= buf->size); // invariant again
278 }
279 // if map_buffer fails, it will have sent a warning
280
281 return ptr;
282}
283
284static U32 end_write_dyn(DynBuffer *buf)
285{
286 unmap_buffer(gdraw->d3d_context, buf->buffer);
287 return buf->write_pos;
288}

295static void stencil_state_cache_clear()
296{
297 S32 i;
298
299 for (i=0; i < STENCIL_STATE_CACHE_SIZE; ++i) {
300 gdraw->stencil_cache_key[i] = 0;
301 safe_release(gdraw->stencil_cache[i]);
302 gdraw->stencil_cache_lru[i] = 0;
303 }
304
305 gdraw->stencil_cache_now = 0;
306}
307
308static ID3D1X(DepthStencilState) *stencil_state_cache_lookup(rrbool set_id, rrbool test_id, U8 read_mask, U8 write_mask)
309{
310 D3D1X_(DEPTH_STENCIL_DESC) desc;
311 S32 i, best = 0;
312 U32 key = (set_id << 1) | test_id | (read_mask << 8) | (write_mask << 16);
313 U32 now, age, highest_age;
314 HRESULT hr;
315
316 // for LRU
317 now = ++gdraw->stencil_cache_now;
318
319 // do we have this in the cache?
320 for (i=0; i < STENCIL_STATE_CACHE_SIZE; ++i) {
321 if (gdraw->stencil_cache_key[i] == key) {
322 gdraw->stencil_cache_lru[i] = now;
323 return gdraw->stencil_cache[i];
324 }
325 }
326
327 // not in the cache, find the best slot to replace it with (LRU)
328 highest_age = 0;
329 for (i=0; i < STENCIL_STATE_CACHE_SIZE; ++i) {
330 if (!gdraw->stencil_cache[i]) { // unused slot!
331 best = i;
332 break;
333 }
334
335 age = now - gdraw->stencil_cache_lru[i];
336 if (age > highest_age) {
337 highest_age = age;
338 best = i;
339 }
340 }
341
342 // release old depth/stencil state at that position and create new one
343 safe_release(gdraw->stencil_cache[best]);
344
345 gdraw->depth_state[set_id][test_id]->GetDesc(&desc); // reference state
346 desc.StencilEnable = TRUE;
347 desc.StencilReadMask = read_mask;
348 desc.StencilWriteMask = write_mask;
349 desc.FrontFace.StencilFailOp = D3D1X_(STENCIL_OP_KEEP);
350 desc.FrontFace.StencilDepthFailOp = D3D1X_(STENCIL_OP_KEEP);
351 desc.FrontFace.StencilPassOp = D3D1X_(STENCIL_OP_REPLACE);
352 desc.FrontFace.StencilFunc = D3D1X_(COMPARISON_EQUAL);
353 desc.BackFace.StencilFailOp = D3D1X_(STENCIL_OP_KEEP);
354 desc.BackFace.StencilDepthFailOp = D3D1X_(STENCIL_OP_KEEP);
355 desc.BackFace.StencilPassOp = D3D1X_(STENCIL_OP_REPLACE);
356 desc.BackFace.StencilFunc = D3D1X_(COMPARISON_EQUAL);
357
358 hr = gdraw->d3d_device->CreateDepthStencilState(&desc, &gdraw->stencil_cache[best]);
359 if (FAILED(hr))
360 report_d3d_error(hr, "CreateDepthStencilState", "");
361
362 gdraw->stencil_cache_key[best] = key;
363 gdraw->stencil_cache_lru[best] = now;
364 return gdraw->stencil_cache[best];
365}

408static rrbool RADLINK gdraw_MakeTextureBegin(void *owner, S32 width, S32 height, gdraw_texture_format format, U32 flags, GDraw_MakeTexture_ProcessingInfo *p, GDrawStats *stats)
409{
410 GDrawHandle *t = nullptr;
411 DXGI_FORMAT dxgi_fmt;
412 S32 bpp, size = 0, nmips = 0;
413
414 if (width >= 16384 || height >= 16384) {
415 IggyGDrawSendWarning(nullptr, "GDraw texture size too large (%d x %d), dimension limit is 16384", width, height);
416 return false;
417 }
418
419 if (format == GDRAW_TEXTURE_FORMAT_rgba32) {
420 dxgi_fmt = DXGI_FORMAT_R8G8B8A8_UNORM;
421 bpp = 4;
422 } else {
423 dxgi_fmt = DXGI_FORMAT_R8_UNORM;
424 bpp = 1;
425 }
426
427 // compute estimated size of texture in video memory
428 do {
429 size += RR_MAX(width >> nmips, 1) * RR_MAX(height >> nmips, 1) * bpp;
430 ++nmips;
431 } while ((flags & GDRAW_MAKETEXTURE_FLAGS_mipmap) && ((width >> nmips) || (height >> nmips)));
432
433 // try to allocate memory for the client to write to
434 p->texture_data = (U8 *) IggyGDrawMalloc(size);
435 if (!p->texture_data) {
436 IggyGDrawSendWarning(nullptr, "GDraw out of memory to store texture data to pass to D3D for %d x %d texture", width, height);
437 return false;
438 }
439
440 // allocate a handle and make room in the cache for this much data
441 t = gdraw_res_alloc_begin(gdraw->texturecache, size, stats);
442 if (!t) {
443 IggyGDrawFree(p->texture_data);
444 return false;
445 }
446
447 t->handle.tex.w = width;
448 t->handle.tex.h = height;
449 t->handle.tex.d3d = nullptr;
450 t->handle.tex.d3d_view = nullptr;
451 t->handle.tex.d3d_rtview = nullptr;
452
453 p->texture_type = GDRAW_TEXTURE_TYPE_rgba;
454 p->p0 = t;
455 p->p1 = owner;
456 p->i0 = width;
457 p->i1 = height;
458 p->i2 = flags;
459 p->i3 = dxgi_fmt;
460 p->i4 = size;
461 p->i5 = nmips;
462 p->i6 = bpp;
463
464 p->stride_in_bytes = width * bpp;
465 p->num_rows = height;
466
467 return true;
468}

475static GDrawTexture * RADLINK gdraw_MakeTextureEnd(GDraw_MakeTexture_ProcessingInfo *p, GDrawStats *stats)
476{
477 GDrawHandle *t = (GDrawHandle *) p->p0;
478 D3D1X_(SUBRESOURCE_DATA) mipdata[24];
479 S32 i, w, h, nmips, bpp;
480 HRESULT hr = S_OK;
481 char *failed_call;
482 U8 *ptr;
483
484 // generate mip maps and set up descriptors for them
485 assert(p->i5 <= 24);
486 ptr = p->texture_data;
487 w = p->i0;
488 h = p->i1;
489 nmips = p->i5;
490 bpp = p->i6;
491
492 for (i=0; i < nmips; ++i) {
493 mipdata[i].pSysMem = ptr;
494 mipdata[i].SysMemPitch = RR_MAX(w >> i, 1) * bpp;
495 mipdata[i].SysMemSlicePitch = 0;
496 ptr += mipdata[i].SysMemPitch * RR_MAX(h >> i, 1);
497
498 // create mip data by downsampling
499 if (i)
500 gdraw_Downsample((U8 *) mipdata[i].pSysMem, mipdata[i].SysMemPitch, w >> i, h >> i,
501 (U8 *) mipdata[i-1].pSysMem, mipdata[i-1].SysMemPitch, bpp);
502 }
503
504 // actually create texture
505 D3D1X_(TEXTURE2D_DESC) desc = { w, h, nmips, 1, (DXGI_FORMAT) p->i3, { 1, 0 },
506 (p->i2 & GDRAW_MAKETEXTURE_FLAGS_updatable) ? D3D1X_(USAGE_DEFAULT) : D3D1X_(USAGE_IMMUTABLE),
507 D3D1X_(BIND_SHADER_RESOURCE), 0, 0 };
508
509 failed_call = "CreateTexture2D";
510 hr = gdraw->d3d_device->CreateTexture2D(&desc, mipdata, &t->handle.tex.d3d);
511 if (FAILED(hr)) goto done;
512
513 // and create a corresponding shader resource view
514 failed_call = "CreateShaderResourceView";
515 hr = gdraw->d3d_device->CreateShaderResourceView(t->handle.tex.d3d, nullptr, &t->handle.tex.d3d_view);
516
517done:
518 if (!FAILED(hr)) {
519 gdraw_HandleCacheAllocateEnd(t, p->i4, p->p1, (p->i2 & GDRAW_MAKETEXTURE_FLAGS_never_flush) ? GDRAW_HANDLE_STATE_pinned : GDRAW_HANDLE_STATE_locked);
520 stats->nonzero_flags |= GDRAW_STATS_alloc_tex;
521 stats->alloc_tex += 1;
522 stats->alloc_tex_bytes += p->i4;
523 } else {
524 safe_release(t->handle.tex.d3d);
525 safe_release(t->handle.tex.d3d_view);
526
527 gdraw_HandleCacheAllocateFail(t);
528 t = nullptr;
529 report_d3d_error(hr, failed_call, " while creating texture");
530 }
531
532 IggyGDrawFree(p->texture_data);
533 return (GDrawTexture *) t;
534}

701static GDrawHandle *get_color_rendertarget(GDrawStats *stats)
702{
703 char *failed_call;
704
705 // try to recycle LRU rendertarget
706 GDrawHandle *t = gdraw_HandleCacheGetLRU(&gdraw->rendertargets);
707 if (t) {
708 gdraw_HandleCacheLock(t, (void *) 1);
709 return t;
710 }
711
712 // ran out of RTs, allocate a new one
713 S32 size = gdraw->frametex_width * gdraw->frametex_height * 4;
714 if (gdraw->rendertargets.bytes_free < size) {
715 IggyGDrawSendWarning(nullptr, "GDraw rendertarget allocation failed: hit size limit of %d bytes", gdraw->rendertargets.total_bytes);
716 return nullptr;
717 }
718
719 t = gdraw_HandleCacheAllocateBegin(&gdraw->rendertargets);
720 if (!t) {
721 IggyGDrawSendWarning(nullptr, "GDraw rendertarget allocation failed: hit handle limit");
722 return t;
723 }
724
725 D3D1X_(TEXTURE2D_DESC) desc = { gdraw->frametex_width, gdraw->frametex_height, 1, 1, DXGI_FORMAT_R8G8B8A8_UNORM, { 1, 0 },
726 D3D1X_(USAGE_DEFAULT), D3D1X_(BIND_SHADER_RESOURCE) | D3D1X_(BIND_RENDER_TARGET), 0, 0 };
727
728 t->handle.tex.d3d = nullptr;
729 t->handle.tex.d3d_view = nullptr;
730 t->handle.tex.d3d_rtview = nullptr;
731
732 HRESULT hr = gdraw->d3d_device->CreateTexture2D(&desc, nullptr, &t->handle.tex.d3d);
733 failed_call = "CreateTexture2D";
734 if (!FAILED(hr)) {
735 hr = gdraw->d3d_device->CreateShaderResourceView(t->handle.tex.d3d, nullptr, &t->handle.tex.d3d_view);
736 failed_call = "CreateTexture2D";
737 }
738 if (!FAILED(hr)) {
739 hr = gdraw->d3d_device->CreateRenderTargetView(t->handle.tex.d3d, nullptr, &t->handle.tex.d3d_rtview);
740 failed_call = "CreateRenderTargetView";
741 }
742
743 if (FAILED(hr)) {
744 safe_release(t->handle.tex.d3d);
745 safe_release(t->handle.tex.d3d_view);
746 safe_release(t->handle.tex.d3d_rtview);
747 gdraw_HandleCacheAllocateFail(t);
748
749 report_d3d_error(hr, failed_call, " creating rendertarget");
750
751 return nullptr;
752 }
753
754 gdraw_HandleCacheAllocateEnd(t, size, (void *) 1, GDRAW_HANDLE_STATE_locked);
755 stats->nonzero_flags |= GDRAW_STATS_alloc_tex;
756 stats->alloc_tex += 1;
757 stats->alloc_tex_bytes += size;
758
759 return t;
760}

1311static int set_renderstate_full(S32 vertex_format, GDrawRenderState *r, GDrawStats * /* stats */, const F32 *rescale1)
1312{
1313 ID3D1XContext *d3d = gdraw->d3d_context;
1314
1315 // set vertex shader
1316 set_vertex_shader(d3d, gdraw->vert[vertex_format].vshader);
1317
1318 // set vertex shader constants
1319 if (VertexVars *vvars = (VertexVars *) map_buffer(gdraw->d3d_context, gdraw->cb_vertex, true)) {
1320 F32 depth = depth_from_id(r->id);
1321 if (!r->use_world_space)
1322 gdraw_ObjectSpace(vvars->world[0], r->o2w, depth, 0.0f);
1323 else
1324 gdraw_WorldSpace(vvars->world[0], gdraw->world_to_pixel, depth, 0.0f);
1325
1326 memcpy(&vvars->x_off, r->edge_matrix, 4*sizeof(F32));
1327
1328 if (r->texgen0_enabled) {
1329 memcpy(&vvars->texgen_s, r->s0_texgen, 4*sizeof(F32));
1330 memcpy(&vvars->texgen_t, r->t0_texgen, 4*sizeof(F32));
1331 }
1332
1333 if (gdraw->use_3d)
1334 memcpy(vvars->x3d, gdraw->xform_3d, 12*sizeof(F32));
1335 else
1336 memcpy(vvars->x3d, gdraw->projmat, 12*sizeof(F32));
1337
1338 unmap_buffer(gdraw->d3d_context, gdraw->cb_vertex);
1339
1340 d3d->VSSetConstantBuffers(0, 1, &gdraw->cb_vertex);
1341 }
1342
1343 // set the blend mode
1344 int blend_mode = r->blend_mode;
1345 if (blend_mode != gdraw->blend_mode) {
1346 gdraw->blend_mode = blend_mode;
1347 d3d->OMSetBlendState(gdraw->blend_state[blend_mode], four_zeros, ~0u);
1348 }
1349
1350 // set the fragment program
1351 if (blend_mode != GDRAW_BLEND_special) {
1352 int which = r->tex0_mode;
1353 assert(which >= 0 && which < sizeof(gdraw->fprog) / sizeof(*gdraw->fprog));
1354
1355 int additive = 0;
1356 if (r->cxf_add) {
1357 additive = 1;
1358 if (r->cxf_add[3]) additive = 2;
1359 }
1360
1361 ID3D1X(PixelShader) *program = gdraw->fprog[which][additive].pshader;
1362 if (r->stencil_set) {
1363 // in stencil set mode, prefer not doing any shading at all
1364 // but if alpha test is on, we need to make an exception
1365
1366#ifndef GDRAW_D3D11_LEVEL9 // level9 can't do nullptr PS it seems
1367 if (which != GDRAW_TEXTURE_alpha_test)
1368 program = nullptr;
1369 else
1370#endif
1371 {
1372 gdraw->blend_mode = -1;
1373 d3d->OMSetBlendState(gdraw->blend_no_color_write, four_zeros, ~0u);
1374 }
1375 }
1376
1377 set_pixel_shader(d3d, program);
1378 } else
1379 set_pixel_shader(d3d, gdraw->exceptional_blend[r->special_blend].pshader);
1380
1381 set_texture(0, r->tex[0], r->nearest0, r->wrap0);
1382
1383 // pixel shader constants
1384 if (PixelCommonVars *pvars = (PixelCommonVars *) map_buffer(gdraw->d3d_context, gdraw->cb_ps_common, true)) {
1385 memcpy(pvars->color_mul, r->color, 4*sizeof(float));
1386
1387 if (r->cxf_add) {
1388 pvars->color_add[0] = r->cxf_add[0] / 255.0f;
1389 pvars->color_add[1] = r->cxf_add[1] / 255.0f;
1390 pvars->color_add[2] = r->cxf_add[2] / 255.0f;
1391 pvars->color_add[3] = r->cxf_add[3] / 255.0f;
1392 } else
1393 pvars->color_add[0] = pvars->color_add[1] = pvars->color_add[2] = pvars->color_add[3] = 0.0f;
1394
1395 if (r->tex0_mode == GDRAW_TEXTURE_focal_gradient) memcpy(pvars->focal, r->focal_point, 4*sizeof(float));
1396 if (r->blend_mode == GDRAW_BLEND_special) memcpy(pvars->rescale1, rescale1, 4*sizeof(float));
1397 unmap_buffer(gdraw->d3d_context, gdraw->cb_ps_common);
1398 d3d->PSSetConstantBuffers(0, 1, &gdraw->cb_ps_common);
1399 }
1400
1401 // Set pixel operation states
1402 if (r->scissor) {
1403 D3D1X_(RECT) s;
1404 gdraw->scissor_state = 1;
1405 if (gdraw->cur == gdraw->frame) {
1406 s.left = r->scissor_rect.x0 + gdraw->vx - gdraw->tx0;
1407 s.top = r->scissor_rect.y0 + gdraw->vy - gdraw->ty0;
1408 s.right = r->scissor_rect.x1 + gdraw->vx - gdraw->tx0;
1409 s.bottom = r->scissor_rect.y1 + gdraw->vy - gdraw->ty0;
1410 } else {
1411 s.left = r->scissor_rect.x0 - gdraw->tx0p;
1412 s.top = r->scissor_rect.y0 - gdraw->ty0p;
1413 s.right = r->scissor_rect.x1 - gdraw->tx0p;
1414 s.bottom = r->scissor_rect.y1 - gdraw->ty0p;
1415 }
1416 d3d->RSSetScissorRects(1, &s);
1417 } else if (r->scissor != gdraw->scissor_state)
1418 disable_scissor(0);
1419
1420 if (r->stencil_set | r->stencil_test)
1421 d3d->OMSetDepthStencilState(stencil_state_cache_lookup(r->set_id, r->test_id, r->stencil_test, r->stencil_set), 255);
1422 else
1423 d3d->OMSetDepthStencilState(gdraw->depth_state[r->set_id][r->test_id], 0);
1424
1425 return 1;
1426}

1487static void RADLINK gdraw_DrawIndexedTriangles(GDrawRenderState *r, GDrawPrimitive *p, GDrawVertexBuffer *buf, GDrawStats *stats)
1488{
1489 ID3D1XContext *d3d = gdraw->d3d_context;
1490 GDrawHandle *vb = (GDrawHandle *) buf;
1491 int vfmt = p->vertex_format;
1492 assert(vfmt >= 0 && vfmt < GDRAW_vformat__count);
1493
1494 if (!set_renderstate(vfmt, r, stats))
1495 return;
1496
1497 UINT stride = vertsize[vfmt];
1498 d3d->IASetInputLayout(gdraw->inlayout[vfmt]);
1499
1500 if (vb) {
1501 UINT offs = (UINT) (UINTa) p->vertices;
1502
1503 d3d->IASetVertexBuffers(0, 1, &vb->handle.vbuf.verts, &stride, &offs);
1504 d3d->IASetIndexBuffer(vb->handle.vbuf.inds, DXGI_FORMAT_R16_UINT, (UINT) (UINTa) p->indices);
1505 d3d->DrawIndexed(p->num_indices, 0, 0);
1506 } else if (p->indices) {
1507 U32 vbytes = p->num_vertices * stride;
1508 U32 ibytes = p->num_indices * 2;
1509
1510 if (void *vbptr = start_write_dyn(&gdraw->dyn_vb, vbytes)) {
1511 memcpy(vbptr, p->vertices, vbytes);
1512 UINT vboffs = end_write_dyn(&gdraw->dyn_vb);
1513
1514 if (void *ibptr = start_write_dyn(&gdraw->dyn_ib, ibytes)) {
1515 memcpy(ibptr, p->indices, ibytes);
1516 UINT iboffs = end_write_dyn(&gdraw->dyn_ib);
1517
1518 d3d->IASetVertexBuffers(0, 1, &gdraw->dyn_vb.buffer, &stride, &vboffs);
1519 d3d->IASetIndexBuffer(gdraw->dyn_ib.buffer, DXGI_FORMAT_R16_UINT, iboffs);
1520 d3d->DrawIndexed(p->num_indices, 0, 0);
1521 }
1522 }
1523 } else { // dynamic quads
1524 assert(p->num_vertices % 4 == 0);
1525 d3d->IASetIndexBuffer(gdraw->quad_ib, DXGI_FORMAT_R16_UINT, 0);
1526
1527 if (gdraw->max_quad_vert_count) {
1528 S32 pos = 0;
1529 while (pos < p->num_vertices) {
1530 S32 vert_count = RR_MIN(p->num_vertices - pos, gdraw->max_quad_vert_count);
1531 UINT chunk_bytes = vert_count * stride;
1532
1533 if (void *vbptr = start_write_dyn(&gdraw->dyn_vb, chunk_bytes)) {
1534 memcpy(vbptr, (U8 *)p->vertices + pos*stride, chunk_bytes);
1535 UINT offs = end_write_dyn(&gdraw->dyn_vb);
1536
1537 d3d->IASetVertexBuffers(0, 1, &gdraw->dyn_vb.buffer, &stride, &offs);
1538 d3d->DrawIndexed((vert_count >> 2) * 6, 0, 0);
1539 }
1540 pos += vert_count;
1541 }
1542 }
1543 }
1544
1545 tag_resources(vb, r->tex[0], r->tex[1]);
1546
1547 stats->nonzero_flags |= GDRAW_STATS_batches;
1548 stats->num_batches += 1;
1549 stats->drawn_indices += p->num_indices;
1550 stats->drawn_vertices += p->num_vertices;
1551}

1702 set_pixel_constant(para->color2, r->shader_data[8], r->shader_data[9], r->shader_data[10], r->shader_data[11]);

1704 end_ps_constants(gdraw->cb_filter);

1709}

1711static void RADLINK gdraw_FilterQuad(GDrawRenderState *r, S32 x0, S32 y0, S32 x1, S32 y1, GDrawStats *stats)
1712{
1713 ID3D1XContext *d3d = gdraw->d3d_context;
1714 F32 tc[4];
1715 gswf_recti s;
1716
1717 // clip to tile boundaries
1718 s.x0 = RR_MAX(x0, gdraw->tx0p);
1719 s.y0 = RR_MAX(y0, gdraw->ty0p);
1720 s.x1 = RR_MIN(x1, gdraw->tx0p + gdraw->tpw);
1721 s.y1 = RR_MIN(y1, gdraw->ty0p + gdraw->tph);
1722 if (s.x1 < s.x0 || s.y1 < s.y0)
1723 return;
1724
1725 tc[0] = (s.x0 - gdraw->tx0p) / (F32) gdraw->frametex_width;
1726 tc[1] = (s.y0 - gdraw->ty0p) / (F32) gdraw->frametex_height;
1727 tc[2] = (s.x1 - gdraw->tx0p) / (F32) gdraw->frametex_width;
1728 tc[3] = (s.y1 - gdraw->ty0p) / (F32) gdraw->frametex_height;
1729
1730 // clear to known render state
1731 d3d->OMSetBlendState(gdraw->blend_state[GDRAW_BLEND_none], four_zeros, ~0u);
1732 d3d->OMSetDepthStencilState(gdraw->depth_state[0][0], 0);
1733 disable_scissor(0);
1734 gdraw->blend_mode = GDRAW_BLEND_none;
1735
1736 if (r->blend_mode == GDRAW_BLEND_filter) {
1737 switch (r->filter) {
1738 case GDRAW_FILTER_blur: {
1739 GDrawBlurInfo b;
1740 gswf_recti bounds = *get_valid_rect(r->tex[0]);
1741 gdraw_ShiftRect(&s, &s, -gdraw->tx0p, -gdraw->ty0p); // blur uses physical rendertarget coordinates
1742
1743 b.BlurPass = gdraw_DriverBlurPass;
1744 b.w = gdraw->tpw;
1745 b.h = gdraw->tph;
1746 b.frametex_width = gdraw->frametex_width;
1747 b.frametex_height = gdraw->frametex_height;
1748
1749 // blur needs to draw with multiple passes, so set up special state
1750 gdraw->in_blur = true;
1751
1752 // do the blur
1753 gdraw_Blur(&gdraw_funcs, &b, r, &s, &bounds, stats);
1754
1755 // restore the normal state
1756 gdraw->in_blur = false;
1757 set_viewport();
1758 set_projection();
1759 break;
1760 }
1761
1762 case GDRAW_FILTER_colormatrix:
1763 gdraw_Colormatrix(r, &s, tc, stats);
1764 break;
1765
1766 case GDRAW_FILTER_dropshadow:
1767 gdraw_Filter(r, &s, tc, 0, stats);
1768 break;
1769
1770 case GDRAW_FILTER_bevel:
1771 gdraw_Filter(r, &s, tc, 1, stats);
1772 break;
1773
1774 default:
1775 assert(0);
1776 }
1777 } else {
1778 GDrawHandle *blend_tex = nullptr;
1779
1780 // for crazy blend modes, we need to read back from the framebuffer
1781 // and do the blending in the pixel shader. we do this with copies
1782 // rather than trying to render-to-texture-all-along, because we want
1783 // to be able to render over the user's existing framebuffer, which might
1784 // not be a texture. note that this isn't optimal when MSAA is on!
1785 F32 rescale1[4] = { 1.0f, 1.0f, 0.0f, 0.0f };
1786 if (r->blend_mode == GDRAW_BLEND_special) {
1787 ID3D1XContext *d3d = gdraw->d3d_context;
1788 ID3D1X(Resource) *cur_rt_rsrc;
1789 get_active_render_target()->GetResource(&cur_rt_rsrc);
1790
1791 if (gdraw->cur == gdraw->frame && gdraw->main_msaa) {
1792 // source surface is main framebuffer and it uses MSAA. just resolve it first.
1793 D3D1X_(SHADER_RESOURCE_VIEW_DESC) desc;
1794 D3D1X_(TEXTURE2D_DESC) texdesc;
1795 ID3D1X(Texture2D) *resolve_tex;
1796
1797 gdraw->main_resolve_target->GetDesc(&desc);
1798 gdraw->main_resolve_target->GetResource((ID3D1X(Resource) **) &resolve_tex);
1799 resolve_tex->GetDesc(&texdesc);
1800 d3d->ResolveSubresource(resolve_tex, 0, cur_rt_rsrc, 0, desc.Format);
1801 resolve_tex->Release();
1802
1803 stats->nonzero_flags |= GDRAW_STATS_blits;
1804 stats->num_blits += 1;
1805 stats->num_blit_pixels += texdesc.Width * texdesc.Height;
1806
1807 d3d->PSSetShaderResources(1, 1, &gdraw->main_resolve_target);
1808 d3d->PSSetSamplers(1, 1, &gdraw->sampler_state[0][GDRAW_WRAP_clamp]);
1809
1810 // calculate texture coordinate remapping
1811 rescale1[0] = gdraw->frametex_width / (F32) texdesc.Width;
1812 rescale1[1] = gdraw->frametex_height / (F32) texdesc.Height;
1813 rescale1[2] = (gdraw->vx - gdraw->tx0 + gdraw->tx0p) / (F32) texdesc.Width;
1814 rescale1[3] = (gdraw->vy - gdraw->ty0 + gdraw->ty0p) / (F32) texdesc.Height;
1815 } else {
1816 D3D1X_(BOX) box = { 0,0,0,0,0,1 };
1817 S32 dx = 0, dy = 0;
1818 blend_tex = get_color_rendertarget(stats);
1819
1820 if (gdraw->cur != gdraw->frame)
1821 box.right=gdraw->tpw, box.bottom=gdraw->tph;
1822 else {
1823 box.left=gdraw->vx, box.top=gdraw->vy, box.right=gdraw->vx+gdraw->tw, box.bottom=gdraw->vy+gdraw->th;
1824 dx = gdraw->tx0 - gdraw->tx0p;
1825 dy = gdraw->ty0 - gdraw->ty0p;
1826 }
1827
1828 d3d->CopySubresourceRegion(blend_tex->handle.tex.d3d, 0, dx, dy, 0,
1829 cur_rt_rsrc, 0, &box);
1830
1831 stats->nonzero_flags |= GDRAW_STATS_blits;
1832 stats->num_blits += 1;
1833 stats->num_blit_pixels += (box.right - box.left) * (box.bottom - box.top);
1834
1835 set_texture(1, (GDrawTexture *) blend_tex, false, GDRAW_WRAP_clamp);
1836 }
1837
1838 cur_rt_rsrc->Release();
1839 }
1840
1841 if (!set_renderstate_full(GDRAW_vformat_v2tc2, r, stats, rescale1))
1842 return;
1843
1844 do_screen_quad(&s, tc, stats);
1845 tag_resources(r->tex[0], r->tex[1]);
1846 if (blend_tex)
1847 gdraw_FreeTexture((GDrawTexture *) blend_tex, 0, stats);
1848 }
1849}

2109static bool alloc_dynbuffer(U32 size)
2110{
2111 // specified input size is vertex buffer size. determine sensible size for the
2112 // corresponding index buffer. iggy always uses 16-bit indices and has three
2113 // primary types of geometry it sends:
2114 //
2115 // 1. filled polygons. these are triangulated simple polygons and thus have
2116 // roughly as many triangles as they have vertices. they use either 8- or
2117 // 16-byte vertex formats; this makes a worst case of 6 bytes of indices
2118 // for every 8 bytes of vertex data.
2119 // 2. strokes and edge antialiasing. they use a 16-byte vertex format and
2120 // worst-case write a "double quadstrip" which has 4 triangles for every
2121 // 3 vertices, which means 24 bytes of index data for every 48 bytes
2122 // of vertex data.
2123 // 3. textured quads. they use a 16-byte vertex format, have exactly 2
2124 // triangles for every 4 vertices, and use either a static index buffer
2125 // (quad_ib) or a single triangle strip, so for our purposes they need no
2126 // space to store indices at all.
2127 //
2128 // 1) argues for allocating index buffers at 3/4 the size of the corresponding
2129 // vertex buffer, while 2) and 3) need 1/2 the size of the vertex buffer or less.
2130 // 2) and 3) are the most common types of vertex data, while 1) is used only for
2131 // morphed shapes and in certain cases when the RESOURCE_vertexbuffer pool is full.
2132 // we just play it safe anyway and make sure we size the IB large enough to cover
2133 // the worst case for 1). this is conservative, but it probably doesn't matter much.
2134 U32 ibsize = (size * 3) / 4;
2135
2136 init_dyn_buffer(&gdraw->dyn_vb, size, D3D1X_(BIND_VERTEX_BUFFER));
2137 init_dyn_buffer(&gdraw->dyn_ib, ibsize, D3D1X_(BIND_INDEX_BUFFER));
2138
2139 gdraw->max_quad_vert_count = RR_MIN(size / sizeof(gswf_vertex_xyst), QUAD_IB_COUNT * 4);
2140 gdraw->max_quad_vert_count &= ~3; // must be multiple of four
2141
2142 return gdraw->dyn_vb.buffer != nullptr && gdraw->dyn_ib.buffer != nullptr;
2143}

2145int gdraw_D3D1X_(SetResourceLimits)(gdraw_resourcetype type, S32 num_handles, S32 num_bytes)
2146{
2147 GDrawStats stats={0};

2150 num_handles = MAX_RENDER_STACK_DEPTH + 1;

2152 assert(type >= GDRAW_D3D1X_(RESOURCE_rendertarget) && type < GDRAW_D3D1X_(RESOURCE__count));
2153 assert(num_handles >= 0);
2154 assert(num_bytes >= 0);
2155
2156 // nothing to do if the values are unchanged
2157 if (gdraw_limits[type].num_handles == num_handles &&
2158 gdraw_limits[type].num_bytes == num_bytes)
2159 return 1;
2160
2161 gdraw_limits[type].num_handles = num_handles;
2162 gdraw_limits[type].num_bytes = num_bytes;
2163

2165 if (!gdraw)
2166 return 1;
2167
2168 // resize the appropriate pool
2169 switch (type) {
2170 case GDRAW_D3D1X_(RESOURCE_rendertarget):
2171 flush_rendertargets(&stats);
2172 gdraw_HandleCacheInit(&gdraw->rendertargets, num_handles, num_bytes);
2173 return 1;
2174
2175 case GDRAW_D3D1X_(RESOURCE_texture):
2176 if (gdraw->texturecache) {
2177 gdraw_res_flush(gdraw->texturecache, &stats);
2178 IggyGDrawFree(gdraw->texturecache);
2179 }
2180 gdraw->texturecache = make_handle_cache(GDRAW_D3D1X_(RESOURCE_texture));
2181 return gdraw->texturecache != nullptr;
2182
2183 case GDRAW_D3D1X_(RESOURCE_vertexbuffer):
2184 if (gdraw->vbufcache) {
