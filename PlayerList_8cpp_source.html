893void PlayerList::repositionAcrossDimension(shared_ptr<Entity> entity, int lastDimension, ServerLevel *oldLevel, ServerLevel *newLevel)
894{
895 double xt = entity->x;
896 double zt = entity->z;
897 double xOriginal = entity->x;

899 double zOriginal = entity->z;
900 float yRotOriginal = entity->yRot;
901 double scale = newLevel->getLevelData()->getHellScale(); // 4J Scale was 8 but this is all we can fit in
902 if (entity->dimension == -1)
903 {
904 xt /= scale;
905 zt /= scale;
906 entity->moveTo(xt, entity->y, zt, entity->yRot, entity->xRot);
907 if (entity->isAlive())
908 {
909 oldLevel->tick(entity, false);
910 }
911 }
912 else if (entity->dimension == 0)
913 {
914 xt *= scale;
915 zt *= scale;
916 entity->moveTo(xt, entity->y, zt, entity->yRot, entity->xRot);
917 if (entity->isAlive())
918 {
919 oldLevel->tick(entity, false);
920 }
921 }
922 else
923 {
924 Pos *p;
925
926 if (lastDimension == 1)
927 {
928 // Coming from the end
929 p = newLevel->getSharedSpawnPos();
930 }
931 else
932 {
933 // Going to the end
934 p = newLevel->getDimensionSpecificSpawn();
935 }
936
937 xt = p->x;
938 entity->y = p->y;
939 zt = p->z;
940 delete p;
941 entity->moveTo(xt, entity->y, zt, 90, 0);
942 if (entity->isAlive())
943 {
944 oldLevel->tick(entity, false);
945 }
946 }

950 shared_ptr<ServerPlayer> player = dynamic_pointer_cast<ServerPlayer>(entity);
951 removePlayerFromReceiving(player, false, lastDimension);
952 addPlayerToReceiving(player);

954
955 if (lastDimension != 1)
956 {
957 xt = static_cast<double>(Mth::clamp(static_cast<int>(xt), -Level::MAX_LEVEL_SIZE + 128, Level::MAX_LEVEL_SIZE - 128));
958 zt = static_cast<double>(Mth::clamp(static_cast<int>(zt), -Level::MAX_LEVEL_SIZE + 128, Level::MAX_LEVEL_SIZE - 128));

960 {
961 newLevel->addEntity(entity);

965 newLevel->getPortalForcer()->force(entity, xOriginal, yOriginal, zOriginal, yRotOriginal);
966 newLevel->cache->autoCreate = false;
967 }
968 }
969
970 entity->setLevel(newLevel);
