390// simple function to load a file completely into memory
391static void * load_file( char const * name )
392{
393 void * ret;
394 int f;
395 unsigned int size;
396
397 f = open( name, O_RDONLY );
398 if ( f == -1 ) return 0;
399
400 size = lseek( f, 0, SEEK_END );
401
402 ret = malloc( size + 127 );
403 if ( ret == 0 )
404 {
405 close( f );
406 return 0;
407 }
408
409 ret = (void*) ( ( ( (unsigned int) ret ) + 127 ) & ~127 );
410
411 lseek( f, 0, SEEK_SET );
412 read( f, ret, size );
413 close( f );
414
415 return ret;
416}
