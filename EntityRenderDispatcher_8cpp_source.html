246void EntityRenderDispatcher::render(shared_ptr<Entity> entity, float a)
247{
248 if (entity == nullptr)
249 {
250 return;
251 }
252
253 double x = entity->xOld + (entity->x - entity->xOld) * a;
254 double y = entity->yOld + (entity->y - entity->yOld) * a;
255 double z = entity->zOld + (entity->z - entity->zOld) * a;
256
257 // Fix for #61057 - TU7: Gameplay: Boat is glitching when player float forward and turning.
258 // Fix to handle the case that yRot and yRotO wrap over the 0/360 line
259 float rotDiff = entity->yRot - entity->yRotO;
260 if( rotDiff > 180 || rotDiff < -180)
261 {
262 if(entity->yRot > entity->yRotO)
263 {
264 rotDiff = (entity->yRot - 360) - entity->yRotO;
265 }
266 else
267 {
268 rotDiff = entity->yRot - (entity->yRotO - 360);
269 }
270 }
271 float r = entity->yRotO + (rotDiff) * a;
272
273 int col = entity->getLightColor(a);
274 if (entity->isOnFire())
275 {
276 col = SharedConstants::FULLBRIGHT_LIGHTVALUE;
277 }
278 int u = col % 65536;
279 int v = col / 65536;
280 glMultiTexCoord2f(GL_TEXTURE1, u / 1.0f, v / 1.0f);
281 glColor4f(1, 1, 1, 1);
282
283 render(entity, x - xOff, y - yOff, z - zOff, r, a);
284}
