3// gdraw_d3d11.cpp - author: Fabian Giesen - copyright 2011 RAD Game Tools
4//
5// This implements the Iggy graphics driver layer for D3D 11.
6
7// GDraw consists of several components that interact fairly loosely with each other;
8// e.g. the resource management, drawing and filtering parts are all fairly independent
9// of each other. If you want to modify some aspect of GDraw - say the texture allocation
10// logic - your best bet is usually to just look for one of the related entry points,
11// e.g. MakeTextureBegin, and take it from there. There's a bunch of code in this file,
12// but none of it is really complicated.
13//
14// The one bit you might want to change that's not that localized is to integrate
15// GDraw with an existing state caching system. The following bits all modify D3D state
16// in some way:
17// - The rendering helpers (set_viewport_raw, set_projection_raw, set_*_renderstate)
18// - RenderTile*/TextureDrawBuffer* may change the active rendertarget and depth/stencil surface,
19// as do D3D1X_(NoMoreGDrawThisFrame) and set_render_target
20// - set_texture
21// - set_renderstate and set_renderstate_full. These are the main places where render state changes occur;
22// you should probably start here.
23// - DrawIndexedTriangles sets the active vertex/index buffers and vertex declaration
24// - Most of the functions in the "filter effects" section modify D3D state, mostly
25// pixel shader constants and textures

87static ID3D11BlendState *create_blend_state(ID3D11Device *dev, BOOL blend, D3D11_BLEND src, D3D11_BLEND dst)
88{
89 D3D11_BLEND_DESC desc = {};
90 desc.RenderTarget[0].BlendEnable = blend;
91 desc.RenderTarget[0].SrcBlend = src;
92 desc.RenderTarget[0].DestBlend = dst;
93 desc.RenderTarget[0].BlendOp = D3D11_BLEND_OP_ADD;
94 desc.RenderTarget[0].SrcBlendAlpha = (src == D3D11_BLEND_DEST_COLOR ) ? D3D11_BLEND_DEST_ALPHA : src;
95 desc.RenderTarget[0].DestBlendAlpha = dst;
96 desc.RenderTarget[0].BlendOpAlpha = D3D11_BLEND_OP_ADD;
97 desc.RenderTarget[0].RenderTargetWriteMask = D3D11_COLOR_WRITE_ENABLE_ALL;
98
99 ID3D11BlendState *res;
100 HRESULT hr = dev->CreateBlendState(&desc, &res);
101 if (FAILED(hr)) {
102 report_d3d_error(hr, "CreateBlendState", "");
103 res = nullptr;
104 }
105
