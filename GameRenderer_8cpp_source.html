857void GameRenderer::updateLightTexture(float a)
858{
859 CachePlayerGammas();
860
861 for (int j = 0; j < XUSER_MAX_COUNT; j++)
862 {
863 shared_ptr<MultiplayerLocalPlayer> player = Minecraft::GetInstance()->localplayers[j];
864 if (player == nullptr)
865 continue;
866
867 Level *level = player->level;
868
869 const float skyDarken1 = level->getSkyDarken(1.0f);
870 for (int i = 0; i < 256; i++)
871 {
872 const float darken = skyDarken1 * 0.95f + 0.05f;
873 float sky = level->dimension->brightnessRamp[i / 16] * darken;
874 const float block = level->dimension->brightnessRamp[i % 16] * (blr * 0.1f + 1.5f);
875
876 if (level->skyFlashTime < 0)
877 {
878 sky = level->dimension->brightnessRamp[i / 16];
879 }
880
881 const float rs = sky * (skyDarken1 * 0.65f + 0.35f);
882 const float gs = sky * (skyDarken1 * 0.65f + 0.35f);
883 const float bs = sky;
884
885 const float rb = block;
886 const float gb = block * ((block * 0.6f + 0.4f) * 0.6f + 0.4f);
887 const float bb = block * ((block * block) * 0.6f + 0.4f);
888
889 float _r = (rs + rb);
890 float _g = (gs + gb);
891 float _b = (bs + bb);
892
893 _r = _r * 0.96f + 0.03f;
894 _g = _g * 0.96f + 0.03f;
895 _b = _b * 0.96f + 0.03f;
896
897 if (darkenWorldAmount > 0)
898 {
899 const float amount = darkenWorldAmountO + (darkenWorldAmount - darkenWorldAmountO) * a;
900 _r = _r * (1.0f - amount) + (_r * 0.7f) * amount;
901 _g = _g * (1.0f - amount) + (_g * 0.6f) * amount;
902 _b = _b * (1.0f - amount) + (_b * 0.6f) * amount;
903 }
904
905 if (level->dimension->id == 1)
906 {
907 _r = (0.22f + rb * 0.75f);
908 _g = (0.28f + gb * 0.75f);
909 _b = (0.25f + bb * 0.75f);
910 }
911
912 if (player->hasEffect(MobEffect::nightVision))
913 {
914 const float scale = getNightVisionScale(player, a);
915 float dist = 1.0f / _r;
916 if (dist > (1.0f / _g))
917 dist = (1.0f / _g);
918 if (dist > (1.0f / _b))
919 dist = (1.0f / _b);
920 _r = _r * (1.0f - scale) + (_r * dist) * scale;
921 _g = _g * (1.0f - scale) + (_g * dist) * scale;
922 _b = _b * (1.0f - scale) + (_b * dist) * scale;
923 }
924
925 if (_r > 1.0f)
926 _r = 1.0f;
927 if (_r < 0.0f)
928 _r = 0.0f;
929 if (_g > 1.0f)
930 _g = 1.0f;
931 if (_g < 0.0f)
932 _g = 0.0f;
933 if (_b > 1.0f)
934 _b = 1.0f;
935 if (_b < 0.0f)
936 _b = 0.0f;
937
938 constexpr int alpha = 255;
939 const int r = static_cast<int>(_r * 255);
940 const int g = static_cast<int>(_g * 255);
941 const int b = static_cast<int>(_b * 255);
942
943#if ( defined _DURANGO || defined _WIN64 || __PSVITA__ )
944 lightPixels[j][i] = alpha << 24 | b << 16 | g << 8 | r;
945#elif ( defined _XBOX || defined __ORBIS__ )
946 lightPixels[j][i] = alpha << 24 | r << 16 | g << 8 | b;
947#else
948 lightPixels[j][i] = r << 24 | g << 16 | b << 8 | alpha;
949#endif
950 }
951
952 mc->textures->replaceTextureDirect(lightPixels[j], 16, 16, getLightTexture(j, level));
953 }
