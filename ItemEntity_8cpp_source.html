149bool ItemEntity::merge(shared_ptr<ItemEntity> target)
150{
151 if (target == shared_from_this()) return false;
152 if (!target->isAlive() || !this->isAlive()) return false;
153 shared_ptr<ItemInstance> myItem = this->getItem();
154 shared_ptr<ItemInstance> targetItem = target->getItem();
155
156 if (targetItem->getItem() != myItem->getItem()) return false;
157 if (targetItem->hasTag() ^ myItem->hasTag()) return false;
158 if (targetItem->hasTag() && !targetItem->getTag()->equals(myItem->getTag())) return false;
159 if (targetItem->getItem()->isStackedByData() && targetItem->getAuxValue() != myItem->getAuxValue()) return false;
160 if (targetItem->count < myItem->count) return target->merge(dynamic_pointer_cast<ItemEntity>(shared_from_this()));
161 if (targetItem->count + myItem->count > targetItem->getMaxStackSize()) return false;
162
163 targetItem->count += myItem->count;
164 target->throwTime = max(target->throwTime, this->throwTime);
165 target->age = min(target->age, this->age);
166 target->setItem(targetItem);
167 remove();
168
169 return true;
170}

190bool ItemEntity::hurt(DamageSource *source, float damage)
191{
192 // 4J - added next line: found whilst debugging an issue with item entities getting into a bad state when being created by a cactus, since entities insides cactuses get hurt

195 if (level->isClientSide ) return false;
196
197 if (isInvulnerable()) return false;
198 if (getItem() != nullptr && getItem()->id == Item::netherStar_Id && source->isExplosion()) return false;
