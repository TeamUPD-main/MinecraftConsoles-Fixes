105void JukeboxTile::dropRecording(Level *level, int x, int y, int z)
106{
107 if (level->isClientSide) return;
108
109 shared_ptr<JukeboxTile::Entity> rte = dynamic_pointer_cast<JukeboxTile::Entity>( level->getTileEntity(x, y, z) );
110 if( rte == nullptr ) return;
111
112 shared_ptr<ItemInstance> oldRecord = rte->getRecord();
113 if (oldRecord == nullptr) return;
114
115
116 level->levelEvent(LevelEvent::SOUND_PLAY_RECORDING, x, y, z, 0);
117 // 4J-PB- the level event will play the music
118 //level->playStreamingMusic(L"", x, y, z);
119 rte->setRecord(nullptr);
120 rte->setChanged();
121 level->setData(x, y, z, 0, Tile::UPDATE_CLIENTS);
122
123 float s = 0.7f;
124 double xo = level->random->nextFloat() * s + (1 - s) * 0.5;
125 double yo = level->random->nextFloat() * s + (1 - s) * 0.2 + 0.6;
126 double zo = level->random->nextFloat() * s + (1 - s) * 0.5;
127
128 shared_ptr<ItemInstance> itemInstance = oldRecord->copy();
129
130 shared_ptr<ItemEntity> item = std::make_shared<ItemEntity>(level, x + xo, y + yo, z + zo, itemInstance);
131 item->throwTime = 10;
132 level->addEntity(item);
