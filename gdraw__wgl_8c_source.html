121GDrawFunctions *gdraw_GL_CreateContext(S32 w, S32 h, S32 msaa_samples)
122{
123 static const TextureFormatDesc tex_formats[] = {
124 { IFT_FORMAT_rgba_8888, 1, 1, 4, GL_RGBA, GL_RGBA, GL_UNSIGNED_BYTE },
125 { IFT_FORMAT_rgba_4444_LE, 1, 1, 2, GL_RGBA4, GL_RGBA, GL_UNSIGNED_SHORT_4_4_4_4 },
126 { IFT_FORMAT_rgba_5551_LE, 1, 1, 2, GL_RGB5_A1, GL_RGBA, GL_UNSIGNED_SHORT_5_5_5_1 },
127 { IFT_FORMAT_la_88, 1, 1, 2, GL_LUMINANCE8_ALPHA8, GL_LUMINANCE_ALPHA, GL_UNSIGNED_BYTE },
128 { IFT_FORMAT_la_44, 1, 1, 1, GL_LUMINANCE4_ALPHA4, GL_LUMINANCE_ALPHA, GL_UNSIGNED_BYTE },
129 { IFT_FORMAT_i_8, 1, 1, 1, GL_INTENSITY8, GL_ALPHA, GL_UNSIGNED_BYTE },
130 { IFT_FORMAT_i_4, 1, 1, 1, GL_INTENSITY4, GL_ALPHA, GL_UNSIGNED_BYTE },
131 { IFT_FORMAT_l_8, 1, 1, 1, GL_LUMINANCE8, GL_LUMINANCE, GL_UNSIGNED_BYTE },
132 { IFT_FORMAT_l_4, 1, 1, 1, GL_LUMINANCE4, GL_LUMINANCE, GL_UNSIGNED_BYTE },
133 { IFT_FORMAT_DXT1, 4, 4, 8, GL_COMPRESSED_RGBA_S3TC_DXT1_EXT, 0, GL_UNSIGNED_BYTE },
134 { IFT_FORMAT_DXT3, 4, 4, 16, GL_COMPRESSED_RGBA_S3TC_DXT3_EXT, 0, GL_UNSIGNED_BYTE },
135 { IFT_FORMAT_DXT5, 4, 4, 16, GL_COMPRESSED_RGBA_S3TC_DXT5_EXT, 0, GL_UNSIGNED_BYTE },
136 { 0, 0, 0, 0, 0, 0, 0 },
137 };
138
139 GDrawFunctions *funcs;
140 const char *s;
141 GLint n;
142
143 // check for the extensions we need
144 s = (const char *) glGetString(GL_EXTENSIONS);
145 if (s == NULL) {
146 assert(s != NULL); // if this is NULL, you're probably trying to create the device too early
147 return NULL;
148 }
149
150 // check for the extensions we won't work without
151 if (!hasext(s, "GL_ARB_multitexture") ||
152 !hasext(s, "GL_ARB_texture_compression") ||
153 !hasext(s, "GL_ARB_texture_mirrored_repeat") ||
154 !hasext(s, "GL_ARB_texture_non_power_of_two") || // with caveats - see below!
155 !hasext(s, "GL_ARB_vertex_buffer_object") ||
156 !hasext(s, "GL_EXT_framebuffer_object") ||
157 !hasext(s, "GL_ARB_shader_objects") ||
158 !hasext(s, "GL_ARB_vertex_shader") ||
159 !hasext(s, "GL_ARB_fragment_shader"))
160 return NULL;
161
162 // if user requests multisampling and HW doesn't support it, bail
163 if (!hasext(s, "GL_EXT_framebuffer_multisample") && msaa_samples > 1)
164 return NULL;
165
166 load_extensions();
167 funcs = create_context(w, h);
168 if (!funcs)
169 return NULL;
170
171 gdraw->tex_formats = tex_formats;
172
173 // check for optional extensions
174 gdraw->has_mapbuffer = true; // part of core VBO extension on regular GL
175 gdraw->has_depth24 = true; // we just assume.
176 gdraw->has_texture_max_level = true; // core on regular GL
177
178 if (hasext(s, "GL_EXT_packed_depth_stencil")) gdraw->has_packed_depth_stencil = true;
179
180 // we require ARB_texture_non_power_of_two - on actual HW, this may either give us
181 // "full" non-power-of-two support, or "conditional" non-power-of-two (wrap mode must
182 // be CLAMP_TO_EDGE, no mipmaps). figure out which it is using this heuristic by
183 // Unity's Aras Pranckevicius (thanks!):
184 // http://www.aras-p.info/blog/2012/10/17/non-power-of-two-textures/
185 //
186 // we use the second heuristic (texture size <8192 for cards without full NPOT support)
187 // since we don't otherwise use ARB_fragment_program and don't want to create a program
188 // just to be able to query MAX_PROGRAM_NATIVE_INSTRUCTIONS_ARB!
189 glGetIntegerv(GL_MAX_TEXTURE_SIZE, &n);
190 gdraw->has_conditional_non_power_of_two = n < 8192;
191
192 // clamp number of multisampling levels to max supported
193 if (msaa_samples > 1) {
194 glGetIntegerv(GL_MAX_SAMPLES, &n);
195 gdraw->multisampling = RR_MIN(msaa_samples, n);
196 }
197
198 opengl_check();
199
200 return funcs;
201}
