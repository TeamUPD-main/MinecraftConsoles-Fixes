97 bool Initialize(const std::string &baseDirectory, bool whitelistEnabled)
98 {
99 std::lock_guard<std::mutex> writeLock(g_accessState.writeLock);
100
101 // Build the replacement manager privately so readers keep using the last published snapshot during disk I/O.
102 std::shared_ptr<BanManager> banManager = std::make_shared<BanManager>(baseDirectory);
103 std::shared_ptr<WhitelistManager> whitelistManager = std::make_shared<WhitelistManager>(baseDirectory);
104 if (!banManager->EnsureBanFilesExist())

106 LogError("access", "failed to ensure dedicated server ban files exist");

108 }
109 if (!whitelistManager->EnsureWhitelistFileExists())
110 {
111 LogError("access", "failed to ensure dedicated server whitelist file exists");
112 return false;
113 }
114
115 if (!banManager->Reload())
116 {
117 LogError("access", "failed to load dedicated server ban files");
118 return false;
119 }
120 if (!whitelistManager->Reload())
121 {
122 LogError("access", "failed to load dedicated server whitelist file");
123 return false;
124 }
125
126 std::vector<BannedPlayerEntry> playerEntries;
127 std::vector<BannedIpEntry> ipEntries;
128 std::vector<WhitelistedPlayerEntry> whitelistEntries;
129 banManager->SnapshotBannedPlayers(&playerEntries);
130 banManager->SnapshotBannedIps(&ipEntries);
131 whitelistManager->SnapshotWhitelistedPlayers(&whitelistEntries);
132 PublishBanManagerSnapshot(banManager);
133 PublishWhitelistManagerSnapshot(whitelistManager);
134 {
135 std::lock_guard<std::mutex> stateLock(g_accessState.stateLock);
136 g_accessState.whitelistEnabled = whitelistEnabled;
137 }
138
139 LogInfof(
140 "access",
141 "loaded %u player bans, %u ip bans, and %u whitelist entries (whitelist=%s)",
142 (unsigned)playerEntries.size(),
143 (unsigned)ipEntries.size(),
144 (unsigned)whitelistEntries.size(),
145 whitelistEnabled ? "enabled" : "disabled");
146 return true;
