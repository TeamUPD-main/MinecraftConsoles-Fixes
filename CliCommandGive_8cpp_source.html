35 bool CliCommandGive::Execute(const ServerCliParsedLine &line, ServerCliEngine *engine)
36 {

38 {
39 engine->LogWarn(std::string("Usage: ") + kGiveUsage);

42

47 return false;

49
50 int itemId = 0;

64 {

66 return false;

68 if (line.tokens.size() >= 5 && !CommandParsing::TryParseInt(line.tokens[4], &aux))
69 {
70 engine->LogWarn("Invalid aux value: " + line.tokens[4]);
71 return false;
72 }
73 if (amount <= 0)
74 {
75 engine->LogWarn("Amount must be greater than 0.");
76 return false;
77 }
78
79 std::shared_ptr<Player> player = std::dynamic_pointer_cast<Player>(target);
80 if (player == nullptr)
81 {
82 engine->LogWarn("Cannot resolve target player entity.");
